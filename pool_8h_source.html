153 /// Create a PooledSet wih all elements in the given range.
154 template<class I> [[nodiscard]] PooledSet<T> create(I begin, I end) {
155 auto size = std::distance(begin, end); // max space needed - may be less; see actual_size below
156 auto [data, state] = allocate(size);
157 auto db = data->elems, de = data->elems + size;
158
159 std::copy(begin, end, db);
160 std::sort(db, de, GIDLt<T>());
161 auto di = std::unique(db, de);
162 auto actual_size = std::distance(db, di);
163 data->size = actual_size; // correct size
164 return unify(data, state, size - actual_size);
165 }

167 /// Yields @f$a \cup b@f$.
168 [[nodiscard]] PooledSet<T> merge(PooledSet<T> a, PooledSet<T> b) {
169 if (a == b || !b) return a;
170 if (!a) return b;
171
172 auto size = a.size() + b.size(); // max space needed - may be less; see actual_size below
173 auto [data, state] = allocate(size);
174
175 auto di = data->elems;
176 auto ai = a.begin(), ae = a.end(), bi = b.begin(), be = b.end();
177
178 *di = (*ai)->gid() < (*bi)->gid() ? *ai++ : *bi++; // di points to the last valid elem, so init first one
179
180 while (ai != ae && bi != be)
181 if ((*ai)->gid() < (*bi)->gid())
182 copy_if_unique_and_inc(di, ai);
183 else
184 copy_if_unique_and_inc(di, bi);
185
186 // copy the rest of a/b (if needed):
187 while (ai != ae) copy_if_unique_and_inc(di, ai);
188 while (bi != be) copy_if_unique_and_inc(di, bi);
189
190 auto actual_size = std::distance(data->elems, di + 1);
191 data->size = actual_size; // correct size
192 return unify(data, state, size - actual_size);
193 }

195 /// Yields @f$a \cap b@f$.
196 [[nodiscard]] PooledSet<T> intersect(PooledSet<T> a, PooledSet<T> b) {
197 if (a == b) return a;
198 if (!a || !b) return {};
199
200 auto size = std::min(a.size(), b.size()); // max space needed - may be less; see actual_size below
201 auto [data, state] = allocate(size);
202
203 auto di = data->elems;
204 for (auto ai = a.begin(), ae = a.end(), bi = b.begin(), be = b.end(); ai != ae && bi != be;)
205 if (*ai == *bi)
206 *di++ = ++ai, *bi++;
207 else if ((*ai)->gid() < (*bi)->gid())
208 ++ai;
209 else
210 ++bi;
211
212 auto actual_size = std::distance(data->elems, di);
213 data->size = actual_size; // correct size
214 return unify(data, state, size - actual_size);
215 }

220 /// Yields @f$a \setminus elem@f$.
221 [[nodiscard]] PooledSet<T> erase(PooledSet<T> a, const T& elem) {
222 if (!a) return a;
223 auto i = binary_find(a.begin(), a.end(), elem, GIDLt<T>());
224 if (i == a.end()) return a;
225
226 auto size = a.size() - 1;
227 if (size == 0) return PooledSet<T>(); // empty Set is not hashed
228
229 auto [data, state] = allocate(size);
230 std::copy(i + 1, a.end(), std::copy(a.begin(), i, data->elems)); // copy over, skip i
231 return unify(data, state);
232 }

251 PooledSet<T> unify(Data* data, fe::Arena::State state, size_t excess = 0) {
252 auto [i, ins] = pool_.emplace(data);
253 if (ins) {
254 ++size_;
255 arena_.deallocate(excess * SizeOf<T>); // release excess memory
256 return PooledSet<T>(data);
257 }
258
259 arena_.deallocate(state);
260 return PooledSet<T>(*i);
261 }

265 ++ai;
266 }
