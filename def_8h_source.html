258 /// Splits this Def via Def::proj%ections into an Arr%ay (if `A == -1_s`) or `std::array` (otherwise).
259 /// Applies @p f to each element.
260 /// ```
261 /// std::array<const Def*, 2> ab = def->projs<2>();
262 /// std::array<u64, 2> xy = def->projs<2>(as_lit<nat_t>);
263 /// auto [a, b] = def->projs<2>();
264 /// auto [x, y] = def->projs<2>(as_lit<nat_t>);
265 /// Array<const Def*> projs = def->projs(); // projs has def->num_projs() many elements
266 /// Array<const Def*> projs = def->projs(n); // projs has n elements - asserts if incorrect
267 /// Array<const Lit*> lits = def->projs(as_lit<nat_t>); // same as above but applies as_lit<nat_t> to each element
268 /// Array<const Lit*> lits = def->projs(n, as_lit<nat_t>);// same as above but applies as_lit<nat_t> to each element
269 /// ```
270 template<nat_t A = -1_s, class F>
271 auto projs(F f, Defs dbgs = {}) const {
272 using R = std::decay_t<decltype(f(this))>;
273 if constexpr (A == -1_s) {
274 return projs(num_projs(), f, dbgs);
275 } else {
276 assert(A == as_lit(arity()));
277 std::array<R, A> array;
278 for (nat_t i = 0; i != A; ++i) array[i] = f(proj(A, i, dbgs.empty() ? nullptr : dbgs[i]));
279 return array;
280 }
