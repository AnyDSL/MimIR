308 /// @name proj
309 ///@{
310 /// Splits this Def via Extract%s or directly accessing the Def::ops in the case of Sigma%s or Arr%ays.
311
312 /// Yields Def::arity as_lit, if it is in fact a Lit, or `1` otherwise.
313 nat_t num_projs() const {
314 if (auto a = isa_lit_arity()) return *a;
315 return 1;

317 /// Similar to World::extract while assuming an arity of @p a but also works on Sigma%s, and Arr%ays.
318 /// If `this` is a Sort::Term (see Def::sort), Def::proj resorts to World::extract.
319 const Def* proj(nat_t a, nat_t i) const;

324 /// Splits this Def via Def::proj%ections into an Arr%ay (if `A == -1_s`) or `std::array` (otherwise).
325 /// Applies @p f to each element.
326 /// ```
327 /// std::array<const Def*, 2> ab = def->projs<2>();
328 /// std::array<u64, 2> xy = def->projs<2>(as_lit<nat_t>);
329 /// auto [a, b] = def->projs<2>();
330 /// auto [x, y] = def->projs<2>(as_lit<nat_t>);
331 /// Array<const Def*> projs = def->projs(); // projs has def->num_projs() many elements
332 /// Array<const Def*> projs = def->projs(n); // projs has n elements - asserts if incorrect
333 /// Array<const Lit*> lits = def->projs(as_lit<nat_t>); // same as above but applies as_lit<nat_t> to each element
334 /// Array<const Lit*> lits = def->projs(n, as_lit<nat_t>);// same as above but applies as_lit<nat_t> to each element
335 /// ```
336 template<nat_t A = -1_s, class F>
337 auto projs(F f) const {
338 using R = std::decay_t<decltype(f(this))>;
339 if constexpr (A == -1_s) {
340 return projs(num_projs(), f);
341 } else {
342 assert(A == as_lit_arity());
343 std::array<R, A> array;
344 for (nat_t i = 0; i != A; ++i) array[i] = f(proj(A, i));
345 return array;
346 }
347 }

351 using R = std::decay_t<decltype(f(this))>;
352 return Array<R>(a, [&](nat_t i) { return f(proj(a, i)); });
353 }
