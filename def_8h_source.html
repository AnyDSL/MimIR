353 /// Splits this Def via Def::proj%ections into an Array (if `A == -1_n`) or `std::array` (otherwise).
354 /// Applies @p f to each element.
355 template<nat_t A = -1_n, class F> auto projs(F f) const {
356 using R = std::decay_t<decltype(f(this))>;
357 if constexpr (A == -1_n) {
358 return projs(num_projs(), f);
359 } else {
360 assert(A == as_lit_arity());
361 std::array<R, A> array;
362 for (nat_t i = 0; i != A; ++i) array[i] = f(proj(A, i));
363 return array;
364 }
365 }
