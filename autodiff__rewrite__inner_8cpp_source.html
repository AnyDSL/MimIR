32const Def* AutoDiffEval::augment_lam(Lam* lam, Lam* f, Lam* f_diff) {
33 auto& world = lam->world();
34 // TODO: we need partial pullbacks for tuples (higher-order / ret-cont application)
35 // also for higher-order args, ret_cont (at another point)
36 // the pullback is not important but formally required by tuple rule
37 if (augmented.count(lam)) {
38 // We already know the function:
39 // * recursion
40 // * higher order arguments
41 // * new encounter of previous function
42 world.DLOG("already augmented {} : {} to {} : {}", lam, lam->type(), augmented[lam], augmented[lam]->type());
43 return augmented[lam];
44 }
45 // TODO: better fix (another pass as analysis?)
46 // TODO: handle open functions
47 if (is_open_continuation(lam) || lam->name().find("ret") != std::string::npos ||
48 lam->name().find("_cont") != std::string::npos) {
49 // A open continuation behaves the same as return:
50 // ```
51 // cont: Cn[X]
52 // cont': Cn[X,Cn[X,A]]
53 // ```
54 // There is dependency on the closed function context.
55 // (All derivatives are with respect to the arguments of a closed function.)
56
57 world.DLOG("found an open continuation {} : {}", lam, lam->type());
58 auto cont_dom = lam->type()->dom(); // not only 0 but all

60 auto aug_dom = autodiff_type_fun(cont_dom);
61 world.DLOG("augmented domain {}", aug_dom);
62 world.DLOG("pb type is {}", pb_ty);

64 world.DLOG("augmented type is {}", aug_ty);
65 auto aug_lam = world.nom_lam(aug_ty, world.dbg("aug_" + lam->name()));
66 auto aug_var = aug_lam->var((nat_t)0);
67 augmented[lam->var()] = aug_var;

69 derived[lam] = aug_lam;
70 auto pb = aug_lam->var(1);
71 partial_pullback[aug_var] = pb;
72 // We are still in same closed function.
73 auto new_body = augment(lam->body(), f, f_diff);
74 aug_lam->set_filter(lam->filter());
75 aug_lam->set_body(new_body);
76

78 partial_pullback[aug_lam] = lam_pb;
