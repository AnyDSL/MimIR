30Ref AutoDiffEval::augment_lam(Lam* lam, Lam* f, Lam* f_diff) {
31 auto& world = lam->world();
32 // TODO: we need partial pullbacks for tuples (higher-order / ret-cont application)
33 // also for higher-order args, ret_cont (at another point)
34 // the pullback is not important but formally required by tuple rule
35 if (augmented.count(lam)) {
36 // We already know the function:
37 // * recursion
38 // * higher order arguments
39 // * new encounter of previous function
40 world.DLOG("already augmented {} : {} to {} : {}", lam, lam->type(), augmented[lam], augmented[lam]->type());
41 return augmented[lam];
42 }
43 // TODO: better fix (another pass as analysis?)
44 // TODO: handle open functions
45 if (Lam::isa_basicblock(lam) || lam->sym()->find("ret") != std::string::npos
46 || lam->sym()->find("_cont") != std::string::npos) {
47 // A open continuation behaves the same as return:
48 // ```
49 // cont: Cn[X]
50 // cont': Cn[X,Cn[X,A]]
51 // ```
52 // There is dependency on the closed function context.
53 // (All derivatives are with respect to the arguments of a closed function.)
54
55 world.DLOG("found an open continuation {} : {}", lam, lam->type());
56 auto cont_dom = lam->type()->dom(); // not only 0 but all
57 auto pb_ty = pullback_type(cont_dom, f->dom(2, 0));
58 auto aug_dom = autodiff_type_fun(cont_dom);
59 world.DLOG("augmented domain {}", aug_dom);
60 world.DLOG("pb type is {}", pb_ty);
61 auto aug_ty = world.cn({aug_dom, pb_ty});
62 world.DLOG("augmented type is {}", aug_ty);
63 auto aug_lam = world.mut_lam(aug_ty)->set("aug_"s + *lam->sym());
64 auto aug_var = aug_lam->var((nat_t)0);
65 augmented[lam->var()] = aug_var;
66 augmented[lam] = aug_lam; // TODO: only one of these two
67 derived[lam] = aug_lam;
68 auto pb = aug_lam->var(1);
69 partial_pullback[aug_var] = pb;
70 // We are still in same closed function.
71 auto new_body = augment(lam->body(), f, f_diff);
72 // TODO we also need to rewrite the filter
73 aug_lam->set(lam->filter(), new_body);
74
75 auto lam_pb = zero_pullback(lam->type(), f->dom(2, 0));
76 partial_pullback[aug_lam] = lam_pb;
77 world.DLOG("augmented {} : {}", lam, lam->type());

82 }
