34Ref AutoDiffEval::augment_lam(Lam* lam, Lam* f, Lam* f_diff) {
35 auto& world = lam->world();
36 // TODO: we need partial pullbacks for tuples (higher-order / ret-cont application)
37 // also for higher-order args, ret_cont (at another point)
38 // the pullback is not important but formally required by tuple rule
39 if (augmented.count(lam)) {
40 // We already know the function:
41 // * recursion
42 // * higher order arguments
43 // * new encounter of previous function
44 world.DLOG("already augmented {} : {} to {} : {}", lam, lam->type(), augmented[lam], augmented[lam]->type());
45 return augmented[lam];
46 }
47 // TODO: better fix (another pass as analysis?)
48 // TODO: handle open functions
49 if (is_open_continuation(lam) || lam->sym()->find("ret") != std::string::npos ||
50 lam->sym()->find("_cont") != std::string::npos) {
51 // A open continuation behaves the same as return:
52 // ```
53 // cont: Cn[X]
54 // cont': Cn[X,Cn[X,A]]
55 // ```
56 // There is dependency on the closed function context.
57 // (All derivatives are with respect to the arguments of a closed function.)
58
59 world.DLOG("found an open continuation {} : {}", lam, lam->type());
60 auto cont_dom = lam->type()->dom(); // not only 0 but all
61 auto pb_ty = pullback_type(cont_dom, f_arg_ty);
62 auto aug_dom = autodiff_type_fun(cont_dom);
63 world.DLOG("augmented domain {}", aug_dom);
64 world.DLOG("pb type is {}", pb_ty);
65 auto aug_ty = world.cn({aug_dom, pb_ty});
66 world.DLOG("augmented type is {}", aug_ty);
67 auto aug_lam = world.mut_lam(aug_ty)->set("aug_"s + *lam->sym());
68 auto aug_var = aug_lam->var((nat_t)0);
69 augmented[lam->var()] = aug_var;
70 augmented[lam] = aug_lam; // TODO: only one of these two
71 derived[lam] = aug_lam;

74 // We are still in same closed function.
