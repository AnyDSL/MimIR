108/// Returns the (first) element of type mem::M from the given tuple.
109static Ref mem_def(Ref def) {
110 if (match<mem::M>(def->type())) return def;

112 if (def->num_projs() > 1) {

136/// Removes recusively all occurences of mem from a type (sigma).
137static Ref strip_mem_ty(Ref def) {
138 auto& world = def->world();
139
140 if (auto sigma = def->isa<Sigma>()) {
141 DefVec new_ops;
142 for (auto op : sigma->ops())
143 if (auto new_op = strip_mem_ty(op); new_op != world.sigma()) new_ops.push_back(new_op);
144
145 return world.sigma(new_ops);
146 } else if (match<mem::M>(def)) {
147 return world.sigma();
148 }
149
150 return def;
151}

153/// Removes recusively all occurences of mem from a tuple.
154/// Returns an empty tuple if applied with mem alone.
155static Ref strip_mem(Ref def) {
156 auto& world = def->world();
157
158 if (auto tuple = def->isa<Tuple>()) {
159 DefVec new_ops;
160 for (auto op : tuple->ops())
161 if (auto new_op = strip_mem(op); new_op != world.tuple()) new_ops.push_back(new_op);
162
163 return world.tuple(new_ops);
164 } else if (match<mem::M>(def->type())) {
165 return world.tuple();
166 } else if (auto extract = def->isa<Extract>()) {
167 // The case that this one element is a mem and should return () is handled above.
168 if (extract->num_projs() == 1) return extract;
169
170 DefVec new_ops;
171 for (auto op : extract->projs())
172 if (auto new_op = strip_mem(op); new_op != world.tuple()) new_ops.push_back(new_op);
173
174 return world.tuple(new_ops);
175 }
176
177 return def;
178}
