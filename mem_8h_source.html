33/// Returns the (first) element of type mem::M from the given tuple.
34inline Ref mem_def(Ref def) {
35 if (match<mem::M>(def->type())) return def;
36 if (def->type()->isa<Arr>()) return {}; // don't look into possibly gigantic arrays

62/// Removes recusively all occurences of mem from a type (sigma).
63inline Ref strip_mem_ty(Ref def) {
64 auto& world = def->world();
65
66 if (auto sigma = def->isa<Sigma>()) {

79/// Recursively removes all occurrences of mem from a tuple.
80/// Returns an empty tuple if applied with mem alone.
81inline Ref strip_mem(Ref def) {
82 auto& world = def->world();
83
84 if (auto tuple = def->isa<Tuple>()) {
85 DefVec new_ops;
86 for (auto op : tuple->ops())
87 if (auto new_op = strip_mem(op); new_op != world.tuple()) new_ops.push_back(new_op);
88
89 return world.tuple(new_ops);
90 } else if (match<mem::M>(def->type())) {
91 return world.tuple();
92 } else if (auto extract = def->isa<Extract>()) {
93 // The case that this one element is a mem and should return () is handled above.
94 if (extract->num_projs() == 1) return extract;
95
96 DefVec new_ops;
97 for (auto op : extract->projs())
98 if (auto new_op = strip_mem(op); new_op != world.tuple()) new_ops.push_back(new_op);
99
100 return world.tuple(new_ops);
101 }
102
103 return def;
104}
