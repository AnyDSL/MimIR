17bool Def::unwrap() const {

77Stream& Def::unwrap(Stream& s) const {
78 if (false) {
79 } else if (isa<Space>())
80 return s.fmt("□");
81 else if (isa<Kind>())
82 return s.fmt("★");
83 else if (isa<Nat>())
84 return s.fmt("nat");
85 else if (auto bot = isa<Bot>())
86 return s.fmt("⊥∷{}", bot->type());
87 else if (auto top = isa<Top>())
88 return s.fmt("⊤∷{}", top->type());
89 else if (auto axiom = isa<Axiom>())
90 return s.fmt(":{}", axiom->debug().name);
91 else if (auto lit = isa<Lit>()) {
92 if (auto real = thorin::isa<Tag::Real>(lit->type())) {
93 switch (as_lit(real->arg())) {
94 case 16: return s.fmt("{}∷r16", lit->get<r16>());
95 case 32: return s.fmt("{}∷r32", lit->get<r32>());
96 case 64: return s.fmt("{}∷r64", lit->get<r64>());
97 default: unreachable();
98 }
99 }
100 return s.fmt("{}∷{}", lit->get(), lit->type());
101 } else if (auto ex = isa<Extract>()) {
102 if (ex->tuple()->isa<Var>() && ex->index()->isa<Lit>()) return s.fmt("{}", ex->unique_name());
103 return s.fmt("{}#{}", ex->tuple(), ex->index());
104 } else if (auto var = isa<Var>()) {
105 if (var->nom()->num_vars() == 1) return s.fmt("{}", var->unique_name());
106 return s.fmt("@{}", var->nom());
107 } else if (auto pi = isa<Pi>()) {
108 if (pi->is_cn()) {
109 return s.fmt("Cn {}", pi->dom());
110 } else {
111 return s.fmt("{} -> {}", pi->dom(), pi->codom());
112 }
113 } else if (auto lam = isa<Lam>()) {
114 return s.fmt("λ@({}) {}", lam->filter(), lam->body());
115 } else if (auto app = isa<App>()) {
116 if (auto size = isa_lit(isa_sized_type(app))) {
117 if (auto real = thorin::isa<Tag::Real>(app)) return s.fmt("r{}", *size);
118 if (auto _int = thorin::isa<Tag::Int>(app)) {
119 if (auto width = mod2width(*size)) return s.fmt("i{}", *width);
120
121 // append utf-8 subscripts in reverse order
122 std::string str;
123 for (size_t mod = *size; mod > 0; mod /= 10)
124 ((str += char(char(0x80) + char(mod % 10))) += char(0x82)) += char(0xe2);
125 std::ranges::reverse(str);
126
127 return s.fmt("i{}", str);
128 }
129 unreachable();
130 }
131 return s.fmt("{} {}", LPrec(app->callee(), app), RPrec(app, app->arg()));
132 } else if (auto sigma = isa<Sigma>()) {
133 return s.fmt("[{, }]", sigma->ops());
134 } else if (auto tuple = isa<Tuple>()) {
135 s.fmt("({, })", tuple->ops());
136 return tuple->type()->isa_nom() ? s.fmt("∷{}", tuple->type()) : s;
137 } else if (auto arr = isa<Arr>()) {
138 return s.fmt("«{}; {}»", arr->shape(), arr->body());
139 } else if (auto pack = isa<Pack>()) {
140 return s.fmt("‹{}; {}›", pack->shape(), pack->body());
141 } else if (auto proxy = isa<Proxy>()) {
142 return s.fmt(".proxy#{}#{} {, }", proxy->index(), proxy->flags(), proxy->ops());
143 } else if (auto bound = isa_bound(this)) {
144 auto op = bound->isa<Join>() ? "∪" : "∩";
145 if (isa_nom()) s.fmt("{}{}: {}", op, unique_name(), type());
146 return s.fmt("{}({, })", op, ops());
147 }
148
149 // other
150 if (fields() == 0) return s.fmt(".{} {, }", node_name(), ops());
151 return s.fmt(".{}#{} {, }", node_name(), fields(), ops());
152}

171void RecStreamer::run(const Def* def) {
172 if (!defs.emplace(def).second) return;
173
174 for (auto op : def->ops()) { // for now, don't include debug info and type
175 if (auto nom = op->isa_nom()) {
176 if (max != 0) {
177 if (noms.push(nom)) --max;
178 }
179 } else {
180 run(op);
181 }
182 }
183
184 if (auto nom = def->isa_nom())
185 nom->unwrap(s.endl());
186 else if (!def->unwrap())
187 def->let(s.endl());
188}

190void RecStreamer::run() {
191 while (!noms.empty()) {
192 auto nom = noms.pop();
193 s.endl().endl();
194
195 if (nom->is_set()) {
196 s.fmt("{}: {}", nom->unique_name(), nom->type());
197 if (nom->has_var()) {
198 auto e = nom->num_vars();
199 s.fmt(": {}", e == 1 ? "" : "(");
200 s.range(nom->vars(), [&](const Def* def) { s << def->unique_name(); });
201 if (e != 1) s.fmt(")");
202 }
203 s.fmt(" = {{\t");
204 run(nom);
205 s.fmt("\b\n}}");
206 } else {
207 s.fmt("{}: {} = {{ <unset> }}", nom->unique_name(), nom->type());
208 }
209 }
210}
