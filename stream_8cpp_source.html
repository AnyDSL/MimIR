171void RecStreamer::run(const Def* def) {
172 if (!defs.emplace(def).second) return;
173
174 for (auto op : def->ops()) { // for now, don't include debug info and type
175 if (auto nom = op->isa_nom()) {
176 if (max != 0) {
177 if (noms.push(nom)) --max;
178 }
179 } else {
180 run(op);
181 }
182 }
183
184 if (auto nom = def->isa_nom()) {
185 tab.print(os << std::endl, "{}", Unwrap(nom));
186 } else if (!Unwrap(def)) {
187 def->let(os, tab);
188 }
189}

191void RecStreamer::run() {
192 while (!noms.empty()) {
193 auto nom = noms.pop();
194 os << std::endl << std::endl;
195
196 if (nom->is_set()) {
197 tab.print(os, "{}: {}", nom->unique_name(), nom->type());
198 if (nom->has_var()) {
199 auto e = nom->num_vars();
200 print(os, ": {}", e == 1 ? "" : "(");
201 range(os, nom->vars(), [&](auto def) { os << def->unique_name(); });
202 if (e != 1) print(os, ")");
203 }
204 print(os, " = {{");
205 ++tab;
206 run(nom);
207 --tab;
208 tab.print(os << std::endl, "}}");
209 } else {
210 tab.print(os, "{}: {} = {{ <unset> }}", nom->unique_name(), nom->type());
211 }
212 }
213}
