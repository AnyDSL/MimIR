121void Cleaner::eliminate_vars() {
122 for (auto old_lam : world().copy_lams()) {
123 if (!old_lam->is_set()) continue;
124
125 std::vector<size_t> proxy_idx; // indices of vars we eliminate
126 std::vector<size_t> var_idx; // indices of vars we keep
127
128 auto old_app = old_lam->body()->isa<App>();
129 if (old_app == nullptr || world().is_external(old_lam)) continue;
130
131 for (auto use : old_lam->uses()) {
132 if (use->isa<Var>()) continue; // ignore old_lam's Var
133 if (use.index() != 0 || !use->isa<App>())
134 goto next_lam;
135 }
136
137 // maybe the whole var tuple is passed somewhere?
138 if (old_lam->num_vars() != 1) {
139 for (auto use : old_lam->var()->uses()) {
140 if (!use->isa<Extract>())
141 goto next_lam;
142 }
143 }
144
145 for (size_t i = 0, e = old_lam->num_vars(); i != e; ++i) {
146 auto var = old_lam->var(i);
147 if (var->num_uses() == 0)
148 proxy_idx.push_back(i);
149 else
150 var_idx.push_back(i);
151 }
152
153 if (!proxy_idx.empty()) {
154 auto old_dom = old_lam->type()->dom();
155
156 const Def* new_dom;
157 if (auto sigma = old_dom->isa<Sigma>())
158 new_dom = world().sigma(sigma->ops().cut(proxy_idx));
159 else {
160 assert(proxy_idx.size() == 1 && proxy_idx[0] == 0);
161 new_dom = world().sigma();
162 }
163
164 auto cn = world().cn(new_dom);
165 auto new_lam = world().nom_lam(cn, old_lam->cc(), old_lam->debug_history());
166 size_t j = 0;
167 for (auto i : var_idx) {
168 old_lam->var(i)->replace(new_lam->var(j));
169 new_lam->var(j++, old_lam->var(i)->debug_history());
170 }
171
172 new_lam->set_filter(old_lam->filter());
173 new_lam->app(old_app->callee(), old_app->args(), old_app->dbg());
174 old_lam->unset();
175
176 for (auto use : old_lam->copy_uses()) {
177 if (use->isa<Var>()) continue; // ignore old_lam's Var
178 auto use_app = use->as<App>();
179 assert(use.index() == 0);
180 use_app->replace(world().app(new_lam, use_app->args().cut(proxy_idx), use_app->dbg()));
181 }
182
183 todo_ = true;
184 }
185next_lam:;

189void Cleaner::verify_closedness() {
