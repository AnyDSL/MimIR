13bool Scalerize::should_expand(Lam* lam) {
14 if (!isa_workable(lam)) return false;
15 if (auto i = tup2sca_.find(lam); i != tup2sca_.end() && i->second && i->second == lam) return false;
16
17 auto pi = lam->type();
18 if (lam->num_doms() > 1 && pi->is_cn() && !pi->isa_nom()) return true; // no ugly dependent pis
19
20 tup2sca_[lam] = lam;
21 return false;
22}

24Lam* Scalerize::make_scalar(const Def* def) {
25 auto tup_lam = def->isa_nom<Lam>();
26 assert(tup_lam);

35 todo |= n != 1 || types.back() != tup_lam->dom(i);
36 }

42 if (eta_exp_) eta_exp_->new2old(sca_lam, tup_lam);

47 return unflatten(tuple, tup_lam->dom(i), false);
48 }));
49 sca_lam->set(tup_lam->reduce(new_vars));

51 tup2sca_.emplace(tup_lam, sca_lam);

56const Def* Scalerize::rewrite(const Def* def) {
57 auto& w = world();
58 if (auto app = def->isa<App>()) {
59 const Def* sca_callee = app->callee();
60
61 if (auto tup_lam = sca_callee->isa_nom<Lam>(); should_expand(tup_lam)) {
62 sca_callee = make_scalar(tup_lam);
63
64 } else if (auto proj = sca_callee->isa<Extract>()) {
65 auto tuple = proj->tuple()->isa<Tuple>();
66 if (tuple && std::all_of(tuple->ops().begin(), tuple->ops().end(),
67 [&](const Def* op) { return should_expand(op->isa_nom<Lam>()); })) {
68 auto new_tuple = w.tuple(DefArray(tuple->num_ops(), [&](auto i) { return make_scalar(tuple->op(i)); }));
69 sca_callee = w.extract(new_tuple, proj->index());
70 w.DLOG("Expand tuple: {, } ~> {, }", tuple->ops(), new_tuple->ops());
71 }
72 }
73
74 if (sca_callee != app->callee()) {
75 auto new_args = DefVec();
76 flatten(new_args, app->arg(), false);
77 return world().app(sca_callee, new_args);
78 }
79 }
