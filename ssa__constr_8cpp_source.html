28const Def* SSAConstr::rewrite(const Def* def) {
29 if (auto slot = match<mem::slot>(def)) {
30 auto [mem, id] = slot->args<2>();
31 auto [_, ptr] = slot->projs<2>();
32 auto sloxy = proxy(ptr->type(), {curr_nom(), id}, Sloxy, slot->dbg());
33 world().DLOG("sloxy: '{}'", sloxy);
34 if (!keep_.contains(sloxy)) {
35 set_val(curr_nom(), sloxy, world().bot(get_sloxy_type(sloxy)));
36 data(curr_nom()).writable.emplace(sloxy);
37 return world().tuple({mem, sloxy});
38 }
39 } else if (auto load = match<mem::load>(def)) {
40 auto [mem, ptr] = load->args<2>();
41 if (auto sloxy = isa_proxy(ptr, Sloxy)) return world().tuple({mem, get_val(curr_nom(), sloxy)});
42 } else if (auto store = match<mem::store>(def)) {
43 auto [mem, ptr, val] = store->args<3>();
44 if (auto sloxy = isa_proxy(ptr, Sloxy)) {
45 if (data(curr_nom()).writable.contains(sloxy)) {
46 set_val(curr_nom(), sloxy, val);
47 return op_remem(mem, store->dbg());
48 }
49 }
50 } else if (auto [app, mem_lam] = isa_apped_nom_lam(def); isa_workable(mem_lam)) {
51 return mem2phi(app, mem_lam);
52 } else {
53 for (size_t i = 0, e = def->num_ops(); i != e; ++i) {
54 if (auto lam = def->op(i)->isa_nom<Lam>(); isa_workable(lam)) {
55 if (mem2phi_.contains(lam)) return def->refine(i, eta_exp_->proxy(lam));
56 }
57 }
58 }
59
60 return def;

63const Def* SSAConstr::get_val(Lam* lam, const Proxy* sloxy) {
64 auto& sloxy2val = lam2sloxy2val_[lam];
65 if (auto i = sloxy2val.find(sloxy); i != sloxy2val.end()) {
66 auto val = i->second;
67 world().DLOG("get_val found: '{}': '{}': '{}'", sloxy, val, lam);
68 return val;
69 } else if (lam->is_external()) {
70 world().DLOG("cannot install phi for '{}' in '{}'", sloxy, lam);

74 return get_val(pred, sloxy);

77 phixy->set_debug_name(std::string("phi_") + phixy->debug().name);

88const Def* SSAConstr::mem2phi(const App* app, Lam* mem_lam) {
89 auto&& sloxys = lam2sloxys_[mem_lam];

91
92 DefVec types, phis;
93 for (auto i = sloxys.begin(), e = sloxys.end(); i != e;) {
94 auto sloxy = *i;
95 if (keep_.contains(sloxy)) {
96 i = sloxys.erase(i);
97 } else {
98 phis.emplace_back(sloxy);
99 types.emplace_back(get_sloxy_type(sloxy));
100 ++i;
101 }
102 }
103
104 size_t num_phis = phis.size();
105 if (num_phis == 0) return app;
106

108 if (phi_lam == nullptr || old_phis != phis) {
109 old_phis = phis;
110 auto new_type = world().pi(merge_sigma(mem_lam->dom(), types), mem_lam->codom());
111 phi_lam = world().nom_lam(new_type, mem_lam->dbg());
112 eta_exp_->new2old(phi_lam, mem_lam);
113 world().DLOG("new phi_lam '{}'", phi_lam);
114
115 auto num_mem_vars = mem_lam->num_vars();
116 size_t i = 0;
117 DefArray traxy_ops(2 * num_phis + 1);
118 traxy_ops[0] = phi_lam->var();
119 for (auto sloxy : sloxys) {
120 traxy_ops[2 * i + 1] = sloxy;
121 traxy_ops[2 * i + 2] = phi_lam->var(num_mem_vars + i);
122 ++i;
123 }
124 auto traxy = proxy(phi_lam->var()->type(), traxy_ops, Traxy);
125
126 DefArray new_vars(num_mem_vars, [&](size_t i) { return traxy->proj(i); });
127 phi_lam->set(mem_lam->reduce(world().tuple(mem_lam->dom(), new_vars)));
128 } else {
129 world().DLOG("reuse phi_lam '{}'", phi_lam);
130 }
131
132 world().DLOG("mem_lam => phi_lam: '{}': '{}' => '{}': '{}'", mem_lam, mem_lam->type()->dom(), phi_lam,
133 phi_lam->dom());
134 auto sloxy = sloxys.begin();
135 DefArray args(num_phis, [&](auto) { return get_val(curr_nom(), *sloxy++); });
136 return world().app(phi_lam, merge_tuple(app->arg(), args));

139undo_t SSAConstr::analyze(const Proxy* proxy) {
140 if (proxy->tag() == Sloxy) {
141 auto sloxy_lam = proxy->op(0)->as_nom<Lam>();
142
143 if (keep_.emplace(proxy).second) {
144 world().DLOG("keep: '{}'; pointer needed", proxy);
145 return undo_enter(sloxy_lam);
146 }
147 }
148
149 assert(proxy->tag() == Phixy);
150 auto [sloxy, mem_lam] = split_phixy(proxy);
151 if (lam2sloxys_[mem_lam].emplace(sloxy).second) {
152 world().DLOG("phi needed: phixy '{}' for sloxy '{}' for mem_lam '{}'", proxy, sloxy, mem_lam);
153 return undo_visit(mem_lam);
154 }
155
156 return No_Undo;
157}

159undo_t SSAConstr::analyze(const Def* def) {
160 for (size_t i = 0, e = def->num_ops(); i != e; ++i) {
161 if (auto succ_lam = isa_workable(def->op(i)->isa_nom<Lam>())) {
162 auto& succ_info = data(succ_lam);
163
164 // TODO this is a bit scruffy - maybe we can do better
165 if (succ_lam->is_basicblock() && succ_lam != curr_nom()) {
166 auto writable = data(curr_nom()).writable;
167 for (auto&& w : writable) succ_info.writable.insert(w);
168 }
169
170 if (!isa_callee(def, i)) {
171 if (succ_info.pred) {
172 world().DLOG("several preds in non-callee position; wait for EtaExp");
173 succ_info.pred = nullptr;
174 } else {
175 world().DLOG("'{}' -> '{}'", curr_nom(), succ_lam);
176 succ_info.pred = curr_nom();
177 }
178 }
179 }
180 }
