54const App* ClosConvPrep::rewrite_arg(const App* app) {
55 auto& w = world();
56 auto arg = app->arg();
57
58 if (arg->isa<Var>()) return app;
59
60 for (auto i = 0u; i < arg->num_projs(); i++) {
61 auto op = arg->proj(i);
62
63 auto refine = [&](const Def* new_op) {
64 const Def* args;
65 if (arg == op) {
66 args = new_op;
67 } else {
68 args = arg->refine(i, new_op);
69 }
70 return app->refine(1, args)->as<App>();
71 };
72
73 if (auto lam = isa_retvar(op); lam && from_outer_scope(lam)) {
74 w.DLOG("found return var from enclosing scope: {}", op);
75 return refine(eta_wrap(op, attr::freeBB)->set("free_ret"));
76 }
77 if (auto bb_lam = op->isa_nom<Lam>(); bb_lam && bb_lam->is_basicblock() && from_outer_scope(bb_lam)) {
78 w.DLOG("found BB from enclosing scope {}", op);
79 return refine(thorin::clos::op(attr::freeBB, op));
80 }
81 if (isa_cont(app, arg, i)) {
82 if (match<attr>(attr::ret, op) || isa_retvar(op)) {
83 return app;
84 } else if (auto contlam = op->isa_nom<Lam>()) {
85 return refine(thorin::clos::op(attr::ret, contlam));
86 } else {
87 auto wrapper = eta_wrap(op, attr::ret)->set("eta_cont");
88 w.DLOG("eta expanded return cont: {} -> {}", op, wrapper);
89 return refine(wrapper);
90 }
91 }
92
93 if (!isa_callee_br(app, arg, i)) {
94 if (auto bb_lam = op->isa_nom<Lam>(); bb_lam && bb_lam->is_basicblock()) {
95 w.DLOG("found firstclass use of BB: {}", bb_lam);
96 return refine(thorin::clos::op(attr::fstclassBB, bb_lam));
97 }
98 // TODO: If EtaRed eta-reduces branches, we have to wrap them again!
99 if (isa_retvar(op)) {
100 w.DLOG("found firstclass use of return var: {}", op);
101 return refine(eta_wrap(op, attr::fstclassBB)->set("fstclass_ret"));
102 }
103 }
104 }

106 return app;
