55const App* ClosConvPrep::rewrite_arg(const App* app) {
56 auto& w = world();
57 auto arg = app->arg();
58
59 if (arg->isa<Var>()) return app;
60
61 for (auto i = 0u; i < arg->num_projs(); i++) {
62 auto op = arg->proj(i);
63
64 auto refine = [&](const Def* new_op) {
65 const Def* args;
66 if (arg == op) {
67 args = new_op;
68 } else {
69 args = arg->refine(i, new_op);
70 }
71 return app->refine(1, args)->as<App>();
72 };
73
74 if (auto lam = isa_retvar(op); lam && from_outer_scope(lam)) {
75 w.DLOG("found return var from enclosing scope: {}", op);
76 return refine(eta_wrap(op, attr::freeBB, "free_ret"));
77 }
78 if (auto bb_lam = op->isa_nom<Lam>(); bb_lam && bb_lam->is_basicblock() && from_outer_scope(bb_lam)) {
79 w.DLOG("found BB from enclosing scope {}", op);
80 return refine(thorin::clos::op(attr::freeBB, op));
81 }
82 if (isa_cont(app, arg, i)) {
83 if (match<attr>(attr::ret, op) || isa_retvar(op)) {
84 return app;
85 } else if (auto contlam = op->isa_nom<Lam>()) {
86 return refine(thorin::clos::op(attr::ret, contlam));
87 } else {
88 auto wrapper = eta_wrap(op, attr::ret, "eta_cont");
89 w.DLOG("eta expanded return cont: {} -> {}", op, wrapper);
90 return refine(wrapper);
91 }
92 }
93
94 if (!isa_callee_br(app, arg, i)) {
95 if (auto bb_lam = op->isa_nom<Lam>(); bb_lam && bb_lam->is_basicblock()) {
96 w.DLOG("found firstclass use of BB: {}", bb_lam);
97 return refine(thorin::clos::op(attr::fstclassBB, bb_lam));
98 }
99 // TODO: If EtaRed eta-reduces branches, we have to wrap them again!
100 if (isa_retvar(op)) {
101 w.DLOG("found firstclass use of return var: {}", op);
102 return refine(eta_wrap(op, attr::fstclassBB, "fstclass_ret"));
103 }
104 }
105 }
106
107 return app;
108}
