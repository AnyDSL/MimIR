109const Sigma* isa_clos_type(Ref def) {
110 auto& w = def->world();
111 auto sig = def->isa_mut<Sigma>();
112 if (!sig || sig->num_ops() < 3 || sig->op(0_u64) != w.type()) return nullptr;
113 auto var = sig->var(0_u64);
114 if (sig->op(2_u64) != var) return nullptr;
115 auto pi = sig->op(1_u64)->isa<Pi>();
116 return (pi && Pi::isa_cn(pi) && pi->num_ops() > 1_u64 && pi->dom(Clos_Env_Param) == var) ? sig : nullptr;
117}
