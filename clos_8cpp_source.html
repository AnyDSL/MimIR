92const Sigma* isa_clos_type(const Def* def) {
93 auto& w = def->world();
94 auto sig = def->isa_nom<Sigma>();
95 if (!sig || sig->num_ops() < 3 || sig->op(0_u64) != w.type()) return nullptr;
96 auto var = sig->var(0_u64);
97 if (sig->op(2_u64) != var) return nullptr;
98 auto pi = sig->op(1_u64)->isa<Pi>();
99 return (pi && pi->is_cn() && pi->num_ops() > 1_u64 && pi->dom(Clos_Env_Param) == var) ? sig : nullptr;
100}
