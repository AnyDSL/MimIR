8/*
9 * The implementation is based on Steensgard's algorithm to find loops in irreducible CFGs.
10 *
11 * In short, Steensgard's algorithm recursively applies Tarjan's SCC algorithm to find nested SCCs.
12 * In the next recursion, backedges from the prior run are ignored.
13 * Please, check out
14 * http://en.wikipedia.org/wiki/Tarjan%27s_strongly_connected_components_algorithm
15 * for more details on Tarjan's SCC algorithm
16 */

21enum {
22 InSCC = 1, // is in current walk_scc run?
23 OnStack = 2, // is in current SCC stack?
24 IsHead = 4, // all heads are marked, so subsequent runs can ignore backedges when searching for SCCs

103void LoopTreeBuilder::build() {
104 for (const auto& n : cfg().reverse_post_order()) // clear all flags
105 states_[n] = 0;
106
107 auto head = new Head(nullptr, 0, std::vector<const CFNode*>(0));
108 looptree_.root_.reset(head);
109 recurse(head, {cfg().entry()}, 1);
110}

112void LoopTreeBuilder::recurse(Head* parent, View<const CFNode*> heads, int depth) {
113 size_t curr_new_child = 0;
114 for (const auto& head : heads) {
115 set_.clear();
116 walk_scc(head, parent, depth, 0);
117
118 // now mark all newly found heads globally as head
119 for (size_t e = parent->num_children(); curr_new_child != e; ++curr_new_child)
120 for (const auto& head : parent->child(curr_new_child)->cf_nodes()) states_[head] |= IsHead;
121 }
122
123 for (const auto& node : parent->children())
124 if (auto new_parent = node->template isa<Head>()) recurse(new_parent, new_parent->cf_nodes(), depth + 1);
125}

127int LoopTreeBuilder::walk_scc(const CFNode* curr, Head* parent, int depth, int scc_counter) {
128 scc_counter = visit(curr, scc_counter);
129
130 for (const auto& succ : curr->succs()) {
131 if (is_head(succ)) continue; // this is a backedge
132 if (!set_.contains(succ)) {
133 scc_counter = walk_scc(succ, parent, depth, scc_counter);
134 lowlink(curr) = std::min(lowlink(curr), lowlink(succ));
135 } else if (on_stack(succ))
136 lowlink(curr) = std::min(lowlink(curr), lowlink(succ));
137 }
138
139 // root of SCC
140 if (lowlink(curr) == dfs(curr)) {
141 std::vector<const CFNode*> heads;
142
143 // mark all cf_nodes in current SCC (all cf_nodes from back to cur on the stack) as 'InSCC'
144 size_t num = 0, e = stack_.size(), b = e - 1;
145 do {
146 states_[stack_[b]] |= InSCC;
147 ++num;
148 } while (stack_[b--] != curr);
149
150 // for all cf_nodes in current SCC
151 for (size_t i = ++b; i != e; ++i) {
152 const CFNode* n = stack_[i];
153
154 if (cfg().entry() == n)
155 heads.emplace_back(n); // entries are axiomatically heads
156 else {
157 for (const auto& pred : n->preds()) {
158 // all backedges are also inducing heads
159 // but do not yet mark them globally as head -- we are still running through the SCC
160 if (!in_scc(pred)) {
161 heads.emplace_back(n);
162 break;
163 }
164 }
165 }
166 }
167
168 if (is_leaf(curr, num)) {
169 assert(heads.size() == 1);
170 looptree_.leaves_[heads.front()] = new Leaf(index_++, parent, depth, heads);
171 } else
172 new Head(parent, depth, heads);
173
174 // reset InSCC and OnStack flags
175 for (size_t i = b; i != e; ++i) states_[stack_[i]] &= ~(OnStack | InSCC);
176
177 // pop whole SCC
178 stack_.resize(b);
179 }
180
181 return scc_counter;
182}
