11/*
12 * The implementation is based on Steensgard's algorithm to find loops in irreducible CFGs.
13 *
14 * In short, Steensgard's algorithm recursively applies Tarjan's SCC algorithm to find nested SCCs.
15 * In the next recursion, backedges from the prior run are ignored.
16 * Please, check out
17 * http://en.wikipedia.org/wiki/Tarjan%27s_strongly_connected_components_algorithm
18 * for more details on Tarjan's SCC algorithm
19 */

23enum {
24 InSCC = 1, // is in current walk_scc run?
25 OnStack = 2, // is in current SCC stack?
26 IsHead = 4, // all heads are marked, so subsequent runs can ignore backedges when searching for SCCs
27};

107void LoopTreeBuilder<forward>::build() {
108 for (const auto& n : cfg().reverse_post_order()) // clear all flags
109 states_[n] = 0;

112 looptree_.root_.reset(head);

116template<bool forward>
117void LoopTreeBuilder<forward>::recurse(Head* parent, ArrayRef<const CFNode*> heads, int depth) {
118 size_t curr_new_child = 0;
119 for (const auto& head : heads) {
120 set_.clear();
121 walk_scc(head, parent, depth, 0);
122
123 // now mark all newly found heads globally as head
124 for (size_t e = parent->num_children(); curr_new_child != e; ++curr_new_child) {
125 for (const auto& head : parent->child(curr_new_child)->cf_nodes()) states_[head] |= IsHead;
126 }
127 }
128
129 for (const auto& node : parent->children()) {
130 if (auto new_parent = node->template isa<Head>()) recurse(new_parent, new_parent->cf_nodes(), depth + 1);
131 }
132}

134template<bool forward>
135int LoopTreeBuilder<forward>::walk_scc(const CFNode* cur, Head* parent, int depth, int scc_counter) {
136 scc_counter = visit(cur, scc_counter);
137
138 for (const auto& succ : cfg().succs(cur)) {
139 if (is_head(succ)) continue; // this is a backedge
140 if (!set_.contains(succ)) {
141 scc_counter = walk_scc(succ, parent, depth, scc_counter);
142 lowlink(cur) = std::min(lowlink(cur), lowlink(succ));
143 } else if (on_stack(succ))
144 lowlink(cur) = std::min(lowlink(cur), lowlink(succ));
145 }
146
147 // root of SCC
148 if (lowlink(cur) == dfs(cur)) {
149 std::vector<const CFNode*> heads;
150
151 // mark all cf_nodes in current SCC (all cf_nodes from back to cur on the stack) as 'InSCC'
152 size_t num = 0, e = stack_.size(), b = e - 1;
153 do {
154 states_[stack_[b]] |= InSCC;
155 ++num;
156 } while (stack_[b--] != cur);
157
158 // for all cf_nodes in current SCC
159 for (size_t i = ++b; i != e; ++i) {
160 const CFNode* n = stack_[i];
161
162 if (cfg().entry() == n)
163 heads.emplace_back(n); // entries are axiomatically heads
164 else {
165 for (const auto& pred : cfg().preds(n)) {
166 // all backedges are also inducing heads
167 // but do not yet mark them globally as head -- we are still running through the SCC
168 if (!in_scc(pred)) {
169 heads.emplace_back(n);
170 break;
171 }
172 }
173 }
174 }
175
176 if (is_leaf(cur, num)) {
177 assert(heads.size() == 1);
178 looptree_.leaves_[heads.front()] = new Leaf(index_++, parent, depth, heads);
179 } else
180 new Head(parent, depth, heads);
181
182 // reset InSCC and OnStack flags
183 for (size_t i = b; i != e; ++i) states_[stack_[i]] &= ~(OnStack | InSCC);
184
185 // pop whole SCC
186 stack_.resize(b);
187 }
188
189 return scc_counter;
190}
