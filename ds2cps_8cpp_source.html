51/// wrap rewrite calls to avoid code duplication
52/// memoize produced rewrite results
53/// and return the memoized result if available
54const Def* DS2CPS::rewrite_(const Def* def) {
55 if (auto i = rewritten_.find(def); i != rewritten_.end()) return i->second;
56
57 if (auto lam = def->isa_nom<Lam>()) {
58 if (!isa_workable(lam)) return lam;
59
60 auto ty = lam->type();
61 if (!ty->is_cn()) {
62 // extend ds function with return continuation
63 auto doms = ty->as<Pi>()->doms();
64 DefArray cps_dom(doms.size() + 1, [&](size_t i) -> const Def* {
65 if (i == doms.size()) {
66 return world().cn(ty->codom());
67 } else {
68 return doms[i];
69 }
70 });
71
72 // cps version of function
73 auto lam_cps = world().nom_lam(world().cn(cps_dom), world().dbg(lam->name() + "_cps"));
74 lam_cps->set_filter(lam->filter());
75
76 // each argument is linked to its corresponding argument in the cps function
77 for (size_t i = 0; i < lam->num_vars(); ++i) { rewritten_[lam->var((nat_t)i)] = lam_cps->var(i); }
78 // reconstruct var -> tuple of var without cont
79 // @fun -> (@fun_cps#0,...,@fun_cps#n)
80 DefArray real_vars(lam->num_vars(), [&](size_t i) { return lam_cps->var((nat_t)i); });
81 rewritten_[lam->var()] = world().tuple(real_vars);
82 rewritten_[lam] = lam_cps;
83
84 // the body of the ds function is the computation result
85 lam_cps->set_body(world().app(lam_cps->ret_var(world().dbg("cps_return")), lam->body()));
86
87 lam = lam_cps;
88 }
89 rewrite_lam(lam);
90 return lam;
91 }
92
93 return rewritten_[def] = rewrite_inner(def);
94}
