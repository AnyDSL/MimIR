93template<class T>
94hash_t hash_combine(hash_t seed, T v) {
95 static_assert(std::is_signed<T>::value || std::is_unsigned<T>::value,
96 "please provide your own hash function");
97
98 hash_t val = v;
99 for (hash_t i = 0; i < sizeof(T); ++i) {
100 hash_t octet = val & 0xff_u32; // extract lower 8 bits
101 seed ^= octet;
102 seed *= FNV1::prime;
103 val >>= 8_u32;
104 }
105 return seed;
106}

322 hash_t s = size() + std::distance(begin, end);
323 hash_t c = round_to_power_of_2(s);

325 if (s > c/4_u32 + c/2_u32) c *= 4_u32;

425 void rehash(hash_t new_capacity) {
426 using std::swap;
427
428 assert(is_power_of_2(new_capacity));
429
430 auto old_capacity = capacity_;
431 capacity_ = std::max(new_capacity, hash_t(MinHeapCapacity));
432 auto old_nodes = alloc();
433 swap(old_nodes, nodes_);
434
435 for (hash_t i = 0; i != old_capacity; ++i) {
436 auto& old = old_nodes[i];
437 if (!is_invalid(&old)) {
438 for (hash_t j = desired_pos(key(&old)), distance = 0; true; j = mod(j+1), ++distance) {
439 if (is_invalid(j)) {
440 swap(nodes_[j], old);
441 break;
442 } else {
443 hash_t cur_distance = probe_distance(j);
444 if (cur_distance < distance) {
445 distance = cur_distance;
446 swap(nodes_[j], old);
447 }
448 debug(j);
449 }
450 }
451 }
452 }
453
454 if (old_capacity != StackCapacity) delete[] old_nodes;
455 }
