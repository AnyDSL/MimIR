95/// Returns a new hash by combining the hash @p seed with @p val.
96template<class T>
97hash_t hash_combine(hash_t seed, T v) {
98 static_assert(std::is_signed<T>::value || std::is_unsigned<T>::value, "please provide your own hash function");
99
100 hash_t val = v;
101 for (hash_t i = 0; i < sizeof(T); ++i) {
102 hash_t octet = val & 0xff_u32; // extract lower 8 bits
103 seed ^= octet;
104 seed *= FNV1::prime;
105 val >>= 8_u32;
106 }
107 return seed;
108}

376 template<class I>
377 bool insert(I begin, I end) {
378 hash_t s = size() + std::distance(begin, end);
379 hash_t c = std::bit_ceil(s);
380
381 if (s > c / 4_u32 + c / 2_u32) c *= 4_u32;
382
383 c = std::max(c, hash_t(capacity_));

385 if (c != capacity_) rehash(c);

397 void rehash(hash_t new_capacity) {
398 using std::swap;
399
400 assert(std::has_single_bit(new_capacity));
401
402 auto old_capacity = capacity_;
403 capacity_ = std::max(new_capacity, hash_t(MinHeapCapacity));
404 auto old_nodes = alloc();
405 swap(old_nodes, nodes_);
406
407 for (hash_t i = 0; i != old_capacity; ++i) {
408 auto& old = old_nodes[i];
409 if (!is_invalid(&old)) {
410 for (hash_t j = desired_pos(key(&old)), distance = 0; true; j = mod(j + 1), ++distance) {
411 if (is_invalid(j)) {
412 swap(nodes_[j], old);
413 break;
414 } else {
415 hash_t cur_distance = probe_distance(j);
416 if (cur_distance < distance) {
417 distance = cur_distance;
418 swap(nodes_[j], old);
419 }
420 debug(j);
421 }
422 }
423 }
424 }
425
426 if (old_capacity != StackCapacity) delete[] old_nodes;
427 }
