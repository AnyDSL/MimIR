82 ///@name Aux Tree (Splay Tree)
83 ///@{
84
85 /// [Splays](https://hackmd.io/@CharlieChuang/By-UlEPFS#Operation1) `this` to the root of its splay tree.
86 constexpr void splay() noexcept {
87 while (auto p = aux_parent()) {
88 if (auto pp = p->aux_parent()) {
89 if (p->aux.bot == this && pp->aux.bot == p) { // zig-zig
90 pp->ror();
91 p->ror();
92 } else if (p->aux.top == this && pp->aux.top == p) { // zag-zag
93 pp->rol();
94 p->rol();
95 } else if (p->aux.bot == this && pp->aux.top == p) { // zig-zag
96 p->ror();
97 pp->rol();
98 } else { // zag-zig
99 assert(p->aux.top == this && pp->aux.bot == p);
100 p->rol();
101 pp->ror();
102 }
103 } else if (p->aux.bot == this) { // zig
104 p->ror();
105 } else { // zag
106 assert(p->aux.top == this);
107 p->rol();
108 }
109 }
110 }

112 /// Helpfer for Splay-Tree: rotate left/right:
113 /// ```
114 /// | Left | Right |
115 /// |-----------------------|------------------------|
116 /// | p p | p p |
117 /// | | | | | | |
118 /// | x c | x c |
119 /// | / \ -> / \ | / \ -> / \ |
120 /// | a c x d | c a d x |
121 /// | / \ / \ | / \ / \ |
122 /// | b d a b | d b b a |
123 /// ```
124 template<size_t l> constexpr void rotate() noexcept {
125 constexpr size_t r = (l + 1) % 2;
126 constexpr auto child = [](Node* n, size_t i) -> Node*& { return i == 0 ? n->aux.bot : n->aux.top; };
127
128 auto x = this;
129 auto p = x->aux.parent;
130 auto c = child(x, r);
131 auto b = child(c, l);
132
133 if (b) b->aux.parent = x;
134
135 if (p) {
136 if (child(p, l) == x) {
137 child(p, l) = c;
138 } else if (child(p, r) == x) {
139 child(p, r) = c;
140 } else {
141 /* only path parent */;
142 }
143 }
144
145 x->aux.parent = c;
146 c->aux.parent = p;
147 child(x, r) = b;
148 child(c, l) = x;
149 }
150
151 constexpr void rol() noexcept { return rotate<0>(); }
152 constexpr void ror() noexcept { return rotate<1>(); }

169 /// Make a preferred path from `this` to root while putting `this` at the root of the *aux* tree.
170 /// @returns the last valid path_parent().
171 constexpr Node* expose() noexcept {
172 Node* prev = nullptr;
173 for (auto curr = this; curr; prev = curr, curr = curr->aux.parent) {
174 curr->splay();
175 assert(!prev || prev->aux.parent == curr);
176 curr->aux.bot = prev;
177 }
178 splay();
179 return prev;
180 }

182 /// Least Common Ancestor of `this` and @p other in the *rep* tree; leaves @p other expose%d.
183 /// @returns `nullptr`, if @p a and @p b are in different trees.
184 constexpr Node* lca(Node* other) noexcept { return this->expose(), other->expose(); }
185
186 /// Is `this` a descendant of `other` in the *rep* tree?
187 /// Also `true`, if `this == other`.
188 constexpr bool is_descendant_of(Node* other) noexcept {
189 if (this == other) return true;
190 this->expose();
191 other->splay();
192 auto curr = this;
193 while (auto p = curr->aux_parent()) curr = p;
194 return curr == other;
195 }

387 /// Is @f$this \cap other \neq \emptyset@f$?.
388 [[nodiscard]] bool has_intersection(Set other) const noexcept {
389 if (this->empty() || other.empty()) return false;
390 if (auto u = this->isa_uniq()) return other.contains(u);
391 if (auto u = other.isa_uniq()) return this->contains(u);
392
393 auto d1 = this->isa_data(), d2 = other.isa_data();
394 auto n1 = this->isa_node(), n2 = other.isa_node();
395
396 if (d1 && d2) {
397 if (d1 == d2) return true;
398
399 for (auto ai = d1->begin(), ae = d1->end(), bi = d2->begin(), be = d2->end(); ai != ae && bi != be;) {
400 if (*ai == *bi) return true;
401
402 if ((*ai)->gid() < (*bi)->gid())
403 ++ai;
404 else
405 ++bi;
406 }
407
408 return false;
409 }
410
411 if (n1 && n2) {
412 if (!n1->lca(n2)->is_root()) return true;
413 if (n1->min > n2->def->tid() || n1->def->tid() < n2->min) return false;
414
415 // if one set is way smaller, iterate over this one and check with `contains` the other one
416 if (auto n1_lt = n1->size <= n2->size >> 3; n1_lt || n1->size <= n2->size >> 3) {
417 auto m = n1_lt ? n2 : n1;
418 for (auto n = n1_lt ? n1 : n2; !n->is_root(); n = n->parent)
419 if (m->contains(n->def)) return true;
420 return false;
421 }
422
423 while (!n1->is_root() && !n2->is_root()) {
424 if (n1->def == n2->def) {
425 n1->splay();
426 n2->splay();
427 return true;
428 }
429
430 if (n1->def->tid() > n2->def->tid()) {
431 n1 = n1->parent;
432 if (n1->def->tid() < n2->min) return false;
433 } else {
434 n2 = n2->parent;
435 if (n1->min > n2->def->tid()) return false;
436 }
437 }
438
439 return false;
440 }
441
442 auto n = n1 ? n1 : n2;
443 for (auto e : *(d1 ? d1 : d2))
444 if (n->contains(e)) return true;
445
446 return false;
447 }

523 /// Create a Set wih all elements in @p v.
524 [[nodiscard]] Set create(Vector<D*> v) {
525 auto vb = v.begin();
526 auto ve = v.end();
527 std::sort(vb, ve, [](D* d1, D* d2) { return d1->gid() < d2->gid(); });
528 auto vu = std::unique(vb, ve);
529 auto size = std::distance(vb, vu);
530
531 if (size == 0) return {};
532 if (size == 1) return {*vb};
533
534 if (size_t(size) <= N) {
535 auto [data, state] = allocate(size);
536 std::copy(vb, vu, data->begin());
537 return unify(data, state);
538 }
539
540 // sort in ascending tids but 0 goes last
541 std::sort(vb, vu, [](D* d1, D* d2) { return d1->tid() != 0 && (d2->tid() == 0 || d1->tid() < d2->tid()); });
542
543 auto res = root();
544 for (auto i = vb; i != vu; ++i) res = insert(res, *i);
545 return res;
546 }

548 /// Yields @f$s \cup d@f$.
549 [[nodiscard]] Set insert(Set s, D* d) {
550 if (auto u = s.isa_uniq()) {
551 if (d == u) return {d};
552
553 auto [data, state] = allocate(2);
554 if (d->gid() < u->gid())
555 data->elems[0] = d, data->elems[1] = u;
556 else
557 data->elems[0] = u, data->elems[1] = d;
558 return unify(data, state);
559 }
560
561 if (auto src = s.isa_data()) {
562 auto size = src->size;
563 if (size + 1 <= N) {
564 auto [dst, state] = allocate(size + 1);
565
566 // copy over and insert new element d
567 bool ins = false;
568 for (auto si = src->begin(), di = dst->begin(), se = src->end(); si != se || !ins; ++di) {
569 if (si != se && d == *si) { // already here
570 data_arena_.deallocate(state);
571 return s;
572 }
573
574 if (!ins && (si == se || d->gid() < (*si)->gid()))
575 *di = d, ins = true;
576 else
577 *di = *si++;
578 }
579
580 return unify(dst, state);
581 } else { // we need to switch from Data to Node
582 auto [dst, state] = allocate(size + 1);
583
584 // copy over
585 auto di = dst->begin();
586 for (auto si = src->begin(), se = src->end(); si != se; ++si, ++di) {
587 if (d == *si) { // already here
588 data_arena_.deallocate(state);
589 return s;
590 }
591
592 *di = *si;
593 }
594 *di = d; // put new element at last into dst->elems
595
596 // sort in ascending tids but 0 goes last
597 std::sort(dst->begin(), di,
598 [](D* d1, D* d2) { return d1->tid() != 0 && (d2->tid() == 0 || d1->tid() < d2->tid()); });
599
600 auto res = root();
601 for (auto i = dst->begin(), e = dst->end(); i != e; ++i) res = insert(res, *i);
602 return res;
603 }
604 }
605
606 if (auto n = s.isa_node()) {
607 if (n->contains(d)) return n;
608 return insert(n, d);
609 }
610
611 return {d};
612 }

614 /// Yields @f$s_1 \cup s_2@f$.
615 [[nodiscard]] Set merge(Set s1, Set s2) {
616 if (s1.empty() || s1 == s2) return s2;
617 if (s2.empty()) return s1;
618
619 if (auto u = s1.isa_uniq()) return insert(s2, u);
620 if (auto u = s2.isa_uniq()) return insert(s1, u);
621
622 auto d1 = s1.isa_data(), d2 = s2.isa_data();
623 auto n1 = s1.isa_node(), n2 = s2.isa_node();
624
625 if (d1 && d2) {
626 auto v = Vector<D*>();
627 v.reserve(d1->size + d2->size);
628
629 for (auto d : *d1) v.emplace_back(d);
630 for (auto d : *d2) v.emplace_back(d);
631
632 return create(std::move(v));
633 }
634
635 if (n1 && n2) {
636 if (n1->is_descendant_of(n2)) return n1;
637 if (n2->is_descendant_of(n1)) return n2;
638 return merge(n1, n2);
639 }
640
641 auto n = n1 ? n1 : n2;
642 for (auto d : *(d1 ? d1 : d2)) n = insert(n, d);
643 return n;
644 }

646 /// Yields @f$s \setminus \{d\}@f$.
647 [[nodiscard]] Set erase(Set s, D* d) {
648 if (auto u = s.isa_uniq()) return d == u ? Set() : s;
649
650 if (auto data = s.isa_data()) {
651 size_t i = 0, size = data->size;
652 for (; i != size; ++i)
653 if (data->elems[i] == d) break;
654
655 if (i == size--) return s;
656 if (size == 0) return {};
657 if (size == 1) return {i == 0 ? data->elems[1] : data->elems[0]};
658
659 assert(size <= N);
660 auto [new_data, state] = allocate(size);
661 auto db = data->begin();
662 std::copy(db + i + 1, data->end(), std::copy(db, db + i, new_data->elems)); // copy over, skip i
663
664 return unify(new_data, state);
665 }
666
667 if (auto n = s.isa_node()) {
668 if (d->tid() == 0 || d->tid() < n->min) return n;
669 if (!n->contains(d)) return n;
670
671 auto res = erase(n, d);
672 if (res->size > N) return res;
673
674 auto v = Vector<D*>();
675 v.reserve(res->size);
676 for (auto i = res; !i->is_root(); i = i->parent) v.emplace_back(i->def);
677 return create(std::move(v));
678 }
679
680 return {};
681 }
