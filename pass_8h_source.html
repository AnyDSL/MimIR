127 /// Add a pass to this PassMan.
128 /// If a pass of the same class has been added already, returns the earlier added instance.
129 template<class P, class... Args>
130 P* add(Args&&... args) {
131 if (auto it = registered_passes_.find(reinterpret_cast<u64>(P::ID())); it != registered_passes_.end())
132 return static_cast<P*>(it->second);
133 auto p = std::make_unique<P>(*this, std::forward<Args>(args)...);
134 auto res = p.get();
135 fixed_point_ |= res->fixed_point();
136 passes_.emplace_back(std::move(p));
137 registered_passes_.emplace(reinterpret_cast<u64>(P::ID()), res);
138 return res;
139 }

143 static void run(World& world, Args&&... args) {
144 PassMan man(world);

146 man.run();
