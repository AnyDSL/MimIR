120 /// Add a pass to this PassMan.
121 /// If a pass of the same class has been added already, returns the earlier added instance.
122 template<class P, class... Args>
123 P* add(Args&&... args) {
124 auto key = std::type_index(typeid(P));
125 if (auto it = registry_.find(key); it != registry_.end()) return static_cast<P*>(it->second);
126 auto p = std::make_unique<P>(*this, std::forward<Args>(args)...);
127 auto res = p.get();
128 fixed_point_ |= res->fixed_point();
129 passes_.emplace_back(std::move(p));
130 registry_.emplace(key, res);
131 return res;
132 }
133
134 /// Runs a single Pass.
135 template<class P, class... Args>
136 static void run(World& world, Args&&... args) {
137 PassMan man(world);
138 man.add<P>(std::forward<Args>(args)...);
139 man.run();
140 }
141 ///@}

143private:

276 /// @name undo
277 ///@{
278 undo_t curr_undo() const { return Super::man().curr_undo(); } ///< Current undo point.
279
280 /// Retrieves the point to backtrack to just **before** @p mut was seen the very first time.
281 undo_t undo_visit(Def* mut) const {
282 const auto& mut2visit = Super::man().curr_state().mut2visit;
283 if (auto i = mut2visit.find(mut); i != mut2visit.end()) return i->second;
284 return No_Undo;
285 }
286
287 /// Retrieves the point to backtrack to just **before** rewriting @p mut%'s body.
288 undo_t undo_enter(Def* mut) const {
289 for (auto i = states().size(); i-- != 0;)
290 if (states()[i].curr_mut == mut) return i;
291 return No_Undo;
292 }
293 ///@}
294
295private:

299 void* copy(const void* p) override { return new typename P::Data(*static_cast<const typename P::Data*>(p)); }
300 void dealloc(void* state) override { delete static_cast<typename P::Data*>(state); }
