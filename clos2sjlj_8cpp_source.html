10std::array<Ref, 3> split(Ref def) {
11 auto new_ops = DefArray(def->num_projs() - 2, nullptr);
12 auto& w = def->world();
13 const Def *mem, *env;
14 auto j = 0;
15 for (size_t i = 0; i < def->num_projs(); i++) {
16 auto op = def->proj(i);
17 if (op == w.annex<mem::M>() || op->type() == w.annex<mem::M>())
18 mem = op;
19 else if (i == Clos_Env_Param)
20 env = op;
21 else
22 new_ops[j++] = op;
23 }
24 assert(mem && env);
25 auto remaining = def->is_term() ? w.tuple(new_ops) : w.sigma(new_ops);
26 if (new_ops.size() == 1 && remaining != new_ops[0]) {
27 // FIXME: For some reason this is not constant folded away??
28 remaining = new_ops[0];
29 }
30 return {mem, env, remaining};
31}

33Ref rebuild(Ref mem, Ref env, Defs remaining) {
34 auto& w = mem->world();
35 auto new_ops = DefArray(remaining.size() + 2, [&](auto i) -> const Def* {
36 static_assert(Clos_Env_Param == 1);
37 if (i == 0) return mem;
38 if (i == 1) return env;
39 return remaining[i - 2];
40 });
41 return w.tuple(new_ops);

46void Clos2SJLJ::get_exn_closures(Ref def, DefSet& visited) {
47 if (!def->is_term() || def->isa_mut<Lam>() || visited.contains(def)) return;
48 visited.emplace(def);
49 if (auto c = isa_clos_lit(def)) {
50 auto lam = c.fnc_as_lam();
51 if (c.is_basicblock() && !ignore_.contains(lam)) {
52 def->world().DLOG("FOUND exn closure: {}", c.fnc_as_lam());
53 lam2tag_[c.fnc_as_lam()] = {lam2tag_.size() + 1, c.env()};
54 }
55 get_exn_closures(c.env(), visited);
56 } else {
57 for (auto op : def->ops()) get_exn_closures(op, visited);
58 }
59}

61void Clos2SJLJ::get_exn_closures() {
62 lam2tag_.clear();
63 if (!curr_mut()->is_set() || !Lam::isa_cn(curr_mut())) return;
64 auto app = curr_mut()->body()->isa<App>();
65 if (!app) return;
66 if (auto p = app->callee()->isa<Extract>(); p && isa_clos_type(p->tuple()->type())) {
67 auto p2 = p->tuple()->isa<Extract>();
68 if (p2 && p2->tuple()->isa<Tuple>()) {
69 // branch: Check the closure environments, but be careful not to traverse
70 // the closures themselves
71 auto branches = p2->tuple()->ops();
72 for (auto b : branches) {
73 auto c = isa_clos_lit(b);
74 if (c) {
75 ignore_.emplace(c.fnc_as_lam());
76 world().DLOG("IGNORE {}", c.fnc_as_lam());
77 }
78 }
79 }
80 }
81 auto visited = DefSet();
82 get_exn_closures(app->arg(), visited);
83}

85Lam* Clos2SJLJ::get_throw(Ref dom) {
86 auto& w = world();
87 auto [p, inserted] = dom2throw_.emplace(dom, nullptr);
88 auto& tlam = p->second;
89 if (inserted || !tlam) {
90 auto pi = w.cn(clos_sub_env(dom, w.sigma({jb_type(), rb_type(), tag_type()})));
91 tlam = w.mut_lam(pi)->set("throw");
92 auto [m0, env, var] = split(tlam->var());
93 auto [jbuf, rbuf, tag] = env->projs<3>();
94 auto [m1, r] = mem::op_alloc(var->type(), m0)->projs<2>();
95 auto m2 = w.call<mem::store>(Defs{m1, r, var});
96 rbuf = w.call<core::bitcast>(world().call<mem::Ptr0>(world().call<mem::Ptr0>(var->type())), rbuf);
97 auto m3 = w.call<mem::store>(Defs{m2, rbuf, r});
98 tlam->set(false, w.call<longjmp>(Defs{m3, jbuf, tag}));
99 ignore_.emplace(tlam);
100 }

123void Clos2SJLJ::enter() {
124 auto& w = world();
125 get_exn_closures();
126 if (lam2tag_.empty()) return;
127
128 {
129 auto m0 = mem::mem_var(curr_mut());
130 auto [m1, jb] = w.call<clos::alloc_jmpbuf>(m0)->projs<2>();
131 auto [m2, rb] = mem::op_slot(void_ptr(), m1)->projs<2>();
132 auto new_args = curr_mut()->vars();
133 new_args[0] = m2;
134 curr_mut()->reset(curr_mut()->reduce(w.tuple(new_args)));
135
136 cur_jbuf_ = jb;
137 cur_rbuf_ = rb;
138
139 // apparently apply() can change the id of the closures, so we have to do it again :(
140 get_exn_closures();
141 }
142
143 auto body = curr_mut()->body()->as<App>();
144
145 auto branch_type = clos_type(w.cn(w.annex<mem::M>()));
146 auto branches = DefVec(lam2tag_.size() + 1);
147 {
148 auto env = w.tuple(body->args().skip_front());
149 auto new_callee = w.mut_lam(w.cn({w.annex<mem::M>(), env->type()}))->set("sjlj_wrap");
150 auto [m, env_var, _] = split(new_callee->var());
151 auto new_args = DefArray(env->num_projs() + 1,
152 [&](auto i) -> const Def* { return (i == 0) ? *m : env_var->proj(i - 1); });
153 new_callee->app(false, body->callee(), new_args);
154 branches[0] = clos_pack(env, new_callee, branch_type);
155 }
156
157 for (auto [exn_lam, p] : lam2tag_) {
158 auto [i, env] = p;
159 branches[i] = clos_pack(env, get_lpad(exn_lam, cur_rbuf_), branch_type);
160 }
161
162 auto m0 = body->arg(0);
163 assert(m0->type() == w.annex<mem::M>());
164 auto [m1, tag] = w.call<setjmp>(Defs{m0, cur_jbuf_})->projs<2>();
165 tag = w.call(core::conv::s, branches.size(), tag);
166 auto filter = curr_mut()->filter();
167 auto branch = w.extract(w.tuple(branches), tag);
168 curr_mut()->reset({filter, clos_apply(branch, m1)});
169}
