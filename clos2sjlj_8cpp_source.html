7void Clos2SJLJ::get_exn_closures(const Def* def, DefSet& visited) {
8 if (def->sort() != Sort::Term || def->isa_nom<Lam>() || visited.contains(def)) return;
9 visited.emplace(def);
10 if (auto c = isa_clos_lit(def)) {
11 auto lam = c.fnc_as_lam();
12 if (c.is_basicblock() && !ignore_.contains(lam)) {
13 def->world().DLOG("FOUND exn closure: {}", c.fnc_as_lam());
14 lam2tag_[c.fnc_as_lam()] = {lam2tag_.size() + 1, c.env()};
15 }
16 get_exn_closures(c.env(), visited);
17 } else {
18 for (auto op : def->ops()) get_exn_closures(op, visited);
19 }
20}

22void Clos2SJLJ::get_exn_closures() {
23 lam2tag_.clear();
24 if (!curr_nom()->is_set() || !curr_nom()->type()->is_cn()) return;
25 auto app = curr_nom()->body()->isa<App>();
26 if (!app) return;
27 if (auto p = app->callee()->isa<Extract>(); p && isa_clos_type(p->tuple()->type())) {
28 auto p2 = p->tuple()->isa<Extract>();
29 if (p2 && p2->tuple()->isa<Tuple>()) {
30 // branch: Check the closure environments, but be careful not to traverse
31 // the closures themselves
32 auto branches = p2->tuple()->ops();
33 for (auto b : branches) {
34 auto c = isa_clos_lit(b);
35 if (c) {
36 ignore_.emplace(c.fnc_as_lam());
37 world().DLOG("IGNORE {}", c.fnc_as_lam());
38 }
39 }
40 }
41 }
42 auto visited = DefSet();
43 get_exn_closures(app->arg(), visited);
44}

46static std::array<const Def*, 3> split(const Def* def) {
47 auto new_ops = DefArray(def->num_projs() - 2, nullptr);
48 auto& w = def->world();
49 const Def *mem, *env;
50 auto j = 0;
51 for (size_t i = 0; i < def->num_projs(); i++) {
52 auto op = def->proj(i);
53 if (op == mem::type_mem(w) || op->type() == mem::type_mem(w))
54 mem = op;
55 else if (i == Clos_Env_Param)
56 env = op;
57 else
58 new_ops[j++] = op;
59 }
60 assert(mem && env);
61 auto remaining = (def->sort() == Sort::Term) ? w.tuple(new_ops) : w.sigma(new_ops);
62 if (new_ops.size() == 1 && remaining != new_ops[0]) {
63 // FIXME: For some reason this is not constant folded away??
64 remaining = new_ops[0];
65 }
66 return {mem, env, remaining};
67}

69static const Def* rebuild(const Def* mem, const Def* env, Defs remaining) {
70 auto& w = mem->world();
71 auto new_ops = DefArray(remaining.size() + 2, [&](auto i) {
72 static_assert(Clos_Env_Param == 1);
73 if (i == 0) return mem;
74 if (i == 1) return env;
75 return remaining[i - 2];
76 });
77 return w.tuple(new_ops);

80Lam* Clos2SJLJ::get_throw(const Def* dom) {
81 auto& w = world();
82 auto [p, inserted] = dom2throw_.emplace(dom, nullptr);
83 auto& tlam = p->second;
84 if (inserted || !tlam) {
85 auto pi = w.cn(clos_sub_env(dom, w.sigma({jb_type(), rb_type(), tag_type()})));
86 tlam = w.nom_lam(pi, w.dbg("throw"));
87 auto [m0, env, var] = split(tlam->var());
88 auto [jbuf, rbuf, tag] = env->projs<3>();
89 auto [m1, r] = mem::op_alloc(var->type(), m0)->projs<2>();
90 auto m2 = w.call<mem::store>(Defs{m1, r, var});
91 rbuf = core::op_bitcast(mem::type_ptr(mem::type_ptr(var->type())), rbuf);
92 auto m3 = w.call<mem::store>(Defs{m2, rbuf, r});
93 tlam->set(false, op_longjmp(m3, jbuf, tag));
94 ignore_.emplace(tlam);
95 }
96 return tlam;
97}

99Lam* Clos2SJLJ::get_lpad(Lam* lam, const Def* rb) {
100 auto& w = world();
101 auto [p, inserted] = lam2lpad_.emplace(w.tuple({lam, rb}), nullptr);
102 auto& lpad = p->second;
103 if (inserted || !lpad) {
104 auto [_, env_type, dom] = split(lam->dom());
105 auto pi = w.cn(w.sigma({mem::type_mem(w), env_type}));
106 lpad = w.nom_lam(pi, w.dbg("lpad"));
107 auto [m, env, __] = split(lpad->var());
108 auto [m1, arg_ptr] = w.call<mem::load>(Defs{m, rb})->projs<2>();
109 arg_ptr = core::op_bitcast(mem::type_ptr(dom), arg_ptr);
110 auto [m2, args] = w.call<mem::load>(Defs{m1, arg_ptr})->projs<2>();
111 auto full_args = (lam->num_doms() == 3) ? rebuild(m2, env, {args}) : rebuild(m2, env, args->ops());
112 lpad->app(false, lam, full_args);
113 ignore_.emplace(lpad);
114 }
115 return lpad;
116}

118void Clos2SJLJ::enter() {
119 auto& w = world();
120 get_exn_closures();
121 if (lam2tag_.empty()) return;
122
123 {
124 auto m0 = mem::mem_var(curr_nom());
125 auto [m1, jb] = op_alloc_jumpbuf(m0)->projs<2>();
126 auto [m2, rb] = mem::op_slot(void_ptr(), m1)->projs<2>();
127
128 auto new_args = curr_nom()->vars();
129 new_args[0] = m2;
130 curr_nom()->set(curr_nom()->reduce(w.tuple(new_args)));
131
132 cur_jbuf_ = jb;
133 cur_rbuf_ = rb;
134
135 // apparently apply() can change the id of the closures, so we have to do it again :(
136 get_exn_closures();
137 }
138
139 auto body = curr_nom()->body()->as<App>();
140
141 auto branch_type = clos_type(w.cn(mem::type_mem(w)));
142 auto branches = DefVec(lam2tag_.size() + 1);
143 {
144 auto env = w.tuple(body->args().skip_front());
145 auto new_callee = w.nom_lam(w.cn({mem::type_mem(w), env->type()}), w.dbg("sjlj_wrap"));
146 auto [m, env_var, _] = split(new_callee->var());
147 auto new_args = DefArray(env->num_projs() + 1, [&](auto i) { return (i == 0) ? m : env_var->proj(i - 1); });
148 new_callee->app(false, body->callee(), new_args, body->dbg());
149 branches[0] = clos_pack(env, new_callee, branch_type);
150 }
151
152 for (auto [exn_lam, p] : lam2tag_) {
153 auto [i, env] = p;
154 branches[i] = clos_pack(env, get_lpad(exn_lam, cur_rbuf_), branch_type);
155 }
156
157 auto m0 = body->arg(0);
158 assert(m0->type() == mem::type_mem(w));
159 auto [m1, tag] = op_setjmp(m0, cur_jbuf_)->projs<2>();
160 tag = w.call(core::conv::s, branches.size(), tag);
161 auto branch = w.extract(w.tuple(branches), tag);
162 curr_nom()->set_body(clos_apply(branch, m1));
163}
