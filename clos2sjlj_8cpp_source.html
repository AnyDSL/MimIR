9std::array<Ref, 3> split(Ref def) {
10 auto new_ops = DefArray(def->num_projs() - 2, nullptr);
11 auto& w = def->world();
12 const Def *mem, *env;
13 auto j = 0;
14 for (size_t i = 0; i < def->num_projs(); i++) {
15 auto op = def->proj(i);
16 if (op == mem::type_mem(w) || op->type() == mem::type_mem(w))
17 mem = op;
18 else if (i == Clos_Env_Param)
19 env = op;
20 else
21 new_ops[j++] = op;
22 }
23 assert(mem && env);
24 auto remaining = def->is_term() ? w.tuple(new_ops) : w.sigma(new_ops);
25 if (new_ops.size() == 1 && remaining != new_ops[0]) {
26 // FIXME: For some reason this is not constant folded away??
27 remaining = new_ops[0];
28 }
29 return {mem, env, remaining};
30}

32Ref rebuild(Ref mem, Ref env, Defs remaining) {
33 auto& w = mem->world();
34 auto new_ops = DefArray(remaining.size() + 2, [&](auto i) -> const Def* {
35 static_assert(Clos_Env_Param == 1);
36 if (i == 0) return mem;
37 if (i == 1) return env;
38 return remaining[i - 2];
39 });
40 return w.tuple(new_ops);
41}

45void Clos2SJLJ::get_exn_closures(Ref def, DefSet& visited) {
46 if (!def->is_term() || def->isa_mut<Lam>() || visited.contains(def)) return;
47 visited.emplace(def);
48 if (auto c = isa_clos_lit(def)) {
49 auto lam = c.fnc_as_lam();
50 if (c.is_basicblock() && !ignore_.contains(lam)) {
51 def->world().DLOG("FOUND exn closure: {}", c.fnc_as_lam());
52 lam2tag_[c.fnc_as_lam()] = {lam2tag_.size() + 1, c.env()};
53 }
54 get_exn_closures(c.env(), visited);
55 } else {
56 for (auto op : def->ops()) get_exn_closures(op, visited);
57 }
58}

60void Clos2SJLJ::get_exn_closures() {
61 lam2tag_.clear();
62 if (!curr_mut()->is_set() || !Lam::isa_cn(curr_mut())) return;
63 auto app = curr_mut()->body()->isa<App>();
64 if (!app) return;
65 if (auto p = app->callee()->isa<Extract>(); p && isa_clos_type(p->tuple()->type())) {
66 auto p2 = p->tuple()->isa<Extract>();
67 if (p2 && p2->tuple()->isa<Tuple>()) {
68 // branch: Check the closure environments, but be careful not to traverse
69 // the closures themselves
70 auto branches = p2->tuple()->ops();
71 for (auto b : branches) {
72 auto c = isa_clos_lit(b);
73 if (c) {
74 ignore_.emplace(c.fnc_as_lam());
75 world().DLOG("IGNORE {}", c.fnc_as_lam());
76 }
77 }
78 }
79 }
80 auto visited = DefSet();
81 get_exn_closures(app->arg(), visited);
82}

84Lam* Clos2SJLJ::get_throw(Ref dom) {
85 auto& w = world();
86 auto [p, inserted] = dom2throw_.emplace(dom, nullptr);
87 auto& tlam = p->second;
88 if (inserted || !tlam) {
89 auto pi = w.cn(clos_sub_env(dom, w.sigma({jb_type(), rb_type(), tag_type()})));
90 tlam = w.mut_lam(pi)->set("throw");
91 auto [m0, env, var] = split(tlam->var());
92 auto [jbuf, rbuf, tag] = env->projs<3>();
93 auto [m1, r] = mem::op_alloc(var->type(), m0)->projs<2>();
94 auto m2 = w.call<mem::store>(Defs{m1, r, var});
95 rbuf = w.call<core::bitcast>(mem::type_ptr(mem::type_ptr(var->type())), rbuf);
96 auto m3 = w.call<mem::store>(Defs{m2, rbuf, r});
97 tlam->set(false, op_longjmp(m3, jbuf, tag));
98 ignore_.emplace(tlam);
99 }
100 return tlam;

103Lam* Clos2SJLJ::get_lpad(Lam* lam, Ref rb) {
104 auto& w = world();
105 auto [p, inserted] = lam2lpad_.emplace(w.tuple({lam, rb}), nullptr);
106 auto& lpad = p->second;
107 if (inserted || !lpad) {
108 auto [_, env_type, dom] = split(lam->dom());
109 auto pi = w.cn(w.sigma({mem::type_mem(w), env_type}));
110 lpad = w.mut_lam(pi)->set("lpad");
111 auto [m, env, __] = split(lpad->var());
112 auto [m1, arg_ptr] = w.call<mem::load>(Defs{m, rb})->projs<2>();
113 arg_ptr = w.call<core::bitcast>(mem::type_ptr(dom), arg_ptr);
114 auto [m2, args] = w.call<mem::load>(Defs{m1, arg_ptr})->projs<2>();
115 auto full_args = (lam->num_doms() == 3) ? rebuild(m2, env, {args}) : rebuild(m2, env, args->ops());
116 lpad->app(false, lam, full_args);
117 ignore_.emplace(lpad);
118 }
119 return lpad;

122void Clos2SJLJ::enter() {
123 auto& w = world();
124 get_exn_closures();
125 if (lam2tag_.empty()) return;
126
127 {
128 auto m0 = mem::mem_var(curr_mut());
129 auto [m1, jb] = op_alloc_jumpbuf(m0)->projs<2>();
130 auto [m2, rb] = mem::op_slot(void_ptr(), m1)->projs<2>();
131
132 auto new_args = curr_mut()->vars();
133 new_args[0] = m2;
134 curr_mut()->set(curr_mut()->reduce(w.tuple(new_args)));
135
136 cur_jbuf_ = jb;
137 cur_rbuf_ = rb;
138
139 // apparently apply() can change the id of the closures, so we have to do it again :(
140 get_exn_closures();
141 }
142
143 auto body = curr_mut()->body()->as<App>();
144
145 auto branch_type = clos_type(w.cn(mem::type_mem(w)));
146 auto branches = DefVec(lam2tag_.size() + 1);
147 {
148 auto env = w.tuple(body->args().skip_front());
149 auto new_callee = w.mut_lam(w.cn({mem::type_mem(w), env->type()}))->set("sjlj_wrap");
150 auto [m, env_var, _] = split(new_callee->var());
151 auto new_args =
152 DefArray(env->num_projs() + 1, [&](auto i) -> const Def* { return (i == 0) ? *m : env_var->proj(i - 1); });
153 new_callee->app(false, body->callee(), new_args);
154 branches[0] = clos_pack(env, new_callee, branch_type);
155 }
156
157 for (auto [exn_lam, p] : lam2tag_) {
158 auto [i, env] = p;
159 branches[i] = clos_pack(env, get_lpad(exn_lam, cur_rbuf_), branch_type);
160 }
161
162 auto m0 = body->arg(0);
163 assert(m0->type() == mem::type_mem(w));
164 auto [m1, tag] = op_setjmp(m0, cur_jbuf_)->projs<2>();
165 tag = w.call(core::conv::s, branches.size(), tag);
166 auto branch = w.extract(w.tuple(branches), tag);
167 curr_mut()->set_body(clos_apply(branch, m1));
168}
