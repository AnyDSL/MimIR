28 const Def* rewrite(const Def* def) final {
29 if (auto hole = def->isa_mut<Hole>()) {
30 auto [last, op] = hole->find();
31 return op ? rewrite(op) : last;
32 }
33
34 return def == root_ || needs_zonk(def) ? Rewriter::rewrite(def) : def;
35 }
36
37 const Def* rewrite_mut(Def* root) final {
38 // Don't create a new stub, instead rewrire the ops of the old mutable root.
39 assert(root == root_);
40 map(root, root);
41
42 auto old_type = root->type();
43 auto old_ops = absl::FixedArray<const Def*>(root->ops().begin(), root->ops().end());
44
45 root->unset()->set_type(rewrite(old_type));
46
47 for (size_t i = 0, e = root->num_ops(); i != e; ++i) root->set(i, rewrite(old_ops[i]));
48 if (auto new_imm = root->immutabilize()) return map(root, new_imm);
49
50 return root;
51 }
52
53private:
54 Def* root_; // Always rewrite this one!
55};

61const Def* Def::zonk_mut() const {
62 if (!is_set()) return this;
63
64 if (auto mut = isa_mut()) {
65 if (auto hole = mut->isa<Hole>()) {
66 auto [last, op] = hole->find();
67 return op ? op->zonk() : last;
68 }
69
70 bool zonk = false;
71 for (auto def : deps())
72 if (needs_zonk(def)) {
73 zonk = true;
74 break;
75 }
76
77 if (zonk) return Zonker(world(), mut).rewrite(mut);
78
79 if (auto imm = mut->immutabilize()) return imm;
80 return this;
81 }
82
83 return zonk();
84}

94std::pair<Hole*, const Def*> Hole::find() {
95 auto def = Def::op(0);
96 auto last = this;
97
98 for (; def;) {
99 if (auto h = def->isa_mut<Hole>()) {
100 def = h->op();
101 last = h;
102 } else {
103 break;
104 }
105 }
106
107 auto root = def ? def : last;
108
109 // path compression
110 for (auto h = this; h != last;) {
111 auto next = h->op()->as_mut<Hole>();
112 h->unset()->set(root);
113 h = next;
114 }
115
116 return {last, def};
117}

119const Def* Hole::tuplefy(nat_t n) {
120 if (is_set()) return this;
121
122 auto& w = world();
123 auto holes = absl::FixedArray<const Def*>(n);
124 if (auto sigma = type()->isa_mut<Sigma>(); sigma && n >= 1 && sigma->has_var()) {
125 auto var = sigma->has_var();
126 auto rw = VarRewriter(var, this);
127 holes[0] = w.mut_hole(sigma->op(0));
128 for (size_t i = 1; i != n; ++i) {
129 rw.map(sigma->var(n, i - 1), holes[i - 1]);
130 holes[i] = w.mut_hole(rw.rewrite(sigma->op(i)));
131 }
132 } else {
133 for (size_t i = 0; i != n; ++i) holes[i] = w.mut_hole(type()->proj(n, i));
134 }
135
136 auto tuple = w.tuple(holes);
137 set(tuple);
138 return tuple;
139}

165const Def* Checker::assignable_(const Def* type, const Def* val) {
166 auto val_ty = val->type()->zonk();
167 if (type == val_ty) return val;
168
169 auto& w = world();
170 if (auto sigma = type->isa<Sigma>()) {
171 if (!alpha_<Check>(type->arity(), val_ty->arity())) return fail();
172
173 size_t a = sigma->num_ops();
174 auto red = sigma->reduce(val);
175 auto new_ops = absl::FixedArray<const Def*>(red.size());
176 for (size_t i = 0; i != a; ++i) {
177 auto new_val = assignable_(red[i], val->proj(a, i));
178 if (new_val)
179 new_ops[i] = new_val;
180 else
181 return fail();
182 }
183 return w.tuple(new_ops);
184 } else if (auto uniq = val->type()->isa<Uniq>()) {
185 if (auto new_val = assignable(type, uniq->inhabitant())) return new_val;
186 return fail();
187 }
188
189 return alpha_<Check>(type, val_ty) ? val : fail();
190}

192template<Checker::Mode mode> bool Checker::alpha_(const Def* d1, const Def* d2) {
193 for (bool todo = true; todo;) {
194 // below we check type and arity which may in turn open up more opportunities for zonking
195 todo = false;
196 d1 = d1->zonk_mut();
197 d2 = d2->zonk_mut();
198
199 // It is only safe to check for pointer equality if there are no Vars involved.
200 // Otherwise, we have to look more thoroughly.
201 // Example: λx.x - λz.x
202 if (!d1->has_dep(Dep::Var) && !d2->has_dep(Dep::Var) && d1 == d2) return true;
203
204 auto h1 = d1->isa_mut<Hole>();
205 auto h2 = d2->isa_mut<Hole>();
206
207 if constexpr (mode == Check) {
208 if (h1) return h1->set(d2), true;
209 if (h2) return h2->set(d1), true;
210 } else if (h1 || h2) // mode == Test and h1 or h2 is an unresolved Hole
211 return fail<Test>();
212
213 if (!d1->is_set() || !d2->is_set()) return fail<mode>();
214
215 auto mut1 = d1->isa_mut();
216 auto mut2 = d2->isa_mut();
217
218 if (mut1 && mut2 && mut1 == mut2) return true;
219
220 // Globals are HACKs and require additionaly HACKs:
221 // Unless they are pointer equal (above) always consider them unequal.
222 if (d1->isa<Global>() || d2->isa<Global>()) return false;
223
224 if (auto [i, ins] = bind(mut1, d2); !ins) return i->second == d2;
225 if (auto [i, ins] = bind(mut2, d1); !ins) return i->second == d1;
226
227 if (d1->isa<Top>() || d2->isa<Top>()) return mode == Check;
228
229 if (auto t1 = d1->type()) {
230 if (auto t2 = d2->type()) {
231 if (!alpha_<mode>(t1, t2)) return fail<mode>();
232 }
233 }
234
235 if (!alpha_<mode>(d1->arity(), d2->arity())) return fail<mode>();
236
237 auto new_d1 = d1->zonk_mut();
238 auto new_d2 = d2->zonk_mut();
239 if (new_d1 != d1 || new_d2 != d2) {
240 todo = true;
241 d1 = new_d1;
242 d2 = new_d2;
243 }
244 }
245
246 auto seq1 = d1->isa<Seq>();
247 auto seq2 = d2->isa<Seq>();
248
249 if constexpr (mode == Mode::Check) {
250 if (auto umax = d1->isa<UMax>(); umax && !d2->isa<UMax>()) return check(umax, d2);
251 if (auto umax = d2->isa<UMax>(); umax && !d1->isa<UMax>()) return check(umax, d1);
252
253 if (seq1 && seq1->shape() == world().lit_nat_1() && !seq2) return check1(seq1, d2);
254 if (seq2 && seq2->shape() == world().lit_nat_1() && !seq1) return check1(seq2, d1);
255
256 if (seq1 && seq2) {
257 if (auto mut_seq = seq1->isa_mut<Seq>(); mut_seq && seq2->isa_imm()) return check(mut_seq, seq2);
258 if (auto mut_seq = seq2->isa_mut<Seq>(); mut_seq && seq1->isa_imm()) return check(mut_seq, seq1);
259 }
260 }
261
262 if (auto prod = d1->isa<Prod>()) return check<mode>(prod, d2);
263 if (auto prod = d2->isa<Prod>()) return check<mode>(prod, d1);
264 if (seq1 && seq2) return alpha_<mode>(seq1->body(), seq2->body());
265
266 if (d1->node() != d2->node() || d1->flags() != d2->flags()) return fail<mode>();
267
268 if (auto var1 = d1->isa<Var>()) {
269 auto var2 = d2->as<Var>();
270 if (auto i = binders_.find(var1->mut()); i != binders_.end()) return i->second == var2->mut();
271 if (auto i = binders_.find(var2->mut()); i != binders_.end()) return fail<mode>(); // var2 is bound
272 // both var1 and var2 are free: OK, when they are the same or in Check mode
273 return var1 == var2 || mode == Check;
274 }
275
276 for (size_t i = 0, e = d1->num_ops(); i != e; ++i)
277 if (!alpha_<mode>(d1->op(i), d2->op(i))) return fail<mode>();
278 return true;
279}
