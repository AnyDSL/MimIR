63Ref Infer::explode() {
64 if (auto a = type()->isa_lit_arity(); a && !is_set()) {
65 auto n = *a;
66 auto infers = DefVec(n);
67 if (auto sigma = type()->isa_mut<Sigma>(); sigma && n >= 1 && sigma->has_var()) {
68 auto var = sigma->has_var();
69 auto rw = VarRewriter(var, this);
70 infers[0] = world().mut_infer(sigma->op(0));
71 for (size_t i = 1; i != n; ++i) {
72 rw.map(sigma->var(n, i - 1), infers[i - 1]);
73 infers[i] = world().mut_infer(rw.rewrite(sigma->op(i)));
74 }
75 } else {
76 for (size_t i = 0; i != n; ++i) infers[i] = world().mut_infer(type()->proj(n, i));
77 }
78
79 auto tuple = world().tuple(infers);
80 set(tuple);
81 return tuple;
82 }
83 return this;
84}

102template<Checker::Mode mode> bool Checker::alpha_(Ref r1, Ref r2) {
103 auto d1 = *r1; // find
104 auto d2 = *r2; // find
105
106 if (!d1 && !d2) return true;
107 if (!d1 || !d2) return fail<mode>();
108
109 // It is only safe to check for pointer equality if there are no Vars involved.
110 // Otherwise, we have to look more thoroughly.
111 // Example: λx.x - λz.x
112 if (!d1->has_dep(Dep::Var) && !d2->has_dep(Dep::Var) && d1 == d2) return true;
113
114 auto i1 = d1->isa_mut<Infer>();
115 auto i2 = d2->isa_mut<Infer>();
116
117 if ((!i1 && !d1->is_set()) || (!i2 && !d2->is_set())) return fail<mode>();
118
119 if (mode == Check) {
120 if (i1 && i2) {
121 // union by rank
122 if (i1->rank() < i2->rank()) std::swap(i1, i2); // make sure i1 is heavier or equal
123 i2->set(i1); // make i1 new root
124 if (i1->rank() == i2->rank()) ++i1->rank();
125 return true;
126 } else if (i1) {
127 i1->set(d2);
128 return true;
129 } else if (i2) {
130 i2->set(d1);
131 return true;
132 }
133 }
134
135 auto mut1 = d1->isa_mut();
136 auto mut2 = d2->isa_mut();
137 if (mut1 && mut2 && mut1 == mut2) return true;
138 // Globals are HACKs and require additionaly HACKs:
139 // Unless they are pointer equal (above) always consider them unequal.
140 if (d1->isa<Global>() || d2->isa<Global>()) return false;
141
142 if (mut1) {
143 if (auto [i, ins] = done_.emplace(mut1, d2); !ins) return i->second == d2;
144 }
145 if (mut2) {
146 if (auto [i, ins] = done_.emplace(mut2, d1); !ins) return i->second == d1;
147 }
148
149 // normalize:
150 if ((d1->isa<Lit>() && !d2->isa<Lit>()) // Lit to right
151 || (!d1->isa<UMax>() && d2->isa<UMax>()) // UMax to left
152 || (!d1->isa<Extract>() && d2->isa<Extract>())) // Extract to left
153 std::swap(d1, d2);
154
155 return alpha_internal<mode>(d1, d2);
156}

158template<Checker::Mode mode> bool Checker::alpha_internal(Ref d1, Ref d2) {
159 if (!alpha_<mode>(d1->type(), d2->type())) return fail<mode>();
160 if (d1->isa<Top>() || d2->isa<Top>()) return mode == Check;
161 if (mode == Opt && (d1->isa_mut<Infer>() || d2->isa_mut<Infer>())) return fail<mode>();
162 if (!alpha_<mode>(d1->arity(), d2->arity())) return fail<mode>();
163
164 // vars are equal if they appeared under the same binder
165 if (auto mut1 = d1->isa_mut()) assert_emplace(vars_, mut1, d2->isa_mut());
166 if (auto mut2 = d2->isa_mut()) assert_emplace(vars_, mut2, d1->isa_mut());
167
168 if (auto ts = d1->isa<Tuple, Sigma>()) {
169 size_t a = ts->num_ops();
170 for (size_t i = 0; i != a; ++i)
171 if (!alpha_<mode>(ts->op(i), d2->proj(a, i))) return fail<mode>();
172 return true;
173 } else if (auto pa = d1->isa<Pack, Arr>()) {
174 if (pa->node() == d2->node()) return alpha_<mode>(pa->ops().back(), d2->ops().back());
175 if (auto a = pa->isa_lit_arity()) {
176 for (size_t i = 0; i != *a; ++i)
177 if (!alpha_<mode>(pa->proj(*a, i), d2->proj(*a, i))) return fail<mode>();
178 return true;
179 }
180 } else if (auto umax = d1->isa<UMax>(); umax && umax->has_dep(Dep::Infer) && !d2->isa<UMax>()) {
181 // .umax(a, ?) == x => .umax(a, x)
182 for (auto op : umax->ops())
183 if (auto inf = op->isa_mut<Infer>(); inf && !inf->is_set()) inf->set(d2);
184 d1 = umax->rebuild(umax->type(), umax->ops());
185 }
186
187 if (d1->node() != d2->node() || d1->flags() != d2->flags() || d1->num_ops() != d2->num_ops()) return fail<mode>();
188
189 if (auto var1 = d1->isa<Var>()) {
190 auto var2 = d2->as<Var>();
191 if (auto i = vars_.find(var1->mut()); i != vars_.end()) return i->second == var2->mut();
192 if (auto i = vars_.find(var2->mut()); i != vars_.end()) return fail<mode>(); // var2 is bound
193 // both var1 and var2 are free: OK, when they are the same or in Check mode
194 return var1 == var2 || mode == Check;
195 }
196
197 for (size_t i = 0, e = d1->num_ops(); i != e; ++i)
198 if (!alpha_<mode>(d1->op(i), d2->op(i))) return fail<mode>();
199 return true;
200}

202Ref Checker::assignable_(Ref type, Ref val) {
203 auto val_ty = Ref::refer(val->type());
204 if (type == val_ty) return val;
205
206 if (auto sigma = type->isa<Sigma>()) {
207 if (!alpha_<Check>(type->arity(), val_ty->arity())) return fail();
208
209 size_t a = sigma->num_ops();
210 auto red = sigma->reduce(val);
211 auto new_ops = DefVec(red.size());
212 for (size_t i = 0; i != a; ++i) {
213 auto new_val = assignable_(red[i], val->proj(a, i));
214 if (new_val)
215 new_ops[i] = new_val;
216 else
217 return fail();
218 }
219 return world().tuple(new_ops);
220 } else if (auto arr = type->isa<Arr>()) {
221 if (!alpha_<Check>(type->arity(), val_ty->arity())) return fail();
222
223 // TODO ack sclarize threshold
224 if (auto a = Lit::isa(arr->arity())) {
225 auto new_ops = DefVec(*a);
226 for (size_t i = 0; i != *a; ++i) {
227 auto new_val = assignable_(arr->proj(*a, i), val->proj(*a, i));
228 if (new_val)
229 new_ops[i] = new_val;
230 else
231 return fail();
232 }
233 return world().tuple(new_ops);
234 }
235 } else if (auto vel = val->isa<Vel>()) {
236 if (auto new_val = assignable_(type, vel->value())) return world().vel(type, new_val);
237 return fail();
238 } else if (auto uniq = val->type()->isa<Uniq>()) {
239 if (auto new_val = assignable(type, uniq->inhabitant())) return new_val;
240 return fail();
241 }
242
243 return alpha_<Check>(type, val_ty) ? val : fail();
244}
