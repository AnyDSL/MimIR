13const Def* Infer::find(const Def* def) {
14 // find root
15 auto res = def;
16 for (auto infer = res->isa_nom<Infer>(); infer && infer->op(); infer = res->isa_nom<Infer>()) res = infer->op();
17
18 // path compression: set all Infers along the chain to res
19 for (auto infer = def->isa_nom<Infer>(); infer && infer->op(); infer = def->isa_nom<Infer>()) {
20 def = infer->op();
21 infer->set(res);
22 }
23
24 assert((!res->isa<Infer>() || res != res->op(0)) && "an Infer shouldn't point to itself");
25
26 // If we have an Infer as operand, try to get rid of it now.
27 // TODO why does this not work?
28 // if (res->isa_structural() && res->has_dep(Dep::Infer)) {
29 if (res->isa<Tuple>() || res->isa<Type>()) {
30 auto new_type = refer(res->type());
31 bool update = new_type != res->type();
32
33 auto new_ops = DefArray(res->num_ops(), [res, &update](size_t i) {
34 auto r = refer(res->op(i));
35 update |= r != res->op(i);
36 return r;
37 });
38
39 if (update) return res->rebuild(res->world(), new_type, new_ops);
40 }

42 return res;

72bool Checker::equiv(Ref r1, Ref r2) {
73 auto d1 = *r1; // find
74 auto d2 = *r2; // find
75
76 if (d1 == d2) return true;
77 if (!d1 || !d2) return false;
78
79 auto i1 = d1->isa_nom<Infer>();
80 auto i2 = d2->isa_nom<Infer>();
81
82 if ((!i1 && !d1->is_set()) || (!i2 && !d2->is_set())) return false;
83
84 if (i1 && i2) {
85 // union by rank
86 if (i1->rank() < i2->rank()) std::swap(i1, i2); // make sure i1 is heavier or equal
87 i2->set(i1); // make i1 new root
88 if (i1->rank() == i2->rank()) ++i1->rank();
89 return true;
90 } else if (i1) {
91 i1->set(d2);
92 return true;
93 } else if (i2) {
94 i2->set(d1);
95 return true;
96 }
97
98 assert(!i1 && !i2);
99 if (d1->gid() > d2->gid()) std::swap(d1, d2); // normalize
100
101 if (auto [it, ins] = equiv_.emplace(std::pair(d1, d2), Equiv::Unknown); !ins) {
102 switch (it->second) {
103 case Equiv::Distinct: return false;
104 case Equiv::Unknown:
105 case Equiv::Equiv: return true;
106 default: unreachable();
107 }
108 }
109
110 bool res = equiv_internal(d1, d2);
111 equiv_[std::pair(d1, d2)] = res ? Equiv::Equiv : Equiv::Distinct;
112 return res;
113}

115bool Checker::equiv_internal(Ref d1, Ref d2) {
116 if (!equiv(d1->type(), d2->type())) return false;
117 if (d1->isa<Top>() || d2->isa<Top>()) return equiv(d1->type(), d2->type());
118
119 if (auto n1 = d1->isa_nom()) {
120 if (auto n2 = d2->isa_nom()) vars_.emplace_back(n1, n2);
121 }
122
123 if (d1->isa<Sigma, Arr>()) {
124 if (!equiv(d1->arity(), d2->arity())) return false;
125
126 if (auto a = d1->isa_lit_arity()) {
127 for (size_t i = 0; i != a; ++i) {
128 if (!equiv(d1->proj(*a, i), d2->proj(*a, i))) return false;
129 }
130 return true;
131 }
132 }
133
134 if (d1->node() != d2->node() || d1->flags() != d2->flags() || d1->num_ops() != d2->num_ops()) return false;
135
136 if (auto var = d1->isa<Var>()) { // vars are equal if they appeared under the same binder
137 for (auto [n1, n2] : vars_) {
138 if (var->nom() == n1) return d2->as<Var>()->nom() == n2;
139 }
140 // TODO what if Var is free?
141 return false;
142 }
143
144 return std::ranges::equal(d1->ops(), d2->ops(), [&](auto op1, auto op2) { return equiv(op1, op2); });
145}

147bool Checker::assignable(Ref type, Ref val) {
148 auto val_ty = refer(val->type());
149 if (type == val_ty) return true;
150
151 auto infer = val->isa_nom<Infer>();
152 if (auto sigma = type->isa<Sigma>()) {
153 if (!infer && !equiv(type->arity(), val_ty->arity())) return false;
154
155 size_t a = sigma->num_ops();
156 auto red = sigma->reduce(val);
157
158 if (infer) infer->inflate(sigma, red);
159
160 for (size_t i = 0; i != a; ++i) {
161 if (!assignable(red[i], val->proj(a, i))) return false;
162 }
163
164 return true;
165 } else if (auto arr = type->isa<Arr>()) {
166 if (!infer && !equiv(type->arity(), val_ty->arity())) return false;
167
168 if (auto a = isa_lit(arr->arity())) {
169 if (infer) infer->inflate(arr, *a, arr->body());
170
171 for (size_t i = 0; i != *a; ++i) {
172 if (!assignable(arr->proj(*a, i), val->proj(*a, i))) return false;
173 }
174
175 return true;
176 }
177 } else if (auto vel = val->isa<Vel>()) {
178 if (assignable(type, vel->value())) return true;
179 }
180
181 return equiv(type, val_ty);
182}
