13const Def* Infer::find(const Def* def) {
14 // find root
15 auto res = def;
16 for (auto infer = res->isa_mut<Infer>(); infer && infer->op(); infer = res->isa_mut<Infer>()) res = infer->op();
17
18 // path compression: set all Infers along the chain to res
19 for (auto infer = def->isa_mut<Infer>(); infer && infer->op(); infer = def->isa_mut<Infer>()) {
20 def = infer->op();
21 infer->reset(res);
22 }
23
24 assert((!res->isa<Infer>() || res != res->op(0)) && "an Infer shouldn't point to itself");
25
26 // If we have an Infer as operand, try to get rid of it now.
27 // TODO why does this not work?
28 // if (res->isa_imm() && res->has_dep(Dep::Infer)) {
29 if (res->isa<Tuple>() || res->isa<Type>()) {
30 auto new_type = Ref::refer(res->type());
31 bool update = new_type != res->type();
32
33 auto new_ops = DefArray(res->num_ops(), [res, &update](size_t i) {
34 auto r = Ref::refer(res->op(i));
35 update |= r != res->op(i);
36 return r;
37 });
38
39 if (update) return res->rebuild(res->world(), new_type, new_ops);
40 }
41
42 return res;
43}

49bool Checker::equiv(Ref r1, Ref r2) {
50 auto d1 = *r1; // find
51 auto d2 = *r2; // find
52
53 if (d1 == d2) return true;
54 if (!d1 || !d2) return false;
55
56 auto i1 = d1->isa_mut<Infer>();
57 auto i2 = d2->isa_mut<Infer>();
58
59 if ((!i1 && !d1->is_set()) || (!i2 && !d2->is_set())) return false;
60
61 if (i1 && i2) {
62 // union by rank
63 if (i1->rank() < i2->rank()) std::swap(i1, i2); // make sure i1 is heavier or equal
64 i2->set(i1); // make i1 new root
65 if (i1->rank() == i2->rank()) ++i1->rank();
66 return true;
67 } else if (i1) {
68 i1->set(d2);
69 return true;
70 } else if (i2) {
71 i2->set(d1);
72 return true;
73 }
74
75 assert(!i1 && !i2);
76 // normalize: Lit to right; then sort by gid
77 if ((d1->isa<Lit>() && !d2->isa<Lit>()) || (d1->gid() > d2->gid())) std::swap(d1, d2);
78
79 if (auto [it, ins] = equiv_.emplace(std::pair(d1, d2), Equiv::Unknown); !ins) {
80 switch (it->second) {
81 case Equiv::Distinct: return false;
82 case Equiv::Unknown:
83 case Equiv::Equiv: return true;
84 default: unreachable();
85 }
86 }
87
88 bool res = equiv_internal(d1, d2);
89 equiv_[std::pair(d1, d2)] = res ? Equiv::Equiv : Equiv::Distinct;
90 return res;
91}

93bool Checker::equiv_internal(Ref d1, Ref d2) {
94 if (!equiv(d1->type(), d2->type())) return false;
95 if (d1->isa<Top>() || d2->isa<Top>()) return equiv(d1->type(), d2->type());
96
97 struct Pop {
98 ~Pop() {
99 if (vars) vars->pop_back();
100 }
101
102 Vars* vars = nullptr;
103 } pop;
104
105 if (auto n1 = d1->isa_mut()) {
106 if (auto n2 = d2->isa_mut()) {
107 vars_.emplace_back(n1, n2);
108 pop.vars = &vars_; // make sure vars_ is popped again
109 }
110 }
111
112 if (d1->isa<Sigma, Arr>()) {
113 if (!equiv(d1->arity(), d2->arity())) return false;
114
115 if (auto a = d1->isa_lit_arity()) {
116 for (size_t i = 0; i != a; ++i)
117 if (!equiv(d1->proj(*a, i), d2->proj(*a, i))) return false;
118 return true;
119 }
120 }
121
122 if (auto umax = d1->isa<UMax>(); umax && umax->has_dep(Dep::Infer)) {
123 if (auto l = d2->isa<Lit>()) {
124 for (auto op : umax->ops())
125 if (auto infer = op->isa_mut<Infer>(); infer && !infer->is_set()) infer->set(l);
126 }
127 d1 = umax->rebuild(world(), umax->type(), umax->ops());
128 }
129
130 if (d1->node() != d2->node() || d1->flags() != d2->flags() || d1->num_ops() != d2->num_ops()) return false;
131
132 if (auto var1 = d1->isa<Var>()) { // vars are equal if they appeared under the same binder
133 auto var2 = d2->as<Var>();
134 bool bound1 = false, bound2 = false;
135 for (auto [n1, n2] : vars_) {
136 if (var1->mut() == n1) {
137 bound1 = true;
138 return d2->as<Var>()->mut() == n2;
139 }
140 assert(var1->mut() != n2);
141 if (var2->mut() == n1 || var2->mut() == n2) bound2 = true;
142 }
143 if (!bound1 && !bound2) return true; // both var1 and var2 are free
144 return false;
145 }
146
147 return std::ranges::equal(d1->ops(), d2->ops(), [&](auto op1, auto op2) { return equiv(op1, op2); });
148}

150bool Checker::assignable(Ref type, Ref val) {
151 auto val_ty = Ref::refer(val->type());
152 if (type == val_ty) return true;
153
154 if (auto infer = val->isa_mut<Infer>()) return equiv(type, infer->type());
155
156 if (auto sigma = type->isa<Sigma>()) {
157 if (!equiv(type->arity(), val_ty->arity())) return false;
158
159 size_t a = sigma->num_ops();
160 auto red = sigma->reduce(val);
161
162 for (size_t i = 0; i != a; ++i)
163 if (!assignable(red[i], val->proj(a, i))) return false;
164
165 return true;
166 } else if (auto arr = type->isa<Arr>()) {
167 if (!equiv(type->arity(), val_ty->arity())) return false;
168
169 if (auto a = Lit::isa(arr->arity())) {
170 for (size_t i = 0; i != *a; ++i)
171 if (!assignable(arr->proj(*a, i), val->proj(*a, i))) return false;
172
173 return true;
174 }
175 } else if (auto vel = val->isa<Vel>()) {
176 if (assignable(type, vel->value())) return true;
177 }
178
179 return equiv(type, val_ty);
180}
