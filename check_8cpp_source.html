23bool Checker::equiv(const Def* d1, const Def* d2, const Def* dbg /*= {}*/) {
24 if (d1 == d2) return true;
25 if (!d1 || !d2 || d1->is_unfinished() || d2->is_unfinished()) return false;
26
27 // normalize: always put smaller gid to the left
28 if (d1->gid() > d2->gid()) std::swap(d1, d2);
29
30 if (auto [it, ins] = equiv_.emplace(std::pair(d1, d2), Equiv::Unknown); !ins) {
31 switch (it->second) {
32 case Equiv::Distinct: return false;
33 case Equiv::Unknown:
34 case Equiv::Equiv: return true;
35 default: unreachable();
36 }
37 }
38
39 bool res = equiv_internal(d1, d2, dbg);
40 equiv_[std::pair(d1, d2)] = res ? Equiv::Equiv : Equiv::Distinct;
41 return res;
42}

44bool Checker::equiv_internal(const Def* d1, const Def* d2, const Def* dbg) {
45 if (!equiv(d1->type(), d2->type(), dbg)) return false;
46 if (d1->isa<Top>() || d2->isa<Top>()) return equiv(d1->type(), d2->type(), dbg);
47
48 if (is_sigma_or_arr(d1)) {
49 if (!equiv(d1->arity(), d2->arity(), dbg)) return false;
50
51 if (auto a = isa_lit(d1->arity())) {
52 for (size_t i = 0; i != a; ++i) {
53 if (!equiv(d1->proj(*a, i), d2->proj(*a, i), dbg)) return false;
54 }
55 return true;
56 }
57 }
58
59 if (d1->node() != d2->node() || d1->flags() != d2->flags() || d1->num_ops() != d2->num_ops()) return false;
60
61 if (auto var = d1->isa<Var>()) { // vars are equal if they appeared under the same binder
62 for (auto [n1, n2] : vars_) {
63 if (var->nom() == n1) return d2->as<Var>()->nom() == n2;
64 }
65 return false;
66 }
67
68 if (auto n1 = d1->isa_nom()) {
69 if (auto n2 = d2->isa_nom()) vars_.emplace_back(n1, n2);
70 }
71
72 return std::ranges::equal(d1->ops(), d2->ops(), [this, dbg](auto op1, auto op2) { return equiv(op1, op2, dbg); });

75bool Checker::assignable(const Def* type, const Def* val, const Def* dbg /*= {}*/) {
76 if (type == val->type()) return true;
77
78 if (auto sigma = type->isa<Sigma>()) {

80
81 auto red = sigma->reduce(val);
82 for (size_t i = 0, a = red.size(); i != a; ++i) {
83 if (!assignable(red[i], val->proj(a, i, dbg), dbg)) return false;
84 }
85
86 return true;
87 } else if (auto arr = type->isa<Arr>()) {
88 if (!equiv(type->arity(), val->type()->arity(), dbg)) return false;
89
90 if (auto a = isa_lit(arr->arity())) {
91 for (size_t i = 0; i != *a; ++i) {

96 }
