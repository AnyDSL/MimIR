39std::vector<std::filesystem::path> get_plugin_search_paths(ArrayRef<std::string> user_paths) {
40 std::vector<std::filesystem::path> paths{user_paths.begin(), user_paths.end()};
41
42 add_paths_from_env(paths);
43
44 // make user paths absolute paths.
45 const auto cwd = std::filesystem::current_path();
46 std::ranges::transform(paths, paths.begin(), [&](auto path) { return path.is_relative() ? cwd / path : path; });
47
48 // add path/to/thorin.exe/../../lib/thorin
49 if (auto path = sys::path_to_curr_exe()) paths.emplace_back(path->parent_path().parent_path() / "lib" / "thorin");
50
51 // add default install path
52 const auto install_prefixed_path = std::filesystem::path{THORIN_INSTALL_PREFIX} / "lib" / "thorin";
53
54 if (paths.empty() ||
55 (std::filesystem::is_directory(install_prefixed_path) &&
56 !std::filesystem::equivalent(install_prefixed_path, paths.back()))) // avoid double checking install dir
57 paths.emplace_back(std::move(install_prefixed_path));
58
59 // add current working directory
60 paths.emplace_back(std::move(cwd));

74Dialect Dialect::load(const std::string& name, ArrayRef<std::string> search_paths) {
75 std::unique_ptr<void, decltype(&dl::close)> handle{nullptr, dl::close};
76 std::string plugin_path = name;
77 if (auto path = std::filesystem::path{name}; path.is_absolute() && std::filesystem::is_regular_file(path))
78 handle.reset(dl::open(name));
79 if (!handle) {
80 auto paths = get_plugin_search_paths(search_paths);
81 auto name_variants = get_plugin_name_variants(name);
82 for (const auto& path : paths) {
83 for (const auto& name_variant : name_variants) {
84 auto full_path = path / name_variant;
85 plugin_path = full_path.string();
86
87 std::error_code ignore;
88 if (bool reg_file = std::filesystem::is_regular_file(full_path, ignore); reg_file && !ignore)
89 if (handle.reset(dl::open(full_path.string())); handle) break;
90 }
91 if (handle) break;
92 }
93 }
94
95 if (!handle) throw std::runtime_error("cannot open plugin '" + name + "'");
96
97 return Dialect{plugin_path, std::move(handle)};
98}
