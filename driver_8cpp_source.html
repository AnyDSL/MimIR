52void Driver::load(Sym name) {
53 ILOG("loading plugin: '{}'", name);
54
55 if (is_loaded(name)) {

60 Plugin::Handle handle{nullptr, dl::close};
61 if (auto path = fs::path{*name}; path.is_absolute() && fs::is_regular_file(path)) handle.reset(dl::open(*name));
62 if (!handle) {
63 for (const auto& path : search_paths()) {
64 for (auto name_variants = get_plugin_name_variants(name); const auto& name_variant : name_variants) {
65 auto full_path = path / name_variant;
66 std::error_code ignore;
67 if (bool reg_file = fs::is_regular_file(full_path, ignore); reg_file && !ignore) {
68 auto path_str = full_path.string();
69 if (handle.reset(dl::open(path_str)); handle) break;
70 }
71 }
72 if (handle) break;
73 }
74 }
75
76 if (!handle) error("cannot open plugin '{}'", name);
77
78 if (auto get_info = reinterpret_cast<decltype(&thorin_get_plugin)>(dl::get(handle.get(), "thorin_get_plugin"))) {
79 assert_emplace(plugins_, name, std::move(handle));
80 auto info = get_info();
81 if (auto reg = info.register_passes) reg(passes_);
82 if (auto reg = info.register_normalizers) reg(normalizers_);
83 if (auto reg = info.register_backends) reg(backends_);
84 } else {
