40std::vector<Lam*> succs(Lam* lam) {
41 std::vector<Lam*> succs;
42 std::queue<const Def*> queue;
43 DefSet done;
44
45 auto enqueue = [&](const Def* def) {
46 if (done.find(def) == done.end()) {
47 queue.push(def);
48 done.insert(def);
49 }
50 };
51
52 done.insert(lam);
53 enqueue(lam->body());
54
55 while (!queue.empty()) {
56 auto def = pop(queue);
57 if (auto lam = def->isa_nom<Lam>()) {
58 succs.push_back(lam);
59 continue;
60 }
61
62 for (auto op : def->ops()) enqueue(op);
63 }

179bool PartialEvaluator::run() {
180 bool todo = false;
181
182 for (const auto& [name, nom] : world().externals()) {
183 if (auto lam = nom->isa<Lam>()) {
184 enqueue(lam);
185 top_level_[lam] = true;
186 }
187 }
188
189 while (!queue_.empty()) {
190 auto lam = pop(queue_);
191
192 auto app = lam->body()->isa<App>();
193 if (app == nullptr) continue;
194
195 bool force_fold = false;
196 auto callee_def = app->callee();
197
198 if (auto run = isa<Tag::PE>(PE::run, app->callee())) {
199 force_fold = true;
200 callee_def = run->arg();
201 }
202
203 if (auto callee = callee_def->isa_nom<Lam>()) {
204 // TODO
205 // if (callee->intrinsic() == Lam::Intrinsic::PeInfo) {
206 // eat_pe_info(lam);
207 // continue;
208 //}
209
210 if (callee->is_set()) {
211 size_t num_args = app->num_args();
212 DefArray args(num_args);
213
214 CondEval cond_eval(callee, app->args(), top_level_);
215
216 bool fold = false;
217 for (size_t i = 0; i != num_args; ++i) {
218 if (force_fold || cond_eval.eval(i, lower2cff_)) {
219 args[i] = app->arg(i);
220 fold = true;
221 } else {
222 args[i] = world().top(callee->var(i)->type());
223 }
224 }
225
226 if (fold) {
227 auto app = world().app(callee, args)->as<App>();
228 const auto& p = cache_.emplace(app, nullptr);
229 Lam*& target = p.first->second;
230 // create new specialization if not found in cache
231 if (p.second) {
232 target = drop(app);
233 todo = true;
234 }
235
236 app_to_dropped_app(lam, target, app);
237
238 if (lower2cff_ && fold) {
239 // re-examine next iteration:
240 // maybe the specialization is not top-level anymore which might need further specialization
241 queue_.push(lam);
242 continue;
243 }
244 }
245 }
246 }
247
248 for (auto succ : succs(lam)) enqueue(succ);
249 }
250
251 return todo;
252}
