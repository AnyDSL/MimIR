54namespace RMode {
55enum RMode : nat_t {
56 none = 0,
57 nnan = 1 << 0, ///< No NaNs - Allow optimizations to assume the arguments and result are not NaN. Such optimizations are required to retain defined behavior over NaNs, but the value of the result is undefined.
58 ninf = 1 << 1, ///< No Infs - Allow optimizations to assume the arguments and result are not +/-Inf. Such optimizations are required to retain defined behavior over +/-Inf, but the value of the result is undefined.
59 nsz = 1 << 2, ///< No Signed Zeros - Allow optimizations to treat the sign of a zero argument or result as insignificant.
60 arcp = 1 << 3, ///< Allow Reciprocal - Allow optimizations to use the reciprocal of an argument rather than perform division.
61 contract = 1 << 4, ///< Allow floating-point contraction (e.g. fusing a multiply followed by an addition into a fused multiply-and-add).
62 afn = 1 << 5, ///< Approximate functions - Allow substitution of approximate calculations for functions (sin, log, sqrt, etc). See floating-point intrinsic definitions for places where this can apply to LLVM’s intrinsic math functions.
63 reassoc = 1 << 6, ///< Allow reassociation transformations for floating-point operations. This may dramatically change results in floating point.
64 finite = nnan | ninf,
65 unsafe = nsz | arcp | reassoc,
66 fast = nnan | ninf | nsz | arcp | contract | afn | reassoc,
67 bot = fast,
68 top = none,
69};

88/// The 5 relations are disjoint and are organized as follows:
89/// ```
90/// ----
91/// 4321
92/// 01234567
93/// ////////
94/// 00001111
95/// y→ 00110011
96/// 01010101
97///
98/// x 0/000 ELLLXXXX
99/// ↓ 1/001 GELLXXXX
100/// 2/010 GGELXXXX
101/// 3/011 GGGEXXXX
102/// -4/4/100 YYYYELLL
103/// -3/5/101 YYYYGELL
104/// -2/6/110 YYYYGGEL X = plus, minus
105/// -1/7/111 YYYYGGGE Y = minus, plus
106/// ```
107/// The more obscure combinations are prefixed with @c _.
108/// The standard comparisons front ends want to use, don't have this prefix.
109#define THORIN_I_CMP(m) /* X Y G L E */ \
110 m(ICmp, _f) /* o o o o o - always false */ \
111 m(ICmp, e) /* o o o o x - equal */ \
112 m(ICmp, _l) /* o o o x o - less (same sign) */ \
113 m(ICmp, _le) /* o o o x x - less or equal */ \
114 m(ICmp, _g) /* o o x o o - greater (same sign) */ \
115 m(ICmp, _ge) /* o o x o x - greater or equal */ \
116 m(ICmp, _gl) /* o o x x o - greater or less */ \
117 m(ICmp, _gle) /* o o x x x - greater or less or equal == same sign */ \
118 m(ICmp, _y) /* o x o o o - minus plus */ \
119 m(ICmp, _ye) /* o x o o x - minus plus or equal */ \
120 m(ICmp, sl) /* o x o x o - signed less */ \
121 m(ICmp, sle) /* o x o x x - signed less or equal */ \
122 m(ICmp, ug) /* o x x o o - unsigned greater */ \
123 m(ICmp, uge) /* o x x o x - unsigned greater or equal */ \
124 m(ICmp, _ygl) /* o x x x o - minus plus or greater or less */ \
125 m(ICmp, _nx) /* o x x x x - not plus minus */ \
126 m(ICmp, _x) /* x o o o o - plus minus */ \
127 m(ICmp, _xe) /* x o o o x - plus minus or equal */ \
128 m(ICmp, ul) /* x o o x o - unsigned less */ \
129 m(ICmp, ule) /* x o o x x - unsigned less or equal */ \
130 m(ICmp, sg) /* x o x o o - signed greater */ \
131 m(ICmp, sge) /* x o x o x - signed greater or equal */ \
132 m(ICmp, _xgl) /* x o x x o - greater or less or plus minus */ \
133 m(ICmp, _ny) /* x o x x x - not minus plus */ \
134 m(ICmp, _xy) /* x x o o o - different sign */ \
135 m(ICmp, _xye) /* x x o o x - different sign or equal */ \
136 m(ICmp, _xyl) /* x x o x o - signed or unsigned less */ \
137 m(ICmp, _ng) /* x x o x x - signed or unsigned less or equal == not greater */ \
138 m(ICmp, _xyg) /* x x x o o - signed or unsigned greater */ \
139 m(ICmp, _nl) /* x x x o x - signed or unsigned greater or equal == not less */ \
140 m(ICmp, ne) /* x x x x o - not equal */ \
141 m(ICmp, _t) /* x x x x x - always true */

143#define THORIN_R_CMP(m) /* U G L E */ \
144 m(RCmp, f) /* o o o o - always false */ \
145 m(RCmp, e) /* o o o x - ordered and equal */ \
146 m(RCmp, l) /* o o x o - ordered and less */ \
147 m(RCmp, le) /* o o x x - ordered and less or equal */ \
148 m(RCmp, g) /* o x o o - ordered and greater */ \
149 m(RCmp, ge) /* o x o x - ordered and greater or equal */ \
150 m(RCmp, ne) /* o x x o - ordered and not equal */ \
151 m(RCmp, o) /* o x x x - ordered (no NaNs) */ \
152 m(RCmp, u) /* x o o o - unordered (either NaNs) */ \
153 m(RCmp, ue) /* x o o x - unordered or equal */ \
154 m(RCmp, ul) /* x o x o - unordered or less */ \
155 m(RCmp, ule) /* x o x x - unordered or less or equal */ \
156 m(RCmp, ug) /* x x o o - unordered or greater */ \
157 m(RCmp, uge) /* x x o x - unordered or greater or equal */ \
158 m(RCmp, une) /* x x x o - unordered or not equal */ \
159 m(RCmp, t) /* x x x x - always true */
