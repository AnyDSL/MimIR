21Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags, const Def* dbg)
22 : world_(w)
23 , flags_(flags)
24 , node_(unsigned(node))
25 , nom_(false)
26 , dep_(unsigned(node == Node::Axiom ? Dep::Axiom
27 : node == Node::Infer ? Dep::Infer
28 : node == Node::Proxy ? Dep::Proxy
29 : node == Node::Var ? Dep::Var
30 : Dep::None))
31 , num_ops_(ops.size())
32 , dbg_(dbg)
33 , type_(type) {
34 std::ranges::copy(ops, ops_ptr());
35 gid_ = world().next_gid();
36
37 if (node == Node::Univ) {
38 hash_ = murmur3(gid());
39 } else {
40 hash_ = type ? type->gid() : 0;
41 for (auto op : ops) hash_ = murmur3(hash_, u32(op->gid()));
42 hash_ = murmur3(hash_, flags_);
43 hash_ = murmur3_rest(hash_, u8(node));
44 hash_ = murmur3_finalize(hash_, num_ops());
45 }
46}

200const Var* Def::var(const Def* dbg) {
201 auto& w = world();
202
203 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
204 for (auto u : uses()) {
205 if (auto var = u->isa<Var>(); var && var->nom() == this) return var;
206 }
207
208 if (w.is_frozen()) return nullptr;
209 }
210
211 if (auto lam = isa<Lam >()) return w.var(lam ->dom(), lam, dbg);
212 if (auto pi = isa<Pi >()) return w.var(pi ->dom(), pi, dbg);
213 if (auto sig = isa<Sigma>()) return w.var(sig, sig, dbg);
214 if (auto arr = isa<Arr >()) return w.var(w.type_idx(arr ->shape()), arr, dbg); // TODO shapes like (2, 3)
215 if (auto pack = isa<Pack >()) return w.var(w.type_idx(pack->shape()), pack, dbg); // TODO shapes like (2, 3)
216 if (isa<Bound >()) return w.var(this, this, dbg);
217 if (isa<Infer >()) return nullptr;
218 if (isa<Global>()) return nullptr;
219 unreachable();
220}

255bool Def::equal(const Def* other) const {
256 if (isa<Univ>() || this->isa_nom() || other->isa_nom()) return this == other;

285

391const Def* Def::proj(nat_t a, nat_t i, const Def* dbg) const {
392 if (a == 1) {
393 if (!type()) return this;
394 if (!isa_nom<Sigma>() && !type()->isa_nom<Sigma>()) return this;
395 }
396
397 World& w = world();
398
399 if (isa<Tuple>() || isa<Sigma>()) {
400 return op(i);
401 } else if (auto arr = isa<Arr>()) {
402 if (arr->arity()->isa<Top>()) return arr->body();
403 return arr->reduce(w.lit_idx(as_lit(arr->arity()), i));
404 } else if (auto pack = isa<Pack>()) {
405 if (pack->arity()->isa<Top>()) return pack->body();
406 assert(!w.is_frozen() && "TODO");
407 return pack->reduce(w.lit_idx(as_lit(pack->arity()), i));
408 } else if (sort() == Sort::Term) {
409 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
410 for (auto u : uses()) {
411 if (auto ex = u->isa<Extract>(); ex && ex->tuple() == this) {
412 if (auto index = isa_lit(ex->index()); index && *index == i) return ex;
413 }
414 }
415
416 if (w.is_frozen()) return nullptr;
417 }
418
419 return w.extract(this, a, i, dbg);
420 }
421
422 return nullptr;
423}
