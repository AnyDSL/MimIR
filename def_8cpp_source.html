24Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags)
25 : world_(w)
26 , flags_(flags)
27 , node_(unsigned(node))
28 , nom_(false)
29 , external_(false)
30 , dep_(unsigned(node == Node::Axiom ? Dep::Axiom
31 : node == Node::Infer ? Dep::Infer
32 : node == Node::Proxy ? Dep::Proxy
33 : node == Node::Var ? Dep::Var
34 : Dep::None))
35 , num_ops_(ops.size())
36 , type_(type) {
37 std::ranges::copy(ops, ops_ptr());
38 gid_ = world().next_gid();
39
40 if (node == Node::Univ) {
41 hash_ = murmur3(gid());

43 hash_ = type ? type->gid() : 0;
44 for (auto op : ops) hash_ = murmur3(hash_, u32(op->gid()));
45 hash_ = murmur3(hash_, flags_);
46 hash_ = murmur3_rest(hash_, u8(node));
47 hash_ = murmur3_finalize(hash_, num_ops());
48 }

231const Var* Def::var() {
232 auto& w = world();
233
234 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
235 for (auto u : uses()) {
236 if (auto var = u->isa<Var>(); var && var->nom() == this) return var;
237 }
238
239 if (w.is_frozen()) return nullptr;
240 }
241
242 if (auto lam = isa<Lam >()) return w.var(lam ->dom(), lam);
243 if (auto pi = isa<Pi >()) return w.var(pi ->dom(), pi);
244 if (auto sig = isa<Sigma>()) return w.var(sig, sig);
245 if (auto arr = isa<Arr >()) return w.var(w.type_idx(arr ->shape()), arr ); // TODO shapes like (2, 3)
246 if (auto pack = isa<Pack >()) return w.var(w.type_idx(pack->shape()), pack); // TODO shapes like (2, 3)
247 if (isa<Bound >()) return w.var(this, this);
248 if (isa<Infer >()) return nullptr;
249 if (isa<Global>()) return nullptr;
250 unreachable();
251}

280bool Def::equal(const Def* other) const {
281 if (isa<Univ>() || this->isa_nom() || other->isa_nom()) return this == other;
282

396const Def* Def::proj(nat_t a, nat_t i) const {
397 if (a == 1) {
398 if (!type()) return this;
399 if (!isa_nom<Sigma>() && !type()->isa_nom<Sigma>()) return this;
400 }
401
402 World& w = world();
403
404 if (isa<Tuple>() || isa<Sigma>()) {
405 return op(i);
406 } else if (auto arr = isa<Arr>()) {
407 if (arr->arity()->isa<Top>()) return arr->body();
408 return arr->reduce(w.lit_idx(arr->as_lit_arity(), i));
409 } else if (auto pack = isa<Pack>()) {
410 if (pack->arity()->isa<Top>()) return pack->body();
411 assert(!w.is_frozen() && "TODO");
412 return pack->reduce(w.lit_idx(pack->as_lit_arity(), i));
413 }
414
415 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
416 for (auto u : uses()) {
417 if (auto ex = u->isa<Extract>(); ex && ex->tuple() == this) {
418 if (auto index = isa_lit(ex->index()); index && *index == i) return ex;
419 }
420 }
421
422 if (w.is_frozen()) return nullptr;
423 }
424
425 return w.extract(this, a, i);
426}
