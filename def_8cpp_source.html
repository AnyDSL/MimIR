18Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags)
19 : world_(w)
20 , flags_(flags)
21 , node_(unsigned(node))
22 , mut_(false)
23 , external_(false)
24 , dep_(unsigned(node == Node::Axiom ? Dep::Axiom
25 : node == Node::Infer ? Dep::Infer
26 : node == Node::Proxy ? Dep::Proxy
27 : node == Node::Var ? Dep::Var
28 : Dep::None))
29 , valid_(false)
30 , num_ops_(ops.size())
31 , type_(type) {
32 if (type) dep_ |= type->dep();
33 for (size_t i = 0, e = ops.size(); i != e; ++i) {
34 auto op = ops[i];
35 ops_ptr()[i] = op;
36 dep_ |= op->dep();
37 }
38
39 gid_ = world().next_gid();
40
41 if (auto var = isa<Var>()) {
42 vars_.local = world().vars().create(var);
43 muts_.local = Muts();
44 } else {
45 vars_.local = Vars();
46 muts_.local = Muts();
47
48 for (auto op : extended_ops()) {
49 vars_.local = world().vars().merge(vars_.local, op->local_vars());
50 muts_.local = world().muts().merge(muts_.local, op->local_muts());
51 }
52 }
53
54 if (node == Node::Univ) {
55 hash_ = mim::hash(gid());
56 } else {
57 hash_ = type ? hash_begin(type->gid()) : 0;
58 for (auto op : ops) hash_ = hash_combine(hash_, u32(op->gid()));
59 hash_ = hash_combine(hash_, flags_);
60 hash_ = hash_combine(hash_, u8(node));
61 hash_ = hash_combine(hash_, num_ops());
62 }

214DefVec Def::reduce(Ref arg) {

216 auto& cache = world().move_.cache;
217 if (auto i = cache.find({this, arg}); i != cache.end()) return i->second;

248Def* Def::set(size_t i, Ref def) {
249 invalidate();
250 def = check(i, def);
251 assert(def && !op(i) && curr_op_ == i);
252#ifndef NDEBUG
253 curr_op_ = (curr_op_ + 1) % num_ops();
254#endif

311Vars Def::free_vars() {
312 if (!is_set()) return {};
313
314 if (!valid_) {
315 // fixed-point interation to recompute vars_.free
316 uint32_t run = 1;
317 for (bool todo = true; todo;) {
318 todo = false;
319 free_vars(todo, ++run);
320 }
321 validate();
322 }
323
324 return vars_.free;
325}
326
327Vars Def::free_vars(bool& todo, uint32_t run) {
328 // Recursively recompute vars_.free. If
329 // * valid_ == true: no recomputation is necessary
330 // * mark_ == run: We are running in cycles within the *current* iteration of our fixed-point loop
331 if (valid_ || mark_ == run) return vars_.free;
332 mark_ = run;
333
334 auto fvs0 = vars_.free;
335 auto fvs = fvs0;
336
337 for (auto op : extended_ops()) fvs = world().vars().merge(fvs, op->local_vars());
338
339 for (auto op : extended_ops()) {
340 for (auto local_mut : op->local_muts()) {
341 local_mut->muts_.users = world().muts().insert(local_mut->muts_.users, this);
342 fvs = world().vars().merge(fvs, local_mut->free_vars(todo, run));
343 }
344 }
345
346 if (auto var = has_var()) fvs = world().vars().erase(fvs, var); // FV(λx.e) = FV(e) \ {x}
347
348 todo |= fvs0 != fvs;
349 return vars_.free = fvs;
350}

496 World& w = world();
497
498 if (auto arr = isa<Arr>()) {
499 if (arr->arity()->isa<Top>()) return arr->body();
