26Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags)
27 : world_(w)
28 , flags_(flags)
29 , node_(unsigned(node))
30 , mut_(false)
31 , external_(false)
32 , dep_(unsigned(node == Node::Axiom ? Dep::Axiom
33 : node == Node::Infer ? Dep::Infer
34 : node == Node::Proxy ? Dep::Proxy
35 : node == Node::Var ? Dep::Var
36 : Dep::None))
37 , num_ops_(ops.size())
38 , type_(type) {
39 std::ranges::copy(ops, ops_ptr());
40 gid_ = world().next_gid();
41
42 if (node == Node::Univ) {
43 hash_ = murmur3(gid());
44 } else {
45 hash_ = type ? type->gid() : 0;
46 for (auto op : ops) hash_ = murmur3(hash_, u32(op->gid()));
47 hash_ = murmur3(hash_, flags_);
48 hash_ = murmur3_rest(hash_, u8(node));
49 hash_ = murmur3_finalize(hash_, num_ops());
50 }
51}

277Ref Def::var() {
278 auto& w = world();
279
280 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
281 for (auto u : uses()) {
282 if (auto var = u->isa<Var>(); var && var->mut() == this) return var;
283 }
284
285 if (w.is_frozen()) return nullptr;
286 }
287
288 if (auto lam = isa<Lam >()) return w.var(lam ->dom(), lam);
289 if (auto pi = isa<Pi >()) return w.var(pi ->dom(), pi);
290 if (auto sig = isa<Sigma>()) return w.var(sig, sig);
291 if (auto arr = isa<Arr >()) return w.var(w.type_idx(arr ->shape()), arr ); // TODO shapes like (2, 3)
292 if (auto pack = isa<Pack >()) return w.var(w.type_idx(pack->shape()), pack); // TODO shapes like (2, 3)
293 if (isa<Bound >()) return w.var(this, this);
294 if (isa<Infer >()) return nullptr;
295 if (isa<Global>()) return nullptr;
296 fe::unreachable();
297}

425const Def* Def::proj(nat_t a, nat_t i) const {
426 if (a == 1) {
427 if (!type()) return this;
428 if (!isa_mut<Sigma>() && !type()->isa_mut<Sigma>()) return this;
429 }
430
431 World& w = world();
432
433 if (isa<Tuple>() || isa<Sigma>()) {
434 return op(i);
435 } else if (auto arr = isa<Arr>()) {
436 if (arr->arity()->isa<Top>()) return arr->body();
437 return arr->reduce(w.lit_idx(a, i));
438 } else if (auto pack = isa<Pack>()) {
439 if (pack->arity()->isa<Top>()) return pack->body();
440 assert(!w.is_frozen() && "TODO");
441 return pack->reduce(w.lit_idx(a, i));
442 }
443
444 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
445 for (auto u : uses()) {
446 if (auto ex = u->isa<Extract>(); ex && ex->tuple() == this) {
447 if (auto index = Lit::isa(ex->index()); index && *index == i) return ex;
448 }
449 }
450
451 if (w.is_frozen()) return nullptr;
452 }
453
454 return w.extract(this, a, i);
455}
