21Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags, const Def* dbg)
22 : world_(w)
23 , flags_(flags)
24 , node_(unsigned(node))
25 , nom_(false)
26 , dep_(node == Node::Axiom ? Dep::Axiom
27 : node == Node::Proxy ? Dep::Proxy
28 : node == Node::Var ? Dep::Var
29 : Dep::None)
30 , num_ops_(ops.size())
31 , dbg_(dbg)
32 , type_(type) {
33 std::ranges::copy(ops, ops_ptr());
34 gid_ = world().next_gid();
35
36 if (node == Node::Univ) {
37 hash_ = murmur3(gid());
38 } else {
39 hash_ = type ? type->gid() : 0;
40 for (auto op : ops) hash_ = murmur3(hash_, u32(op->gid()));
41 hash_ = murmur3(hash_, flags_);
42 hash_ = murmur3_rest(hash_, u8(node));
43 hash_ = murmur3_finalize(hash_, num_ops());
44 }

189const Var* Def::var(const Def* dbg) {
190 auto& w = world();
191
192 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
193 for (auto u : uses()) {
194 if (auto var = u->isa<Var>(); var && var->nom() == this) return var;
195 }
196
197 if (w.is_frozen()) return nullptr;
198 }
199
200 if (auto lam = isa<Lam >()) return w.var(lam ->dom(), lam, dbg);
201 if (auto pi = isa<Pi >()) return w.var(pi ->dom(), pi, dbg);
202 if (auto sig = isa<Sigma>()) return w.var(sig, sig, dbg);
203 if (auto arr = isa<Arr >()) return w.var(w.type_int(arr ->shape()), arr, dbg); // TODO shapes like (2, 3)
204 if (auto pack = isa<Pack >()) return w.var(w.type_int(pack->shape()), pack, dbg); // TODO shapes like (2, 3)
205 if (isa_bound(this)) return w.var(this, this, dbg);
206 if (isa<Infer >()) return nullptr;
207 if (isa<Global>()) return nullptr;
208 unreachable();
209}

380const Def* Def::proj(nat_t a, nat_t i, const Def* dbg) const {
381 if (a == 1 && (!isa_nom<Sigma>() && !type()->isa_nom<Sigma>())) return this;
382 World& w = world();
383
384 if (isa<Tuple>() || isa<Sigma>()) {
385 return op(i);
386 } else if (auto arr = isa<Arr>()) {
387 if (arr->arity()->isa<Top>()) return arr->body();
388 if (!w.type_int()) return arr->op(i); // hack for alpha equiv check of sigma (dbg of %Int..)
389 return arr->reduce(w.lit_int(as_lit(arr->arity()), i)).back();
390 } else if (auto pack = isa<Pack>()) {
391 if (pack->arity()->isa<Top>()) return pack->body();
392 assert(!w.is_frozen() && "TODO");
393 return pack->reduce(w.lit_int(as_lit(pack->arity()), i)).back();
394 } else if (sort() == Sort::Term) {
395 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
396 for (auto u : uses()) {
397 if (auto ex = u->isa<Extract>(); ex && ex->tuple() == this) {
398 if (auto index = isa_lit(ex->index()); *index == i) return ex;
399 }
400 }
401
402 if (w.is_frozen()) return nullptr;
403 }
404
405 return w.extract(this, a, i, dbg);
406 }
407
408 return nullptr;
409}
