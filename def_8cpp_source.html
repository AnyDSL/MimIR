240void Def::finalize() {
241 for (size_t i = 0, e = num_ops(); i != e; ++i) {
242 if (auto dep = op(i)->dep(); dep != Dep::Bot) {
243 dep_ |= dep;
244 const auto& p = op(i)->uses_.emplace(this, i);
245 assert_unused(p.second);
246 }
247 order_ = std::max(order_, op(i)->order_);
248 }
249
250 if (!isa<Space>() && !isa<Axiom>()) {
251 if (auto dep = type()->dep(); dep != Dep::Bot) {
252 dep_ |= dep;
253 const auto& p = type()->uses_.emplace(this, -1);
254 assert_unused(p.second);
255 }
256 }
257
258 assert(!dbg() || dbg()->no_dep());
259 if (isa<Pi>()) ++order_;
260 if (auto var = isa<Var>()) {
261 var->nom()->var_ = true;
262 dep_ = Dep::Var;
263 }
264
265 if (isa<Proxy>()) {
266 proxy_ = true;
267 } else {
268 for (auto op : extended_ops()) proxy_ |= op->contains_proxy();
269 }
270}

362const Def* Def::proj(nat_t a, nat_t i, const Def* dbg) const {
363 if (a == 1 && (!isa_nom<Sigma>() && !type()->isa_nom<Sigma>())) return this;
364
365 if (isa<Tuple>() || isa<Sigma>()) {
366 return op(i);
367 } else if (auto arr = isa<Arr>()) {
368 if (arr->arity()->isa<Top>()) return arr->body();
369 return arr->reduce(world().lit_int(as_lit(arr->arity()), i)).back();
370 } else if (auto pack = isa<Pack>()) {
371 if (pack->arity()->isa<Top>()) return pack->body();
372 return pack->reduce(world().lit_int(as_lit(pack->arity()), i)).back();
373 } else if (sort() == Sort::Term) {
374 return world().extract(this, a, i, dbg);
375 }
376
377 return nullptr;
378}
