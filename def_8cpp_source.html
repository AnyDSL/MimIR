26Def::Def(World* w, node_t node, const Def* type, Defs ops, flags_t flags)
27 : world_(w)
28 , flags_(flags)
29 , node_(unsigned(node))
30 , mut_(false)
31 , external_(false)
32 , dep_(unsigned(node == Node::Axiom ? Dep::Axiom
33 : node == Node::Infer ? Dep::Infer
34 : node == Node::Proxy ? Dep::Proxy
35 : node == Node::Var ? Dep::Var
36 : Dep::None))
37 , num_ops_(ops.size())
38 , type_(type) {
39 std::ranges::copy(ops, ops_ptr());
40 gid_ = world().next_gid();
41
42 if (node == Node::Univ) {
43 hash_ = murmur3(gid());
44 } else {
45 hash_ = type ? type->gid() : 0;
46 for (auto op : ops) hash_ = murmur3(hash_, u32(op->gid()));
47 hash_ = murmur3(hash_, flags_);
48 hash_ = murmur3_rest(hash_, u8(node));
49 hash_ = murmur3_finalize(hash_, num_ops());
50 }
51}

275Ref Def::var() {
276 auto& w = world();
277
278 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
279 for (auto u : uses()) {
280 if (auto var = u->isa<Var>(); var && var->mut() == this) return var;
281 }
282
283 if (w.is_frozen()) return nullptr;
284 }

287 if (auto pi = isa<Pi >()) return w.var(pi ->dom(), pi);
288 if (auto sig = isa<Sigma>()) return w.var(sig, sig);
289 if (auto arr = isa<Arr >()) return w.var(w.type_idx(arr ->shape()), arr ); // TODO shapes like (2, 3)
290 if (auto pack = isa<Pack >()) return w.var(w.type_idx(pack->shape()), pack); // TODO shapes like (2, 3)
291 if (isa<Bound >()) return w.var(this, this);
292 if (isa<Infer >()) return nullptr;
293 if (isa<Global>()) return nullptr;
294 unreachable();

423const Def* Def::proj(nat_t a, nat_t i) const {
424 if (a == 1) {
425 if (!type()) return this;
426 if (!isa_mut<Sigma>() && !type()->isa_mut<Sigma>()) return this;
427 }
428
429 World& w = world();
430
431 if (isa<Tuple>() || isa<Sigma>()) {
432 return op(i);
433 } else if (auto arr = isa<Arr>()) {
434 if (arr->arity()->isa<Top>()) return arr->body();
435 return arr->reduce(w.lit_idx(a, i));
436 } else if (auto pack = isa<Pack>()) {
437 if (pack->arity()->isa<Top>()) return pack->body();
438 assert(!w.is_frozen() && "TODO");
439 return pack->reduce(w.lit_idx(a, i));
440 }
441
442 if (w.is_frozen() || uses().size() < Search_In_Uses_Threshold) {
443 for (auto u : uses()) {
444 if (auto ex = u->isa<Extract>(); ex && ex->tuple() == this) {
445 if (auto index = Lit::isa(ex->index()); index && *index == i) return ex;
446 }
447 }
448
449 if (w.is_frozen()) return nullptr;
450 }
451
452 return w.extract(this, a, i);
453}
