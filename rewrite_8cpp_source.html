22Ref Rewriter::rewrite_imm(Ref old_def) {
23 // Extracts are used as conditional branches: make sure that we don't rewrite unreachable stuff.
24 if (auto extract = old_def->isa<Extract>()) {
25 if (auto index = Lit::isa(rewrite(extract->index()))) {
26 if (auto tuple = extract->tuple()->isa<Tuple>()) return rewrite(tuple->op(*index));
27 if (auto pack = extract->tuple()->isa_imm<Pack>(); pack && pack->shape()->dep_const())
28 return rewrite(pack->body());

30 }
31
32 auto new_type = old_def->isa<Type>() ? nullptr : rewrite(old_def->type());

34 for (size_t i = 0, e = new_ops.size(); i != e; ++i) new_ops[i] = rewrite(old_def->op(i));
