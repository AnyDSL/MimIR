122template<Sort sort>
123Ref World::umax(DefArray ops) {
124 level_t lvl = 0;
125 for (auto& op : ops) {
126 Ref r = op;
127 if (sort == Sort::Term) r = r->unfold_type();
128 if (sort <= Sort::Type) r = r->unfold_type();
129 if (sort <= Sort::Kind) {
130 if (auto type = r->isa<Type>())
131 r = type->level();
132 else
133 error(r, "operand '{}' must be a .Type of some level", r);
134 }
135
136 if (!r->type()->isa<Univ>())
137 error(r, "operand '{}' of a universe max must be of type '.Univ' but is of type '{}'", r, r->type());
138
139 op = r;
140
141 if (auto l = Lit::isa(op))
142 lvl = std::max(lvl, *l);
143 else
144 lvl = level_t(-1);
145 }
146
147 auto ldef = lvl == level_t(-1) ? (const Def*)unify<UMax>(ops.size(), *this, ops) : lit_univ(lvl);
148 std::ranges::sort(ops, [](auto op1, auto op2) { return op1->gid() < op2->gid(); });
149 return sort == Sort::Univ ? ldef : type(ldef);
150}

162Ref World::iapp(Ref callee, Ref arg) {
163 while (auto pi = callee->type()->isa<Pi>()) {
164 if (pi->is_implicit()) {
165 auto infer = mut_infer_entity();
166 auto a = app(callee, infer);
167 callee = a;
168 } else {
169 // resolve Infers now if possible before normalizers are run
170 if (auto app = callee->isa<App>(); app && app->curry() == 1) {
171 checker().assignable(callee->type()->as<Pi>()->dom(), arg);
172 auto apps = decurry(app);
173 callee = apps.front()->callee();
174 for (auto app : apps) callee = this->app(callee, Ref::refer(app->arg()));
175 }
176 break;
177 }
178 }
179
180 return app(callee, arg);
181}

183Ref World::app(Ref callee, Ref arg) {
184 // try to eliminate Infers if present - TODO better place for this?
185 if (callee->has_dep(Dep::Infer) || arg->has_dep(Dep::Infer)) {
186 InferRewriter rw(*this);
187 callee = rw.rewrite(callee);
188 arg = rw.rewrite(arg);
189 }
190
191 auto pi = callee->type()->isa<Pi>();
192
193 // (a, b)#i arg where a = A -> B; b = A -> B
194 if (auto extract = callee->type()->isa<Extract>()) {
195 if (auto tuple = extract->tuple()->isa<Tuple>()) {
196 if (auto uni = checker().is_uniform(tuple->ops())) pi = uni->isa<Pi>();
197 }
198 }
199
200 if (!pi) error(callee, "called expression '{}' : '{}' is not of function type", callee, callee->type());
201 if (!checker().assignable(pi->dom(), arg))
202 error(arg, "cannot pass argument \n'{}' of type \n'{}' to \n'{}' of domain \n'{}'", arg, arg->type(), callee,
203 pi->dom());
204
205 if (auto imm = callee->isa_imm<Lam>()) return imm->body();
206 if (auto lam = callee->isa<Lam>(); lam && lam->is_set() && !lam->is_term()) return lam->reduce(arg).back();
207
208 auto type = pi->reduce(arg).back();
209 return raw_app<true>(type, callee, arg);
210}

214 auto [axiom, curry, trip] = Axiom::get(callee);
215 if (axiom) {

217 curry = curry == Axiom::Trip_End ? curry : curry - 1;
218
219 if (auto normalize = axiom->normalizer(); Normalize && normalize && curry == 0)
220 return normalize(type, callee, arg);
221 }

223 return unify<App>(2, axiom, curry, trip, type, callee, arg);

246Ref World::tuple(Ref type, Defs ops) {
247 // TODO type-check type vs inferred type
248
249 auto n = ops.size();
250 if (!type->isa_mut<Sigma>()) {
251 if (n == 0) return tuple();
252 if (n == 1) return ops[0];
253 auto front = ops.front();
254 if (std::ranges::all_of(ops.skip_front(), [front](auto op) { return front == op; })) return pack(n, front);
255 }
256
257 if (n != 0) {
258 // eta rule for tuples:
259 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
260 if (auto extract = ops[0]->isa<Extract>()) {
261 auto tup = extract->tuple();
262 bool eta = tup->type() == type;
263 for (size_t i = 0; i != n && eta; ++i) {
264 if (auto extract = ops[i]->isa<Extract>()) {
265 if (auto index = Lit::isa(extract->index())) {
266 if (eta &= u64(i) == *index) {
267 eta &= extract->tuple() == tup;
268 continue;
269 }
270 }
271 }
272 eta = false;
273 }
274
275 if (eta) return tup;
276 }
277 }
278
279 return unify<Tuple>(ops.size(), type, ops);
280}

288Ref World::extract(Ref d, Ref index) {
289 assert(d);
290 if (index->isa<Tuple>()) {
291 auto n = index->num_ops();
292 DefArray idx(n, [&](size_t i) { return index->op(i); });
293 DefArray ops(n, [&](size_t i) { return d->proj(n, Lit::as(idx[i])); });
294 return tuple(ops);
295 } else if (index->isa<Pack>()) {
296 DefArray ops(index->as_lit_arity(), [&](size_t) { return extract(d, index->ops().back()); });
297 return tuple(ops);
298 }
299
300 Ref size = Idx::size(index->type());
301 Ref type = d->unfold_type();
302
303 if (auto l = Lit::isa(size); l && *l == 1) {
304 if (auto l = Lit::isa(index); !l || *l != 0) WLOG("unknown Idx of size 1: {}", index);
305 if (auto sigma = type->isa_mut<Sigma>(); sigma && sigma->num_ops() == 1) {
306 // mut sigmas can be 1-tuples; TODO mutables Arr?
307 } else {
308 return d;
309 }
310 }
311
312 if (auto pack = d->isa_imm<Pack>()) return pack->body();
313
314 if (!checker().equiv(type->arity(), size))
315 error(index, "index '{}' does not fit within arity '{}'", index, type->arity());
316
317 // extract(insert(x, index, val), index) -> val
318 if (auto insert = d->isa<Insert>()) {
319 if (index == insert->index()) return insert->value();
320 }
321
322 if (auto i = Lit::isa(index)) {
323 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
324
325 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
326 if (auto insert = d->isa<Insert>()) {
327 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index);
328 }
329
330 if (auto sigma = type->isa<Sigma>()) {
331 if (auto mut_sigma = sigma->isa_mut<Sigma>()) {
332 Scope scope(mut_sigma);
333 auto t = rewrite(sigma->op(*i), mut_sigma->var(), d, scope);
334 return unify<Extract>(2, t, d, index);
335 }
336
337 return unify<Extract>(2, sigma->op(*i), d, index);
338 }
339 }
340
341 const Def* elem_t;
342 if (auto arr = type->isa<Arr>())
343 elem_t = arr->reduce(index);
344 else
345 elem_t = extract(tuple(type->as<Sigma>()->ops()), index);
346
347 assert(d);
348 return unify<Extract>(2, elem_t, d, index);
349}

351Ref World::insert(Ref d, Ref index, Ref val) {
352 auto type = d->unfold_type();
353 auto size = Idx::size(index->type());
354
355 if (!checker().equiv(type->arity(), size))
356 error(index, "index '{}' does not fit within arity '{}'", index, type->arity());
357
358 if (auto index_lit = Lit::isa(index)) {
359 auto target_type = type->proj(*index_lit);
360 if (!checker().assignable(target_type, val))
361 error(val, "value of type {} is not assignable to type {}", val->type(), target_type);
362 }
363
364 if (auto l = Lit::isa(size); l && *l == 1)
365 return tuple(d, {val}); // d could be mut - that's why the tuple ctor is needed
366
367 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
368 if (auto t = d->isa<Tuple>()) return t->refine(Lit::as(index), val);
369
370 // insert(4; x, 2, y) -> (x, x, y, x)
371 if (auto pack = d->isa<Pack>()) {
372 if (auto a = pack->isa_lit_arity()) {
373 DefArray new_ops(*a, pack->body());
374 new_ops[Lit::as(index)] = val;
375 return tuple(type, new_ops);
376 }
377 }
378
379 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
380 if (auto insert = d->isa<Insert>()) {
381 if (insert->index() == index) d = insert->tuple();
382 }
383
384 return unify<Insert>(3, d, index, val);
385}

388Ref World::arr(Ref shape, Ref body) {
389 if (!is_shape(shape->type())) error(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
390
391 if (auto a = Lit::isa(shape)) {
392 if (*a == 0) return sigma();
393 if (*a == 1) return body;
394 }
395
396 // (a, b)#i; T -> (a, T, <b, T)#i
397 if (auto ex = shape->isa<Extract>()) {
398 if (auto tup = ex->tuple()->isa<Tuple>()) {
399 DefArray arrs(tup->num_ops(), [&](size_t i) { return arr(tup->op(i), body); });
400 return extract(tuple(arrs), ex->index());
401 }
402 }
403
404 // (a, b, c); body -> a; (b, c); body
405 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body));
406
407 // <n; x>; body -> x; <n-1, x>; body
408 if (auto p = shape->isa<Pack>()) {
409 if (auto s = Lit::isa(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body));
410 }
411
412 return unify<Arr>(2, body->unfold_type(), shape, body);
413}

415Ref World::pack(Ref shape, Ref body) {
416 if (!is_shape(shape->type())) error(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
417
418 if (auto a = Lit::isa(shape)) {
419 if (*a == 0) return tuple();
420 if (*a == 1) return body;
421 }
422
423 // <(a, b, c); body> -> <a; (b, c); body>>
424 if (auto tuple = shape->isa<Tuple>()) return pack(tuple->ops().front(), pack(tuple->ops().skip_front(), body));
425
426 // <<n; x>; body> -> <x; <<n-1, x>; body>>
427 if (auto p = shape->isa<Pack>()) {
428 if (auto s = Lit::isa(p->shape())) return pack(*s, pack(pack(*s - 1, p->body()), body));
429 }
430
431 auto type = arr(shape, body->type());
432 return unify<Pack>(1, type, body);
433}

469template<bool Up>
470Ref World::bound(Defs ops) {
471 auto kind = umax<Sort::Type>(ops);
472
473 // has ext<Up> value?
474 if (std::ranges::any_of(ops, [&](Ref op) { return Up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
475 return ext<Up>(kind);
476
477 // ignore: ext<!Up>
478 DefArray cpy(ops);
479 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](Ref op) { return !op->isa<Ext>(); });
480
481 // sort and remove duplicates
482 std::sort(cpy.begin(), end, GIDLt<const Def*>());
483 end = std::unique(cpy.begin(), end);
484 cpy.shrink(std::distance(cpy.begin(), end));
485
486 if (cpy.size() == 0) return ext<!Up>(kind);
487 if (cpy.size() == 1) return cpy[0];
488
489 // TODO simplify mixed terms with joins and meets
490
491 return unify<TBound<Up>>(cpy.size(), kind, cpy);
492}
