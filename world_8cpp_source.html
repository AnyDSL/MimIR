297const Def* World::app(const Def* callee, const Def* arg, const Def* dbg) {
298 auto pi = callee->type()->as<Pi>();
299
300 if (err()) {
301 if (!checker_->assignable(pi->dom(), arg)) err()->ill_typed_app(callee, arg);
302 }
303
304 auto type = pi->reduce(arg).back();
305 auto [axiom, curry] = Axiom::get(callee); // TODO move down again
306 if (axiom && curry == 1) {

308 }
309
310 return unify<App>(2, axiom, curry - 1, type, callee, arg, dbg);

320const Def* World::sigma(Defs ops, const Def* dbg) {
321 auto n = ops.size();
322 if (n == 0) return sigma();
323 if (n == 1) return ops[0];

345const Def* World::tuple(const Def* type, Defs ops, const Def* dbg) {
346 if (err()) {
347 // TODO type-check type vs inferred type
348 }
349
350 auto n = ops.size();
351 if (!type->isa_nom<Sigma>()) {
352 if (n == 0) return tuple();
353 if (n == 1) return ops[0];
354 if (std::all_of(ops.begin() + 1, ops.end(), [&](auto op) { return ops[0] == op; })) return pack(n, ops[0]);
355 }
356
357 // eta rule for tuples:
358 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
359 if (n == 0) goto out;
360
361 if (auto extract = ops[0]->isa<Extract>()) {
362 auto tup = extract->tuple();
363 bool eta = tup->type() == type;
364 for (size_t i = 0; i != n && eta; ++i) {
365 if (auto extract = ops[i]->isa<Extract>()) {
366 if (auto index = isa_lit(extract->index())) {
367 if (eta &= u64(i) == *index) {
368 eta &= extract->tuple() == tup;
369 continue;
370 }
371 }
372 }
373 eta = false;
374 }
375
376 if (eta) return tup;
377 }
378
379out:
380 return unify<Tuple>(ops.size(), type, ops, dbg);
381}

389const Def* World::extract_(const Def* ex_type, const Def* tup, const Def* index, const Def* dbg) {
390 if (index->isa<Arr>() || index->isa<Pack>()) {
391 DefArray ops(as_lit(index->arity()), [&](size_t) { return extract(tup, index->ops().back()); });
392 return index->isa<Arr>() ? sigma(ops, dbg) : tuple(ops, dbg);
393 } else if (index->isa<Sigma>() || index->isa<Tuple>()) {
394 auto n = index->num_ops();
395 DefArray idx(n, [&](size_t i) { return index->op(i); });
396 DefArray ops(n, [&](size_t i) { return tup->proj(n, as_lit(idx[i])); });
397 return index->isa<Sigma>() ? sigma(ops, dbg) : tuple(ops, dbg);
398 }
399
400 auto type = tup->type()->reduce_rec();
401 if (err()) {
402 if (!checker_->equiv(type->arity(), isa_sized_type(index->type())))
403 err()->index_out_of_range(type->arity(), index);
404 }
405
406 // nom sigmas can be 1-tuples
407 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1 && !tup->type()->isa_nom<Sigma>())
408 return tup;
409 if (auto pack = tup->isa_structural<Pack>()) return pack->body();
410
411 // extract(insert(x, index, val), index) -> val
412 if (auto insert = tup->isa<Insert>()) {
413 if (index == insert->index()) return insert->value();
414 }
415
416 if (auto i = isa_lit(index)) {
417 if (auto tuple = tup->isa<Tuple>()) return tuple->op(*i);
418
419 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
420 if (auto insert = tup->isa<Insert>()) {
421 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index, dbg);
422 }
423
424 if (auto sigma = type->isa<Sigma>()) {
425 if (auto nom_sigma = sigma->isa_nom<Sigma>()) {
426 Scope scope(nom_sigma);
427 auto t = rewrite(sigma->op(*i), nom_sigma->var(), tup, scope);
428 return unify<Extract>(2, ex_type ? ex_type : t, tup, index, dbg);
429 }
430
431 return unify<Extract>(2, ex_type ? ex_type : sigma->op(*i), tup, index, dbg);
432 }
433 }
434
435 type = type->as<Arr>()->body();
436 return unify<Extract>(2, type, tup, index, dbg);
437}

439const Def* World::insert(const Def* tup, const Def* index, const Def* val, const Def* dbg) {
440 auto type = tup->type()->reduce_rec();
441
442 if (err() && !checker_->equiv(type->arity(), isa_sized_type(index->type())))
443 err()->index_out_of_range(type->arity(), index);
444
445 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1)
446 return tuple(tup, {val}, dbg); // tup could be nom - that's why the tuple ctor is needed
447
448 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
449 if (auto t = tup->isa<Tuple>()) return t->refine(as_lit(index), val);
450
451 // insert(4; x, 2, y) -> (x, x, y, x)
452 if (auto pack = tup->isa<Pack>()) {
453 if (auto a = isa_lit(pack->arity())) {
454 DefArray new_ops(*a, pack->body());
455 new_ops[as_lit(index)] = val;
456 return tuple(type, new_ops, dbg);
457 }
458 }
459
460 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
461 if (auto insert = tup->isa<Insert>()) {
462 if (insert->index() == index) tup = insert->tuple();
463 }
464
465 return unify<Insert>(3, tup, index, val, dbg);
466}

468bool is_shape(const Def* s) {
469 if (s->isa<Nat>()) return true;
470 if (auto arr = s->isa<Arr>()) return arr->body()->isa<Nat>();
471 if (auto sig = s->isa_structural<Sigma>())
472 return std::ranges::all_of(sig->ops(), [](const Def* op) { return op->isa<Nat>(); });
473
474 return false;
475}

477const Def* World::arr(const Def* shape, const Def* body, const Def* dbg) {
478 if (err()) {
479 if (!is_shape(shape->type())) err()->expected_shape(shape);
480 }
481
482 if (auto a = isa_lit<u64>(shape)) {
483 if (*a == 0) return sigma();
484 if (*a == 1) return body;
485 }
486
487 // (a, b, c); body -> a; (b, c); body
488 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body), dbg);
489
490 // <n; x>; body -> x; <n-1, x>; body
491 if (auto p = shape->isa<Pack>()) {
492 if (auto s = isa_lit(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body), dbg);
493 }
494
495 return unify<Arr>(2, body->inf_type(), shape, body, dbg);
496}

498const Def* World::pack(const Def* shape, const Def* body, const Def* dbg) {
499 if (err()) {
500 if (!is_shape(shape->type())) err()->expected_shape(shape);
501 }
502
503 if (auto a = isa_lit<u64>(shape)) {
504 if (*a == 0) return tuple();
505 if (*a == 1) return body;
506 }
507
508 // <(a, b, c); body> -> <a; (b, c); body>>
509 if (auto tuple = shape->isa<Tuple>()) return pack(tuple->ops().front(), pack(tuple->ops().skip_front(), body), dbg);
510
511 // <<n; x>; body> -> <x; <<n-1, x>; body>>
512 if (auto p = shape->isa<Pack>()) {
513 if (auto s = isa_lit(p->shape())) return pack(*s, pack(pack(*s - 1, p->body()), body), dbg);
514 }
515
516 auto type = arr(shape, body->type());
517 return unify<Pack>(1, type, body, dbg);
518}

561const Def* World::ext(const Def* type, const Def* dbg) {
562 if (auto arr = type->isa<Arr>()) return pack(arr->shape(), ext<up>(arr->body()), dbg);
563 if (auto sigma = type->isa<Sigma>())

568template<bool up>
569const Def* World::bound(Defs ops, const Def* dbg) {
570 auto kind = infer_type(ops);
571
572 // has ext<up> value?
573 if (std::ranges::any_of(ops, [&](const Def* op) { return up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
574 return ext<up>(kind);
575
576 // ignore: ext<!up>
577 DefArray cpy(ops);
578 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](const Def* op) { return !isa_ext(op); });
579
580 // sort and remove duplicates
581 std::sort(cpy.begin(), end, GIDLt<const Def*>());
582 end = std::unique(cpy.begin(), end);
583 cpy.shrink(std::distance(cpy.begin(), end));
584
585 if (cpy.size() == 0) return ext<!up>(kind, dbg);
586 if (cpy.size() == 1) return cpy[0];
587
588 // TODO simplify mixed terms with joins and meets
589
590 return unify<TBound<up>>(cpy.size(), kind, cpy, dbg);
591}

726template<bool elide_empty>
727void World::visit(VisitFn f) const {
728 unique_queue<NomSet> noms;
729 unique_stack<DefSet> defs;
730
731 for (const auto& [name, nom] : externals()) {
732 assert(nom->is_set() && "external must not be empty");
733 noms.push(nom);
734 }
735
736 while (!noms.empty()) {
737 auto nom = noms.pop();
738 if (elide_empty && nom->is_unset()) continue;
739
740 Scope scope(nom);
741 f(scope);
742
743 for (auto nom : scope.free_noms()) noms.push(nom);
744 }
745}
