99template<Sort sort>
100Ref World::umax(DefArray ops) {
101 level_t lvl = 0;
102 for (auto& op : ops) {
103 Ref r = op;
104 if (sort == Sort::Term) r = r->unfold_type();
105 if (sort <= Sort::Type) r = r->unfold_type();
106 if (sort <= Sort::Kind) {
107 if (auto type = r->isa<Type>())
108 r = type->level();
109 else
110 err(r, "operand '{}' must be a .Type of some level", r);
111 }
112
113 if (!r->type()->isa<Univ>())
114 err(r, "operand '{}' of a universe max must be of type '.Univ' but is of type '{}'", r, r->type());
115
116 op = r;
117
118 if (auto l = isa_lit(op))
119 lvl = std::max(lvl, *l);
120 else
121 lvl = level_t(-1);
122 }
123
124 auto ldef = lvl == level_t(-1) ? (const Def*)unify<UMax>(ops.size(), *this, ops) : lit_univ(lvl);
125 std::ranges::sort(ops, [](auto op1, auto op2) { return op1->gid() < op2->gid(); });
126 return sort == Sort::Univ ? ldef : type(ldef);
127}

129Ref World::app(Ref callee, Ref arg) {
130 auto pi = callee->type()->isa<Pi>();
131
132 // (a, b)#i arg where a = A -> B; b = A -> B
133 if (auto extract = callee->type()->isa<Extract>()) {
134 if (auto tuple = extract->tuple()->isa<Tuple>()) {
135 if (auto uni = checker().is_uniform(tuple->ops())) pi = uni->isa<Pi>();
136 }
137 }
138
139 if (!pi) err(callee, "called expression '{}' : '{}' is not of function type", callee, callee->type());
140 if (!checker().assignable(pi->dom(), arg))
141 err(arg, "cannot pass argument '{}' of type '{}' to '{}' of domain '{}'", arg, arg->type(), callee, pi->dom());
142
143 if (auto lam = callee->isa<Lam>(); lam && lam->is_set() && !lam->is_term()) return lam->reduce(arg).back();
144
145 auto type = pi->reduce(arg).back();
146 return raw_app<true>(type, callee, arg);
147}

149template<bool Normalize>
150Ref World::raw_app(Ref type, Ref callee, Ref arg) {
151 auto [axiom, curry, trip] = Axiom::get(callee);
152 if (axiom) {
153 curry = curry == 0 ? trip : curry;
154 curry = curry == Axiom::Trip_End ? curry : curry - 1;
155
156 if (auto normalize = axiom->normalizer(); Normalize && normalize && curry == 0)
157 return normalize(type, callee, arg);
158 }
159
160 return unify<App>(2, axiom, curry, trip, type, callee, arg);
161}

189Ref World::tuple(Ref type, Defs ops) {
190 // TODO type-check type vs inferred type
191
192 auto n = ops.size();
193 if (!type->isa_nom<Sigma>()) {
194 if (n == 0) return tuple();
195 if (n == 1) return ops[0];
196 auto front = ops.front();
197 if (std::ranges::all_of(ops.skip_front(), [front](auto op) { return front == op; })) return pack(n, front);
198 }
199
200 if (n != 0) {
201 // eta rule for tuples:
202 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
203 if (auto extract = ops[0]->isa<Extract>()) {
204 auto tup = extract->tuple();
205 bool eta = tup->type() == type;
206 for (size_t i = 0; i != n && eta; ++i) {
207 if (auto extract = ops[i]->isa<Extract>()) {
208 if (auto index = isa_lit(extract->index())) {
209 if (eta &= u64(i) == *index) {
210 eta &= extract->tuple() == tup;
211 continue;
212 }
213 }
214 }
215 eta = false;
216 }
217
218 if (eta) return tup;
219 }
220 }
221
222 return unify<Tuple>(ops.size(), type, ops);
223}

225Ref World::extract(Ref d, Ref index) {
226 assert(d);
227 if (index->isa<Tuple>()) {
228 auto n = index->num_ops();
229 DefArray idx(n, [&](size_t i) { return index->op(i); });
230 DefArray ops(n, [&](size_t i) { return d->proj(n, as_lit(idx[i])); });
231 return tuple(ops);
232 } else if (index->isa<Pack>()) {
233 DefArray ops(index->as_lit_arity(), [&](size_t) { return extract(d, index->ops().back()); });
234 return tuple(ops);
235 }
236
237 Ref size = Idx::size(index->type());
238 Ref type = d->unfold_type();
239
240 // nom sigmas can be 1-tuples
241 if (auto l = isa_lit(size); l && *l == 1 && !d->type()->isa_nom<Sigma>()) return d;
242 if (auto pack = d->isa_structural<Pack>()) return pack->body();
243
244 if (!checker().equiv(type->arity(), size))
245 err(index, "index '{}' does not fit within arity '{}'", type->arity(), index);
246
247 // extract(insert(x, index, val), index) -> val
248 if (auto insert = d->isa<Insert>()) {
249 if (index == insert->index()) return insert->value();
250 }
251
252 if (auto i = isa_lit(index)) {
253 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
254
255 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
256 if (auto insert = d->isa<Insert>()) {
257 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index);
258 }
259
260 if (auto sigma = type->isa<Sigma>()) {
261 if (auto nom_sigma = sigma->isa_nom<Sigma>()) {
262 Scope scope(nom_sigma);
263 auto t = rewrite(sigma->op(*i), nom_sigma->var(), d, scope);
264 return unify<Extract>(2, t, d, index);
265 }
266
267 return unify<Extract>(2, sigma->op(*i), d, index);
268 }
269 }
270
271 const Def* elem_t;
272 if (auto arr = type->isa<Arr>())
273 elem_t = arr->reduce(index);
274 else
275 elem_t = extract(tuple(type->as<Sigma>()->ops()), index);
276
277 assert(d);
278 return unify<Extract>(2, elem_t, d, index);
279}

281Ref World::insert(Ref d, Ref index, Ref val) {
282 auto type = d->unfold_type();
283 auto size = Idx::size(index->type());
284
285 if (!checker().equiv(type->arity(), size))
286 err(index, "index '{}' does not fit within arity '{}'", type->arity(), index);
287
288 if (auto index_lit = isa_lit(index)) {
289 auto target_type = type->proj(*index_lit);
290 if (!checker().assignable(target_type, val))
291 err(val, "value of type {} is not assignable to type {}", val->type(), target_type);
292 }
293
294 if (auto l = isa_lit(size); l && *l == 1)
295 return tuple(d, {val}); // d could be nom - that's why the tuple ctor is needed
296
297 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
298 if (auto t = d->isa<Tuple>()) return t->refine(as_lit(index), val);
299
300 // insert(‹4; x›, 2, y) -> (x, x, y, x)
301 if (auto pack = d->isa<Pack>()) {
302 if (auto a = pack->isa_lit_arity()) {
303 DefArray new_ops(*a, pack->body());
304 new_ops[as_lit(index)] = val;
305 return tuple(type, new_ops);
306 }
307 }
308
309 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
310 if (auto insert = d->isa<Insert>()) {
311 if (insert->index() == index) d = insert->tuple();
312 }
313
314 return unify<Insert>(3, d, index, val);
315}

327Ref World::arr(Ref shape, Ref body) {
328 if (!is_shape(shape->type())) err(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
329
330 if (auto a = isa_lit<u64>(shape)) {
331 if (*a == 0) return sigma();
332 if (*a == 1) return body;
333 }
334
335 // «(a, b)#i; T» -> («a, T», <b, T»)#i
336 if (auto ex = shape->isa<Extract>()) {
337 if (auto tup = ex->tuple()->isa<Tuple>()) {
338 DefArray arrs(tup->num_ops(), [&](size_t i) { return arr(tup->op(i), body); });
339 return extract(tuple(arrs), ex->index());
340 }
341 }
342
343 // «(a, b, c); body» -> «a; «(b, c); body»»
344 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body));
345
346 // «<n; x>; body» -> «x; «<n-1, x>; body»»
347 if (auto p = shape->isa<Pack>()) {
348 if (auto s = isa_lit(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body));
349 }
350
351 return unify<Arr>(2, body->unfold_type(), shape, body);
352}

354Ref World::pack(Ref shape, Ref body) {
355 if (!is_shape(shape->type())) err(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
356
357 if (auto a = isa_lit<u64>(shape)) {
358 if (*a == 0) return tuple();
359 if (*a == 1) return body;
360 }
361
362 // <(a, b, c); body> -> <a; «(b, c); body>>
363 if (auto tuple = shape->isa<Tuple>()) return pack(tuple->ops().front(), pack(tuple->ops().skip_front(), body));
364
365 // <<n; x>; body> -> <x; <<n-1, x>; body>>
366 if (auto p = shape->isa<Pack>()) {
367 if (auto s = isa_lit(p->shape())) return pack(*s, pack(pack(*s - 1, p->body()), body));
368 }
369
370 auto type = arr(shape, body->type());
371 return unify<Pack>(1, type, body);
372}

408template<bool Up>
409Ref World::bound(Defs ops) {
410 auto kind = umax<Sort::Type>(ops);
411
412 // has ext<Up> value?
413 if (std::ranges::any_of(ops, [&](Ref op) { return Up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
414 return ext<Up>(kind);
415
416 // ignore: ext<!Up>
417 DefArray cpy(ops);
418 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](Ref op) { return !op->isa<Ext>(); });
419
420 // sort and remove duplicates
421 std::sort(cpy.begin(), end, GIDLt<const Def*>());
422 end = std::unique(cpy.begin(), end);
423 cpy.shrink(std::distance(cpy.begin(), end));
424
425 if (cpy.size() == 0) return ext<!Up>(kind);
426 if (cpy.size() == 1) return cpy[0];
427
428 // TODO simplify mixed terms with joins and meets
429
430 return unify<TBound<Up>>(cpy.size(), kind, cpy);
431}

468// appends a suffix or an increasing number if the suffix already exists
469Sym World::append_suffix(Sym symbol, std::string suffix) {
470 auto name = *symbol;
471
472 auto pos = name.find(suffix);
473 if (pos != std::string::npos) {
474 auto num = name.substr(pos + suffix.size());
475 if (num.empty()) {
476 name += "_1";
477 } else {
478 num = num.substr(1);

493Ref World::iapp(Ref callee, Ref arg) {
494 while (auto pi = callee->type()->isa<Pi>()) {
495 if (pi->implicit()) {
496 auto infer = nom_infer_entity();
497 auto a = app(callee, infer);
498 callee = a;
499 } else {
500 // resolve Infers now if possible before normalizers are run
501 if (auto app = callee->isa<App>(); app && app->curry() == 1) {
502 checker().assignable(callee->type()->as<Pi>()->dom(), arg);
503 auto apps = decurry(app);
504 callee = apps.front()->callee();
505 for (auto app : apps) callee = this->app(callee, refer(app->arg()));
506 }
507 break;
508 }
509 }
510
511 return app(callee, arg);
512}
