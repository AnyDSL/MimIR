133template<int sort> const Def* World::umax(Defs ops_) {
134 DefVec ops;
135 for (auto op : ops_) {
136 op = op->zonk();
137
138 if constexpr (sort == UMax::Term) op = op->unfold_type();
139 if constexpr (sort >= UMax::Type) op = op->unfold_type();
140 if constexpr (sort >= UMax::Kind) {
141 if (auto type = op->isa<Type>())
142 op = type->level();
143 else
144 error(op->loc(), "operand '{}' must be a Type of some level", op); // TODO better error message
145 }
146
147 flatten_umax(ops, op);
148 }
149
150 level_t lvl = 0;
151 DefVec res;
152 for (auto op : ops) {
153 if (!op->type()->isa<Univ>())
154 error(op->loc(), "operand '{}' of a universe max must be of type 'Univ' but is of type '{}'", op,
155 op->type());
156
157 if (auto l = Lit::isa(op))
158 lvl = std::max(lvl, *l);
159 else
160 res.emplace_back(op);
161 }
162
163 const Def* l = lit_univ(lvl);
164 if (res.empty()) return sort == UMax::Univ ? l : type(l);
165 if (lvl > 0) res.emplace_back(l);
166
167 std::ranges::sort(res, [](auto op1, auto op2) { return op1->gid() < op2->gid(); });
168 res.erase(std::unique(res.begin(), res.end()), res.end());
169 const Def* umax = unify<UMax>(res.size(), *this, res);
170 return sort == UMax::Univ ? umax : type(umax);
171}

191template<bool Normalize> const Def* World::app(const Def* callee, const Def* arg) {
192 callee = callee->zonk();
193 arg = arg->zonk();
194
195 if (auto pi = callee->type()->isa<Pi>()) {
196 if (auto new_arg = Checker::assignable(pi->dom(), arg)) {
197 arg = new_arg->zonk();
198 if (auto imm = callee->isa_imm<Lam>()) return imm->body();
199
200 if (auto lam = callee->isa_mut<Lam>(); lam && lam->is_set() && lam->filter() != lit_ff()) {
201 if (auto var = lam->has_var()) {
202 if (auto i = move_.substs.find({var, arg}); i != move_.substs.end()) {
203 // Is there a cached version?
204 auto [filter, body] = i->second->defs<2>();
205 if (filter == lit_tt()) return body;
206 } else {
207 // First check filter, If true, reduce body and cache reduct.
208 auto rw = VarRewriter(var, arg);
209 auto filter = rw.rewrite(lam->filter());
210 if (filter == lit_tt()) {
211 DLOG("partial evaluate: {} ({})", lam, arg);
212 auto body = rw.rewrite(lam->body());
213 auto num_bytes = sizeof(Reduct) + 2 * sizeof(const Def*);
214 auto buf = move_.arena.substs.allocate(num_bytes, alignof(const Def*));
215 auto reduct = new (buf) Reduct(2);
216 reduct->defs_[0] = filter;
217 reduct->defs_[1] = body;
218 assert_emplace(move_.substs, std::pair{var, arg}, reduct);
219 return body;
220 }
221 }
222 } else if (lam->filter() == lit_tt()) {
223 return lam->body();
224 }
225 }
226
227 auto type = pi->reduce(arg)->zonk();
228 callee = callee->zonk();
229 auto [axm, curry, trip] = Axm::get(callee);
230 if (axm) {
231 curry = curry == 0 ? trip : curry;
232 curry = curry == Axm::Trip_End ? curry : curry - 1;
233
234 if (auto normalizer = axm->normalizer(); Normalize && normalizer && curry == 0) {
235 if (auto norm = normalizer(type, callee, arg)) return norm;
236 }
237 }
238
239 return raw_app(axm, curry, trip, type, callee, arg);
240 }
241
242 throw Error()
243 .error(arg->loc(), "cannot apply argument to callee")
244 .note(callee->loc(), "callee: '{}'", callee)
245 .note(arg->loc(), "argument: '{}'", arg)
246 .note(callee->loc(), "vvv domain type vvv\n'{}'\n'{}'", pi->dom(), arg->type())
247 .note(arg->loc(), "^^^ argument type ^^^");
248 }
249
250 throw Error()
251 .error(callee->loc(), "called expression not of function type")
252 .error(callee->loc(), "'{}' <--- callee type", callee->type());
253}

298const Def* World::tuple(const Def* type, Defs ops_) {
299 // TODO type-check type vs inferred type
300 type = type->zonk();
301 auto ops = Def::zonk(ops_);
302
303 auto n = ops.size();
304 if (!type->isa_mut<Sigma>()) {
305 if (n == 0) return tuple();
306 if (n == 1) return ops[0];
307 if (auto uni = Checker::is_uniform(ops)) return pack(n, uni);
308 }
309
310 if (n != 0) {
311 // eta rule for tuples:
312 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
313 if (auto extract = ops[0]->isa<Extract>()) {
314 auto tup = extract->tuple();
315 bool eta = tup->type() == type;
316 for (size_t i = 0; i != n && eta; ++i) {
317 if (auto extract = ops[i]->isa<Extract>()) {
318 if (auto index = Lit::isa(extract->index())) {
319 if (eta &= u64(i) == *index) {
320 eta &= extract->tuple() == tup;
321 continue;
322 }
323 }
324 }
325 eta = false;
326 }
327
328 if (eta) return tup;
329 }
330 }
331
332 return unify<Tuple>(ops.size(), type, ops);
333}

341const Def* World::extract(const Def* d, const Def* index) {
342 d = d->zonk();
343 index = index->zonk();
344
345 if (index->isa<Tuple>()) {
346 auto n = index->num_ops();
347 auto idx = DefVec(n, [&](size_t i) { return index->op(i); });
348 auto ops = DefVec(n, [&](size_t i) { return d->proj(n, Lit::as(idx[i])); });
349 return tuple(ops);
350 } else if (index->isa<Pack>()) {
351 auto ops = DefVec(index->as_lit_arity(), [&](size_t) { return extract(d, index->ops().back()); });
352 return tuple(ops);
353 }
354
355 auto size = Idx::isa(index->type());
356 auto type = d->unfold_type()->zonk();
357
358 if (size) {
359 if (auto l = Lit::isa(size); l && *l == 1) {
360 if (auto l = Lit::isa(index); !l || *l != 0) WLOG("unknown Idx of size 1: {}", index);
361 if (auto sigma = type->isa_mut<Sigma>(); sigma && sigma->num_ops() == 1) {
362 // mut sigmas can be 1-tuples; TODO mutables Arr?
363 } else {
364 return d;
365 }
366 }
367 }
368
369 if (auto pack = d->isa_imm<Pack>()) return pack->body();
370
371 if (size && !Checker::alpha<Checker::Check>(type->arity(), size))
372 error(index->loc(), "index '{}' does not fit within arity '{}'", index, type->arity());
373
374 // extract(insert(x, index, val), index) -> val
375 if (auto insert = d->isa<Insert>()) {
376 if (index == insert->index()) return insert->value();
377 }
378
379 if (auto i = Lit::isa(index)) {
380 if (auto hole = d->isa_mut<Hole>()) d = hole->tuplefy(Idx::as_lit(index->type()));
381 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
382
383 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
384 if (auto insert = d->isa<Insert>()) {
385 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index);
386 }
387
388 if (auto sigma = type->isa<Sigma>()) {
389 if (auto var = sigma->has_var()) {
390 auto t = VarRewriter(var, d).rewrite(sigma->op(*i));
391 return unify<Extract>(2, t, d, index);
392 }
393
394 return unify<Extract>(2, sigma->op(*i), d, index);
395 }
396 }
397
398 const Def* elem_t;
399 if (auto arr = type->isa<Arr>())
400 elem_t = arr->reduce(index);
401 else
402 elem_t = join(type->as<Sigma>()->ops());
403
404 if (index->isa<Top>()) {
405 if (auto hole = Hole::isa_unset(d)) {
406 auto elem_hole = mut_hole(elem_t);
407 hole->set(pack(size, elem_hole));
408 return elem_hole;
409 }
410 }
411
412 assert(d);
413 return unify<Extract>(2, elem_t, d, index);
414}

416const Def* World::insert(const Def* d, const Def* index, const Def* val) {
417 d = d->zonk();
418 index = index->zonk();
419 val = val->zonk();
420
421 auto type = d->unfold_type();
422 auto size = Idx::isa(index->type());
423 auto lidx = Lit::isa(index);
424
425 if (!size) error(d->loc(), "index '{}' must be of type 'Idx' but is of type '{}'", index, index->type());
426
427 if (!Checker::alpha<Checker::Check>(type->arity(), size))
428 error(index->loc(), "index '{}' does not fit within arity '{}'", index, type->arity());
429
430 if (lidx) {
431 auto elem_type = type->proj(*lidx);
432 auto new_val = Checker::assignable(elem_type, val);
433 if (!new_val) {
434 throw Error()
435 .error(val->loc(), "value to be inserted not assignable to element")
436 .note(val->loc(), "vvv value type vvv \n'{}'\n'{}'", val->type(), elem_type)
437 .note(val->loc(), "^^^ element type ^^^", elem_type);
438 }
439 val = new_val;
440 }
441
442 if (auto l = Lit::isa(size); l && *l == 1)
443 return tuple(d, {val}); // d could be mut - that's why the tuple ctor is needed
444
445 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
446 if (auto t = d->isa<Tuple>(); t && lidx) return t->refine(*lidx, val);
447
448 // insert(4; x, 2, y) -> (x, x, y, x)
449 if (auto pack = d->isa<Pack>(); pack && lidx) {
450 if (auto a = pack->isa_lit_arity(); a && *a < flags().scalarize_threshold) {
451 auto new_ops = DefVec(*a, pack->body());
452 new_ops[*lidx] = val;
453 return tuple(type, new_ops);
454 }
455 }
456
457 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
458 if (auto insert = d->isa<Insert>()) {
459 if (insert->index() == index) d = insert->tuple();
460 }
461
462 return unify<Insert>(3, d, index, val);
463}

551template<bool Up> const Def* World::bound(Defs ops_) {
552 auto ops = DefVec();
553 for (size_t i = 0, e = ops_.size(); i != e; ++i) {
554 auto op = ops_[i]->zonk();
555 if (!op->isa<TExt<!Up>>()) ops.emplace_back(op); // ignore: ext<!Up>

571 return unify<TBound<Up>>(ops.size(), kind, ops);

607const Def* World::match(Defs ops_) {
608 auto ops = Def::zonk(ops_);
609 if (ops.size() == 1) return ops.front();
610
611 auto scrutinee = ops.front();
612 auto arms = ops.span().subspan(1);
613 auto join = scrutinee->type()->isa<Join>();
614
615 if (!join) error(scrutinee->loc(), "scrutinee of a test expression must be of union type");
616
617 if (arms.size() != join->num_ops())
618 error(scrutinee->loc(), "test expression has {} arms but union type has {} cases", arms.size(),
619 join->num_ops());
620
621 for (auto arm : arms)
622 if (!arm->type()->isa<Pi>())
623 error(arm->loc(), "arm of test expression does not have a function type but is of type '{}'", arm->type());
624
625 std::ranges::sort(arms, [](const Def* arm1, const Def* arm2) {
626 return arm1->type()->as<Pi>()->dom()->gid() < arm2->type()->as<Pi>()->dom()->gid();
627 });
628
629 const Def* type = nullptr;
630 for (size_t i = 0, e = arms.size(); i != e; ++i) {
631 auto arm = arms[i];
632 auto pi = arm->type()->as<Pi>();
633 if (!Checker::alpha<Checker::Check>(pi->dom(), join->op(i)))
634 error(arm->loc(),
635 "domain type '{}' of arm in a test expression does not match case type '{}' in union type", pi->dom(),
636 join->op(i));

648Sym World::append_suffix(Sym symbol, std::string suffix) {
649 auto name = symbol.str();
650
651 auto pos = name.find(suffix);
652 if (pos != std::string::npos) {
653 auto num = name.substr(pos + suffix.size());
654 if (num.empty()) {
655 name += "_1";
656 } else {
657 num = num.substr(1);
658 num = std::to_string(std::stoi(num) + 1);
659 name = name.substr(0, pos + suffix.size()) + "_" + num;
660 }
661 } else {
662 name += suffix;
663 }
664
665 return sym(std::move(name));
666}

668Defs World::reduce(const Var* var, const Def* arg) {
669 auto mut = var->mut();
670 auto offset = mut->reduction_offset();
671 auto size = mut->num_ops() - offset;
672
673 if (auto i = move_.substs.find({var, arg}); i != move_.substs.end()) return i->second->defs();
674
675 auto buf = move_.arena.substs.allocate(sizeof(Reduct) + size * sizeof(const Def*), alignof(const Def*));
676 auto reduct = new (buf) Reduct(size);
677 auto rw = VarRewriter(var, arg);
678 for (size_t i = 0; i != size; ++i) reduct->defs_[i] = rw.rewrite(mut->op(i + offset));
679 assert_emplace(move_.substs, std::pair{var, arg}, reduct);
680 return reduct->defs();
681}
