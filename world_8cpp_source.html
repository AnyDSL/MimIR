80template<Sort sort>
81const Def* World::umax(DefArray ops, Ref dbg) {
82 level_t lvl = 0;
83 for (auto& op : ops) {
84 Ref r = op;
85 if (sort == Sort::Term) r = r->unfold_type();
86 if (sort <= Sort::Type) r = r->unfold_type();
87 if (sort <= Sort::Kind) {
88 if (auto type = r->isa<Type>())
89 r = type->level();
90 else
91 err(dbg, "operand '{}' must be a .Type of some level", r);
92 }
93
94 if (!r->type()->isa<Univ>())
95 err(dbg, "operand '{}' of a universe max must be of type '.Univ' but is of type '{}'", r, r->type());
96
97 op = r;
98
99 if (auto l = isa_lit(op))
100 lvl = std::max(lvl, *l);
101 else
102 lvl = level_t(-1);
103 }
104
105 auto ldef = lvl == level_t(-1) ? (const Def*)unify<UMax>(ops.size(), *this, ops, dbg) : lit_univ(lvl, dbg);
106 std::ranges::sort(ops, [](auto op1, auto op2) { return op1->gid() < op2->gid(); });
107 return sort == Sort::Univ ? ldef : type(ldef, dbg);
108}

110const Def* World::app(Ref callee, Ref arg, Ref dbg) {
111 auto pi = callee->type()->isa<Pi>();
112
113 // (a, b)#i arg where a = A -> B; b = A -> B
114 if (auto extract = callee->type()->isa<Extract>()) {
115 if (auto tuple = extract->tuple()->isa<Tuple>()) {
116 if (auto uni = checker().is_uniform(tuple->ops(), dbg)) pi = uni->isa<Pi>();
117 }
118 }
119
120 if (!pi) err(dbg, "called expression '{}' : '{}' is not of function type", callee, callee->type());
121 if (!checker().assignable(pi->dom(), arg, dbg))
122 err(dbg, "cannot pass argument '{}' of type '{}' to '{}' of domain '{}'", arg, arg->type(), callee, pi->dom());
123
124 if (auto lam = callee->isa<Lam>(); lam && lam->is_set() && lam->codom()->sort() > Sort::Type)
125 return lam->reduce(arg).back();
126
127 auto type = pi->reduce(arg).back();
128 return raw_app<true>(type, callee, arg, dbg);
129}

172const Def* World::tuple(Ref type, Defs ops, Ref dbg) {
173 // TODO type-check type vs inferred type
174
175 auto n = ops.size();
176 if (!type->isa_nom<Sigma>()) {
177 if (n == 0) return tuple();
178 if (n == 1) return ops[0];
179 auto front = ops.front();
180 if (std::ranges::all_of(ops.skip_front(), [front](auto op) { return front == op; })) return pack(n, front, dbg);
181 }
182
183 if (n != 0) {
184 // eta rule for tuples:
185 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
186 if (auto extract = ops[0]->isa<Extract>()) {
187 auto tup = extract->tuple();
188 bool eta = tup->type() == type;
189 for (size_t i = 0; i != n && eta; ++i) {
190 if (auto extract = ops[i]->isa<Extract>()) {
191 if (auto index = isa_lit(extract->index())) {
192 if (eta &= u64(i) == *index) {
193 eta &= extract->tuple() == tup;
194 continue;
195 }
196 }
197 }
198 eta = false;
199 }
200
201 if (eta) return tup;
202 }
203 }
204
205 return unify<Tuple>(ops.size(), type, ops, dbg);
206}

214const Def* World::extract(Ref d, Ref index, Ref dbg) {
215 if (index->isa<Tuple>()) {
216 auto n = index->num_ops();
217 DefArray idx(n, [&](size_t i) { return index->op(i); });
218 DefArray ops(n, [&](size_t i) { return d->proj(n, as_lit(idx[i])); });
219 return tuple(ops, dbg);
220 } else if (index->isa<Pack>()) {
221 DefArray ops(as_lit(index->arity()), [&](size_t) { return extract(d, index->ops().back()); });
222 return tuple(ops, dbg);
223 }
224
225 Ref size = Idx::size(index->type());
226 Ref type = d->unfold_type();
227
228 // nom sigmas can be 1-tuples
229 if (auto l = isa_lit(size); l && *l == 1 && !d->type()->isa_nom<Sigma>()) return d;
230 if (auto pack = d->isa_structural<Pack>()) return pack->body();
231
232 if (!checker().equiv(type->arity(), size, dbg))
233 err(dbg, "index '{}' does not fit within arity '{}'", type->arity(), index);
234
235 // extract(insert(x, index, val), index) -> val
236 if (auto insert = d->isa<Insert>()) {
237 if (index == insert->index()) return insert->value();
238 }
239
240 if (auto i = isa_lit(index)) {
241 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
242
243 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
244 if (auto insert = d->isa<Insert>()) {
245 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index, dbg);
246 }
247
248 if (auto sigma = type->isa<Sigma>()) {
249 if (auto nom_sigma = sigma->isa_nom<Sigma>()) {
250 Scope scope(nom_sigma);
251 auto t = rewrite(sigma->op(*i), nom_sigma->var(), d, scope);
252 return unify<Extract>(2, t, d, index, dbg);
253 }
254
255 return unify<Extract>(2, sigma->op(*i), d, index, dbg);
256 }
257 }
258
259 const Def* elem_t;
260 if (auto arr = type->isa<Arr>())
261 elem_t = arr->reduce(index);
262 else
263 elem_t = extract(tuple(type->as<Sigma>()->ops(), dbg), index, dbg);
264
265 return unify<Extract>(2, elem_t, d, index, dbg);
266}

268const Def* World::insert(Ref d, Ref index, Ref val, Ref dbg) {
269 auto type = d->unfold_type();
270 auto size = Idx::size(index->type());
271
272 if (!checker().equiv(type->arity(), size, dbg))
273 err(dbg, "index '{}' does not fit within arity '{}'", type->arity(), index);
274
275 if (auto l = isa_lit(size); l && *l == 1)
276 return tuple(d, {val}, dbg); // d could be nom - that's why the tuple ctor is needed
277
278 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
279 if (auto t = d->isa<Tuple>()) return t->refine(as_lit(index), val);
280
281 // insert(‹4; x›, 2, y) -> (x, x, y, x)
282 if (auto pack = d->isa<Pack>()) {
283 if (auto a = isa_lit(pack->arity())) {
284 DefArray new_ops(*a, pack->body());
285 new_ops[as_lit(index)] = val;
286 return tuple(type, new_ops, dbg);
287 }
288 }
289
290 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
291 if (auto insert = d->isa<Insert>()) {
292 if (insert->index() == index) d = insert->tuple();
293 }
294
295 return unify<Insert>(3, d, index, val, dbg);
296}

308const Def* World::arr(Ref shape, Ref body, Ref dbg) {
309 if (!is_shape(shape->type())) err(dbg, "expected shape but got '{}' of type '{}'", shape, shape->type());
310
311 if (auto a = isa_lit<u64>(shape)) {
312 if (*a == 0) return sigma();
313 if (*a == 1) return body;
314 }
315
316 // «(a, b)#i; T» -> («a, T», <b, T»)#i
317 if (auto ex = shape->isa<Extract>()) {
318 if (auto tup = ex->tuple()->isa<Tuple>()) {
319 DefArray arrs(tup->num_ops(), [&](size_t i) { return arr(tup->op(i), body); });
320 return extract(tuple(arrs), ex->index(), dbg);
321 }
322 }
323
324 // «(a, b, c); body» -> «a; «(b, c); body»»
325 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body), dbg);
326
327 // «<n; x>; body» -> «x; «<n-1, x>; body»»
328 if (auto p = shape->isa<Pack>()) {
329 if (auto s = isa_lit(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body), dbg);
330 }
331
332 return unify<Arr>(2, body->unfold_type(), shape, body, dbg);
333}

389template<bool up>
390const Def* World::bound(Defs ops, Ref dbg) {
391 auto kind = umax<Sort::Type>(ops, dbg);
392
393 // has ext<up> value?
394 if (std::ranges::any_of(ops, [&](Ref op) { return up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
395 return ext<up>(kind);
396
397 // ignore: ext<!up>
398 DefArray cpy(ops);
399 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](Ref op) { return !op->isa<Ext>(); });
400
401 // sort and remove duplicates
402 std::sort(cpy.begin(), end, GIDLt<const Def*>());
403 end = std::unique(cpy.begin(), end);
404 cpy.shrink(std::distance(cpy.begin(), end));
405
406 if (cpy.size() == 0) return ext<!up>(kind, dbg);
407 if (cpy.size() == 1) return cpy[0];
408
409 // TODO simplify mixed terms with joins and meets
410
411 return unify<TBound<up>>(cpy.size(), kind, cpy, dbg);
412}

469const Def* World::iapp(Ref callee, Ref arg, Debug debug) {
470 Debug mdebug = debug;
471 while (auto implicit = peel(callee->meta())) {
472 bool dot;
473 std::tie(dot, mdebug.meta) = *implicit;
474
475 if (dot) {
476 auto infer = nom_infer_entity(dbg(debug));
477 auto d = dbg(mdebug);
478 auto a = app(callee, infer, d);
479 callee = a;
480 } else {
481 // resolve Infers now if possible before normalizers are run
482 if (auto app = callee->isa<App>(); app && app->curry() == 1) {
483 checker().assignable(callee->type()->as<Pi>()->dom(), arg, callee->dbg());
484 auto apps = decurry(app);
485 callee = apps.front()->callee();
486 for (auto app : apps) callee = this->app(callee, refer(app->arg()), app->dbg());
487 }
488 break;
489 }
490 }
491
492 return app(callee, arg, dbg(mdebug));
493}
