238const Def* World::app(const Def* callee, const Def* arg, const Def* dbg) {
239 auto pi = callee->type()->as<Pi>();
240
241 if (err()) {
242 if (!checker_->assignable(pi->dom(), arg))
243 err()->ill_typed_app(callee, arg);
244 }
245
246 auto type = pi->apply(arg).back();
247 auto [axiom, currying_depth] = get_axiom(callee); // TODO move down again
248 if (axiom && currying_depth == 1) {
249 if (auto normalize = axiom->normalizer())
250 return normalize(type, callee, arg, dbg);
251 }
252
253 return unify<App>(2, axiom, currying_depth-1, type, callee, arg, dbg);

289const Def* World::tuple(const Def* type, Defs ops, const Def* dbg) {
290 if (err()) {
291 // TODO type-check type vs inferred type
292 }
293
294 auto n = ops.size();
295 if (!type->isa_nom<Sigma>()) {
296 if (n == 0) return tuple();
297 if (n == 1) return ops[0];
298 if (std::all_of(ops.begin()+1, ops.end(), [&](auto op) { return ops[0] == op; })) return pack(n, ops[0]);
299 }
300
301 // eta rule for tuples:
302 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
303 if (n != 0) if (auto extract = ops[0]->isa<Extract>()) {
304 auto tup = extract->tuple();
305 bool eta = tup->type() == type;
306 for (size_t i = 0; i != n && eta; ++i) {
307 if (auto extract = ops[i]->isa<Extract>()) {
308 if (auto index = isa_lit(extract->index())) {
309 if (eta &= u64(i) == *index) {
310 eta &= extract->tuple() == tup;
311 continue;
312 }
313 }
314 }
315 eta = false;
316 }
317
318 if (eta) return tup;
319 }
320
321 return unify<Tuple>(ops.size(), type, ops, dbg);
322}

331const Def* World::extract_(const Def* ex_type, const Def* tup, const Def* index, const Def* dbg) {
332 if (index->isa<Arr>() || index->isa<Pack>()) {
333 DefArray ops(as_lit(index->arity()), [&](size_t) { return extract(tup, index->ops().back()); });
334 return index->isa<Arr>() ? sigma(ops, dbg) : tuple(ops, dbg);
335 } else if (index->isa<Sigma>() || index->isa<Tuple>()) {
336 auto n = index->num_ops();
337 DefArray idx(n, [&](size_t i) { return index->op(i); });
338 DefArray ops(n, [&](size_t i) { return tup->proj(n, as_lit(idx[i])); });
339 return index->isa<Sigma>() ? sigma(ops, dbg) : tuple(ops, dbg);
340 }
341
342 auto type = tup->type()->reduce();
343 if (err()) {
344 if (!checker_->equiv(type->arity(), isa_sized_type(index->type())))
345 err()->index_out_of_range(type->arity(), index);
346 }
347
348 // nom sigmas can be 1-tuples
349 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1 && !tup->type()->isa_nom<Sigma>()) return tup;
350 if (auto pack = tup->isa_structural<Pack>()) return pack->body();
351
352 // extract(insert(x, index, val), index) -> val
353 if (auto insert = tup->isa<Insert>()) {
354 if (index == insert->index())
355 return insert->value();
356 }
357
358 if (auto i = isa_lit(index)) {
359 if (auto tuple = tup->isa<Tuple>()) return tuple->op(*i);
360
361 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
362 if (auto insert = tup->isa<Insert>()) {
363 if (insert->index()->isa<Lit>())
364 return extract(insert->tuple(), index, dbg);
365 }
366
367 if (type->isa<Sigma>())
368 return unify<Extract>(2, ex_type ? ex_type : type->op(*i), tup, index, dbg);
369 }
370
371 type = type->as<Arr>()->body();
372 return unify<Extract>(2, type, tup, index, dbg);
373}

375const Def* World::insert(const Def* tup, const Def* index, const Def* val, const Def* dbg) {
376 auto type = tup->type()->reduce();
377
378 if (err() && !checker_->equiv(type->arity(), isa_sized_type(index->type())))
379 err()->index_out_of_range(type->arity(), index);
380
381 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1)
382 return tuple(tup, {val}, dbg); // tup could be nom - that's why the tuple ctor is needed
383
384 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
385 if (auto t = tup->isa<Tuple>()) return t->refine(as_lit(index), val);
386
387 // insert(‹4; x›, 2, y) -> (x, x, y, x)
388 if (auto pack = tup->isa<Pack>()) {
389 if (auto a = isa_lit(pack->arity())) {
390 DefArray new_ops(*a, pack->body());
391 new_ops[as_lit(index)] = val;
392 return tuple(type, new_ops, dbg);
393 }
394 }
395
396 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
397 if (auto insert = tup->isa<Insert>()) {
398 if (insert->index() == index)
399 tup = insert->tuple();
400 }
401
402 return unify<Insert>(3, tup, index, val, dbg);
403}

408 if (auto sig = s->isa_structural<Sigma>())

410
411 return false;

414const Def* World::arr(const Def* shape, const Def* body, const Def* dbg) {
415 if (err()) {
416 if (!is_shape(shape->type())) err()->expected_shape(shape);
417 }
418
419 if (auto a = isa_lit<u64>(shape)) {
420 if (*a == 0) return sigma();
421 if (*a == 1) return body;
422 }
423
424 // «(a, b, c); body» -> «a; «(b, c); body»»
425 if (auto tuple = shape->isa<Tuple>())
426 return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body), dbg);
427
428 // «<n; x>; body» -> «x; «<n-1, x>; body»»
429 if (auto p = shape->isa<Pack>()) {
430 if (auto s = isa_lit(p->shape()))
431 return arr(*s, arr(pack(*s-1, p->body()), body), dbg);
432 }
433
434 return unify<Arr>(2, kind(), shape, body, dbg);
435}

511template<bool up>
512const Def* World::bound(Defs ops, const Def* dbg) {
513 auto kind = infer_kind(ops);
514
515 // has ext<up> value?
516 if (std::any_of(ops.begin(), ops.end(), [&](const Def* op) { return up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
517 return ext<up>(kind);
518
519 // ignore: ext<!up>
520 DefArray cpy(ops);
521 auto end = std::copy_if(ops.begin(), ops.end(), cpy.begin(), [&](const Def* op) { return !isa_ext(op); });
522
523 // sort and remove duplicates
524 std::sort(cpy.begin(), end, GIDLt<const Def*>());
525 end = std::unique(cpy.begin(), end);
526 cpy.shrink(cpy.begin() - end);
527
528 if (cpy.size() == 0) return ext<!up>(kind, dbg);
529 if (cpy.size() == 1) return cpy[0];
530
531 // TODO simplify mixed terms with joins and meets
532
533 return unify<TBound<up>>(cpy.size(), kind, cpy, dbg);
534}

655template<bool elide_empty>
656void World::visit(VisitFn f) const {
657 unique_queue<NomSet> noms;
658 unique_stack<DefSet> defs;
659
660 for (const auto& [name, nom] : externals()) {
661 assert(nom->is_set() && "external must not be empty");
662 noms.push(nom);
663 }
664
665 while (!noms.empty()) {
666 auto nom = noms.pop();
667 if (elide_empty && !nom->is_set()) continue;
668
669 Scope scope(nom);
670 f(scope);
671
672 for (auto nom : scope.free_noms())
673 noms.push(nom);
674 }
675}
