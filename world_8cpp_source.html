77const Def* World::app(const Def* callee, const Def* arg, const Def* dbg) {
78 auto pi = callee->type()->isa<Pi>();
79
80 if (err()) {
81 if (!pi)
82 err()->err(dbg->loc(), "called expression '{}' : '{}' is not of function type", callee, callee->type());
83 if (!checker().assignable(pi->dom(), arg, dbg)) err()->ill_typed_app(callee, arg, dbg);
84 }
85
86 auto type = pi->reduce(arg).back();
87 auto [axiom, curry] = Axiom::get(callee);
88 if (axiom && curry == 1) {
89 if (auto normalize = axiom->normalizer()) return normalize(type, callee, arg, dbg);
90 }
91
92 if (auto lam = callee->isa<Lam>(); lam && !lam->is_unfinished() && lam->codom()->sort() > Sort::Type)
93 return lam->reduce(arg).back();
94
95 return unify<App>(2, axiom, curry - 1, type, callee, arg, dbg);

130const Def* World::tuple(const Def* type, Defs ops, const Def* dbg) {
131 if (err()) {
132 // TODO type-check type vs inferred type
133 }
134
135 auto n = ops.size();
136 if (!type->isa_nom<Sigma>()) {
137 if (n == 0) return tuple();
138 if (n == 1) return ops[0];
139 if (auto uni = checker().is_uniform(ops, dbg)) return pack(n, uni, dbg);
140 }
141
142 if (n != 0) {
143 // eta rule for tuples:
144 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
145 if (auto extract = ops[0]->isa<Extract>()) {
146 auto tup = extract->tuple();
147 bool eta = tup->type() == type;
148 for (size_t i = 0; i != n && eta; ++i) {
149 if (auto extract = ops[i]->isa<Extract>()) {
150 if (auto index = isa_lit(extract->index())) {
151 if (eta &= u64(i) == *index) {
152 eta &= extract->tuple() == tup;
153 continue;
154 }
155 }
156 }
157 eta = false;
158 }
159
160 if (eta) return tup;
161 }
162 }
163
164 return unify<Tuple>(ops.size(), type, ops, dbg);
165}

173const Def* World::extract(const Def* d, const Def* index, const Def* dbg) {
174 if (index->isa<Tuple>()) {
175 auto n = index->num_ops();
176 DefArray idx(n, [&](size_t i) { return index->op(i); });
177 DefArray ops(n, [&](size_t i) { return d->proj(n, as_lit(idx[i])); });
178 return tuple(ops, dbg);
179 } else if (index->isa<Pack>()) {
180 DefArray ops(as_lit(index->arity()), [&](size_t) { return extract(d, index->ops().back()); });
181 return tuple(ops, dbg);
182 }
183
184 auto size = Idx::size(index->type());
185 auto type = d->unfold_type();
186
187 // nom sigmas can be 1-tuples
188 if (auto l = isa_lit(size); l && *l == 1 && !d->type()->isa_nom<Sigma>()) return d;
189 if (auto pack = d->isa_structural<Pack>()) return pack->body();
190
191 if (err()) {
192 if (!checker().equiv(type->arity(), size, dbg)) err()->index_out_of_range(type->arity(), index, dbg);
193 }
194
195 // extract(insert(x, index, val), index) -> val
196 if (auto insert = d->isa<Insert>()) {
197 if (index == insert->index()) return insert->value();
198 }
199
200 if (auto i = isa_lit(index)) {
201 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
202
203 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
204 if (auto insert = d->isa<Insert>()) {
205 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index, dbg);
206 }
207
208 if (auto sigma = type->isa<Sigma>()) {
209 if (auto nom_sigma = sigma->isa_nom<Sigma>()) {
210 Scope scope(nom_sigma);
211 auto t = rewrite(sigma->op(*i), nom_sigma->var(), d, scope);
212 return unify<Extract>(2, t, d, index, dbg);
213 }
214
215 return unify<Extract>(2, sigma->op(*i), d, index, dbg);
216 }
217 }
218
219 const Def* elem_t;
220 if (auto arr = type->isa<Arr>())
221 elem_t = arr->reduce(index);
222 else
223 elem_t = extract(tuple(type->as<Sigma>()->ops(), dbg), index, dbg);
224
225 return unify<Extract>(2, elem_t, d, index, dbg);
226}

228const Def* World::insert(const Def* d, const Def* index, const Def* val, const Def* dbg) {
229 auto type = d->unfold_type();
230 auto size = Idx::size(index->type());
231
232 if (err() && !checker().equiv(type->arity(), size, dbg)) err()->index_out_of_range(type->arity(), index, dbg);
233
234 if (auto l = isa_lit(size); l && *l == 1)
235 return tuple(d, {val}, dbg); // d could be nom - that's why the tuple ctor is needed
236
237 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
238 if (auto t = d->isa<Tuple>()) return t->refine(as_lit(index), val);
239
240 // insert(‹4; x›, 2, y) -> (x, x, y, x)
241 if (auto pack = d->isa<Pack>()) {
242 if (auto a = isa_lit(pack->arity())) {
243 DefArray new_ops(*a, pack->body());
244 new_ops[as_lit(index)] = val;
245 return tuple(type, new_ops, dbg);
246 }
247 }
248
249 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
250 if (auto insert = d->isa<Insert>()) {
251 if (insert->index() == index) d = insert->tuple();
252 }
253
254 return unify<Insert>(3, d, index, val, dbg);
255}

266const Def* World::arr(const Def* shape, const Def* body, const Def* dbg) {
267 if (err()) {
268 if (!is_shape(shape->type())) err()->expected_shape(shape, dbg);
269 }
270
271 if (auto a = isa_lit<u64>(shape)) {
272 if (*a == 0) return sigma();
273 if (*a == 1) return body;
274 }
275
276 // «(a, b)#i; T» -> («a, T», <b, T»)#i
277 if (auto ex = shape->isa<Extract>()) {
278 if (auto tup = ex->tuple()->isa<Tuple>()) {
279 DefArray arrs(tup->num_ops(), [&](size_t i) { return arr(tup->op(i), body); });
280 return extract(tuple(arrs), ex->index(), dbg);
281 }
282 }
283
284 // «(a, b, c); body» -> «a; «(b, c); body»»
285 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body), dbg);
286
287 // «<n; x>; body» -> «x; «<n-1, x>; body»»
288 if (auto p = shape->isa<Pack>()) {
289 if (auto s = isa_lit(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body), dbg);
290 }
291
292 return unify<Arr>(2, body->unfold_type(), shape, body, dbg);
293}

296 if (err()) {

300 if (auto a = isa_lit<u64>(shape)) {
301 if (*a == 0) return tuple();
302 if (*a == 1) return body;

353template<bool up>
354const Def* World::bound(Defs ops, const Def* dbg) {
355 auto kind = infer_type_level(*this, ops);
356
357 // has ext<up> value?
358 if (std::ranges::any_of(ops, [&](const Def* op) { return up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
359 return ext<up>(kind);
360
361 // ignore: ext<!up>
362 DefArray cpy(ops);
363 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](const Def* op) { return !isa_ext(op); });
364
365 // sort and remove duplicates
366 std::sort(cpy.begin(), end, GIDLt<const Def*>());
367 end = std::unique(cpy.begin(), end);
368 cpy.shrink(std::distance(cpy.begin(), end));
369
370 if (cpy.size() == 0) return ext<!up>(kind, dbg);
371 if (cpy.size() == 1) return cpy[0];
372
373 // TODO simplify mixed terms with joins and meets
374
375 return unify<TBound<up>>(cpy.size(), kind, cpy, dbg);
376}
