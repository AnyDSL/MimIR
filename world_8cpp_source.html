213const Def* World::app(const Def* callee, const Def* arg, const Def* dbg) {
214 auto pi = callee->type()->isa<Pi>();
215
216 if (err()) {
217 if (!pi) err()->err(dbg->loc(), "called expression '{}' is not of function type", callee);
218 if (!checker().assignable(pi->dom(), arg, dbg)) err()->ill_typed_app(callee, arg, dbg);
219 }
220
221 auto type = pi->reduce(arg).back();
222 auto [axiom, curry] = Axiom::get(callee);
223 if (axiom && curry == 1) {
224 if (auto normalize = axiom->normalizer()) return normalize(type, callee, arg, dbg);
225 }
226
227 return unify<App>(2, axiom, curry - 1, type, callee, arg, dbg);
228}

262const Def* World::tuple(const Def* type, Defs ops, const Def* dbg) {
263 if (err()) {
264 // TODO type-check type vs inferred type
265 }
266
267 auto n = ops.size();
268 if (!type->isa_nom<Sigma>()) {
269 if (n == 0) return tuple();
270 if (n == 1) return ops[0];
271 if (std::all_of(ops.begin() + 1, ops.end(), [&](auto op) { return ops[0] == op; })) return pack(n, ops[0]);
272 }
273
274 // eta rule for tuples:
275 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
276 if (n == 0) goto out;
277
278 if (auto extract = ops[0]->isa<Extract>()) {
279 auto tup = extract->tuple();
280 bool eta = tup->type() == type;
281 for (size_t i = 0; i != n && eta; ++i) {
282 if (auto extract = ops[i]->isa<Extract>()) {
283 if (auto index = isa_lit(extract->index())) {
284 if (eta &= u64(i) == *index) {
285 eta &= extract->tuple() == tup;
286 continue;
287 }
288 }
289 }
290 eta = false;
291 }
292
293 if (eta) return tup;
294 }
295
296out:
297 return unify<Tuple>(ops.size(), type, ops, dbg);
298}

306const Def* World::extract(const Def* d, const Def* index, const Def* dbg) {
307 if (index->isa<Arr>() || index->isa<Pack>()) {
308 DefArray ops(as_lit(index->arity()), [&](size_t) { return extract(d, index->ops().back()); });
309 return index->isa<Arr>() ? sigma(ops, dbg) : tuple(ops, dbg);
310 } else if (index->isa<Sigma>() || index->isa<Tuple>()) {
311 auto n = index->num_ops();
312 DefArray idx(n, [&](size_t i) { return index->op(i); });
313 DefArray ops(n, [&](size_t i) { return d->proj(n, as_lit(idx[i])); });
314 return index->isa<Sigma>() ? sigma(ops, dbg) : tuple(ops, dbg);
315 }
316
317 auto type = d->unfold_type();
318 if (err()) {
319 if (!checker().equiv(type->arity(), isa_sized_type(index->type()), dbg))
320 err()->index_out_of_range(type->arity(), index, dbg);
321 }
322
323 // nom sigmas can be 1-tuples
324 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1 && !d->type()->isa_nom<Sigma>()) return d;
325 if (auto pack = d->isa_structural<Pack>()) return pack->body();
326
327 // extract(insert(x, index, val), index) -> val
328 if (auto insert = d->isa<Insert>()) {
329 if (index == insert->index()) return insert->value();
330 }
331
332 if (auto i = isa_lit(index)) {
333 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
334
335 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
336 if (auto insert = d->isa<Insert>()) {
337 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index, dbg);
338 }
339
340 if (auto sigma = type->isa<Sigma>()) {
341 if (auto nom_sigma = sigma->isa_nom<Sigma>()) {
342 Scope scope(nom_sigma);
343 auto t = rewrite(sigma->op(*i), nom_sigma->var(), d, scope);
344 return unify<Extract>(2, t, d, index, dbg);
345 }
346
347 return unify<Extract>(2, sigma->op(*i), d, index, dbg);
348 }
349 }
350
351 // e.g. (t, f)#cond, where t&f's types contain nominals but still are alpha-equiv
352 // for now just use t's type.
353 if (auto sigma = type->isa<Sigma>();
354 sigma && std::all_of(sigma->ops().begin() + 1, sigma->ops().end(),
355 [&](auto op) { return checker().equiv<false>(sigma->op(0), op, dbg); }))
356 return unify<Extract>(2, sigma->op(0), d, index, dbg);
357
358 if (err() && !type->isa<Arr>())
359 err()->err(dbg->loc(), "cannot extract from non-homogeneous sigma with non-literal index");
360
361 type = type->as<Arr>()->body();
362 return unify<Extract>(2, type, d, index, dbg);
363}

365const Def* World::insert(const Def* d, const Def* index, const Def* val, const Def* dbg) {
366 auto type = d->unfold_type();
367
368 if (err() && !checker().equiv(type->arity(), isa_sized_type(index->type()), dbg))
369 err()->index_out_of_range(type->arity(), index, dbg);
370
371 if (auto mod = isa_lit(isa_sized_type(index->type())); mod && *mod == 1)
372 return tuple(d, {val}, dbg); // d could be nom - that's why the tuple ctor is needed
373
374 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
375 if (auto t = d->isa<Tuple>()) return t->refine(as_lit(index), val);
376
377 // insert(4; x, 2, y) -> (x, x, y, x)
378 if (auto pack = d->isa<Pack>()) {
379 if (auto a = isa_lit(pack->arity())) {
380 DefArray new_ops(*a, pack->body());
381 new_ops[as_lit(index)] = val;
382 return tuple(type, new_ops, dbg);
383 }
384 }
385
386 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
387 if (auto insert = d->isa<Insert>()) {
388 if (insert->index() == index) d = insert->tuple();
389 }
390
391 return unify<Insert>(3, d, index, val, dbg);
392}

403const Def* World::arr(const Def* shape, const Def* body, const Def* dbg) {
404 if (err()) {
405 if (!is_shape(shape->type())) err()->expected_shape(shape, dbg);
406 }
407
408 if (auto a = isa_lit<u64>(shape)) {
409 if (*a == 0) return sigma();
410 if (*a == 1) return body;
411 }
412
413 // (a, b, c); body -> a; (b, c); body
414 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body), dbg);
415
416 // <n; x>; body -> x; <n-1, x>; body
417 if (auto p = shape->isa<Pack>()) {
418 if (auto s = isa_lit(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body), dbg);
419 }
420
421 return unify<Arr>(2, body->unfold_type(), shape, body, dbg);

481template<bool up>
482const Def* World::bound(Defs ops, const Def* dbg) {
483 auto kind = infer_type_level(*this, ops);
484
485 // has ext<up> value?
486 if (std::ranges::any_of(ops, [&](const Def* op) { return up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
487 return ext<up>(kind);
488
489 // ignore: ext<!up>
490 DefArray cpy(ops);
491 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](const Def* op) { return !isa_ext(op); });
492
493 // sort and remove duplicates
494 std::sort(cpy.begin(), end, GIDLt<const Def*>());
495 end = std::unique(cpy.begin(), end);
496 cpy.shrink(std::distance(cpy.begin(), end));
497
498 if (cpy.size() == 0) return ext<!up>(kind, dbg);
499 if (cpy.size() == 1) return cpy[0];
500
501 // TODO simplify mixed terms with joins and meets
502
503 return unify<TBound<up>>(cpy.size(), kind, cpy, dbg);
504}
