118template<Sort sort> Ref World::umax(Defs ops_) {
119 // consume nested umax
120 DefVec ops;
121 for (auto op : ops_)
122 if (auto um = op->isa<UMax>())
123 ops.insert(ops.end(), um->ops().begin(), um->ops().end());
124 else
125 ops.emplace_back(op);
126
127 level_t lvl = 0;
128 for (auto& op : ops) {
129 Ref r = op;
130 if (sort == Sort::Term) r = r->unfold_type();
131 if (sort <= Sort::Type) r = r->unfold_type();
132 if (sort <= Sort::Kind) {
133 if (auto type = r->isa<Type>())
134 r = type->level();
135 else
136 error(r, "operand '{}' must be a .Type of some level", r);
137 }
138
139 if (!r->type()->isa<Univ>())
140 error(r, "operand '{}' of a universe max must be of type '.Univ' but is of type '{}'", r, r->type());
141
142 op = r;
143
144 if (auto l = Lit::isa(op))
145 lvl = std::max(lvl, *l);
146 else
147 lvl = level_t(-1);
148 }
149
150 const Def* ldef;
151 if (lvl != level_t(-1))
152 ldef = lit_univ(lvl);
153 else {
154 std::ranges::sort(ops, [](auto op1, auto op2) { return op1->gid() < op2->gid(); });
155 ops.erase(std::unique(ops.begin(), ops.end()), ops.end());
156 ldef = unify<UMax>(ops.size(), *this, ops);
157 }
158
159 return sort == Sort::Univ ? ldef : type(ldef);
160}

172Ref World::iapp(Ref callee, Ref arg) {
173 while (auto pi = callee->type()->isa<Pi>()) {
174 if (pi->is_implicit()) {
175 auto infer = mut_infer(pi->dom());
176 auto a = app(callee, infer);
177 callee = a;
178 } else {
179 // resolve Infers now if possible before normalizers are run
180 if (auto app = callee->isa<App>(); app && app->curry() == 1) {
181 Check::assignable(callee->type()->as<Pi>()->dom(), arg);
182 auto apps = decurry(app);
183 callee = apps.front()->callee();
184 for (auto app : apps) callee = this->app(callee, Ref::refer(app->arg()));
185 }
186 break;
187 }
188 }
189
190 return app(callee, arg);
191}

193Ref World::app(Ref callee, Ref arg) {
194 Infer::eliminate(Array<Ref*>{&callee, &arg});
195 auto pi = callee->type()->isa<Pi>();
196
197 if (!pi) error(callee, "called expression '{}' : '{}' is not of function type", callee, callee->type());
198 if (!Check::assignable(pi->dom(), arg))
199 error(arg, "cannot pass argument \n'{}' of type \n'{}' to \n'{}' of domain \n'{}'", arg, arg->type(), callee,
200 pi->dom());
201
202 if (auto imm = callee->isa_imm<Lam>()) return imm->body();
203 if (auto lam = callee->isa_mut<Lam>(); lam && lam->is_set() && lam->filter() != lit_ff()) {
204 Scope scope(lam);
205 ScopeRewriter rw(scope);
206 rw.map(lam->var(), arg);
207 if (rw.rewrite(lam->filter()) == lit_tt()) {
208 DLOG("partial evaluate: {} ({})", lam, arg);
209 return rw.rewrite(lam->body());
210 }
211 }
212
213 auto type = pi->reduce(arg).back();
214 return raw_app<true>(type, callee, arg);
215}

217template<bool Normalize> Ref World::raw_app(Ref type, Ref callee, Ref arg) {
218 auto [axiom, curry, trip] = Axiom::get(callee);
219 if (axiom) {
220 curry = curry == 0 ? trip : curry;
221 curry = curry == Axiom::Trip_End ? curry : curry - 1;
222
223 if (auto normalize = axiom->normalizer(); Normalize && normalize && curry == 0)
224 return normalize(type, callee, arg);
225 }
226
227 return unify<App>(2, axiom, curry, trip, type, callee, arg);
228}

249Ref World::tuple(Ref type, Defs ops) {
250 // TODO type-check type vs inferred type
251
252 auto n = ops.size();
253 if (!type->isa_mut<Sigma>()) {
254 if (n == 0) return tuple();
255 if (n == 1) return ops[0];
256 if (auto uni = Check::is_uniform(ops)) return pack(n, uni);
257 }
258
259 if (n != 0) {
260 // eta rule for tuples:
261 // (extract(tup, 0), extract(tup, 1), extract(tup, 2)) -> tup
262 if (auto extract = ops[0]->isa<Extract>()) {
263 auto tup = extract->tuple();
264 bool eta = tup->type() == type;
265 for (size_t i = 0; i != n && eta; ++i) {
266 if (auto extract = ops[i]->isa<Extract>()) {
267 if (auto index = Lit::isa(extract->index())) {
268 if (eta &= u64(i) == *index) {
269 eta &= extract->tuple() == tup;
270 continue;
271 }
272 }
273 }
274 eta = false;
275 }
276
277 if (eta) return tup;
278 }
279 }
280
281 return unify<Tuple>(ops.size(), type, ops);
282}

290Ref World::extract(Ref d, Ref index) {
291 assert(d);
292 if (index->isa<Tuple>()) {
293 auto n = index->num_ops();
294 DefArray idx(n, [&](size_t i) { return index->op(i); });
295 DefArray ops(n, [&](size_t i) { return d->proj(n, Lit::as(idx[i])); });
296 return tuple(ops);
297 } else if (index->isa<Pack>()) {
298 DefArray ops(index->as_lit_arity(), [&](size_t) { return extract(d, index->ops().back()); });
299 return tuple(ops);
300 }
301
302 Ref size = Idx::size(index->type());
303 Ref type = d->unfold_type();
304
305 if (auto l = Lit::isa(size); l && *l == 1) {
306 if (auto l = Lit::isa(index); !l || *l != 0) WLOG("unknown Idx of size 1: {}", index);
307 if (auto sigma = type->isa_mut<Sigma>(); sigma && sigma->num_ops() == 1) {
308 // mut sigmas can be 1-tuples; TODO mutables Arr?
309 } else {
310 return d;
311 }
312 }
313
314 if (auto pack = d->isa_imm<Pack>()) return pack->body();
315
316 if (!Check::alpha(type->arity(), size))
317 error(index, "index '{}' does not fit within arity '{}'", index, type->arity());
318
319 // extract(insert(x, index, val), index) -> val
320 if (auto insert = d->isa<Insert>()) {
321 if (index == insert->index()) return insert->value();
322 }
323
324 if (auto i = Lit::isa(index)) {
325 if (auto tuple = d->isa<Tuple>()) return tuple->op(*i);
326
327 // extract(insert(x, j, val), i) -> extract(x, i) where i != j (guaranteed by rule above)
328 if (auto insert = d->isa<Insert>()) {
329 if (insert->index()->isa<Lit>()) return extract(insert->tuple(), index);
330 }
331
332 if (auto sigma = type->isa<Sigma>()) {
333 if (auto mut_sigma = sigma->isa_mut<Sigma>()) {
334 Scope scope(mut_sigma);
335 auto t = rewrite(sigma->op(*i), mut_sigma->var(), d, scope);
336 return unify<Extract>(2, t, d, index);
337 }
338
339 return unify<Extract>(2, sigma->op(*i), d, index);
340 }
341 }
342
343 const Def* elem_t;
344 if (auto arr = type->isa<Arr>())
345 elem_t = arr->reduce(index);
346 else
347 elem_t = extract(tuple(type->as<Sigma>()->ops()), index);
348
349 assert(d);
350 return unify<Extract>(2, elem_t, d, index);
351}

353Ref World::insert(Ref d, Ref index, Ref val) {
354 auto type = d->unfold_type();
355 auto size = Idx::size(index->type());
356 auto lidx = Lit::isa(index);
357
358 if (!Check::alpha(type->arity(), size))
359 error(index, "index '{}' does not fit within arity '{}'", index, type->arity());
360
361 if (lidx) {
362 auto target_type = type->proj(*lidx);
363 if (!Check::assignable(target_type, val))
364 error(val, "value of type {} is not assignable to type {}", val->type(), target_type);
365 }
366
367 if (auto l = Lit::isa(size); l && *l == 1)
368 return tuple(d, {val}); // d could be mut - that's why the tuple ctor is needed
369
370 // insert((a, b, c, d), 2, x) -> (a, b, x, d)
371 if (auto t = d->isa<Tuple>(); t && lidx) return t->refine(*lidx, val);
372
373 // insert(‹4; x›, 2, y) -> (x, x, y, x)
374 if (auto pack = d->isa<Pack>(); pack && lidx) {
375 if (auto a = pack->isa_lit_arity(); a && *a < flags().scalerize_threshold) {
376 DefArray new_ops(*a, pack->body());
377 new_ops[*lidx] = val;
378 return tuple(type, new_ops);
379 }
380 }
381
382 // insert(insert(x, index, y), index, val) -> insert(x, index, val)
383 if (auto insert = d->isa<Insert>()) {
384 if (insert->index() == index) d = insert->tuple();
385 }
386
387 return unify<Insert>(3, d, index, val);
388}

391Ref World::arr(Ref shape, Ref body) {
392 if (!is_shape(shape->type())) error(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
393
394 if (auto a = Lit::isa(shape)) {
395 if (*a == 0) return sigma();
396 if (*a == 1) return body;
397 }
398
399 // «(a, b)#i; T» -> («a, T», <b, T»)#i
400 if (auto ex = shape->isa<Extract>()) {
401 if (auto tup = ex->tuple()->isa<Tuple>()) {
402 DefArray arrs(tup->num_ops(), [&](size_t i) { return arr(tup->op(i), body); });
403 return extract(tuple(arrs), ex->index());
404 }
405 }
406
407 // «(a, b, c); body» -> «a; «(b, c); body»»
408 if (auto tuple = shape->isa<Tuple>()) return arr(tuple->ops().front(), arr(tuple->ops().skip_front(), body));
409
410 // «<n; x>; body» -> «x; «<n-1, x>; body»»
411 if (auto p = shape->isa<Pack>()) {
412 if (auto s = Lit::isa(p->shape())) return arr(*s, arr(pack(*s - 1, p->body()), body));
413 }
414
415 return unify<Arr>(2, body->unfold_type(), shape, body);
416}

418Ref World::pack(Ref shape, Ref body) {
419 if (!is_shape(shape->type())) error(shape, "expected shape but got '{}' of type '{}'", shape, shape->type());
420
421 if (auto a = Lit::isa(shape)) {
422 if (*a == 0) return tuple();
423 if (*a == 1) return body;
424 }
425
426 // <(a, b, c); body> -> <a; «(b, c); body>>
427 if (auto tuple = shape->isa<Tuple>()) return pack(tuple->ops().front(), pack(tuple->ops().skip_front(), body));
428
429 // <<n; x>; body> -> <x; <<n-1, x>; body>>
430 if (auto p = shape->isa<Pack>()) {
431 if (auto s = Lit::isa(p->shape())) return pack(*s, pack(pack(*s - 1, p->body()), body));
432 }
433
434 auto type = arr(shape, body->type());
435 return unify<Pack>(1, type, body);
436}

471template<bool Up> Ref World::bound(Defs ops) {
472 auto kind = umax<Sort::Type>(ops);
473
474 // has ext<Up> value?
475 if (std::ranges::any_of(ops, [&](Ref op) { return Up ? bool(op->isa<Top>()) : bool(op->isa<Bot>()); }))
476 return ext<Up>(kind);
477
478 // ignore: ext<!Up>
479 DefArray cpy(ops);
480 auto [_, end] = std::ranges::copy_if(ops, cpy.begin(), [&](Ref op) { return !op->isa<Ext>(); });
481
482 // sort and remove duplicates
483 std::sort(cpy.begin(), end, GIDLt<const Def*>());
484 end = std::unique(cpy.begin(), end);
485 cpy.shrink(std::distance(cpy.begin(), end));
486
487 if (cpy.size() == 0) return ext<!Up>(kind);
488 if (cpy.size() == 1) return cpy[0];
489
490 // TODO simplify mixed terms with joins and meets
491
492 return unify<TBound<Up>>(cpy.size(), kind, cpy);
493}
