128template<class T, class... Args>
129Stream& Stream::fmt(const char* s, T&& t, Args&&... args) {
130 while (*s != '\0') {
131 auto next = s + 1;
132
133 switch (*s) {
134 case '\n': s++; endl(); break;
135 case '\t': s++; indent(); break;
136 case '\b': s++; dedent(); break;
137 case '{': {
138 if (match2nd(next, s, '{')) continue;
139 s++; // skip opening brace '{'
140
141 std::string spec;
142 while (*s != '\0' && *s != '}') spec.push_back(*s++);
143 assert(*s == '}' && "unmatched closing brace '}' in format string");
144
145 if constexpr (std::ranges::range<decltype(t)>) {
146 range(t, spec.c_str());
147 } else {
148 (*this) << t;
149 }
150
151 ++s; // skip closing brace '}'
152 return fmt(s, std::forward<Args&&>(args)...); // call even when *s == '\0' to detect extra arguments
153 }
154 case '}':
155 if (match2nd(next, s, '}')) continue;
156 assert(false && "unmatched/unescaped closing brace '}' in format string");
157 default:
158 (*this) << *s++;
159 }
160 }
161
162 assert(false && "invalid format string for 's'");
163}
