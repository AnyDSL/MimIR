186template<class T, class... Args>
187Stream& Stream::fmt(const char* s, T&& t, Args&&... args) {
188 while (*s != '\0') {
189 auto next = s + 1;
190
191 switch (*s) {
192 // clang-format off
193 case '\n': s++; endl(); break;
194 case '\t': s++; indent(); break;
195 case '\b': s++; dedent(); break;
196 // clang-format on
197 case '{': {
198 if (match2nd(next, s, '{')) continue;
199 s++; // skip opening brace '{'
200
201 std::string spec;
202 while (*s != '\0' && *s != '}') spec.push_back(*s++);
203 assert(*s == '}' && "unmatched closing brace '}' in format string");
204
205 if constexpr (std::ranges::range<decltype(t)>) {
206 range(t, spec.c_str());
207 } else {
208 (*this) << t;
209 }
210
211 ++s; // skip closing brace '}'
212 return fmt(s, std::forward<Args&&>(args)...); // call even when *s == '\0' to detect extra arguments
213 }
214 case '}':
215 if (match2nd(next, s, '}')) continue;
216 assert(false && "unmatched/unescaped closing brace '}' in format string");
217 default: (*this) << *s++;
218 }
219 }
220
221 unreachable("invalid format string for 's'");
222}

225Stream& Stream::range(const R& r, const char* sep, F f) {
226 const char* cur_sep = "";
