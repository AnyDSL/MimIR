17// nomenclature:
18// c is the character from the string we want to match
19
20// see lit/regex/match_manual.thorin for a hand written state machine impl that this is based on
21
22// the main idea is:
23// for every state in the DFA, we create a lam that checks if the char c at pos is the end of the string \0
24// if not, jump to a checker lam, that consists of a few bit-wise or-ed in-range checks to verify if we can transition
25// to a certain other state with c if any of the checks is true, we jump to the corresponding state lam with the updated
26// position if the check fails, we jump to the next checker lam or the exit lam if we checked all possible transitions
27// Note, since the DFA stores the transitions as chars, not ranges, we have to merge the transitions to ranges first
28// (cf. transitions_to_ranges)

38absl::flat_hash_map<const automaton::DFANode*, std::vector<std::pair<nat_t, nat_t>>>
39transitions_to_ranges(World& w, const automaton::DFANode* state) {
40 absl::flat_hash_map<const automaton::DFANode*, std::vector<std::pair<nat_t, nat_t>>> state2ranges;
41 state->for_transitions([&](std::uint16_t transition, const automaton::DFANode* next_state) {
42 if (!state2ranges.contains(next_state))
43 state2ranges.emplace(next_state, std::vector<std::pair<nat_t, nat_t>>{
44 {transition, transition}
45 });
46 else
47 state2ranges[next_state].emplace_back(transition, transition);
48 });
49 std::pair<nat_t, nat_t> any_range{0, 255};
50 for (auto& [state, ranges] : state2ranges) {
51 if (std::find(ranges.cbegin(), ranges.cend(), any_range) != ranges.cend()) {
52 ranges = {any_range};
53 continue;
54 }
55
56 std::sort(ranges.begin(), ranges.end(), regex::RangeCompare{});
57 ranges = regex::merge_ranges(ranges, [&w](auto&&... args) { w.DLOG(std::forward<decltype(args)>(args)...); });
58 }
59 return state2ranges;
60}

62Ref match_range(Ref c, nat_t lo, nat_t hi) {
63 World& w = c->world();
64 if (lo == 0 && hi == 255) return w.lit_tt();
65
66 // .let in_range = %core.bit2.and_ 0 (%core.icmp.uge (char, lower), %core.icmp.ule (char, upper));
67 auto below_hi = w.call(core::icmp::ule, w.tuple({c, w.lit_idx(256, hi)}));
68 auto above_lo = w.call(core::icmp::uge, w.tuple({c, w.lit_idx(256, lo)}));
69 return w.call(core::bit2::and_, w.lit_nat(2), w.tuple({below_hi, above_lo}));
70}

72absl::flat_hash_map<const automaton::DFANode*, Ref>
73create_check_match_transitions_from(Ref c, const automaton::DFANode* state) {
74 World& w = c->world();
75 absl::flat_hash_map<const automaton::DFANode*, Ref> state2check;
76
77 auto state2ranges = transitions_to_ranges(w, state);
78
79 for (auto& [state, ranges] : state2ranges) {
80 for (auto& [lo, hi] : ranges)
81 if (!state2check.contains(state))
82 state2check.emplace(state, match_range(c, lo, hi));
83 else
84 state2check[state]
85 = w.call(core::bit2::or_, w.lit_nat(2), w.tuple({state2check[state], match_range(c, lo, hi)}));
86 }
87 return state2check;
88}

171 }
