60std::optional<std::filesystem::path> get_path_to_current_executable() {
61#ifdef _WIN32
62 std::vector<char> path_buffer;
63 size_t read = 0;
64 do {
65 // start with 256 (almost MAX_PATH) and grow exp
66 path_buffer.resize(std::max(path_buffer.size(), static_cast<size_t>(128)) * 2);
67 read = GetModuleFileNameA(nullptr, path_buffer.data(), static_cast<DWORD>(path_buffer.size()));
68 } while (read == path_buffer.size()); // if equal, the buffer was too small.
69 if (read != 0) {
70 path_buffer.resize(read);
71 return std::filesystem::path{path_buffer.data()}.parent_path().parent_path() / "lib";
72 }
