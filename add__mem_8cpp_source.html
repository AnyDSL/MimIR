13std::pair<const App*, Array<Lam*>> isa_apped_mut_lam_in_tuple(const Def* def) {
14 if (auto app = def->isa<App>()) {
15 std::vector<Lam*> lams;
16 std::deque<const Def*> wl;
17 wl.push_back(app->callee());
18 while (!wl.empty()) {
19 auto elem = wl.front();
20 wl.pop_front();
21 if (auto mut = elem->isa_mut<Lam>()) {
22 lams.push_back(mut);
23 } else if (auto extract = elem->isa<Extract>()) {
24 if (auto tuple = extract->tuple()->isa<Tuple>())
25 for (auto&& op : tuple->ops()) wl.push_back(op);
26 else
27 return {nullptr, {}};
28 } else {
29 return {nullptr, {}};
30 }
31 }
32 return {app, lams};
33 }
34 return {nullptr, {}};
35}

38template<class F, class H> const Def* rewrite_mut_lam_in_tuple(const Def* def, F&& rewrite, H&& rewrite_idx) {
39 auto& w = def->world();
40 if (auto mut = def->isa_mut<Lam>()) return std::forward<F>(rewrite)(mut);
41
42 auto extract = def->as<Extract>();
43 auto tuple = extract->tuple()->as<Tuple>();
44 DefArray new_ops{tuple->ops(), [&](const Def* op) {
45 return rewrite_mut_lam_in_tuple(op, std::forward<F>(rewrite), std::forward<H>(rewrite_idx));
46 }};
47 return w.extract(w.tuple(new_ops), rewrite_idx(extract->index()));
48}

54 RewriteArg&& rewrite_arg,
55 RewriteIdx&& rewrite_idx) {

75const Def* AddMem::mem_for_lam(Lam* lam) const {
76 if (auto it = mem_rewritten_.find(lam); it != mem_rewritten_.end()) {
77 // We created a new lambda. Therefore, we want to lookup the mem for the new lambda.
78 lam = it->second->as_mut<Lam>();
79 }
80 if (auto it = val2mem_.find(lam); it != val2mem_.end()) {
81 lam->world().DLOG("found mem for {} in val2mem_ : {}", lam, it->second);
82 // We found a (overwritten) memory in the lambda.
83 return it->second;
84 }
85 // As a fallback, we lookup the memory in vars of the lambda.
86 auto mem = mem::mem_var(lam);
87 assert(mem && "mut must have mem!");
88 return mem;
89}

100const Def* AddMem::rewrite_pi(const Pi* pi) {
