11static std::pair<const App*, Array<Lam*>> isa_apped_nom_lam_in_tuple(const Def* def) {
12 if (auto app = def->isa<App>()) {
13 std::vector<Lam*> lams;
14 std::deque<const Def*> wl;
15 wl.push_back(app->callee());
16 while (!wl.empty()) {
17 auto elem = wl.front();
18 wl.pop_front();
19 if (auto nom = elem->isa_nom<Lam>()) {
20 lams.push_back(nom);
21 } else if (auto extract = elem->isa<Extract>()) {
22 if (auto tuple = extract->tuple()->isa<Tuple>()) {
23 for (auto&& op : tuple->ops()) wl.push_back(op);
24 } else {
25 return {nullptr, {}};
26 }
27 } else {
28 return {nullptr, {}};
29 }
30 }
31 return {app, lams};
32 }
33 return {nullptr, {}};
34}

37template<class F, class H>
38static const Def* rewrite_nom_lam_in_tuple(const Def* def, F&& rewrite, H&& rewrite_idx) {
39 auto& w = def->world();
40 if (auto nom = def->isa_nom<Lam>()) { return std::forward<F>(rewrite)(nom); }
41
42 auto extract = def->as<Extract>();
43 auto tuple = extract->tuple()->as<Tuple>();
44 DefArray new_ops{tuple->ops(), [&](const Def* op) {
45 return rewrite_nom_lam_in_tuple(op, std::forward<F>(rewrite), std::forward<H>(rewrite_idx));
46 }};
47 return w.extract(w.tuple(new_ops, tuple->dbg()), rewrite_idx(extract->index()), extract->dbg());
48}
