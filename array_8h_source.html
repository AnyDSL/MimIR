21/// A container-like wrapper for an array.
22/// The array may either stem from a C array, a `std::vector`, a `std::initializer_list`, an Array or another Span.
23/// Span does **not** own the data and, thus, does not destroy any data.
24/// Likewise, you must be carefull to not destroy data a Span is pointing to.
25/// Note that you can often construct a Span inline with an initializer_list: `foo(arg1, {elem1, elem2, elem3}, arg3)`.
26template<class T> class Span {
27public:
28 using value_type = T;
29 using const_iterator = const T*;
30 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
31
32 /// @name Constructor, Destructor & Assignment
33 ///@{
34 Span() noexcept = default;
35 Span(const Span<T>&) noexcept = default;
36 Span(Span&& span) noexcept = default;
37 Span(size_t size, const T* ptr)
38 : size_(size)
39 , ptr_(ptr) {}
40 template<size_t N>
41 Span(const T (&ptr)[N])
42 : size_(N)
43 , ptr_(ptr) {}
44 Span(const Array<T>& array)
45 : size_(array.size())
46 , ptr_(array.begin()) {}
47 template<size_t N>
48 Span(const std::array<T, N>& array)
49 : size_(N)
50 , ptr_(array.data()) {}
51 Span(std::initializer_list<T> list)
52 : size_(std::ranges::distance(list))
53 , ptr_(std::begin(list)) {}
54 Span(const std::vector<T>& vector)

56 , ptr_(vector.data()) {}

135template<typename T, size_t StackSize> class ArrayStorage {
136 // Unions only work with trivial types.
137 static_assert(std::is_trivial<T>::value, "Stack based array storage is only available for trivial types");
138
139public:
140 /// @name Constructor, Destructor & Assignment
141 ///@{
142 ArrayStorage() noexcept
143 : size_(0)
144 , stack_(true) {}
145 ArrayStorage(size_t size) {
146 size_ = size;
147 stack_ = size <= StackSize;
148 if (!stack_) data_.ptr = new T[size]();
149 }
150 ArrayStorage(ArrayStorage&& other) noexcept
151 : ArrayStorage() {
152 swap(*this, other);
153 }
154 ~ArrayStorage() {
155 if (!stack_) delete[] data_.ptr;
156 }
157 ArrayStorage& operator=(ArrayStorage other) noexcept { return swap(*this, other), *this; }
158 ///@}
159
160 /// @name Getters
161 ///@{
162 size_t size() const { return size_; }
163 void shrink(size_t new_size) { size_ = new_size; }
164 T* data() { return stack_ ? data_.elems : data_.ptr; }
165 const T* data() const { return stack_ ? data_.elems : data_.ptr; }
166 ///@}
167
168 friend void swap(ArrayStorage& a, ArrayStorage& b) noexcept {
169 using std::swap;
170 auto size = a.size_;
171 a.size_ = b.size_;
172 b.size_ = size;
173 auto stack = a.stack_;
174 a.stack_ = b.stack_;
175 b.stack_ = stack;
176 swap(a.data_, b.data_);
177 }

179private:
180 size_t size_ : sizeof(size_t) * 8 - 1;
181 bool stack_ : 1;

183 T* ptr;
