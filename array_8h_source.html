22/// A container-like wrapper for an array.
23/// The array may either stem from a C array, a `std::vector`, a `std::initializer_list`, an Array or another ArrayRef.
24/// ArrayRef does **not** own the data and, thus, does not destroy any data.
25/// Likewise, you must be carefull to not destroy data an ArrayRef is pointing to.
26/// Thorin makes use of ArrayRef%s in many places.
27/// Note that you can often construct an ArrayRef inline with an initializer_list: `foo(arg1, {elem1, elem2, elem3},
28/// arg3)`. Useful operations are ArrayRef::skip_front and ArrayRef::skip_back to create other ArrayRef%s.

31public:

33 using const_iterator = const T*;
34 using const_reverse_iterator = std::reverse_iterator<const_iterator>;

162 size_ = size;
163 stack_ = size <= StackSize;

165 }

180 T* data() { return stack_ ? data_.elems : data_.ptr; }

183 friend void swap(ArrayStorage& a, ArrayStorage& b) {
184 auto size = a.size_;
185 a.size_ = b.size_;
186 b.size_ = size;
187 auto stack = a.stack_;
188 a.stack_ = b.stack_;
189 b.stack_ = stack;

193private:

238/// * If the size is small enough, the array resides on the stack.

391template<class T>
392Array<T> ArrayRef<T>::cut(ArrayRef<size_t> indices, size_t reserve) const {
393 size_t num_old = size();
394 size_t num_idx = indices.size();
395 size_t num_res = num_old - num_idx;
396
397 Array<T> result(num_res + reserve);
398
399 for (size_t o = 0, i = 0, r = 0; o < num_old; ++o) {
400 if (i < num_idx && indices[i] == o)
401 ++i;
402 else
403 result[r++] = (*this)[o];
404 }
405
406 return result;
407}
