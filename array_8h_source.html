306Array<T> ArrayRef<T>::cut(ArrayRef<size_t> indices, size_t reserve) const {
307 size_t num_old = size();
308 size_t num_idx = indices.size();
309 size_t num_res = num_old - num_idx;
310
311 Array<T> result(num_res + reserve);
312
313 for (size_t o = 0, i = 0, r = 0; o < num_old; ++o) {
314 if (i < num_idx && indices[i] == o)
315 ++i;
316 else
317 result[r++] = (*this)[o];
318 }
319
320 return result;
321}
