129undo_t PassMan::analyze(const Def* def) {
130 undo_t undo = No_Undo;
131
132 if (def->dep_none() || analyzed(def)) {
133 // do nothing
134 } else if (auto nom = def->isa_nom()) {
135 curr_state().stack.push(nom);
136 } else if (auto proxy = def->isa<Proxy>()) {
137 proxy_ = true;
138 undo = passes_[proxy->pass()]->analyze(proxy);
139 } else {
140 auto var = def->isa<Var>();
141
142 if (!var) {
143 for (auto op : def->extended_ops()) undo = std::min(undo, analyze(op));
144 }
145
146 for (auto&& pass : passes_) {
147 if (pass->inspect()) undo = std::min(undo, var ? pass->analyze(var) : pass->analyze(def));
148 }
149 }
150
151 return undo;
