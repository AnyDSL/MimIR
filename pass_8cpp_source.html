145const Def* PassMan::rewrite(const Def* old_def) {
146 if (!old_def->has_dep()) return old_def;
147
148 if (auto mut = old_def->isa_mut()) {
149 curr_state().mut2visit.emplace(mut, curr_undo());
150 return map(mut, mut);
151 }
152
153 if (auto new_def = lookup(old_def)) {
154 if (old_def == *new_def)
155 return old_def;
156 else
157 return map(old_def, rewrite(*new_def));
158 }
159
160 auto new_type = old_def->type() ? rewrite(old_def->type()) : nullptr;
161 auto new_ops = absl::FixedArray<const Def*>(old_def->num_ops());
162 for (size_t i = 0, e = old_def->num_ops(); i != e; ++i)
163 new_ops[i] = rewrite(old_def->op(i));
164 auto new_def = old_def->rebuild(new_type, new_ops);
165
166 if (auto proxy = new_def->isa<Proxy>()) {
167 if (auto&& pass = passes_[proxy->pass()]; pass->inspect()) {
168 if (auto rw = pass->rewrite(proxy); rw != proxy) return map(old_def, rewrite(rw));
169 }
170 } else {
171 for (auto&& pass : passes_) {
172 if (!pass->inspect()) continue;
173
174 if (auto var = new_def->isa<Var>()) {
175 if (auto rw = pass->rewrite(var); rw != var) return map(old_def, rewrite(rw));
176 } else {
177 if (auto rw = pass->rewrite(new_def); rw != new_def) return map(old_def, rewrite(rw));
178 }
179 }
180 }
181
182 return map(old_def, new_def);
183}

185undo_t PassMan::analyze(const Def* def) {
186 undo_t undo = No_Undo;
187
188 if (!def->has_dep() || analyzed(def)) {
189 // do nothing
190 } else if (auto mut = def->isa_mut()) {
191 if (mut->is_set()) curr_state().stack.push(mut);
192 } else if (auto proxy = def->isa<Proxy>()) {
193 proxy_ = true;
194 undo = passes_[proxy->pass()]->analyze(proxy);
195 } else {
196 auto var = def->isa<Var>();
197 if (!var)
198 for (auto op : def->deps())
199 undo = std::min(undo, analyze(op));
200
201 for (auto&& pass : passes_)
202 if (pass->inspect()) undo = std::min(undo, var ? pass->analyze(var) : pass->analyze(def));
203 }
204
205 return undo;
206}
