130undo_t PassMan::analyze(Ref def) {
131 undo_t undo = No_Undo;
132
133 if (!def->dep() || analyzed(def)) {
134 // do nothing
135 } else if (auto mut = def->isa_mut()) {
136 curr_state().stack.push(mut);
137 } else if (auto proxy = def->isa<Proxy>()) {
138 proxy_ = true;
139 undo = passes_[proxy->pass()]->analyze(proxy);
140 } else {
141 auto var = def->isa<Var>();
