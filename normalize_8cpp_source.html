141template<ICmp cmp, nat_t w>
142struct Fold<ICmp, cmp, w> {
143 inline static Res run(u64 a, u64 b) {
144 using T = w2u<w>;
145 auto x = get<T>(a), y = get<T>(b);
146 bool result = false;
147 auto pm = !(x >> T(w - 1)) && (y >> T(w - 1));
148 auto mp = (x >> T(w - 1)) && !(y >> T(w - 1));
149 result |= ((cmp & ICmp::_x) != ICmp::_f) && pm;
150 result |= ((cmp & ICmp::_y) != ICmp::_f) && mp;
151 result |= ((cmp & ICmp::_g) != ICmp::_f) && x > y && !mp;
152 result |= ((cmp & ICmp::_l) != ICmp::_f) && x < y && !pm;
153 result |= ((cmp & ICmp::e) != ICmp::_f) && x == y;
154 return result;
155 }
156};

187/// Reassociates @p a und @p b according to following rules.
188/// We use the following naming convention while literals are prefixed with an 'l':
189/// ```
190/// a op b
191/// (x op y) op (z op w)
192///
193/// (1) la op (lz op w) -> (la op lz) op w
194/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
195/// (3) a op (lz op w) -> lz op (a op w)
196/// (4) (lx op y) op b -> lx op (y op b)
197/// ```
198template<sub_t sub>
199static const Def* reassociate(Tag2Enum<sub> op,
200 World& world,
201 [[maybe_unused]] const App* ab,
202 const Def* a,
203 const Def* b,
204 const Def* dbg) {
205 if (!is_associative(op)) return nullptr;
206
207 auto la = a->isa<Lit>();
208 auto xy = isa<sub>(op, a);
209 auto zw = isa<sub>(op, b);
210 auto lx = xy ? xy->arg(0)->template isa<Lit>() : nullptr;
211 auto lz = zw ? zw->arg(0)->template isa<Lit>() : nullptr;
212 auto y = xy ? xy->arg(1) : nullptr;
213 auto w = zw ? zw->arg(1) : nullptr;
214
215 std::function<const Def*(const Def*, const Def*)> make_op;
216
217 if constexpr (sub == Tag::ROp) {
218 // build rmode for all new ops by using the least upper bound of all involved apps
219 nat_t rmode = RMode::bot;
220 auto check_mode = [&](const App* app) {
221 auto app_m = isa_lit(app->arg(0));
222 if (!app_m || !(*app_m & RMode::reassoc)) return false;
223 rmode &= *app_m; // least upper bound
224 return true;
225 };
226
227 if (!check_mode(ab)) return nullptr;
228 if (lx && !check_mode(xy->decurry())) return nullptr;
229 if (lz && !check_mode(zw->decurry())) return nullptr;
230
231 make_op = [&](const Def* a, const Def* b) { return world.op(op, rmode, a, b, dbg); };
232 } else if constexpr (sub == Tag::Wrap) {
233 // if we reassociate Wraps, we have to forget about nsw/nuw
234 make_op = [&](const Def* a, const Def* b) { return world.op(op, WMode::none, a, b, dbg); };
235 } else {
236 make_op = [&](const Def* a, const Def* b) { return world.op(op, a, b, dbg); };
237 }
238
239 if (la && lz) return make_op(make_op(la, lz), w); // (1)
240 if (lx && lz) return make_op(make_op(lx, lz), make_op(y, w)); // (2)
241 if (lz) return make_op(lz, make_op(a, w)); // (3)
242 if (lx) return make_op(lx, make_op(y, b)); // (4)
243
244 return nullptr;
245}

251template<tag_t tag>
252static const Def* merge_cmps(std::array<std::array<u64, 2>, 2> tab, const Def* a, const Def* b, const Def* dbg) {
253 static_assert(sizeof(sub_t) == 1, "if this ever changes, please adjust the logic below");
254 static constexpr size_t num_bits = std::bit_width(Num<Tag2Enum<tag>> - 1_u64);
255
256 auto a_cmp = isa<tag>(a);
257 auto b_cmp = isa<tag>(b);
258
259 if (a_cmp && b_cmp && a_cmp->args() == b_cmp->args()) {
260 // push sub bits of a_cmp and b_cmp through truth table
261 sub_t res = 0;
262 sub_t a_sub = a_cmp.axiom()->sub();
263 sub_t b_sub = b_cmp.axiom()->sub();
264 for (size_t i = 0; i != num_bits; ++i, res >>= 1, a_sub >>= 1, b_sub >>= 1)
265 res |= tab[a_sub & 1][b_sub & 1] << 7_u8;
266 res >>= (7_u8 - u8(num_bits));
267
268 auto& world = a->world();
269 if constexpr (tag == Tag::RCmp)
270 return world.op(RCmp(res), /*rmode*/ a_cmp->decurry()->arg(0), a_cmp->arg(0), a_cmp->arg(1), dbg);
271 else
272 return world.op(ICmp(res), a_cmp->arg(0), a_cmp->arg(1), dbg);
273 }
274
275 return nullptr;
276}

428 auto& world = type->world();
429 auto callee = c->as<App>();
430 auto [a, b] = arg->projs<2>();
431 auto w = isa_lit(callee->arg());
432
433 if (auto result = fold<Shr, op>(world, type, callee, a, b, dbg)) return result;
434
435 if (auto la = a->isa<Lit>()) {
436 if (la == world.lit_int(*w, 0)) {
437 switch (op) {
438 case Shr::ashr: return la;
439 case Shr::lshr: return la;
440 default: unreachable();
441 }
442 }
443 }
444
445 if (auto lb = b->isa<Lit>()) {
446 if (lb == world.lit_int(*w, 0)) {
447 switch (op) {
448 case Shr::ashr: return a;
449 case Shr::lshr: return a;
450 default: unreachable();
451 }
452 }
453
454 if (lb->get() > *w) return world.bot(type, dbg);
455 }
456
457 return world.raw_app(callee, {a, b}, dbg);

460template<Wrap op>
461const Def* normalize_Wrap(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
462 auto& world = type->world();
463 auto callee = c->as<App>();
464 auto [a, b] = arg->projs<2>();
465 auto [m, w] = callee->args<2>(isa_lit<nat_t>); // mode and width
466
467 if (auto result = fold<Wrap, op>(world, type, callee, a, b, dbg)) return result;
468
469 // clang-format off
470 if (auto la = a->isa<Lit>()) {
471 if (la == world.lit_int(*w, 0)) {
472 switch (op) {
473 case Wrap::add: return b; // 0 + b -> b
474 case Wrap::sub: break;
475 case Wrap::mul: return la; // 0 * b -> 0
476 case Wrap::shl: return la; // 0 << b -> 0
477 default: unreachable();
478 }
479 }
480
481 if (la == world.lit_int(*w, 1)) {
482 switch (op) {
483 case Wrap::add: break;
484 case Wrap::sub: break;
485 case Wrap::mul: return b; // 1 * b -> b
486 case Wrap::shl: break;
487 default: unreachable();
488 }
489 }
490 }
491
492 if (auto lb = b->isa<Lit>()) {
493 if (lb == world.lit_int(*w, 0)) {
494 switch (op) {
495 case Wrap::sub: return a; // a - 0 -> a
496 case Wrap::shl: return a; // a >> 0 -> a
497 default: unreachable();
498 // add, mul are commutative, the literal has been normalized to the left
499 }
500 }
501
502 if (op == Wrap::sub)
503 return world.op(Wrap::add, *m, a, world.lit_int_mod(*w, ~lb->get() + 1_u64)); // a - lb -> a + (~lb + 1)
504 else if (op == Wrap::shl && lb->get() > *w)
505 return world.bot(type, dbg);
506 }
507
508 if (a == b) {
509 switch (op) {
510 case Wrap::add: return world.op(Wrap::mul, *m, world.lit_int(*w, 2), a, dbg); // a + a -> 2 * a
511 case Wrap::sub: return world.lit_int(*w, 0); // a - a -> 0
512 case Wrap::mul: break;
513 case Wrap::shl: break;
514 default: unreachable();
515 }
516 }
517 // clang-format on
518
519 if (auto res = reassociate<Tag::Wrap>(op, world, callee, a, b, dbg)) return res;
520
521 return world.raw_app(callee, {a, b}, dbg);
522}

550 case ROp::add: break;
551 case ROp::sub: break;
552 case ROp::mul: return b; // 1 * b -> b
553 case ROp::div: break;
554 case ROp::rem: break;
555 default: unreachable();
556 }
557 }
558 }
559
560 if (auto lb = b->isa<Lit>()) {
561 if (lb == world.lit_real(*w, 0.0)) {
562 switch (op) {
563 case ROp::sub: return a; // a - 0 -> a
564 case ROp::div: break;
565 case ROp::rem: break;
566 default: unreachable();
567 // add, mul are commutative, the literal has been normalized to the left
568 }
569 }
570 }
571
572 if (a == b) {
573 switch (op) {
574 case ROp::add: return world.op(ROp::mul, world.lit_real(*w, 2.0), a, dbg); // a + a -> 2 * a
575 case ROp::sub: return world.lit_real(*w, 0.0); // a - a -> 0
576 case ROp::mul: break;
577 case ROp::div: return world.lit_real(*w, 1.0); // a / a -> 1
578 case ROp::rem: break;
579 default: unreachable();
580 }
581 }
582 }
583 // clang-format on
584
585 if (auto res = reassociate<Tag::ROp>(op, world, callee, a, b, dbg)) return res;
586
587 return world.raw_app(callee, {a, b}, dbg);
588}

631template<Trait op>
632const Def* normalize_Trait(const Def*, const Def* callee, const Def* type, const Def* dbg) {
633 auto& world = type->world();
634
635 // todo: figure out a way to normalize traits on dialect types..
636 if (isa<Tag::Ptr>(type)) {
637 return world.lit_nat(8);
638 } else if (type->isa<Pi>()) {
639 return world.lit_nat(8); // Gets lowered to function ptr
640 } else if (auto int_ = isa<Tag::Int>(type)) {
641 if (int_->type()->isa<Top>()) return world.lit_nat(8);
642 if (auto w = isa_lit(int_->arg())) {
643 if (*w == 0) return world.lit_nat(8);
644 if (*w <= 0x0000'0000'0000'0100_u64) return world.lit_nat(1);
645 if (*w <= 0x0000'0000'0001'0000_u64) return world.lit_nat(2);
646 if (*w <= 0x0000'0001'0000'0000_u64) return world.lit_nat(4);
647 return world.lit_nat(8);
648 }
649 } else if (auto real = isa<Tag::Real>(type)) {
650 if (auto w = isa_lit(real->arg())) {
651 switch (*w) {
652 case 16: return world.lit_nat(2);
653 case 32: return world.lit_nat(4);
654 case 64: return world.lit_nat(8);
655 default: unreachable();
656 }
657 }
658 } else if (type->isa<Sigma>() || type->isa<Meet>()) {
659 u64 offset = 0;
660 u64 align = 1;
661 for (auto t : type->ops()) {
662 auto a = isa_lit(world.op(Trait::align, t));
663 auto s = isa_lit(world.op(Trait::size, t));
664 if (!a || !s) goto out;
665
666 align = std::max(align, *a);
667 offset = pad(offset, *a) + *s;
668 }
669
670 offset = pad(offset, align);
671 u64 size = std::max(1_u64, offset);
672
673 switch (op) {
674 case Trait::align: return world.lit_nat(align);
675 case Trait::size: return world.lit_nat(size);
676 }
677 } else if (auto arr = type->isa_structural<Arr>()) {
678 auto align = world.op(Trait::align, arr->body());
679 if constexpr (op == Trait::align) return align;
680
681 if (auto b = isa_lit(world.op(Trait::size, arr->body()))) {
682 auto i64_t = world.type_int_width(64);
683 auto s = world.op_bitcast(i64_t, arr->shape());
684 auto mul = world.op(Wrap::mul, WMode::nsw | WMode::nuw, world.lit_int(i64_t, *b), s);
685 return world.op_bitcast(world.type_nat(), mul);
686 }
687 } else if (auto join = type->isa<Join>()) {
688 if (auto sigma = join->convert()) return world.op(op, sigma, dbg);
689 }
690 return world.lit_nat(8);
691out:
692 return world.raw_app(callee, type, dbg);
693}

703template<nat_t min_sw, nat_t min_dw, Conv op>
704static const Def* fold_Conv(const Def* dst_type, const App* callee, const Def* src, const Def* dbg) {
705 auto& world = dst_type->world();
706 if (src->isa<Bot>()) return world.bot(dst_type, dbg);
707
708 auto [lit_dw, lit_sw] = callee->args<2>(isa_lit<nat_t>);
709 auto lit_src = src->isa<Lit>();
710 if (lit_src && lit_dw && lit_sw) {
711 if (op == Conv::u2u) {
712 if (*lit_dw == 0) return world.lit(dst_type, as_lit(lit_src));
713 return world.lit(dst_type, as_lit(lit_src) % *lit_dw);
714 }
715
716 if (isa<Tag::Int>(src->type())) *lit_sw = *mod2width(*lit_sw);
717 if (isa<Tag::Int>(dst_type)) *lit_dw = *mod2width(*lit_dw);
718
719 Res res;
720#define CODE(sw, dw) \
721 else if (*lit_dw == dw && *lit_sw == sw) { \
722 if constexpr (dw >= min_dw && sw >= min_sw) res = FoldConv<op, dw, sw>::run(lit_src->get()); \
723 }
724 if (false) {}
725 TABLE(CODE)
726#undef CODE
727 if (res) return world.lit(dst_type, *res, dbg);
728 return world.bot(dst_type, dbg);
729 }

734template<Conv op>
735const Def* normalize_Conv(const Def* dst_type, const Def* c, const Def* src, const Def* dbg) {
736 auto& world = dst_type->world();
737 auto callee = c->as<App>();
738
739 static constexpr auto min_sw = op == Conv::r2s || op == Conv::r2u || op == Conv::r2r ? 16 : 1;
740 static constexpr auto min_dw = op == Conv::s2r || op == Conv::u2r || op == Conv::r2r ? 16 : 1;
741 if (auto result = fold_Conv<min_sw, min_dw, op>(dst_type, callee, src, dbg)) return result;
742
743 auto [dw, sw] = callee->args<2>(isa_lit<nat_t>);
744 if (sw == dw && dst_type == src->type()) return src;
745
746 if constexpr (op == Conv::s2s) {
747 if (sw && dw && *sw < *dw) return world.op(Conv::u2u, dst_type, src, dbg);
748 }
749
750 return world.raw_app(callee, src, dbg);
751}
