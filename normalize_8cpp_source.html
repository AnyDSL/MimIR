226/// Reassociates @p a und @p b according to following rules.
227/// We use the following naming convention while literals are prefixed with an 'l':
228/// ```
229/// a op b
230/// (x op y) op (z op w)
231///
232/// (1) la op (lz op w) -> (la op lz) op w
233/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
234/// (3) a op (lz op w) -> lz op (a op w)
235/// (4) (lx op y) op b -> lx op (y op b)
236/// ```
237template<tag_t tag>
238static const Def* reassociate(Tag2Enum<tag> op,
239 World& world,
240 [[maybe_unused]] const App* ab,
241 const Def* a,
242 const Def* b,
243 const Def* dbg) {
244 if (!is_associative(op)) return nullptr;
245
246 auto la = a->isa<Lit>();
247 auto xy = isa<tag>(op, a);
248 auto zw = isa<tag>(op, b);
249 auto lx = xy ? xy->arg(0)->template isa<Lit>() : nullptr;
250 auto lz = zw ? zw->arg(0)->template isa<Lit>() : nullptr;
251 auto y = xy ? xy->arg(1) : nullptr;
252 auto w = zw ? zw->arg(1) : nullptr;
253
254 std::function<const Def*(const Def*, const Def*)> make_op;
255
256 if constexpr (tag == Tag::ROp) {
257 // build rmode for all new ops by using the least upper bound of all involved apps
258 nat_t rmode = RMode::bot;
259 auto check_mode = [&](const App* app) {
260 auto app_m = isa_lit(app->arg(0));
261 if (!app_m || !has(*app_m, RMode::reassoc)) return false;
262 rmode &= *app_m; // least upper bound
263 return true;
264 };
265
266 if (!check_mode(ab)) return nullptr;
267 if (lx && !check_mode(xy->decurry())) return nullptr;
268 if (lz && !check_mode(zw->decurry())) return nullptr;
269
270 make_op = [&](const Def* a, const Def* b) { return world.op(op, rmode, a, b, dbg); };
271 } else if constexpr (tag == Tag::Wrap) {
272 // if we reassociate Wraps, we have to forget about nsw/nuw
273 make_op = [&](const Def* a, const Def* b) { return world.op(op, WMode::none, a, b, dbg); };
274 } else {
275 make_op = [&](const Def* a, const Def* b) { return world.op(op, a, b, dbg); };
276 }
277
278 if (la && lz) return make_op(make_op(la, lz), w); // (1)
279 if (lx && lz) return make_op(make_op(lx, lz), make_op(y, w)); // (2)
280 if (lz) return make_op(lz, make_op(a, w)); // (3)
281 if (lx) return make_op(lx, make_op(y, b)); // (4)
282
283 return nullptr;
284}

287template<class Op, Op op>
288static const Def* fold(World& world, const Def* type, const App* callee, const Def*& a, const Def*& b, const Def* dbg) {
289 static constexpr int min_w = std::is_same_v<Op, ROp> || std::is_same_v<Op, RCmp> ? 16 : 1;
290 auto la = a->isa<Lit>(), lb = b->isa<Lit>();
291
292 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type, dbg);
293
294 if (la && lb) {
295 nat_t width;
296 [[maybe_unused]] bool nsw = false, nuw = false;
297 if constexpr (std::is_same_v<Op, Wrap>) {
298 auto [mode, w] = callee->args<2>(as_lit<nat_t>);
299 nsw = mode & WMode::nsw;
300 nuw = mode & WMode::nuw;
301 width = w;
302 } else {
303 width = as_lit(a->type()->as<App>()->arg());
304 }
305
306 if (is_int<Op>()) width = *mod2width(width);
307
308 Res res;
309 switch (width) {
310#define CODE(i) \
311 case i: \
312 if constexpr (i >= min_w) { \
313 if constexpr (std::is_same_v<Op, Wrap>) \
314 res = Fold<Op, op, i>::run(la->get(), lb->get(), nsw, nuw); \
315 else \
316 res = Fold<Op, op, i>::run(la->get(), lb->get()); \
317 } \
318 break;
319 THORIN_1_8_16_32_64(CODE)
320#undef CODE
321 default: unreachable();
322 }
323
324 return res ? world.lit(type, *res, dbg) : world.bot(type, dbg);
325 }
326
327 commute(op, a, b);
328 return nullptr;
329}

335template<tag_t tag>
336static const Def* merge_cmps(std::array<std::array<uint64_t, 2>, 2> tab, const Def* a, const Def* b, const Def* dbg) {
337 static_assert(sizeof(flags_t) == 4, "if this ever changes, please adjust the logic below");
338 static constexpr size_t num_bits = std::bit_width(Num<Tag2Enum<tag>> - 1_u64);
339
340 auto a_cmp = isa<tag>(a);
341 auto b_cmp = isa<tag>(b);
342
343 if (a_cmp && b_cmp && a_cmp->args() == b_cmp->args()) {
344 // push flags of a_cmp and b_cmp through truth table
345 flags_t res = 0;
346 flags_t a_flags = a_cmp.axiom()->flags();
347 flags_t b_flags = b_cmp.axiom()->flags();
348 for (size_t i = 0; i != num_bits; ++i, res >>= 1, a_flags >>= 1, b_flags >>= 1)
349 res |= tab[a_flags & 1][b_flags & 1] << 31_u32;
350 res >>= (31_u32 - u32(num_bits));
351
352 auto& world = a->world();
353 if constexpr (tag == Tag::RCmp)
354 return world.op(RCmp(res), /*rmode*/ a_cmp->decurry()->arg(0), a_cmp->arg(0), a_cmp->arg(1), dbg);
355 else
356 return world.op(ICmp(res), a_cmp->arg(0), a_cmp->arg(1), dbg);
357 }
358
359 return nullptr;
360}

795 case Trait::size: return world.lit_nat(size);
