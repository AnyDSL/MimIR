42/// @name Formatted Output
43/// @anchor fmt
44///@{
45/// Provides a `printf`-like interface to format @p s with @p args and puts it into @p os.
46/// Use `{}` as a placeholder within your format string @p s.
47/// * By default, `os << t` will be used to stream the appropriate argument:
48/// ```
49/// print(os, "int: {}, float: {}", 23, 42.f);
50/// ```
51/// * You can also place a function as argument to inject arbitrary code:
52/// ```
53/// print(os, "int: {}, fun: {}, float: {}", 23, [&]() { os << "hey :)"}, 42.f);
54/// ```
55/// * There is also a variant to pass @p os to the function, if you don't have easy access to it:
56/// ```
57/// print(os, "int: {}, fun: {}, float: {}", 23, [&](auto& os) { os << "hey :)"}, 42.f);
58/// ```
59/// * You can put a `std::ranges::range` as argument.
60/// This will output a list - using the given specifier as separator.
61/// Each element `elem` of the range will be output via `os << elem`:
62/// ```
63/// std::vector<int> v = {0, 1, 2, 3};
64/// print(os, "v: {, }", v);
65/// ```
66/// * If you have a `std::ranges::range` that needs to be formatted in a more complicated way, bundle it with an Elem:
67/// ```
68/// std::vector<int> v = {3, 2, 1, 0};
69/// size_t i = 0;
70/// print(os, "v: {, }", Elem(v, [&](auto elem) { print(os, "{}: {}", i++, elem); }));
71/// ```
72/// * You again have the option to pass @p os to the bundled function:
73/// ```
74/// std::vector<int> v = {3, 2, 1, 0};
75/// size_t i = 0;
76/// print(os, "v: {, }", Elem(v, [&](auto& os, auto elem) { print(os, "{}: {}", i++, elem); }));
77/// * Use `{{` or `}}` to literally output `{` or `{`:
78/// print(os, "{{{}}}", 23); // "{23}"
79/// ```
80/// @see Tab
81
82/// Use with print to output complicated `std::ranges::range`s.
83template<class R, class F>
84struct Elem {
85 Elem(const R& range, const F& f)
86 : range(range)
87 , f(f) {}
88
89 const R range;
90 const F f;
91};

95template<class T, class... Args>
96std::ostream& print(std::ostream& os, const char* s, T&& t, Args&&... args) {
97 while (*s != '\0') {
98 auto next = s + 1;
99
100 switch (*s) {
101 case '{': {
102 if (detail::match2nd(os, next, s, '{')) continue;
103 s++; // skip opening brace '{'
104
105 std::string spec;
106 while (*s != '\0' && *s != '}') spec.push_back(*s++);
107 assert(*s == '}' && "unmatched closing brace '}' in format string");
108
109 if constexpr (std::is_invocable_v<decltype(t)>) {
110 std::invoke(t);
111 } else if constexpr (std::is_invocable_v<decltype(t), std::ostream&>) {
112 std::invoke(t, os);
113 } else if constexpr (detail::Printable<decltype(t)>) {
114 os << t;
115 } else if constexpr (detail::Elemable<decltype(t)>) {
116 detail::range(os, t.range, t.f, spec.c_str());
117 } else if constexpr (std::ranges::range<decltype(t)>) {
118 detail::range(
119 os, t, [&](const auto& x) { os << x; }, spec.c_str());
120 } else {
121 []<bool flag = false>() { static_assert(flag, "cannot print T t"); }
122 ();
123 }
124
125 ++s; // skip closing brace '}'
126 // call even when *s == '\0' to detect extra arguments
127 return print(os, s, std::forward<Args&&>(args)...);
128 }
129 case '}':
130 if (detail::match2nd(os, next, s, '}')) continue;
131 assert(false && "unmatched/unescaped closing brace '}' in format string");
132 unreachable();
133 default: os << *s++;
134 }
135 }
136
137 assert(false && "invalid format string for 's'");
138 unreachable();
139}
