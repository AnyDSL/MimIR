18// Lessons learned:
19// * **Always** follow all ops - even if you actually want to ignore one.
20// Otherwise, you might end up with an incorrect schedule.
21// This was the case for an Extract of type Mem.
22// While we want to ignore the value obtained from that, since there is no Mem value in LLVM,
23// we still want to **first** recursively emit code for its operands and **then** ignore the Extract itself.
24// * i1 has a different meaning in LLVM then in Thorin:
25// * Thorin: {0, 1} = i1
26// * LLVM: {0, -1} = i1
27// This is a problem when, e.g., using an index of type i1 as LLVM thinks like this:
28// getelementptr ..., i1 1 == getelementptr .., i1 -1

33static bool is_const(const Def* def) {
34 if (def->isa<Bot>()) return true;
35 if (def->isa<Lit>()) return true;
36 if (auto pack = def->isa_structural<Pack>()) return is_const(pack->shape()) && is_const(pack->body());
37
38 if (auto tuple = def->isa<Tuple>()) {
39 auto ops = tuple->ops();
40 return std::ranges::all_of(ops, [](auto def) { return is_const(def); });
41 }
42
43 return false;
44}

51 std::deque<std::ostringstream>& head() { return parts[0]; }
52 std::deque<std::ostringstream>& body() { return parts[1]; }

55 template<class... Args>

57 print(print(body().emplace_back(), "{} = ", name), s, std::forward<Args&&>(args)...);

67 using std::swap;
68 swap(a.phis, b.phis);

121std::string Emitter::convert(const Def* type) {
122 if (auto i = types_.find(type); i != types_.end()) return i->second;
123
124 assert(!match<mem::M>(type));
125 std::ostringstream s;
126 std::string name;
127
128 if (type->isa<Nat>()) {
129 return types_[type] = "i64";
130 } else if (auto size = Idx::size(type)) {
131 if (size->isa<Top>()) return types_[type] = "i64";
132 if (auto width = size2bitwidth(as_lit(size))) {
133 switch (*width) {
134 // clang-format off
135 case 1: return types_[type] = "i1";
136 case 2:
137 case 4:
138 case 8: return types_[type] = "i8";
139 case 16: return types_[type] = "i16";
140 case 32: return types_[type] = "i32";
141 case 64: return types_[type] = "i64";
142 // clang-format on
143 default: unreachable();
144 }
145 } else {
146 return types_[type] = "i64";
147 }
148 } else if (auto real = match<core::Real>(type)) {
149 switch (as_lit<nat_t>(real->arg())) {
150 case 16: return types_[type] = "half";
151 case 32: return types_[type] = "float";
152 case 64: return types_[type] = "double";
153 default: unreachable();
154 }
155 } else if (auto ptr = match<mem::Ptr>(type)) {
156 auto [pointee, addr_space] = ptr->args<2>();
157 // TODO addr_space
158 print(s, "{}*", convert(pointee));
159 } else if (auto arr = type->isa<Arr>()) {
160 auto elem_type = convert(arr->body());
161 u64 size = 0;
162 if (auto arity = isa_lit(arr->shape())) size = *arity;
163 print(s, "[{} x {}]", size, elem_type);
164 } else if (auto pi = type->isa<Pi>()) {
165 assert(pi->is_returning() && "should never have to convert type of BB");
166 print(s, "{} (", convert_ret_pi(pi->ret_pi()));
167
168 std::string_view sep = "";
169 auto doms = pi->doms();
170 for (auto dom : doms.skip_back()) {
171 if (match<mem::M>(dom)) continue;
172 s << sep << convert(dom);
173 sep = ", ";
174 }
175
176 s << ")*";
177 } else if (auto sigma = type->isa<Sigma>()) {
178 if (sigma->isa_nom()) {
179 name = id(sigma);
180 types_[sigma] = name;
181 print(s, "{} = type", name);
182 }
183 print(s, "{{");
184 std::string_view sep = "";
185 for (auto t : sigma->ops()) {
186 if (match<mem::M>(t)) continue;
187 s << sep << convert(t);
188 sep = ", ";
189 }
190 print(s, "}}");
191 } else {
192 unreachable();
193 }
194
195 if (name.empty()) return types_[type] = s.str();
196
197 assert(!s.str().empty());
198 type_decls_ << s.str() << '\n';
199 return types_[type] = name;
200}

202std::string Emitter::convert_ret_pi(const Pi* pi) {
203 switch (pi->num_doms()) {
204 case 0: return "void";
205 case 1:
206 if (match<mem::M>(pi->dom())) return "void";
207 return convert(pi->dom());
208 case 2:
209 if (match<mem::M>(pi->dom(0))) return convert(pi->dom(1));
210 if (match<mem::M>(pi->dom(1))) return convert(pi->dom(0));
211 [[fallthrough]];
212 default: return convert(pi->dom());
213 }

249std::string Emitter::prepare(const Scope& scope) {
250 auto lam = scope.entry()->as_nom<Lam>();
251
252 print(func_impls_, "define {} {}(", convert_ret_pi(lam->type()->ret_pi()), id(lam));
253
254 auto sep = "";
255 auto vars = lam->vars();
256 for (auto var : vars.skip_back()) {
257 if (match<mem::M>(var->type())) continue;
258 auto name = id(var);
259 locals_[var] = name;
260 print(func_impls_, "{}{} {}", sep, convert(var->type()), name);
261 sep = ", ";
262 }
263
264 print(func_impls_, ") {{\n");
265 return lam->unique_name();

268void Emitter::finalize(const Scope& scope) {
269 for (auto& [lam, bb] : lam2bb_) {
270 for (const auto& [phi, args] : bb.phis) {
271 print(bb.head().emplace_back(), "{} = phi {} ", id(phi), convert(phi->type()));
272 auto sep = "";
273 for (const auto& [arg, pred] : args) {
274 print(bb.head().back(), "{}[ {}, {} ]", sep, arg, pred);
275 sep = ", ";
276 }
277 }
278 }
279
280 for (auto nom : schedule(scope)) {
281 if (auto lam = nom->isa_nom<Lam>()) {
282 if (lam == scope.exit()) continue;
283 assert(lam2bb_.contains(lam));
284 auto& bb = lam2bb_[lam];
285 print(func_impls_, "{}:\n", lam->unique_name());
286
287 ++tab;
288 for (const auto& part : bb.parts) {
289 for (const auto& line : part) tab.print(func_impls_, "{}\n", line.str());
290 }
291 --tab;
292 func_impls_ << std::endl;
293 }
294 }
295
296 print(func_impls_, "}}\n\n");
297}

299void Emitter::emit_epilogue(Lam* lam) {
300 auto app = lam->body()->as<App>();
301 auto& bb = lam2bb_[lam];
302
303 if (app->callee() == entry_->ret_var()) { // return
304 std::vector<std::string> values;
305 std::vector<const Def*> types;
306
307 for (auto arg : app->args()) {
308 if (auto val = emit_unsafe(arg); !val.empty()) {
309 values.emplace_back(val);
310 types.emplace_back(arg->type());
311 }
312 }
313
314 switch (values.size()) {
315 case 0: return bb.tail("ret void");
316 case 1: return bb.tail("ret {} {}", convert(types[0]), values[0]);
317 default: {
318 std::string prev = "undef";
319 auto type = convert(world().sigma(types));
320 for (size_t i = 0, n = values.size(); i != n; ++i) {
321 auto elem = values[i];
322 auto elem_t = convert(types[i]);
323 auto namei = "%ret_val." + std::to_string(i);
324 bb.tail("{} = insertvalue {} {}, {} {}, {}", namei, type, prev, elem_t, elem, i);
325 prev = namei;
326 }
327
328 bb.tail("ret {} {}", type, prev);
329 }
330 }
331 } else if (auto ex = app->callee()->isa<Extract>(); ex && app->callee_type()->is_basicblock()) {
332 emit_unsafe(app->arg());
333 auto c = emit(ex->index());
334 if (ex->tuple()->num_projs() == 2) {
335 auto [f, t] = ex->tuple()->projs<2>([this](auto def) { return emit(def); });
336 return bb.tail("br i1 {}, label {}, label {}", c, t, f);
337 } else {
338 auto c_ty = convert(ex->index()->type());
339 bb.tail("switch {} {}, label {} [ ", c_ty, c, emit(ex->tuple()->proj(0)));
340 for (auto i = 1u; i < ex->tuple()->num_projs(); i++)
341 print(bb.tail().back(), "{} {}, label {} ", c_ty, std::to_string(i), emit(ex->tuple()->proj(i)));
342 print(bb.tail().back(), "]");
343 }
344 } else if (app->callee()->isa<Bot>()) {
345 return bb.tail("ret ; bottom: unreachable");
346 } else if (auto callee = app->callee()->isa_nom<Lam>(); callee && callee->is_basicblock()) { // ordinary jump
347 for (size_t i = 0, e = callee->num_vars(); i != e; ++i) {
348 if (auto arg = emit_unsafe(app->arg(i)); !arg.empty()) {
349 auto phi = callee->var(i);
350 assert(!match<mem::M>(phi->type()));
351 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
352 locals_[phi] = id(phi);
353 }
354 }
355 return bb.tail("br label {}", id(callee));
356 } else if (auto longjmp = match<clos::longjmp>(app)) {
357 auto [mem, jbuf, tag] = app->args<3>();
358 emit_unsafe(mem);
359 auto emitted_jb = emit(jbuf);
360 auto emitted_tag = emit(tag);
361 bb.tail("call void @longjmp(i8* {}, i32 {})", emitted_jb, emitted_tag);
362 return bb.tail("unreachable");
363 } else if (app->callee_type()->is_returning()) { // function call
364 auto emmited_callee = emit(app->callee());
365
366 std::vector<std::string> args;
367 auto app_args = app->args();
368 for (auto arg : app_args.skip_back()) {
369 if (auto emitted_arg = emit_unsafe(arg); !emitted_arg.empty())
370 args.emplace_back(convert(arg->type()) + " " + emitted_arg);
371 }
372
373 if (app->args().back()->isa<Bot>()) {
374 // TODO: Perhaps it'd be better to simply η-wrap this prior to the BE...
375 assert(convert_ret_pi(app->callee_type()->ret_pi()) == "void");
376 bb.tail("call void {}({, })", emmited_callee, args);
377 return bb.tail("unreachable");
378 }
379
380 auto ret_lam = app->args().back()->as_nom<Lam>();
381 size_t num_vars = ret_lam->num_vars();
382 size_t n = 0;
383 Array<const Def*> values(num_vars);
384 Array<const Def*> types(num_vars);
385 for (auto var : ret_lam->vars()) {
386 if (match<mem::M>(var->type())) continue;
387 values[n] = var;
388 types[n] = var->type();
389 ++n;
390 }
391
392 if (n == 0) {
393 bb.tail("call void {}({, })", emmited_callee, args);
394 } else {
395 auto name = "%" + app->unique_name() + ".ret";
396 auto ret_ty = convert_ret_pi(ret_lam->type());
397 bb.tail("{} = call {} {}({, })", name, ret_ty, emmited_callee, args);
398
399 for (size_t i = 0, e = ret_lam->num_vars(); i != e; ++i) {
400 auto phi = ret_lam->var(i);
401 if (match<mem::M>(phi->type())) continue;
402
403 auto namei = name;
404 if (e > 2) {
405 namei += '.' + std::to_string(i - 1);
406 bb.tail("{} = extractvalue {} {}, {}", namei, ret_ty, name, i - 1);
407 }
408 assert(!match<mem::M>(phi->type()));
409 lam2bb_[ret_lam].phis[phi].emplace_back(namei, id(lam, true));
410 locals_[phi] = id(phi);
411 }
412 }
413
414 return bb.tail("br label {}", id(ret_lam));
415 }

418std::string Emitter::emit_bb(BB& bb, const Def* def) {
419 if (def->isa<Var>()) return {};
420 if (auto lam = def->isa<Lam>()) return id(lam);
421
422 auto name = id(def);
423 std::string op;
424
425 auto emit_tuple = [&](const Def* tuple) {
426 if (is_const(tuple)) {
427 bool is_array = tuple->type()->isa<Arr>();
428
429 std::string s;
430 s += is_array ? "[" : "{";
431 auto sep = "";
432 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
433 auto e = tuple->proj(n, i);
434 if (auto elem = emit_unsafe(e); !elem.empty()) {
435 auto elem_t = convert(e->type());
436 s += sep + elem_t + " " + elem;
437 sep = ", ";
438 }
439 }
440
441 return s += is_array ? "]" : "}";
442 }
443
444 std::string prev = "undef";
445 auto t = convert(tuple->type());
446 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
447 auto e = tuple->proj(n, i);
448 if (auto elem = emit_unsafe(e); !elem.empty()) {
449 auto elem_t = convert(e->type());
450 auto namei = name + "." + std::to_string(i);
451 prev = bb.assign(namei, "insertvalue {} {}, {} {}, {}", t, prev, elem_t, elem, i);
452 }
453 }
454 return prev;
455 };
456
457 if (auto lit = def->isa<Lit>()) {
458 if (lit->type()->isa<Nat>()) {
459 return std::to_string(lit->get<nat_t>());
460 } else if (auto size = Idx::size(lit->type())) {
461 if (size->isa<Top>()) return std::to_string(lit->get<nat_t>());
462 if (auto mod = size2bitwidth(as_lit(size))) {
463 switch (*mod) {
464 // clang-format off
465 case 0: return {};
466 case 1: return std::to_string(lit->get< u1>());
467 case 2:
468 case 4:
469 case 8: return std::to_string(lit->get< u8>());
470 case 16: return std::to_string(lit->get<u16>());
471 case 32: return std::to_string(lit->get<u32>());
472 case 64: return std::to_string(lit->get<u64>());
473 // clang-format on
474 default: return std::to_string(lit->get<u64>());
475 }
476 } else {
477 return std::to_string(lit->get<u64>());
478 }
479 } else if (auto real = match<core::Real>(lit->type())) {
480 std::stringstream s;
481 u64 hex;
482
483 switch (as_lit<nat_t>(real->arg())) {
484 case 16:
485 s << "0xH" << std::setfill('0') << std::setw(4) << std::right << std::hex << lit->get<u16>();
486 return s.str();
487 case 32: {
488 hex = std::bit_cast<u64>(r64(lit->get<r32>()));
489 break;
490 }
491 case 64: hex = lit->get<u64>(); break;
492 default: unreachable();
493 }
494
495 s << "0x" << std::setfill('0') << std::setw(16) << std::right << std::hex << hex;
496 return s.str();
497 }
498 unreachable();
499 } else if (def->isa<Bot>()) {
500 return "undef";
501 } else if (auto bit1 = match<core::bit1>(def)) {
502 assert(bit1.id() == core::bit1::neg);
503 auto x = emit(bit1->arg());
504 auto t = convert(bit1->type());
505 return bb.assign(name, "xor {} -1, {}", t, x);
506 } else if (auto bit2 = match<core::bit2>(def)) {
507 auto [a, b] = bit2->args<2>([this](auto def) { return emit(def); });
508 auto t = convert(bit2->type());
509
510 auto neg = [&](std::string_view x) { return bb.assign(name + ".neg", "xor {} -1, {}", t, x); };
511
512 switch (bit2.id()) {
513 // clang-format off
514 case core::bit2::and_: return bb.assign(name, "and {} {}, {}", t, a, b);
515 case core::bit2:: or_: return bb.assign(name, "or {} {}, {}", t, a, b);
516 case core::bit2::xor_: return bb.assign(name, "xor {} {}, {}", t, a, b);
517 case core::bit2::nand: return neg(bb.assign(name, "and {} {}, {}", t, a, b));
518 case core::bit2:: nor: return neg(bb.assign(name, "or {} {}, {}", t, a, b));
519 case core::bit2::nxor: return neg(bb.assign(name, "xor {} {}, {}", t, a, b));
520 case core::bit2:: iff: return bb.assign(name, "and {} {}, {}", neg(a), b);
521 case core::bit2::niff: return bb.assign(name, "or {} {}, {}", neg(a), b);
522 // clang-format on
523 default: unreachable();
524 }
525 } else if (auto shr = match<core::shr>(def)) {
526 auto [a, b] = shr->args<2>([this](auto def) { return emit(def); });
527 auto t = convert(shr->type());
528
529 switch (shr.id()) {
530 case core::shr::a: op = "ashr"; break;
531 case core::shr::l: op = "lshr"; break;
532 }
533
534 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
535 } else if (auto wrap = match<core::wrap>(def)) {
536 auto [a, b] = wrap->args<2>([this](auto def) { return emit(def); });
537 auto t = convert(wrap->type());
538 auto [mode, width] = wrap->decurry()->args<2>(as_lit<nat_t>);
539
540 switch (wrap.id()) {
541 case core::wrap::add: op = "add"; break;
542 case core::wrap::sub: op = "sub"; break;
543 case core::wrap::mul: op = "mul"; break;
544 case core::wrap::shl: op = "shl"; break;
545 }
546
547 if (mode & core::WMode::nuw) op += " nuw";
548 if (mode & core::WMode::nsw) op += " nsw";
549
550 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
551 } else if (auto div = match<core::div>(def)) {
552 auto [m, x, y] = div->args<3>();
553 auto t = convert(x->type());
554 emit_unsafe(m);
555 auto a = emit(x);
556 auto b = emit(y);
557
558 switch (div.id()) {
559 case core::div::sdiv: op = "sdiv"; break;
560 case core::div::udiv: op = "udiv"; break;
561 case core::div::srem: op = "srem"; break;
562 case core::div::urem: op = "urem"; break;
563 }
564
565 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
566 } else if (auto rop = match<core::rop>(def)) {
567 auto [a, b] = rop->args<2>([this](auto def) { return emit(def); });
568 auto t = convert(rop->type());
569 auto [mode, width] = rop->decurry()->args<2>(as_lit<nat_t>);
570
571 switch (rop.id()) {
572 case core::rop::add: op = "fadd"; break;
573 case core::rop::sub: op = "fsub"; break;
574 case core::rop::mul: op = "fmul"; break;
575 case core::rop::div: op = "fdiv"; break;
576 case core::rop::rem: op = "frem"; break;
577 }
578
579 if (mode == core::RMode::fast)
580 op += " fast";
581 else {
582 // clang-format off
583 if (mode & core::RMode::nnan ) op += " nnan";
584 if (mode & core::RMode::ninf ) op += " ninf";
585 if (mode & core::RMode::nsz ) op += " nsz";
586 if (mode & core::RMode::arcp ) op += " arcp";
587 if (mode & core::RMode::contract) op += " contract";
588 if (mode & core::RMode::afn ) op += " afn";
589 if (mode & core::RMode::reassoc ) op += " reassoc";
590 // clang-format on
591 }
592
593 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
594 } else if (auto icmp = match<core::icmp>(def)) {
595 auto [a, b] = icmp->args<2>([this](auto def) { return emit(def); });
596 auto t = convert(icmp->arg(0)->type());
597 op = "icmp ";
598
599 switch (icmp.id()) {
600 // clang-format off
601 case core::icmp::e: op += "eq" ; break;
602 case core::icmp::ne: op += "ne" ; break;
603 case core::icmp::sg: op += "sgt"; break;
604 case core::icmp::sge: op += "sge"; break;
605 case core::icmp::sl: op += "slt"; break;
606 case core::icmp::sle: op += "sle"; break;
607 case core::icmp::ug: op += "ugt"; break;
608 case core::icmp::uge: op += "uge"; break;
609 case core::icmp::ul: op += "ult"; break;
610 case core::icmp::ule: op += "ule"; break;
611 // clang-format on
612 default: unreachable();
613 }
614
615 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
616 } else if (auto rcmp = match<core::rcmp>(def)) {
617 auto [a, b] = rcmp->args<2>([this](auto def) { return emit(def); });
618 auto t = convert(rcmp->arg(0)->type());
619 op = "fcmp ";
620
621 switch (rcmp.id()) {
622 // clang-format off
623 case core::rcmp:: e: op += "oeq"; break;
624 case core::rcmp:: l: op += "olt"; break;
625 case core::rcmp:: le: op += "ole"; break;
626 case core::rcmp:: g: op += "ogt"; break;
627 case core::rcmp:: ge: op += "oge"; break;
628 case core::rcmp:: ne: op += "one"; break;
629 case core::rcmp:: o: op += "ord"; break;
630 case core::rcmp:: u: op += "uno"; break;
631 case core::rcmp:: ue: op += "ueq"; break;
632 case core::rcmp:: ul: op += "ult"; break;
633 case core::rcmp::ule: op += "ule"; break;
634 case core::rcmp:: ug: op += "ugt"; break;
635 case core::rcmp::uge: op += "uge"; break;
636 case core::rcmp::une: op += "une"; break;
637 // clang-format on
638 default: unreachable();
639 }
640
641 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
642 } else if (auto conv = match<core::conv>(def)) {
643 auto src = emit(conv->arg());
644 auto src_t = convert(conv->arg()->type());
645 auto dst_t = convert(conv->type());
646
647 auto size2width = [&](const Def* type) {
648 if (auto size = Idx::size(type)) {
649 if (size->isa<Top>()) return 64_u64;
650 if (auto width = size2bitwidth(as_lit(size))) return *width;
651 return 64_u64;
652 }
653 return as_lit(force<core::Real>(type)->arg());
654 };
655
656 nat_t s_src = size2width(conv->arg()->type());
657 nat_t s_dst = size2width(conv->type());
658
659 // this might happen when casting from int top to i64
660 if (s_src == s_dst && (conv.id() == core::conv::s2s || conv.id() == core::conv::u2u)) return src;
661
662 switch (conv.id()) {
663 case core::conv::s2s: op = s_src < s_dst ? "sext" : "trunc"; break;
664 case core::conv::u2u: op = s_src < s_dst ? "zext" : "trunc"; break;
665 case core::conv::r2r: op = s_src < s_dst ? "fpext" : "fptrunc"; break;
666 case core::conv::s2r: op = "sitofp"; break;
667 case core::conv::u2r: op = "uitofp"; break;
668 case core::conv::r2s: op = "fptosi"; break;
669 case core::conv::r2u: op = "fptoui"; break;
670 }
671
672 return bb.assign(name, "{} {} {} to {}", op, src_t, src, dst_t);
673 } else if (auto bitcast = match<core::bitcast>(def)) {
674 auto dst_type_ptr = match<mem::Ptr>(bitcast->type());
675 auto src_type_ptr = match<mem::Ptr>(bitcast->arg()->type());
676 auto src = emit(bitcast->arg());
677 auto src_t = convert(bitcast->arg()->type());
678 auto dst_t = convert(bitcast->type());
679
680 if (auto lit = isa_lit(bitcast->arg()); lit && *lit == 0) return "zeroinitializer";
681 // clang-format off
682 if (src_type_ptr && dst_type_ptr) return bb.assign(name, "bitcast {} {} to {}", src_t, src, dst_t);
683 if (src_type_ptr) return bb.assign(name, "ptrtoint {} {} to {}", src_t, src, dst_t);
684 if (dst_type_ptr) return bb.assign(name, "inttoptr {} {} to {}", src_t, src, dst_t);
685 // clang-format on
686
687 auto size2width = [&](const Def* type) {
688 if (type->isa<Nat>()) return 64_u64;
689 if (auto size = Idx::size(type)) {
690 if (size->isa<Top>() || !size->isa<Lit>()) return 64_u64;
691 if (auto width = size2bitwidth(as_lit(size))) return std::bit_ceil(*width);
692 return 64_u64;
693 }
694 return 0_u64;
695 };
696
697 auto src_size = size2width(bitcast->arg()->type());
698 auto dst_size = size2width(bitcast->type());
699
700 op = "bitcast";
701 if (src_size && dst_size) {
702 if (src_size == dst_size) return src;
703 op = (src_size < dst_size) ? "zext" : "trunc";
704 }
705 return bb.assign(name, "{} {} {} to {}", op, src_t, src, dst_t);
706 } else if (auto lea = match<mem::lea>(def)) {
707 auto [ptr, i] = lea->args<2>();
708 auto ll_ptr = emit(ptr);
709 auto pointee = force<mem::Ptr>(ptr->type())->arg(0);
710 auto t = convert(pointee);
711 auto p = convert(ptr->type());
712 if (pointee->isa<Sigma>())
713 return bb.assign(name, "getelementptr inbounds {}, {} {}, i64 0, i32 {}", t, p, ll_ptr, as_lit<u64>(i));
714
715 assert(pointee->isa<Arr>());
716 auto ll_i = emit(i);
717 auto i_t = convert(i->type());
718
719 if (auto size = Idx::size(i->type())) {
720 if (auto s = isa_lit(size); s && *s == 2) { // mod(2) = width(1)
721 ll_i = bb.assign(name + ".8", "zext i1 {} to i8", ll_i);
722 i_t = "i8";
723 }
724 }
725
726 return bb.assign(name, "getelementptr inbounds {}, {} {}, i64 0, {} {}", t, p, ll_ptr, i_t, ll_i);
727 } else if (match<core::trait>(def)) {
728 unreachable();
729 } else if (auto malloc = match<mem::malloc>(def)) {
730 emit_unsafe(malloc->arg(0));
731 auto size = emit(malloc->arg(1));
732 auto ptr_t = convert(force<mem::Ptr>(def->proj(1)->type()));
733 bb.assign(name + ".i8", "call i8* @malloc(i64 {})", size);
734 return bb.assign(name, "bitcast i8* {} to {}", name + ".i8", ptr_t);
735 } else if (auto mslot = match<mem::mslot>(def)) {
736 emit_unsafe(mslot->arg(0));
737 // TODO array with size
738 // auto size = emit(mslot->arg(1));
739 auto [pointee, addr_space] = mslot->decurry()->args<2>();
740 print(bb.body().emplace_back(), "{} = alloca {}", name, convert(pointee));
741 return name;
742 } else if (auto free = match<mem::free>(def)) {
743 emit_unsafe(free->arg(0));
744 auto ptr = emit(free->arg(1));
745 auto ptr_t = convert(force<mem::Ptr>(free->arg(1)->type()));
746
747 bb.assign(name + ".i8", "bitcast {} {} to i8*", ptr_t, ptr);
748 bb.tail("call void @free(i8* {})", name + ".i8");
749 return {};
750 } else if (auto load = match<mem::load>(def)) {
751 emit_unsafe(load->arg(0));
752 auto ptr = emit(load->arg(1));
753 auto ptr_t = convert(load->arg(1)->type());
754 auto pointee_t = convert(force<mem::Ptr>(load->arg(1)->type())->arg(0));
755 return bb.assign(name, "load {}, {} {}", pointee_t, ptr_t, ptr);
756 } else if (auto store = match<mem::store>(def)) {
757 emit_unsafe(store->arg(0));
758 auto ptr = emit(store->arg(1));
759 auto val = emit(store->arg(2));
760 auto ptr_t = convert(store->arg(1)->type());
761 auto val_t = convert(store->arg(2)->type());
762 print(bb.body().emplace_back(), "store {} {}, {} {}", val_t, val, ptr_t, ptr);
763 return {};
764 } else if (auto q = match<clos::alloc_jmpbuf>(def)) {
765 emit_unsafe(q->arg());
766 auto size = name + ".size";
767 bb.assign(size, "call i64 @jmpbuf_size()");
768 return bb.assign(name, "alloca i8, i64 {}", size);
769 } else if (auto setjmp = match<clos::setjmp>(def)) {
770 auto [mem, jmpbuf] = setjmp->arg()->projs<2>();
771 emit_unsafe(mem);
772 auto emitted_jb = emit(jmpbuf);
773 return bb.assign(name, "call i32 @_setjmp(i8* {})", emitted_jb);
774 } else if (auto tuple = def->isa<Tuple>()) {
775 return emit_tuple(tuple);
776 } else if (auto pack = def->isa<Pack>()) {
777 if (auto lit = isa_lit(pack->body()); lit && *lit == 0) return "zeroinitializer";
778 return emit_tuple(pack);
779 } else if (auto extract = def->isa<Extract>()) {
780 auto tuple = extract->tuple();
781 auto index = extract->index();
782
783#if 0
784 // TODO this was von closure-conv branch which I need to double-check
785 if (tuple->isa<Var>()) {
786 // computing the index may crash, so we bail out
787 assert(match<core::Mem>(extract->type()) && "only mem-var should not be mapped");
788 return {};
789 }
790#endif
791
792 auto ll_tup = emit_unsafe(tuple);
793
794 // this exact location is important: after emitting the tuple -> ordering of mem ops
795 // before emitting the index, as it might be a weird value for mem vars.
796 if (match<mem::M>(extract->type())) return {};
797
798 auto ll_idx = emit_unsafe(index);
799
800 if (tuple->num_projs() == 2) {
801 if (match<mem::M>(tuple->proj(2, 0_s)->type())) return ll_tup;
802 if (match<mem::M>(tuple->proj(2, 1_s)->type())) return ll_tup;
803 }
804
805 auto tup_t = convert(tuple->type());
806 if (isa_lit(index)) {
807 assert(!ll_tup.empty());
808 return bb.assign(name, "extractvalue {} {}, {}", tup_t, ll_tup, ll_idx);
809 } else {
810 auto elem_t = convert(extract->type());
811 print(lam2bb_[entry_].body().emplace_front(),
812 "{}.alloca = alloca {} ; copy to alloca to emulate extract with store + gep + load", name, tup_t);
813 print(bb.body().emplace_back(), "store {} {}, {}* {}.alloca", tup_t, ll_tup, tup_t, name);
814 print(bb.body().emplace_back(), "{}.gep = getelementptr inbounds {}, {}* {}.alloca, i64 0, i64 {}", name,
815 tup_t, tup_t, name, ll_idx);
816 return bb.assign(name, "load {}, {}* {}.gep", elem_t, elem_t, name);
817 }
818 } else if (auto insert = def->isa<Insert>()) {
819 auto tuple = emit(insert->tuple());
820 auto index = emit(insert->index());
821 auto value = emit(insert->value());
822 auto tup_t = convert(insert->tuple()->type());
823 auto val_t = convert(insert->value()->type());
824 return bb.assign(name, "insertvalue {} {}, {} {}, {}", tup_t, tuple, val_t, value, index);
825 } else if (auto global = def->isa<Global>()) {
826 auto init = emit(global->init());
827 auto [pointee, addr_space] = force<mem::Ptr>(global->type())->args<2>();
828 print(vars_decls_, "{} = global {} {}\n", name, convert(pointee), init);
829 return globals_[global] = name;
830 }
831
832 unreachable(); // not yet implemented
833}
