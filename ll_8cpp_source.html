11// Lessons learned:
12// * **Always** follow all ops - even if you actually want to ignore it.
13// Otherwise, you might end up with an incorrect schedule.
14// This was the case for an Extract of type Mem.
15// While we want to ignore the value obtained from that, since there is no Mem value in LLVM,
16// we still want to **first** recursively emit code for its operands and **then** ignore the Extract itself.
17// * i1 has a different meaning in LLVM then in Thorin:
18// * Thorin: {0, 1} = i1
19// * LLVM: {0, -1} = i1
20// This is a problem when, e.g., using an index of type i1 as LLVM thinks like this:
21// getelementptr ..., i1 1 == getelementptr .., i1 -1

25static bool is_const(const Def* def) {
26 if (def->isa<Bot>()) return true;
27 if (def->isa<Lit>()) return true;
28 if (auto pack = def->isa_structural<Pack>()) return is_const(pack->shape()) && is_const(pack->body());
29
30 if (auto tuple = def->isa<Tuple>()) {
31 auto ops = tuple->ops();
32 return std::ranges::all_of(ops, [](auto def) { return is_const(def); });
33 }
34
35 return false;
36}

60 using std::swap;
61 swap(a.phis, b.phis);

111std::string CodeGen::convert(const Def* type) {
112 if (auto ll_type = types_.lookup(type)) return *ll_type;
113
114 assert(!isa<Tag::Mem>(type));
115 StringStream s;
116 std::string name;
117
118 if (type->isa<Nat>()) {
119 return types_[type] = "i64";
120 } else if (isa<Tag::Int>(type)) {
121 auto size = isa_sized_type(type);
122 if (size->isa<Top>()) return types_[type] = "i64";
123 if (auto width = mod2width(as_lit(size))) {
124 switch (*width) {
125 case 1: return types_[type] = "i1";
126 case 2:
127 case 4:
128 case 8: return types_[type] = "i8";
129 case 16: return types_[type] = "i16";
130 case 32: return types_[type] = "i32";
131 case 64: return types_[type] = "i64";
132 default: THORIN_UNREACHABLE;
133 }
134 } else {
135 return types_[type] = "i64";
136 }
137 } else if (auto real = isa<Tag::Real>(type)) {
138 switch (as_lit<nat_t>(real->arg())) {
139 case 16: return types_[type] = "half";
140 case 32: return types_[type] = "float";
141 case 64: return types_[type] = "double";
142 default: THORIN_UNREACHABLE;
143 }
144 } else if (auto ptr = isa<Tag::Ptr>(type)) {
145 auto [pointee, addr_space] = ptr->args<2>();
146 // TODO addr_space
147 s.fmt("{}*", convert(pointee));
148 } else if (auto arr = type->isa<Arr>()) {
149 auto elem_type = convert(arr->body());
150 u64 size = 0;
151 if (auto arity = isa_lit(arr->shape())) size = *arity;
152 s.fmt("[{} x {}]", size, elem_type);
153 } else if (auto pi = type->isa<Pi>()) {
154 s.fmt("{} (", convert(pi->doms().back()->as<Pi>()->dom()));
155
156 std::string_view sep = "";
157 for (auto dom : pi->doms().skip_back()) {
158 if (isa<Tag::Mem>(dom)) continue;
159 s << sep << convert(dom);
160 sep = ", ";
161 }
162 } else if (auto sigma = type->isa<Sigma>()) {
163 if (sigma->isa_nom()) {
164 name = id(sigma);
165 types_[sigma] = name;
166 s.fmt("{} = type", name);
167 }
168 s.fmt("{{");
169 std::string_view sep = "";
170 for (auto t : sigma->ops()) {
171 if (isa<Tag::Mem>(t)) continue;
172 s << sep << convert(t);
173 sep = ", ";
174 }
175 s.fmt("}}");
176 } else {
177 THORIN_UNREACHABLE;
178 }
179
180 if (name.empty()) return types_[type] = s.str();
181
182 assert(!s.str().empty());
183 type_decls_ << s.str() << '\n';
184 return types_[type] = name;
185}

187std::string CodeGen::convert_ret_pi(const Pi* pi) {
188 switch (pi->num_doms()) {
189 case 0:
190 return "void";
191 case 1:
192 if (isa<Tag::Mem>(pi->dom())) return "void";
193 return convert(pi->dom());
194 case 2:
195 if (isa<Tag::Mem>(pi->dom(0))) return convert(pi->dom(1));
196 if (isa<Tag::Mem>(pi->dom(1))) return convert(pi->dom(0));
197 [[fallthrough]];
198 default:
199 return convert(pi->dom());
200 }

231std::string CodeGen::prepare(const Scope& scope) {
232 auto lam = scope.entry()->as_nom<Lam>();
233
234 func_impls_.fmt("define {} {}(", convert_ret_pi(lam->type()->ret_pi()), id(lam));
235
236 auto sep = "";
237 auto vars = lam->vars();
238 for (auto var : vars.skip_back()) {
239 if (isa<Tag::Mem>(var->type())) continue;
240 auto name = id(var);
241 locals_[var] = name;
242 func_impls_.fmt("{}{} {}", sep, convert(var->type()), name);
243 sep = ", ";
244 }
245
246 func_impls_.fmt(") {{\n");
247
248 return lam->unique_name();

251void CodeGen::finalize(const Scope& scope) {
252 for (auto& [lam, bb] : lam2bb_) {
253 for (const auto& [phi, args] : bb.phis) {
254 bb.head().emplace_back().fmt("{} = phi {} ", id(phi), convert(phi->type()));
255 auto sep = "";
256 for (const auto& [arg, pred] : args) {
257 bb.head().back().fmt("{}[ {}, {} ]", sep, arg, pred);
258 sep = ", ";
259 }
260 }
261 }
262
263 for (auto nom : schedule(scope)) {
264 if (auto lam = nom->isa_nom<Lam>()) {
265 if (lam == scope.exit()) continue;
266 assert(lam2bb_.contains(lam));
267 auto& bb = lam2bb_[lam];
268 func_impls_.fmt("{}:\t\n", lam->unique_name());
269
270 for (const auto& part : bb.parts) {
271 for (const auto& line : part)
272 (func_impls_ << line.str()).endl();
273 }
274
275 func_impls_.dedent().endl();
276 }
277 }
278
279 func_impls_.fmt("}}\n\n");
280}

282void CodeGen::emit_epilogue(Lam* lam) {
283 auto app = lam->body()->as<App>();
284 auto& bb = lam2bb_[lam];
285
286 if (app->callee() == entry_->ret_var()) { // return
287 std::vector<std::string> values;
288 std::vector<const Def*> types;
289
290 for (auto arg : app->args()) {
291 if (auto val = emit_unsafe(arg); !val.empty()) {
292 values.emplace_back(val);
293 types.emplace_back(arg->type());
294 }
295 }
296
297 switch (values.size()) {
298 case 0: return bb.tail("ret void");
299 case 1: return bb.tail("ret {} {}", convert(types[0]), values[0]);
300 default:
301 auto tuple = convert(world().sigma(types));
302 bb.tail("{} ret_val\n", tuple);
303 for (size_t i = 0, e = types.size(); i != e; ++i)
304 bb.tail("ret_val.e{} = {};\n", i, values[i]);
305 return bb.tail("ret ret_val");
306 }
307 } else if (auto ex = app->callee()->isa<Extract>()) {
308 emit_unsafe(app->arg());
309 auto c = emit(ex->index());
310 auto [f, t] = ex->tuple()->projs<2>([this](auto def) { return emit(def); });
311 return bb.tail("br i1 {}, label {}, label {}", c, t, f);
312 } else if (app->callee()->isa<Bot>()) {
313 return bb.tail("ret ; bottom: unreachable");
314 } else if (auto callee = app->callee()->isa_nom<Lam>(); callee && callee->is_basicblock()) { // ordinary jump
315 for (size_t i = 0, e = callee->num_vars(); i != e; ++i) {
316 if (auto arg = emit_unsafe(app->arg(i)); !arg.empty()) {
317 auto phi = callee->var(i);
318 assert(!isa<Tag::Mem>(phi->type()));
319 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
320 locals_[phi] = id(phi);
321 }
322 }
323 return bb.tail("br label {}", id(callee));
324 } else if (auto callee = app->callee()->isa_nom<Lam>()) { // function call
325 auto ret_lam = app->args().back()->as_nom<Lam>();
326
327 std::vector<std::string> args;
328 auto app_args = app->args();
329 for (auto arg : app_args.skip_back()) {
330 if (auto emitted_arg = emit_unsafe(arg); !emitted_arg.empty())
331 args.emplace_back(convert(arg->type()) + " " + emitted_arg);
332 }
333
334 size_t num_vars = ret_lam->num_vars();
335 size_t n = 0;
336 Array<const Def*> values(num_vars);
337 Array<const Def*> types(num_vars);
338 for (auto var : ret_lam->vars()) {
339 if (isa<Tag::Mem>(var->type())) continue;
340 values[n] = var;
341 types[n] = var->type();
342 ++n;
343 }
344
345 if (n == 0) {
346 bb.tail("call void {}({, })", id(callee), args);
347 } else {
348 auto name = "%" + app->unique_name() + ".ret";
349 auto ret_ty = convert_ret_pi(ret_lam->type());
350 auto phi = ret_lam->var(1);
351 assert(!isa<Tag::Mem>(phi->type()));
352 lam2bb_[ret_lam].phis[phi].emplace_back(name, id(lam, true));
353 locals_[phi] = id(phi);
354 bb.tail("{} = call {} {}({, })", name, ret_ty, id(callee), args);
355 }
356
357 return bb.tail("br label {}", id(ret_lam));
358 }
359}

361std::string CodeGen::emit_bb(BB& bb, const Def* def) {
362 if (def->isa<Var>()) return {};
363 if (auto lam = def->isa<Lam>()) return id(lam);
364
365 auto name = id(def);
366 std::string op;
367
368 auto emit_tuple = [&](const Def* tuple) {
369 if (is_const(tuple)) {
370 bool is_array = tuple->type()->isa<Arr>();
371
372 std::string s;
373 s += is_array ? "[" : "{";
374 auto sep = "";
375 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
376 auto e = tuple->proj(n, i);
377 if (auto elem = emit_unsafe(e); !elem.empty()) {
378 auto elem_t = convert(e->type());
379 s += sep + elem_t + " " + elem;
380 sep = ", ";
381 }
382 }
383
384 return s += is_array ? "]" : "}";
385 }
386
387 std::string prev = "undef";
388 auto t = convert(tuple->type());
389 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
390 auto e = tuple->proj(n, i);
391 if (auto elem = emit_unsafe(e); !elem.empty()) {
392 auto elem_t = convert(e->type());
393 prev = bb.assign(name + "." + std::to_string(i), "insertvalue {} {}, {} {}, {}", t, prev, elem_t, elem, i);
394 }
395 }
396 return prev;
397 };
398
399 if (auto lit = def->isa<Lit>()) {
400 if (lit->type()->isa<Nat>()) {
401 return std::to_string(lit->get<nat_t>());
402 } else if (isa<Tag::Int>(lit->type())) {
403 auto size = isa_sized_type(lit->type());
404 if (size->isa<Top>()) return std::to_string(lit->get<nat_t>());
405 if (auto mod = mod2width(as_lit(size))) {
406 switch (*mod) {
407 case 1: return std::to_string(lit->get< u1>());
408 case 2:
409 case 4:
410 case 8: return std::to_string(lit->get< u8>());
411 case 16: return std::to_string(lit->get<u16>());
412 case 32: return std::to_string(lit->get<u32>());
413 case 64: return std::to_string(lit->get<u64>());
414 default: THORIN_UNREACHABLE;
415 }
416 } else {
417 return std::to_string(lit->get<u64>());
418 }
419 } else if (auto real = isa<Tag::Real>(lit->type())) {
420 std::stringstream s;
421 u64 hex;
422
423 switch (as_lit<nat_t>(real->arg())) {
424 case 16:
425 s << "0xH" << std::setfill('0') << std::setw(4) << std::right << std::hex << lit->get<u16>();
426 return s.str();
427 case 32: {
428 hex = std::bit_cast<u64>(r64(lit->get<r32>()));
429 break;
430 }
431 case 64:
432 hex = lit->get<u64>();
433 break;
434 default: THORIN_UNREACHABLE;
435 }
436
437 s << "0x" << std::setfill('0') << std::setw(16) << std::right << std::hex << hex;
438 return s.str();
439 }
440 THORIN_UNREACHABLE;
441 } else if (def->isa<Bot>()) {
442 return "undef";
443 } else if (auto bit = isa<Tag::Bit>(def)) {
444 auto [a, b] = bit->args<2>([this](auto def) { return emit(def); });
445 auto t = convert(bit->type());
446
447 auto neg = [&](std::string_view x) {
448 return bb.assign(name + ".neg", "xor {} 0, {}", t, x);
449 };
450
451 switch (bit.flags()) {
452 case Bit::_and: return bb.assign(name, "and {} {}, {}", t, a, b);
453 case Bit:: _or: return bb.assign(name, "or {} {}, {}", t, a, b);
454 case Bit::_xor: return bb.assign(name, "xor {} {}, {}", t, a, b);
455 case Bit::nand: return neg(bb.assign(name, "and {} {}, {}", t, a, b));
456 case Bit:: nor: return neg(bb.assign(name, "or {} {}, {}", t, a, b));
457 case Bit::nxor: return neg(bb.assign(name, "xor {} {}, {}", t, a, b));
458 case Bit:: iff: return bb.assign(name, "and {} {}, {}", neg(a), b);
459 case Bit::niff: return bb.assign(name, "or {} {}, {}", neg(a), b);
460 default: THORIN_UNREACHABLE;
461 }
462 } else if (auto shr = isa<Tag::Shr>(def)) {
463 auto [a, b] = shr->args<2>([this](auto def) { return emit(def); });
464 auto t = convert(shr->type());
465
466 switch (shr.flags()) {
467 case Shr::ashr: op = "ashr"; break;
468 case Shr::lshr: op = "lshr"; break;
469 default: THORIN_UNREACHABLE;
470 }
471
472 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
473 } else if (auto wrap = isa<Tag::Wrap>(def)) {
474 auto [a, b] = wrap->args<2>([this](auto def) { return emit(def); });
475 auto t = convert(wrap->type());
476 auto [mode, width] = wrap->decurry()->args<2>(as_lit<nat_t>);
477
478 switch (wrap.flags()) {
479 case Wrap::add: op = "add"; break;
480 case Wrap::sub: op = "sub"; break;
481 case Wrap::mul: op = "mul"; break;
482 case Wrap::shl: op = "shl"; break;
483 default: THORIN_UNREACHABLE;
484 }
485
486 if (mode & WMode::nuw) op += " nuw";
487 if (mode & WMode::nsw) op += " nsw";
488
489 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
490 } else if (auto div = isa<Tag::Div>(def)) {
491 auto [m, x, y] = div->args<3>();
492 auto t = convert(x->type());
493 emit_unsafe(m);
494 auto a = emit(x);
495 auto b = emit(y);
496
497 switch (div.flags()) {
498 case Div::sdiv: op = "sdiv"; break;
499 case Div::udiv: op = "udiv"; break;
500 case Div::srem: op = "srem"; break;
501 case Div::urem: op = "urem"; break;
502 default: THORIN_UNREACHABLE;
503 }
504
505 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
506 } else if (auto rop = isa<Tag::ROp>(def)) {
507 auto [a, b] = rop->args<2>([this](auto def) { return emit(def); });
508 auto t = convert(rop->type());
509 auto [mode, width] = rop->decurry()->args<2>(as_lit<nat_t>);
510
511 switch (rop.flags()) {
512 case ROp::add: op = "fadd"; break;
513 case ROp::sub: op = "fsub"; break;
514 case ROp::mul: op = "fmul"; break;
515 case ROp::div: op = "fdiv"; break;
516 case ROp::rem: op = "frem"; break;
517 default: THORIN_UNREACHABLE;
518 }
519
520 if (mode == RMode::fast)
521 op += " fast";
522 else {
523 if (mode & RMode::nnan ) op += " nnan";
524 if (mode & RMode::ninf ) op += " ninf";
525 if (mode & RMode::nsz ) op += " nsz";
526 if (mode & RMode::arcp ) op += " arcp";
527 if (mode & RMode::contract) op += " contract";
528 if (mode & RMode::afn ) op += " afn";
529 if (mode & RMode::reassoc ) op += " reassoc";
530 }
531
532 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
533 } else if (auto icmp = isa<Tag::ICmp>(def)) {
534 auto [a, b] = icmp->args<2>([this](auto def) { return emit(def); });
535 auto t = convert(icmp->arg(0)->type());
536 op = "icmp ";
537
538 switch (icmp.flags()) {
539 case ICmp::e: op += "eq" ; break;
540 case ICmp::ne: op += "ne" ; break;
541 case ICmp::sg: op += "sgt"; break;
542 case ICmp::sge: op += "sge"; break;
543 case ICmp::sl: op += "slt"; break;
544 case ICmp::sle: op += "sle"; break;
545 case ICmp::ug: op += "ugt"; break;
546 case ICmp::uge: op += "uge"; break;
547 case ICmp::ul: op += "ult"; break;
548 case ICmp::ule: op += "ule"; break;
549 default: THORIN_UNREACHABLE;
550 }
551
552 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
553 } else if (auto rcmp = isa<Tag::RCmp>(def)) {
554 auto [a, b] = rcmp->args<2>([this](auto def) { return emit(def); });
555 auto t = convert(rcmp->arg(0)->type());
556 op = "fcmp ";
557
558 switch (rcmp.flags()) {
559 case RCmp:: e: op += "oeq"; break;
560 case RCmp:: l: op += "olt"; break;
561 case RCmp:: le: op += "ole"; break;
562 case RCmp:: g: op += "ogt"; break;
563 case RCmp:: ge: op += "oge"; break;
564 case RCmp:: ne: op += "one"; break;
565 case RCmp:: o: op += "ord"; break;
566 case RCmp:: u: op += "uno"; break;
567 case RCmp:: ue: op += "ueq"; break;
568 case RCmp:: ul: op += "ult"; break;
569 case RCmp::ule: op += "ule"; break;
570 case RCmp:: ug: op += "ugt"; break;
571 case RCmp::uge: op += "uge"; break;
572 case RCmp::une: op += "une"; break;
573 default: THORIN_UNREACHABLE;
574 }
575
576 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
577 } else if (auto conv = isa<Tag::Conv>(def)) {
578 auto src = emit(conv->arg());
579 auto src_t = convert(conv->arg()->type());
580 auto dst_t = convert(conv->type());
581
582 auto size2width = [&](const Def* type) {
583 if (auto int_ = isa<Tag::Int>(type)) {
584 if (int_->arg()->isa<Top>()) return 64_u64;
585 if (auto width = mod2width(as_lit(int_->arg()))) return *width;
586 return 64_u64;
587 }
588 return as_lit(as<Tag::Real>(type)->arg());
589 };
590
591 nat_t s_src = size2width(conv->arg()->type());
592 nat_t s_dst = size2width(conv->type());
593
594 // this might happen when casting from int top to i64
595 if (s_src == s_dst && (conv.flags() == Conv::s2s || conv.flags() == Conv::u2u)) return src;
596
597 switch (conv.flags()) {
598 case Conv::s2s: op = s_src < s_dst ? "sext" : "trunc"; break;
599 case Conv::u2u: op = s_src < s_dst ? "zext" : "trunc"; break;
600 case Conv::r2r: op = s_src < s_dst ? "fpext" : "fptrunc"; break;
601 case Conv::s2r: op = "sitofp"; break;
602 case Conv::u2r: op = "uitofp"; break;
603 case Conv::r2s: op = "fptosi"; break;
604 case Conv::r2u: op = "fptoui"; break;
605 default: THORIN_UNREACHABLE;
606 }
607
608 return bb.assign(name, "{} {} {} to {}", op, src_t, src, dst_t);
609 } else if (auto bitcast = isa<Tag::Bitcast>(def)) {
610 auto dst_type_ptr = isa<Tag::Ptr>(bitcast->type());
611 auto src_type_ptr = isa<Tag::Ptr>(bitcast->arg()->type());
612 auto src = emit(bitcast->arg());
613 auto src_t = convert(bitcast->arg()->type());
614 auto dst_t = convert(bitcast->type());
