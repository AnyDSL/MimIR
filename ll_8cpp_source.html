19// Lessons learned:
20// * **Always** follow all ops - even if you actually want to ignore one.
21// Otherwise, you might end up with an incorrect schedule.
22// This was the case for an Extract of type Mem.
23// While we want to ignore the value obtained from that, since there is no Mem value in LLVM,
24// we still want to **first** recursively emit code for its operands and **then** ignore the Extract itself.
25// * i1 has a different meaning in LLVM then in Thorin:
26// * Thorin: {0, 1} = i1
27// * LLVM: {0, -1} = i1
28// This is a problem when, e.g., using an index of type i1 as LLVM thinks like this:
29// getelementptr ..., i1 1 == getelementptr .., i1 -1

158std::string Emitter::convert(const Def* type) {
159 if (auto i = types_.find(type); i != types_.end()) return i->second;
160
161 assert(!match<mem::M>(type));
162 std::ostringstream s;
163 std::string name;
164
165 if (type->isa<Nat>()) {
166 return types_[type] = "i64";
167 } else if (auto size = Idx::size(type)) {
168 return types_[type] = "i" + std::to_string(*Idx::size2bitwidth(size));
169 } else if (auto w = math::isa_f(type)) {
170 switch (*w) {
171 case 16: return types_[type] = "half";
172 case 32: return types_[type] = "float";
173 case 64: return types_[type] = "double";
174 default: unreachable();
175 }
176 } else if (auto ptr = match<mem::Ptr>(type)) {
177 auto [pointee, addr_space] = ptr->args<2>();
178 // TODO addr_space
179 print(s, "{}*", convert(pointee));
180 } else if (auto arr = type->isa<Arr>()) {
181 auto t_elem = convert(arr->body());
182 u64 size = 0;
183 if (auto arity = Lit::isa(arr->shape())) size = *arity;
184 print(s, "[{} x {}]", size, t_elem);
185 } else if (auto pi = type->isa<Pi>()) {
186 assert(Pi::isa_returning(pi) && "should never have to convert type of BB");
187 print(s, "{} (", convert_ret_pi(pi->ret_pi()));
188
189 if (auto t = isa_mem_sigma_2(pi->dom()))
190 s << convert(t);
191 else {
192 auto doms = pi->doms();
193 for (auto sep = ""; auto dom : doms.skip_back()) {
194 if (match<mem::M>(dom)) continue;
195 s << sep << convert(dom);
196 sep = ", ";
197 }
198 }
199 s << ")*";
200 } else if (auto t = isa_mem_sigma_2(type)) {
201 return convert(t);
202 } else if (auto sigma = type->isa<Sigma>()) {
203 if (sigma->isa_mut()) {
204 name = id(sigma);
205 types_[sigma] = name;
206 print(s, "{} = type", name);
207 }
208
209 print(s, "{{");
210 for (auto sep = ""; auto t : sigma->ops()) {
211 if (match<mem::M>(t)) continue;
212 s << sep << convert(t);
213 sep = ", ";
214 }
215 print(s, "}}");
216 } else {
217 unreachable();
218 }
219
220 if (name.empty()) return types_[type] = s.str();
221
222 assert(!s.str().empty());
223 type_decls_ << s.str() << '\n';
224 return types_[type] = name;
225}

308void Emitter::emit_epilogue(Lam* lam) {
309 auto app = lam->body()->as<App>();
310 auto& bb = lam2bb_[lam];
311
312 if (app->callee() == entry_->ret_var()) { // return
313 std::vector<std::string> values;
314 std::vector<const Def*> types;
315
316 for (auto arg : app->args()) {
317 if (auto val = emit_unsafe(arg); !val.empty()) {
318 values.emplace_back(val);
319 types.emplace_back(arg->type());
320 }
321 }
322
323 switch (values.size()) {
324 case 0: return bb.tail("ret void");
325 case 1: return bb.tail("ret {} {}", convert(types[0]), values[0]);
326 default: {
327 std::string prev = "undef";
328 auto type = convert(world().sigma(types));
329 for (size_t i = 0, n = values.size(); i != n; ++i) {
330 auto v_elem = values[i];
331 auto t_elem = convert(types[i]);
332 auto namei = "%ret_val." + std::to_string(i);
333 bb.tail("{} = insertvalue {} {}, {} {}, {}", namei, type, prev, t_elem, v_elem, i);
334 prev = namei;
335 }
336
337 bb.tail("ret {} {}", type, prev);
338 }
339 }
340 } else if (auto ex = app->callee()->isa<Extract>(); ex && Pi::isa_basicblock(app->callee_type())) {
341 // emit_unsafe(app->arg());
342 // A call to an extract like constructed for conditionals (else,then)#cond (args)
343 // TODO: we can not rely on the structure of the extract (it might be a nested extract)
344 for (auto callee_def : ex->tuple()->projs()) {
345 // dissect the tuple of lambdas
346 auto callee = callee_def->as_mut<Lam>();
347 // each callees type should agree with the argument type (should be checked by type checking).
348 // Especially, the number of vars should be the number of arguments.
349 // TODO: does not hold for complex arguments that are not tuples.
350 assert(callee->num_tvars() == app->num_targs());
351 size_t n = callee->num_tvars();
352 for (size_t i = 0; i != n; ++i) {
353 // emits the arguments one by one (TODO: handle together like before)
354 if (auto arg = emit_unsafe(app->arg(n, i)); !arg.empty()) {
355 auto phi = callee->var(n, i);
356 assert(!match<mem::M>(phi->type()));
357 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
358 locals_[phi] = id(phi);
359 }
360 }
361 }
362
363 auto c = emit(ex->index());
364 if (ex->tuple()->num_projs() == 2) {
365 auto [f, t] = ex->tuple()->projs<2>([this](auto def) { return emit(def); });
366 return bb.tail("br i1 {}, label {}, label {}", c, t, f);
367 } else {
368 auto t_c = convert(ex->index()->type());
369 bb.tail("switch {} {}, label {} [ ", t_c, c, emit(ex->tuple()->proj(0)));
370 for (auto i = 1u; i < ex->tuple()->num_projs(); i++)
371 print(bb.tail().back(), "{} {}, label {} ", t_c, std::to_string(i), emit(ex->tuple()->proj(i)));
372 print(bb.tail().back(), "]");
373 }
374 } else if (app->callee()->isa<Bot>()) {
375 return bb.tail("ret ; bottom: unreachable");
376 } else if (auto callee = Lam::isa_mut_basicblock(app->callee())) { // ordinary jump
377 size_t n = callee->num_tvars();
378 for (size_t i = 0; i != n; ++i) {
379 if (auto arg = emit_unsafe(app->arg(n, i)); !arg.empty()) {
380 auto phi = callee->var(n, i);
381 assert(!match<mem::M>(phi->type()));
382 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
383 locals_[phi] = id(phi);
384 }
385 }
386 return bb.tail("br label {}", id(callee));
387 } else if (auto longjmp = match<clos::longjmp>(app)) {
388 declare("void @longjmp(i8*, i32) noreturn");
389
390 auto [mem, jbuf, tag] = app->args<3>();
391 emit_unsafe(mem);
392 auto v_jb = emit(jbuf);
393 auto v_tag = emit(tag);
394 bb.tail("call void @longjmp(i8* {}, i32 {})", v_jb, v_tag);
395 return bb.tail("unreachable");
396 } else if (Pi::isa_returning(app->callee_type())) { // function call
397 auto v_callee = emit(app->callee());
398
399 std::vector<std::string> args;
400 auto app_args = app->args();
401 for (auto arg : app_args.skip_back())
402 if (auto v_arg = emit_unsafe(arg); !v_arg.empty()) args.emplace_back(convert(arg->type()) + " " + v_arg);
403
404 if (app->args().back()->isa<Bot>()) {
405 // TODO: Perhaps it'd be better to simply η-wrap this prior to the BE...
406 assert(convert_ret_pi(app->callee_type()->ret_pi()) == "void");
407 bb.tail("call void {}({, })", v_callee, args);
408 return bb.tail("unreachable");
409 }
410
411 auto ret_lam = app->args().back()->as_mut<Lam>();
412 size_t num_vars = ret_lam->num_vars();
413 size_t n = 0;
414 Array<const Def*> values(num_vars);
415 Array<const Def*> types(num_vars);
416 for (auto var : ret_lam->vars()) {
417 if (match<mem::M>(var->type())) continue;
418 values[n] = var;
419 types[n] = var->type();
420 ++n;
421 }
422
423 if (n == 0) {
424 bb.tail("call void {}({, })", v_callee, args);
425 } else {
426 auto name = "%" + app->unique_name() + ".ret";
427 auto t_ret = convert_ret_pi(ret_lam->type());
428 bb.tail("{} = call {} {}({, })", name, t_ret, v_callee, args);
429
430 for (size_t i = 0, e = ret_lam->num_vars(); i != e; ++i) {
431 auto phi = ret_lam->var(i);
432 if (match<mem::M>(phi->type())) continue;
433
434 auto namei = name;
435 if (e > 2) {
436 namei += '.' + std::to_string(i - 1);
437 bb.tail("{} = extractvalue {} {}, {}", namei, t_ret, name, i - 1);
438 }
439 assert(!match<mem::M>(phi->type()));
440 lam2bb_[ret_lam].phis[phi].emplace_back(namei, id(lam, true));
441 locals_[phi] = id(phi);
442 }
443 }
444
445 return bb.tail("br label {}", id(ret_lam));
446 }

449std::string Emitter::emit_bb(BB& bb, const Def* def) {
450 if (auto lam = def->isa<Lam>()) return id(lam);
451
452 auto name = id(def);
453 std::string op;
454
455 auto emit_tuple = [&](const Def* tuple) {
456 if (isa_mem_sigma_2(tuple->type())) {
457 emit_unsafe(tuple->proj(2, 0));
458 return emit(tuple->proj(2, 1));
459 }
460
461 if (is_const(tuple)) {
462 bool is_array = tuple->type()->isa<Arr>();
463
464 std::string s;
465 s += is_array ? "[" : "{";
466 auto sep = "";
467 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
468 auto e = tuple->proj(n, i);
469 if (auto v_elem = emit_unsafe(e); !v_elem.empty()) {
470 auto t_elem = convert(e->type());
471 s += sep + t_elem + " " + v_elem;
472 sep = ", ";
473 }
474 }
475
476 return s += is_array ? "]" : "}";
477 }
478
479 std::string prev = "undef";
480 auto t = convert(tuple->type());
481 for (size_t src = 0, dst = 0, n = tuple->num_projs(); src != n; ++src) {
482 auto e = tuple->proj(n, src);
483 if (auto elem = emit_unsafe(e); !elem.empty()) {
484 auto elem_t = convert(e->type());
485 // TODO: check dst vs src
486 auto namei = name + "." + std::to_string(dst);
487 prev = bb.assign(namei, "insertvalue {} {}, {} {}, {}", t, prev, elem_t, elem, dst);
488 dst++;
489 }
490 }
491 return prev;
492 };
493
494 if (def->isa<Var>()) {
495 auto ts = def->type()->projs();
496 if (std::ranges::any_of(ts, [](auto t) { return match<mem::M>(t); })) return {};
497 return emit_tuple(def);
498 }
499
500 auto emit_gep_index = [&](const Def* index) {
501 auto v_i = emit(index);
502 auto t_i = convert(index->type());
503
504 if (auto size = Idx::size(index->type())) {
505 if (auto w = Idx::size2bitwidth(size); w && *w < 64) {
506 v_i = bb.assign(name + ".zext",
507 "zext {} {} to i{} ; add one more bit for gep index as it is treated as signed value",
508 t_i, v_i, *w + 1);
509 t_i = "i" + std::to_string(*w + 1);
510 }
511 }
512
513 return std::pair(v_i, t_i);
514 };
515
516 if (auto lit = def->isa<Lit>()) {
517 if (lit->type()->isa<Nat>() || Idx::size(lit->type())) {
518 return std::to_string(lit->get());
519 } else if (auto w = math::isa_f(lit->type())) {
520 std::stringstream s;
521 u64 hex;
522
523 switch (*w) {
524 case 16:
525 s << "0xH" << std::setfill('0') << std::setw(4) << std::right << std::hex << lit->get<u16>();
526 return s.str();
527 case 32: {
528 hex = std::bit_cast<u64>(f64(lit->get<f32>()));
529 break;
530 }
531 case 64: hex = lit->get<u64>(); break;
532 default: unreachable();
533 }
534
535 s << "0x" << std::setfill('0') << std::setw(16) << std::right << std::hex << hex;
536 return s.str();
537 }
538 unreachable();
539 } else if (def->isa<Bot>()) {
540 return "undef";
541 } else if (auto top = def->isa<Top>()) {
542 if (match<mem::M>(top->type())) return {};
543 // bail out to error below
544 } else if (auto tuple = def->isa<Tuple>()) {
545 return emit_tuple(tuple);
546 } else if (auto pack = def->isa<Pack>()) {
547 if (auto lit = Lit::isa(pack->body()); lit && *lit == 0) return "zeroinitializer";
548 return emit_tuple(pack);
549 } else if (auto extract = def->isa<Extract>()) {
550 auto tuple = extract->tuple();
551 auto index = extract->index();
552
553 // use select when extracting from 2-element integral tuples
554 // literal indices would be normalized away already, if it was possible
555 // As they aren't they likely depend on a var, which is implemented as array -> need extractvalue
556 if (auto app = extract->type()->isa<App>();
557 app && app->callee()->isa<Idx>() && !index->isa<Lit>() && tuple->type()->isa<Arr>()) {
558 if (auto arity = tuple->type()->isa_lit_arity(); arity && *arity == 2) {
559 auto t = convert(extract->type());
560 auto [elem_a, elem_b] = tuple->projs<2>([&](auto e) { return emit_unsafe(e); });
561
562 return bb.assign(name, "select {} {}, {} {}, {} {}", convert(index->type()), emit(index), t, elem_b, t,
563 elem_a);
564 }
565 }
566
567 auto v_tup = emit_unsafe(tuple);
568
569 // this exact location is important: after emitting the tuple -> ordering of mem ops
570 // before emitting the index, as it might be a weird value for mem vars.
571 if (match<mem::M>(extract->type())) return {};
572
573 auto t_tup = convert(tuple->type());
574 if (auto li = Lit::isa(index)) {
575 if (isa_mem_sigma_2(tuple->type())) return v_tup;
576 // Adjust index, if mem is present.
577 auto v_i = match<mem::M>(tuple->proj(0)->type()) ? std::to_string(*li - 1) : std::to_string(*li);
578 return bb.assign(name, "extractvalue {} {}, {}", t_tup, v_tup, v_i);
579 }
580
581 auto t_elem = convert(extract->type());
582 auto [v_i, t_i] = emit_gep_index(index);
583
584 print(lam2bb_[entry_].body().emplace_front(),
585 "{}.alloca = alloca {} ; copy to alloca to emulate extract with store + gep + load", name, t_tup);
586 print(bb.body().emplace_back(), "store {} {}, {}* {}.alloca", t_tup, v_tup, t_tup, name);
587 print(bb.body().emplace_back(), "{}.gep = getelementptr inbounds {}, {}* {}.alloca, i64 0, {} {}", name, t_tup,
588 t_tup, name, t_i, v_i);
589 return bb.assign(name, "load {}, {}* {}.gep", t_elem, t_elem, name);
590 } else if (auto insert = def->isa<Insert>()) {
591 assert(!match<mem::M>(insert->tuple()->proj(0)->type()));
592 auto t_tup = convert(insert->tuple()->type());
593 auto t_val = convert(insert->value()->type());
594 auto v_tup = emit(insert->tuple());
595 auto v_val = emit(insert->value());
596 if (auto idx = Lit::isa(insert->index())) {
597 auto v_idx = emit(insert->index());
598 return bb.assign(name, "insertvalue {} {}, {} {}, {}", t_tup, v_tup, t_val, v_val, v_idx);
599 } else {
600 auto t_elem = convert(insert->value()->type());
601 auto [v_i, t_i] = emit_gep_index(insert->index());
602 print(lam2bb_[entry_].body().emplace_front(),
603 "{}.alloca = alloca {} ; copy to alloca to emulate insert with store + gep + load", name, t_tup);
604 print(bb.body().emplace_back(), "store {} {}, {}* {}.alloca", t_tup, v_tup, t_tup, name);
