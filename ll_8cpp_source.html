18// Lessons learned:
19// * **Always** follow all ops - even if you actually want to ignore one.
20// Otherwise, you might end up with an incorrect schedule.
21// This was the case for an Extract of type Mem.
22// While we want to ignore the value obtained from that, since there is no Mem value in LLVM,
23// we still want to **first** recursively emit code for its operands and **then** ignore the Extract itself.
24// * i1 has a different meaning in LLVM then in Thorin:
25// * Thorin: {0, 1} = i1
26// * LLVM: {0, -1} = i1
27// This is a problem when, e.g., using an index of type i1 as LLVM thinks like this:
28// getelementptr ..., i1 1 == getelementptr .., i1 -1

33static bool is_const(const Def* def) {
34 if (def->isa<Bot>()) return true;
35 if (def->isa<Lit>()) return true;
36 if (auto pack = def->isa_structural<Pack>()) return is_const(pack->shape()) && is_const(pack->body());
37
38 if (auto tuple = def->isa<Tuple>()) {
39 auto ops = tuple->ops();
40 return std::ranges::all_of(ops, [](auto def) { return is_const(def); });
41 }
42
43 return false;
44}

46struct BB {
47 BB() = default;

49 BB(BB&& other) { swap(*this, other); }
50
51 std::deque<std::ostringstream>& head() { return parts[0]; }
52 std::deque<std::ostringstream>& body() { return parts[1]; }
53 std::deque<std::ostringstream>& tail() { return parts[2]; }

55 template<class... Args>
56 std::string assign(std::string_view name, const char* s, Args&&... args) {
57 print(print(body().emplace_back(), "{} = ", name), s, std::forward<Args&&>(args)...);

60
61 template<class... Args>

63 print(tail().emplace_back(), s, std::forward<Args&&>(args)...);
64 }
65

67 using std::swap;
68 swap(a.phis, b.phis);
69 swap(a.parts, b.parts);

119std::string CodeGen::convert(const Def* type) {
120 if (auto i = types_.find(type); i != types_.end()) return i->second;
121
122 assert(!match<mem::M>(type));
123 std::ostringstream s;
124 std::string name;
125
126 if (type->isa<Nat>()) {
127 return types_[type] = "i64";
128 } else if (isa<Tag::Int>(type)) {
129 auto size = isa_sized_type(type);
130 if (size->isa<Top>()) return types_[type] = "i64";
131 if (auto width = mod2width(as_lit(size))) {
132 switch (*width) {
133 // clang-format off
134 case 1: return types_[type] = "i1";
135 case 2:
136 case 4:
137 case 8: return types_[type] = "i8";
138 case 16: return types_[type] = "i16";
139 case 32: return types_[type] = "i32";
140 case 64: return types_[type] = "i64";
141 // clang-format on
142 default: unreachable();
143 }
144 } else {
145 return types_[type] = "i64";
146 }
147 } else if (auto real = isa<Tag::Real>(type)) {
148 switch (as_lit<nat_t>(real->arg())) {
149 case 16: return types_[type] = "half";
150 case 32: return types_[type] = "float";
151 case 64: return types_[type] = "double";
152 default: unreachable();
153 }
154 } else if (auto ptr = match<mem::Ptr>(type)) {
155 auto [pointee, addr_space] = ptr->args<2>();
156 // TODO addr_space
157 print(s, "{}*", convert(pointee));
158 } else if (auto arr = type->isa<Arr>()) {
159 auto elem_type = convert(arr->body());
160 u64 size = 0;
161 if (auto arity = isa_lit(arr->shape())) size = *arity;
162 print(s, "[{} x {}]", size, elem_type);
163 } else if (auto pi = type->isa<Pi>()) {
164 assert(pi->is_returning() && "should never have to convert type of BB");
165 print(s, "{} (", convert_ret_pi(pi->ret_pi()));
166
167 std::string_view sep = "";
168 auto doms = pi->doms();
169 for (auto dom : doms.skip_back()) {
170 if (match<mem::M>(dom)) continue;
171 s << sep << convert(dom);
172 sep = ", ";
173 }
174
175 s << ")*";
176 } else if (auto sigma = type->isa<Sigma>()) {
177 if (sigma->isa_nom()) {
178 name = id(sigma);
179 types_[sigma] = name;
180 print(s, "{} = type", name);
181 }
182 print(s, "{{");
183 std::string_view sep = "";
184 for (auto t : sigma->ops()) {
185 if (match<mem::M>(t)) continue;
186 s << sep << convert(t);
187 sep = ", ";
188 }
189 print(s, "}}");
190 } else {
191 unreachable();
192 }
193
194 if (name.empty()) return types_[type] = s.str();
195
196 assert(!s.str().empty());
197 type_decls_ << s.str() << '\n';
198 return types_[type] = name;
199}

201std::string CodeGen::convert_ret_pi(const Pi* pi) {
202 switch (pi->num_doms()) {
203 case 0: return "void";
204 case 1:
205 if (match<mem::M>(pi->dom())) return "void";
206 return convert(pi->dom());
207 case 2:
208 if (match<mem::M>(pi->dom(0))) return convert(pi->dom(1));
209 if (match<mem::M>(pi->dom(1))) return convert(pi->dom(0));
210 [[fallthrough]];
211 default: return convert(pi->dom());
212 }

247std::string CodeGen::prepare(const Scope& scope) {
248 auto lam = scope.entry()->as_nom<Lam>();
249
250 print(func_impls_, "define {} {}(", convert_ret_pi(lam->type()->ret_pi()), id(lam));
251
252 auto sep = "";
253 auto vars = lam->vars();
254 for (auto var : vars.skip_back()) {
255 if (match<mem::M>(var->type())) continue;
256 auto name = id(var);
257 locals_[var] = name;
258 print(func_impls_, "{}{} {}", sep, convert(var->type()), name);
259 sep = ", ";
260 }
261
262 print(func_impls_, ") {{\n");
263 return lam->unique_name();
264}

266void CodeGen::finalize(const Scope& scope) {
267 for (auto& [lam, bb] : lam2bb_) {
268 for (const auto& [phi, args] : bb.phis) {
269 print(bb.head().emplace_back(), "{} = phi {} ", id(phi), convert(phi->type()));
270 auto sep = "";
271 for (const auto& [arg, pred] : args) {
272 print(bb.head().back(), "{}[ {}, {} ]", sep, arg, pred);
273 sep = ", ";
274 }
275 }
276 }
277
278 for (auto nom : schedule(scope)) {
279 if (auto lam = nom->isa_nom<Lam>()) {
280 if (lam == scope.exit()) continue;
281 assert(lam2bb_.contains(lam));
282 auto& bb = lam2bb_[lam];
283 print(func_impls_, "{}:\n", lam->unique_name());
284
285 ++tab;
286 for (const auto& part : bb.parts) {
287 for (const auto& line : part) tab.print(func_impls_, "{}\n", line.str());
288 }
289 --tab;
290 func_impls_ << std::endl;
291 }
292 }
293
294 print(func_impls_, "}}\n\n");
295}

297void CodeGen::emit_epilogue(Lam* lam) {
298 auto app = lam->body()->as<App>();
299 auto& bb = lam2bb_[lam];
300
301 if (app->callee() == entry_->ret_var()) { // return
302 std::vector<std::string> values;
303 std::vector<const Def*> types;
304
305 for (auto arg : app->args()) {
306 if (auto val = emit_unsafe(arg); !val.empty()) {
307 values.emplace_back(val);
308 types.emplace_back(arg->type());
309 }
310 }
311
312 switch (values.size()) {
313 case 0: return bb.tail("ret void");
314 case 1: return bb.tail("ret {} {}", convert(types[0]), values[0]);
315 default: {
316 std::string prev = "undef";
317 auto type = convert(world().sigma(types));
318 for (size_t i = 0, n = values.size(); i != n; ++i) {
319 auto elem = values[i];
320 auto elem_t = convert(types[i]);
321 auto namei = "%ret_val." + std::to_string(i);
322 bb.tail("{} = insertvalue {} {}, {} {}, {}", namei, type, prev, elem_t, elem, i);
323 prev = namei;
324 }
325
326 bb.tail("ret {} {}", type, prev);
327 }
328 }
329 } else if (auto ex = app->callee()->isa<Extract>(); ex && app->callee_type()->is_basicblock()) {
330 emit_unsafe(app->arg());
331 auto c = emit(ex->index());
332 if (ex->tuple()->num_projs() == 2) {
333 auto [f, t] = ex->tuple()->projs<2>([this](auto def) { return emit(def); });
334 return bb.tail("br i1 {}, label {}, label {}", c, t, f);
335 } else {
336 auto c_ty = convert(ex->index()->type());
337 bb.tail("switch {} {}, label {} [ ", c_ty, c, emit(ex->tuple()->proj(0)));
338 for (auto i = 1u; i < ex->tuple()->num_projs(); i++)
339 print(bb.tail().back(), "{} {}, label {} ", c_ty, std::to_string(i), emit(ex->tuple()->proj(i)));
340 print(bb.tail().back(), "]");
341 }
342 } else if (app->callee()->isa<Bot>()) {
343 return bb.tail("ret ; bottom: unreachable");
344 } else if (auto callee = app->callee()->isa_nom<Lam>(); callee && callee->is_basicblock()) { // ordinary jump
345 for (size_t i = 0, e = callee->num_vars(); i != e; ++i) {
346 if (auto arg = emit_unsafe(app->arg(i)); !arg.empty()) {
347 auto phi = callee->var(i);
348 assert(!match<mem::M>(phi->type()));
349 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
350 locals_[phi] = id(phi);
351 }
352 }
353 return bb.tail("br label {}", id(callee));
354 } else if (auto longjmp = match<clos::longjmp>(app)) {
355 auto [mem, jbuf, tag] = app->args<3>();
356 emit_unsafe(mem);
357 auto emitted_jb = emit(jbuf);
358 auto emitted_tag = emit(tag);
359 bb.tail("call void @longjmp(i8* {}, i32 {})", emitted_jb, emitted_tag);
360 return bb.tail("unreachable");
361 } else if (app->callee_type()->is_returning()) { // function call
362 auto emmited_callee = emit(app->callee());
363
364 std::vector<std::string> args;
365 auto app_args = app->args();
366 for (auto arg : app_args.skip_back()) {
367 if (auto emitted_arg = emit_unsafe(arg); !emitted_arg.empty())
368 args.emplace_back(convert(arg->type()) + " " + emitted_arg);
369 }
370
371 if (app->args().back()->isa<Bot>()) {
372 // TODO: Perhaps it'd be better to simply -wrap this prior to the BE...
373 assert(convert_ret_pi(app->callee_type()->ret_pi()) == "void");
374 bb.tail("call void {}({, })", emmited_callee, args);
375 return bb.tail("unreachable");
376 }
377
378 auto ret_lam = app->args().back()->as_nom<Lam>();
379 size_t num_vars = ret_lam->num_vars();
380 size_t n = 0;
381 Array<const Def*> values(num_vars);
382 Array<const Def*> types(num_vars);
383 for (auto var : ret_lam->vars()) {
384 if (match<mem::M>(var->type())) continue;
385 values[n] = var;
386 types[n] = var->type();
387 ++n;
388 }
389
390 if (n == 0) {
391 bb.tail("call void {}({, })", emmited_callee, args);
392 } else {
393 auto name = "%" + app->unique_name() + ".ret";
394 auto ret_ty = convert_ret_pi(ret_lam->type());
395 bb.tail("{} = call {} {}({, })", name, ret_ty, emmited_callee, args);
396
397 for (size_t i = 1, e = ret_lam->num_vars(); i != e; ++i) {
398 auto phi = ret_lam->var(i);
399 auto namei = name;
400 if (e > 2) {
401 namei += '.' + std::to_string(i - 1);
402 bb.tail("{} = extractvalue {} {}, {}", namei, ret_ty, name, i - 1);
403 }
404 assert(!match<mem::M>(phi->type()));
405 lam2bb_[ret_lam].phis[phi].emplace_back(namei, id(lam, true));
406 locals_[phi] = id(phi);
407 }
408 }
409
410 return bb.tail("br label {}", id(ret_lam));
411 }

414std::string CodeGen::emit_bb(BB& bb, const Def* def) {
415 if (def->isa<Var>()) return {};
416 if (auto lam = def->isa<Lam>()) return id(lam);
417
418 auto name = id(def);
419 std::string op;
420
421 auto emit_tuple = [&](const Def* tuple) {
422 if (is_const(tuple)) {
423 bool is_array = tuple->type()->isa<Arr>();
424
425 std::string s;
426 s += is_array ? "[" : "{";
427 auto sep = "";
428 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
429 auto e = tuple->proj(n, i);
430 if (auto elem = emit_unsafe(e); !elem.empty()) {
431 auto elem_t = convert(e->type());
432 s += sep + elem_t + " " + elem;
433 sep = ", ";
434 }
435 }
436
437 return s += is_array ? "]" : "}";
438 }
439
440 std::string prev = "undef";
441 auto t = convert(tuple->type());
442 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
443 auto e = tuple->proj(n, i);
444 if (auto elem = emit_unsafe(e); !elem.empty()) {
445 auto elem_t = convert(e->type());
446 auto namei = name + "." + std::to_string(i);
447 prev = bb.assign(namei, "insertvalue {} {}, {} {}, {}", t, prev, elem_t, elem, i);
448 }
449 }
450 return prev;
451 };
452
453 if (auto lit = def->isa<Lit>()) {
454 if (lit->type()->isa<Nat>()) {
455 return std::to_string(lit->get<nat_t>());
456 } else if (isa<Tag::Int>(lit->type())) {
457 auto size = isa_sized_type(lit->type());
458 if (size->isa<Top>()) return std::to_string(lit->get<nat_t>());
459 if (auto mod = mod2width(as_lit(size))) {
460 switch (*mod) {
461 // clang-format off
462 case 0: return {};
463 case 1: return std::to_string(lit->get< u1>());
464 case 2:
465 case 4:
466 case 8: return std::to_string(lit->get< u8>());
467 case 16: return std::to_string(lit->get<u16>());
468 case 32: return std::to_string(lit->get<u32>());
469 case 64: return std::to_string(lit->get<u64>());
470 // clang-format on
471 default: return std::to_string(lit->get<u64>());
472 }
473 } else {
474 return std::to_string(lit->get<u64>());
475 }
476 } else if (auto real = isa<Tag::Real>(lit->type())) {
477 std::stringstream s;
478 u64 hex;
479
480 switch (as_lit<nat_t>(real->arg())) {
481 case 16:
482 s << "0xH" << std::setfill('0') << std::setw(4) << std::right << std::hex << lit->get<u16>();
483 return s.str();
484 case 32: {
485 hex = std::bit_cast<u64>(r64(lit->get<r32>()));
486 break;
487 }
488 case 64: hex = lit->get<u64>(); break;
489 default: unreachable();
490 }
491
492 s << "0x" << std::setfill('0') << std::setw(16) << std::right << std::hex << hex;
493 return s.str();
494 }
495 unreachable();
496 } else if (def->isa<Bot>()) {
497 return "undef";
498 } else if (auto bit = isa<Tag::Bit>(def)) {
499 auto [a, b] = bit->args<2>([this](auto def) { return emit(def); });
500 auto t = convert(bit->type());
501
502 auto neg = [&](std::string_view x) { return bb.assign(name + ".neg", "xor {} 0, {}", t, x); };
503
504 switch (bit.sub()) {
505 // clang-format off
506 case Bit::_and: return bb.assign(name, "and {} {}, {}", t, a, b);
507 case Bit:: _or: return bb.assign(name, "or {} {}, {}", t, a, b);
508 case Bit::_xor: return bb.assign(name, "xor {} {}, {}", t, a, b);
509 case Bit::nand: return neg(bb.assign(name, "and {} {}, {}", t, a, b));
510 case Bit:: nor: return neg(bb.assign(name, "or {} {}, {}", t, a, b));
511 case Bit::nxor: return neg(bb.assign(name, "xor {} {}, {}", t, a, b));
512 case Bit:: iff: return bb.assign(name, "and {} {}, {}", neg(a), b);
513 case Bit::niff: return bb.assign(name, "or {} {}, {}", neg(a), b);
514 // clang-format on
515 default: unreachable();
516 }
517 } else if (auto bit = match<core::bit2>(def)) {
518 auto [a, b] = bit->args<2>([this](auto def) { return emit(def); });
519 auto t = convert(bit->type());
520
521 auto neg = [&](std::string_view x) { return bb.assign(name + ".neg", "xor {} 0, {}", t, x); };
522
523 switch (bit.flags()) {
524 // clang-format off
525 case core::bit2::_and: return bb.assign(name, "and {} {}, {}", t, a, b);
526 case core::bit2:: _or: return bb.assign(name, "or {} {}, {}", t, a, b);
527 case core::bit2::_xor: return bb.assign(name, "xor {} {}, {}", t, a, b);
528 case core::bit2::nand: return neg(bb.assign(name, "and {} {}, {}", t, a, b));
529 case core::bit2:: nor: return neg(bb.assign(name, "or {} {}, {}", t, a, b));
530 case core::bit2::nxor: return neg(bb.assign(name, "xor {} {}, {}", t, a, b));
531 case core::bit2:: iff: return bb.assign(name, "and {} {}, {}", neg(a), b);
532 case core::bit2::niff: return bb.assign(name, "or {} {}, {}", neg(a), b);
533 // clang-format on
534 default: unreachable();
535 }
536 } else if (auto shr = isa<Tag::Shr>(def)) {
537 auto [a, b] = shr->args<2>([this](auto def) { return emit(def); });
538 auto t = convert(shr->type());
539
540 switch (shr.sub()) {
541 case Shr::ashr: op = "ashr"; break;
542 case Shr::lshr: op = "lshr"; break;
543 default: unreachable();
544 }
545
546 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
547 } else if (auto shr = match<core::shr>(def)) {
548 auto [a, b] = shr->args<2>([this](auto def) { return emit(def); });
549 auto t = convert(shr->type());
550
551 switch (shr.flags()) {
552 case core::shr::ashr: op = "ashr"; break;
553 case core::shr::lshr: op = "lshr"; break;
554 default: unreachable();
555 }
556
557 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
558 } else if (auto wrap = isa<Tag::Wrap>(def)) {
559 auto [a, b] = wrap->args<2>([this](auto def) { return emit(def); });
560 auto t = convert(wrap->type());
561 auto [mode, width] = wrap->decurry()->args<2>(as_lit<nat_t>);
562
563 switch (wrap.sub()) {
564 case Wrap::add: op = "add"; break;
565 case Wrap::sub: op = "sub"; break;
566 case Wrap::mul: op = "mul"; break;
567 case Wrap::shl: op = "shl"; break;
568 default: unreachable();
569 }
570
571 if (mode & WMode::nuw) op += " nuw";
572 if (mode & WMode::nsw) op += " nsw";
573
574 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
575 } else if (auto wrap = match<core::wrap>(def)) {
576 auto [a, b] = wrap->args<2>([this](auto def) { return emit(def); });
577 auto t = convert(wrap->type());
578 auto [mode, width] = wrap->decurry()->args<2>(as_lit<nat_t>);
579
580 switch (wrap.flags()) {
581 case core::wrap::add: op = "add"; break;
582 case core::wrap::sub: op = "sub"; break;
583 case core::wrap::mul: op = "mul"; break;
584 case core::wrap::shl: op = "shl"; break;
585 default: unreachable();
586 }
587
588 if (mode & WMode::nuw) op += " nuw";
589 if (mode & WMode::nsw) op += " nsw";
590
591 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
592 } else if (auto div = match<core::div>(def)) {
593 auto [m, x, y] = div->args<3>();
594 auto t = convert(x->type());
595 emit_unsafe(m);
596 auto a = emit(x);
597 auto b = emit(y);
598
599 switch (div.flags()) {
600 case core::div::sdiv: op = "sdiv"; break;
601 case core::div::udiv: op = "udiv"; break;
602 case core::div::srem: op = "srem"; break;
603 case core::div::urem: op = "urem"; break;
604 default: unreachable();
605 }
606
607 return bb.assign(name, "{} {} {}, {}", op, t, a, b);
