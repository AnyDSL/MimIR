19// Lessons learned:
20// * **Always** follow all ops - even if you actually want to ignore one.
21// Otherwise, you might end up with an incorrect schedule.
22// This was the case for an Extract of type Mem.
23// While we want to ignore the value obtained from that, since there is no Mem value in LLVM,
24// we still want to **first** recursively emit code for its operands and **then** ignore the Extract itself.
25// * i1 has a different meaning in LLVM then in Thorin:
26// * Thorin: {0, 1} = i1
27// * LLVM: {0, -1} = i1
28// This is a problem when, e.g., using an index of type i1 as LLVM thinks like this:
29// getelementptr ..., i1 1 == getelementptr .., i1 -1

34static bool is_const(const Def* def) {
35 if (def->isa<Bot>()) return true;
36 if (def->isa<Lit>()) return true;
37 if (auto pack = def->isa_structural<Pack>()) return is_const(pack->shape()) && is_const(pack->body());
38
39 if (auto tuple = def->isa<Tuple>()) {
40 auto ops = tuple->ops();
41 return std::ranges::all_of(ops, [](auto def) { return is_const(def); });
42 }
43
44 return false;
45}

130std::string Emitter::convert(const Def* type) {
131 if (auto i = types_.find(type); i != types_.end()) return i->second;
132
133 assert(!match<mem::M>(type));
134 std::ostringstream s;
135 std::string name;
136
137 if (type->isa<Nat>()) {
138 return types_[type] = "i64";
139 } else if (auto size = Idx::size(type)) {
140 return types_[type] = "i" + std::to_string(*Idx::size2bitwidth(size));
141 } else if (auto w = math::isa_f(type)) {
142 switch (*w) {
143 case 16: return types_[type] = "half";
144 case 32: return types_[type] = "float";
145 case 64: return types_[type] = "double";
146 default: unreachable();
147 }
148 } else if (auto ptr = match<mem::Ptr>(type)) {
149 auto [pointee, addr_space] = ptr->args<2>();
150 // TODO addr_space
151 print(s, "{}*", convert(pointee));
152 } else if (auto arr = type->isa<Arr>()) {
153 auto t_elem = convert(arr->body());
154 u64 size = 0;
155 if (auto arity = isa_lit(arr->shape())) size = *arity;
156 print(s, "[{} x {}]", size, t_elem);
157 } else if (auto pi = type->isa<Pi>()) {
158 assert(pi->is_returning() && "should never have to convert type of BB");
159 print(s, "{} (", convert_ret_pi(pi->ret_pi()));
160
161 auto doms = pi->doms();
162 for (auto sep = ""; auto dom : doms.skip_back()) {
163 if (match<mem::M>(dom)) continue;
164 s << sep << convert(dom);
165 sep = ", ";
166 }
167
168 s << ")*";
169 } else if (auto sigma = type->isa<Sigma>()) {
170 if (sigma->isa_nom()) {
171 name = id(sigma);
172 types_[sigma] = name;
173 print(s, "{} = type", name);
174 }
175 print(s, "{{");
176 for (auto sep = ""; auto t : sigma->ops()) {
177 if (match<mem::M>(t)) continue;
178 s << sep << convert(t);
179 sep = ", ";
180 }
181 print(s, "}}");
182 } else {
183 unreachable();
184 }
185
186 if (name.empty()) return types_[type] = s.str();
187
188 assert(!s.str().empty());
189 type_decls_ << s.str() << '\n';
190 return types_[type] = name;
191}

245void Emitter::finalize(const Scope& scope) {
246 for (auto& [lam, bb] : lam2bb_) {
247 for (const auto& [phi, args] : bb.phis) {
248 print(bb.head().emplace_back(), "{} = phi {} ", id(phi), convert(phi->type()));
249 for (auto sep = ""; const auto& [arg, pred] : args) {
250 print(bb.head().back(), "{}[ {}, {} ]", sep, arg, pred);
251 sep = ", ";
252 }
253 }
254 }
255
256 for (auto nom : schedule(scope)) {
257 if (auto lam = nom->isa_nom<Lam>()) {
258 if (lam == scope.exit()) continue;
259 assert(lam2bb_.contains(lam));
260 auto& bb = lam2bb_[lam];
261 print(func_impls_, "{}:\n", lam->unique_name());
262
263 ++tab;
264 for (const auto& part : bb.parts)
265 for (const auto& line : part) tab.print(func_impls_, "{}\n", line.str());
266 --tab;
267 func_impls_ << std::endl;
268 }
269 }
270
271 print(func_impls_, "}}\n\n");
272}

274void Emitter::emit_epilogue(Lam* lam) {
275 auto app = lam->body()->as<App>();
276 auto& bb = lam2bb_[lam];
277
278 if (app->callee() == entry_->ret_var()) { // return
279 std::vector<std::string> values;
280 std::vector<const Def*> types;
281
282 for (auto arg : app->args()) {
283 if (auto val = emit_unsafe(arg); !val.empty()) {
284 values.emplace_back(val);
285 types.emplace_back(arg->type());
286 }
287 }
288
289 switch (values.size()) {
290 case 0: return bb.tail("ret void");
291 case 1: return bb.tail("ret {} {}", convert(types[0]), values[0]);
292 default: {
293 std::string prev = "undef";
294 auto type = convert(world().sigma(types));
295 for (size_t i = 0, n = values.size(); i != n; ++i) {
296 auto v_elem = values[i];
297 auto t_elem = convert(types[i]);
298 auto namei = "%ret_val." + std::to_string(i);
299 bb.tail("{} = insertvalue {} {}, {} {}, {}", namei, type, prev, t_elem, v_elem, i);
300 prev = namei;
301 }
302
303 bb.tail("ret {} {}", type, prev);
304 }
305 }
306 } else if (auto ex = app->callee()->isa<Extract>(); ex && app->callee_type()->is_basicblock()) {
307 // emit_unsafe(app->arg());
308 // A call to an extract like constructed for conditionals (else,then)#cond (args)
309 // TODO: we can not rely on the structure of the extract (it might be a nested extract)
310 for (auto callee_def : ex->tuple()->projs()) {
311 // dissect the tuple of lambdas
312 auto callee = callee_def->as_nom<Lam>();
313 // each callees type should agree with the argument type (should be checked by type checking).
314 // Especially, the number of vars should be the number of arguments.
315 // TODO: does not hold for complex arguments that are not tuples.
316 assert(callee->num_vars() == app->num_args());
317 for (size_t i = 0, e = callee->num_vars(); i != e; ++i) {
318 // emits the arguments one by one (TODO: handle together like before)
319 if (auto arg = emit_unsafe(app->arg(i)); !arg.empty()) {
320 auto phi = callee->var(i);
321 assert(!match<mem::M>(phi->type()));
322 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
323 locals_[phi] = id(phi);
324 }
325 }
326 }
327
328 auto c = emit(ex->index());
329 if (ex->tuple()->num_projs() == 2) {
330 auto [f, t] = ex->tuple()->projs<2>([this](auto def) { return emit(def); });
331 return bb.tail("br i1 {}, label {}, label {}", c, t, f);
332 } else {
333 auto t_c = convert(ex->index()->type());
334 bb.tail("switch {} {}, label {} [ ", t_c, c, emit(ex->tuple()->proj(0)));
335 for (auto i = 1u; i < ex->tuple()->num_projs(); i++)
336 print(bb.tail().back(), "{} {}, label {} ", t_c, std::to_string(i), emit(ex->tuple()->proj(i)));
337 print(bb.tail().back(), "]");
338 }
339 } else if (app->callee()->isa<Bot>()) {
340 return bb.tail("ret ; bottom: unreachable");
341 } else if (auto callee = app->callee()->isa_nom<Lam>(); callee && callee->is_basicblock()) { // ordinary jump
342 for (size_t i = 0, e = callee->num_vars(); i != e; ++i) {
343 if (auto arg = emit_unsafe(app->arg(i)); !arg.empty()) {
344 auto phi = callee->var(i);
345 assert(!match<mem::M>(phi->type()));
346 lam2bb_[callee].phis[phi].emplace_back(arg, id(lam, true));
347 locals_[phi] = id(phi);
348 }
349 }
350 return bb.tail("br label {}", id(callee));
351 } else if (auto longjmp = match<clos::longjmp>(app)) {
352 declare("void @longjmp(i8*, i32) noreturn");
353
354 auto [mem, jbuf, tag] = app->args<3>();
355 emit_unsafe(mem);
356 auto v_jb = emit(jbuf);
357 auto v_tag = emit(tag);
358 bb.tail("call void @longjmp(i8* {}, i32 {})", v_jb, v_tag);
359 return bb.tail("unreachable");
360 } else if (app->callee_type()->is_returning()) { // function call
361 auto v_callee = emit(app->callee());
362
363 std::vector<std::string> args;
364 auto app_args = app->args();
365 for (auto arg : app_args.skip_back())
366 if (auto v_arg = emit_unsafe(arg); !v_arg.empty()) args.emplace_back(convert(arg->type()) + " " + v_arg);
367
368 if (app->args().back()->isa<Bot>()) {
369 // TODO: Perhaps it'd be better to simply -wrap this prior to the BE...
370 assert(convert_ret_pi(app->callee_type()->ret_pi()) == "void");
371 bb.tail("call void {}({, })", v_callee, args);
372 return bb.tail("unreachable");
373 }
374
375 auto ret_lam = app->args().back()->as_nom<Lam>();
376 size_t num_vars = ret_lam->num_vars();
377 size_t n = 0;
378 Array<const Def*> values(num_vars);
379 Array<const Def*> types(num_vars);
380 for (auto var : ret_lam->vars()) {
381 if (match<mem::M>(var->type())) continue;
382 values[n] = var;
383 types[n] = var->type();
384 ++n;
385 }
386
387 if (n == 0) {
388 bb.tail("call void {}({, })", v_callee, args);
389 } else {
390 auto name = "%" + app->unique_name() + ".ret";
391 auto t_ret = convert_ret_pi(ret_lam->type());
392 bb.tail("{} = call {} {}({, })", name, t_ret, v_callee, args);
393
394 for (size_t i = 0, e = ret_lam->num_vars(); i != e; ++i) {
395 auto phi = ret_lam->var(i);
396 if (match<mem::M>(phi->type())) continue;
397
398 auto namei = name;
399 if (e > 2) {
400 namei += '.' + std::to_string(i - 1);
401 bb.tail("{} = extractvalue {} {}, {}", namei, t_ret, name, i - 1);
402 }
403 assert(!match<mem::M>(phi->type()));
404 lam2bb_[ret_lam].phis[phi].emplace_back(namei, id(lam, true));
405 locals_[phi] = id(phi);
406 }
407 }
408
409 return bb.tail("br label {}", id(ret_lam));
410 }

433std::string Emitter::emit_bb(BB& bb, const Def* def) {
434 if (def->isa<Var>()) return {};
435 if (auto lam = def->isa<Lam>()) return id(lam);
436
437 auto name = id(def);
438 std::string op;
439
440 auto emit_tuple = [&](const Def* tuple) {
441 if (is_const(tuple)) {
442 bool is_array = tuple->type()->isa<Arr>();
443
444 std::string s;
445 s += is_array ? "[" : "{";
446 auto sep = "";
447 for (size_t i = 0, n = tuple->num_projs(); i != n; ++i) {
448 auto e = tuple->proj(n, i);
449 if (auto v_elem = emit_unsafe(e); !v_elem.empty()) {
450 auto t_elem = convert(e->type());
451 s += sep + t_elem + " " + v_elem;
452 sep = ", ";
453 }
454 }
455
456 return s += is_array ? "]" : "}";
457 }
458
459 std::string prev = "undef";
460 auto t = convert(tuple->type());
461 for (size_t src = 0, dst = 0, n = tuple->num_projs(); src != n; ++src) {
462 auto e = tuple->proj(n, src);
463 if (auto elem = emit_unsafe(e); !elem.empty()) {
464 auto elem_t = convert(e->type());
465 // TODO: check dst vs src
466 auto namei = name + "." + std::to_string(dst);
467 prev = bb.assign(namei, "insertvalue {} {}, {} {}, {}", t, prev, elem_t, elem, dst);
468 dst++;
469 }
470 }
471 return prev;
472 };
473
474 auto emit_gep_index = [&](const Def* index) {
475 auto v_i = emit(index);
476 auto t_i = convert(index->type());
477
478 if (auto size = Idx::size(index->type())) {
479 if (auto w = Idx::size2bitwidth(size); w && *w < 64) {
480 v_i = bb.assign(name + ".zext",
481 "zext {} {} to i{} ; add one more bit for gep index as it is treated as signed value",
482 t_i, v_i, *w + 1);
483 t_i = "i" + std::to_string(*w + 1);
484 }
485 }
486
487 return std::pair(v_i, t_i);
488 };
489
490 if (auto lit = def->isa<Lit>()) {
491 if (lit->type()->isa<Nat>() || Idx::size(lit->type())) {
492 return std::to_string(lit->get());
493 } else if (auto w = math::isa_f(lit->type())) {
494 std::stringstream s;
495 u64 hex;
496
497 switch (*w) {
498 case 16:
499 s << "0xH" << std::setfill('0') << std::setw(4) << std::right << std::hex << lit->get<u16>();
500 return s.str();
501 case 32: {
502 hex = std::bit_cast<u64>(f64(lit->get<f32>()));
503 break;
504 }
505 case 64: hex = lit->get<u64>(); break;
506 default: unreachable();
507 }
508
509 s << "0x" << std::setfill('0') << std::setw(16) << std::right << std::hex << hex;
510 return s.str();
511 }
512 unreachable();
513 } else if (def->isa<Bot>()) {
514 return "undef";
515 } else if (auto tuple = def->isa<Tuple>()) {
516 return emit_tuple(tuple);
517 } else if (auto pack = def->isa<Pack>()) {
518 if (auto lit = isa_lit(pack->body()); lit && *lit == 0) return "zeroinitializer";
519 return emit_tuple(pack);
520 } else if (auto extract = def->isa<Extract>()) {
521 auto tuple = extract->tuple();
522 auto index = extract->index();
523
524#if 0
525 // TODO this was von closure-conv branch which I need to double-check
526 if (tuple->isa<Var>()) {
527 // computing the index may crash, so we bail out
528 assert(match<core::Mem>(extract->type()) && "only mem-var should not be mapped");
529 return {};
530 }
531#endif
532
533 auto v_tup = emit_unsafe(tuple);
534
535 // this exact location is important: after emitting the tuple -> ordering of mem ops
536 // before emitting the index, as it might be a weird value for mem vars.
537 if (match<mem::M>(extract->type())) return {};
538
539 auto v_idx = emit_unsafe(index);
540
541 if (tuple->num_projs() == 2) {
542 if (match<mem::M>(tuple->proj(2, 0_s)->type())) return v_tup;
543 if (match<mem::M>(tuple->proj(2, 1_s)->type())) return v_tup;
544 }
545
546 auto t_tup = convert(tuple->type());
547 if (isa_lit(index)) {
548 assert(!v_tup.empty());
549 return bb.assign(name, "extractvalue {} {}, {}", t_tup, v_tup, v_idx);
550 } else {
551 auto t_elem = convert(extract->type());
552 auto [v_i, t_i] = emit_gep_index(index);
553
554 print(lam2bb_[entry_].body().emplace_front(),
555 "{}.alloca = alloca {} ; copy to alloca to emulate extract with store + gep + load", name, t_tup);
556 print(bb.body().emplace_back(), "store {} {}, {}* {}.alloca", t_tup, v_tup, t_tup, name);
557 print(bb.body().emplace_back(), "{}.gep = getelementptr inbounds {}, {}* {}.alloca, i64 0, {} {}", name,
558 t_tup, t_tup, name, t_i, v_i);
559 return bb.assign(name, "load {}, {}* {}.gep", t_elem, t_elem, name);
560 }
561 } else if (auto insert = def->isa<Insert>()) {
562 auto v_tuple = emit(insert->tuple());
563 auto v_index = emit(insert->index());
564 auto v_value = emit(insert->value());
565 auto t_tuple = convert(insert->tuple()->type());
566 auto t_value = convert(insert->value()->type());
567 return bb.assign(name, "insertvalue {} {}, {} {}, {}", t_tuple, v_tuple, t_value, v_value, v_index);
568 } else if (auto global = def->isa<Global>()) {
569 auto v_init = emit(global->init());
570 auto [pointee, addr_space] = force<mem::Ptr>(global->type())->args<2>();
571 print(vars_decls_, "{} = global {} {}\n", name, convert(pointee), v_init);
572 return globals_[global] = name;
573 } else if (auto nop = match<core::nop>(def)) {
574 auto [a, b] = nop->args<2>([this](auto def) { return emit(def); });
575
576 switch (nop.id()) {
577 case core::nop::add: op = "add"; break;
578 case core::nop::mul: op = "mul"; break;
579 }
580
581 return bb.assign(name, "{} nsw nuw i64 {}, {}", op, a, b);
582 } else if (auto ncmp = match<core::ncmp>(def)) {
583 auto [a, b] = ncmp->args<2>([this](auto def) { return emit(def); });
584 op = "icmp ";
585
586 switch (ncmp.id()) {
587 // clang-format off
588 case core::ncmp::e: op += "eq" ; break;
589 case core::ncmp::ne: op += "ne" ; break;
590 case core::ncmp::g: op += "ugt"; break;
591 case core::ncmp::ge: op += "uge"; break;
592 case core::ncmp::l: op += "ult"; break;
593 case core::ncmp::le: op += "ule"; break;
594 // clang-format on
595 default: unreachable();
596 }
597
598 return bb.assign(name, "{} i64 {}, {}", op, a, b);
599 } else if (auto bit1 = match<core::bit1>(def)) {
600 assert(bit1.id() == core::bit1::neg);
601 auto x = emit(bit1->arg());
602 auto t = convert(bit1->type());
603 return bb.assign(name, "xor {} -1, {}", t, x);
604 } else if (auto bit2 = match<core::bit2>(def)) {
605 auto [a, b] = bit2->args<2>([this](auto def) { return emit(def); });
606 auto t = convert(bit2->type());
607
608 auto neg = [&](std::string_view x) { return bb.assign(name + ".neg", "xor {} -1, {}", t, x); };
609
610 switch (bit2.id()) {
