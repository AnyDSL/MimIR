126template<icmp cmp, nat_t w>
127struct Fold<icmp, cmp, w> {
128 inline static Res run(u64 a, u64 b) {
129 using T = w2u<w>;
130 auto x = get<T>(a), y = get<T>(b);
131 bool result = false;
132 auto pm = !(x >> T(w - 1)) && (y >> T(w - 1));
133 auto mp = (x >> T(w - 1)) && !(y >> T(w - 1));
134 result |= ((cmp & icmp::Xygle) != icmp::f) && pm;
135 result |= ((cmp & icmp::xYgle) != icmp::f) && mp;
136 result |= ((cmp & icmp::xyGle) != icmp::f) && x > y && !mp;
137 result |= ((cmp & icmp::xygLe) != icmp::f) && x < y && !pm;
138 result |= ((cmp & icmp::xyglE) != icmp::f) && x == y;
139 return result;
140 }
141};

234template<icmp sub>
235const Def* normalize_icmp(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
236 auto& world = type->world();
237 auto callee = c->as<App>();
238 auto [a, b] = arg->projs<2>();
239
240 if (auto result = fold<icmp, sub>(world, type, callee, a, b, dbg)) return result;
241 if (sub == icmp::f) return world.lit_false();
242 if (sub == icmp::t) return world.lit_true();
243 if (a == b) {
244 if (sub == icmp::e) return world.lit_true();
245 if (sub == icmp::ne) return world.lit_false();
246 }
247
248 return world.raw_app(callee, {a, b}, dbg);
249}

431template<wrap sub>
432const Def* normalize_wrap(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
433 auto& world = type->world();
434 auto callee = c->as<App>();
435 auto [a, b] = arg->projs<2>();
436 auto [m, w] = callee->args<2>(isa_lit<nat_t>); // mode and width
437
438 if (auto result = fold<wrap, sub, true>(world, type, callee, a, b, dbg)) return result;
439
440 // clang-format off
441 if (auto la = a->isa<Lit>()) {
442 if (la == world.lit_int(*w, 0)) {
443 switch (sub) {
444 case wrap::add: return b; // 0 + b -> b
445 case wrap::sub: break;
446 case wrap::mul: return la; // 0 * b -> 0
447 case wrap::shl: return la; // 0 << b -> 0
448 default: unreachable();
449 }
450 }
451
452 if (la == world.lit_int(*w, 1)) {
453 switch (sub) {
454 case wrap::add: break;
455 case wrap::sub: break;
456 case wrap::mul: return b; // 1 * b -> b
457 case wrap::shl: break;
458 default: unreachable();
459 }
460 }
461 }
462
463 if (auto lb = b->isa<Lit>()) {
464 if (lb == world.lit_int(*w, 0)) {
465 switch (sub) {
466 case wrap::sub: return a; // a - 0 -> a
467 case wrap::shl: return a; // a >> 0 -> a
468 default: unreachable();
469 // add, mul are commutative, the literal has been normalized to the left
470 }
471 }
472
473 if (sub == wrap::sub)
474 return op(wrap::add, *m, a, world.lit_int_mod(*w, ~lb->get() + 1_u64)); // a - lb -> a + (~lb + 1)
475 else if (sub == wrap::shl && lb->get() > *w)
476 return world.bot(type, dbg);
477 }
478
479 if (a == b) {
480 switch (sub) {
481 case wrap::add: return op(wrap::mul, *m, world.lit_int(*w, 2), a, dbg); // a + a -> 2 * a
482 case wrap::sub: return world.lit_int(*w, 0); // a - a -> 0
483 case wrap::mul: break;
484 case wrap::shl: break;
485 default: unreachable();
486 }
487 }
488 // clang-format on
489
490 if (auto res = reassociate<wrap>(sub, world, callee, a, b, dbg)) return res;
491
492 return world.raw_app(callee, {a, b}, dbg);
493}

495template<div op>
496const Def* normalize_div(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
497 auto& world = type->world();
498 auto callee = c->as<App>();
499 auto [mem, a, b] = arg->projs<3>();
500 auto w = isa_lit(callee->arg());
501 type = type->as<Sigma>()->op(1); // peel of actual type
502 auto make_res = [&, mem = mem](const Def* res) { return world.tuple({mem, res}, dbg); };
503
504 if (auto result = normalize::fold<div, op>(world, type, callee, a, b, dbg)) return make_res(result);
505
506 if (auto la = a->isa<Lit>()) {
507 if (la == world.lit_int(*w, 0)) return make_res(la); // 0 / b -> 0 and 0 % b -> 0
508 }
509
510 if (auto lb = b->isa<Lit>()) {
511 if (lb == world.lit_int(*w, 0)) return make_res(world.bot(type)); // a / 0 -> and a % 0 ->
512
513 if (lb == world.lit_int(*w, 1)) {
514 switch (op) {
515 case div::sdiv: return make_res(a); // a / 1 -> a
516 case div::udiv: return make_res(a); // a / 1 -> a
517 case div::srem: return make_res(world.lit_int(*w, 0)); // a % 1 -> 0
518 case div::urem: return make_res(world.lit_int(*w, 0)); // a % 1 -> 0
519 default: unreachable();
520 }
521 }
522 }
523
524 if (a == b) {
525 switch (op) {
526 case div::sdiv: return make_res(world.lit_int(*w, 1)); // a / a -> 1
527 case div::udiv: return make_res(world.lit_int(*w, 1)); // a / a -> 1
528 case div::srem: return make_res(world.lit_int(*w, 0)); // a % a -> 0
529 case div::urem: return make_res(world.lit_int(*w, 0)); // a % a -> 0
530 default: unreachable();
