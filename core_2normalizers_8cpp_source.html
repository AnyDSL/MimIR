96/// @attention Note that @p a and @p b are passed by reference as fold also commutes if possible. @sa commute().
97template<class Id, Id id, bool isa_wrap = std::is_same_v<Id, wrap>>
98static const Def* fold(World& world, const Def* type, const App* callee, const Def*& a, const Def*& b, const Def* dbg) {
99 static constexpr int min_w = std::is_same_v<Id, rop> || std::is_same_v<Id, rcmp> ? 16 : 1;
100 auto la = a->isa<Lit>(), lb = b->isa<Lit>();
101
102 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type, dbg);
103
104 if (la && lb) {
105 nat_t width;
106 [[maybe_unused]] bool nsw = false, nuw = false;
107 if constexpr (std::is_same_v<Id, wrap>) {
108 auto [mode, w] = callee->args<2>(as_lit<nat_t>);
109 nsw = mode & WMode::nsw;
110 nuw = mode & WMode::nuw;
111 width = w;
112 } else if (auto idx = a->type()->isa<Idx>()) {
113 width = as_lit(idx->size());
114 } else {
115 width = as_lit(force<Real>(a->type())->arg());
116 }
117
118 if (is_int<Id>()) width = *size2bitwidth(width);
119
120 Res res;
121 switch (width) {
122#define CODE(i) \
123 case i: \
124 if constexpr (i >= min_w) { \
125 if constexpr (isa_wrap) \
126 res = Fold<Id, id, i>::run(la->get(), lb->get(), nsw, nuw); \
127 else \
128 res = Fold<Id, id, i>::run(la->get(), lb->get()); \
129 } \
130 break;
131 THORIN_1_8_16_32_64(CODE)
132#undef CODE
133 default: unreachable();
134 }
135
136 return res ? world.lit(type, *res, dbg) : world.bot(type, dbg);
137 }
138
139 commute(id, a, b);
140 return nullptr;
141}

212template<icmp cmp, nat_t w>
213struct Fold<icmp, cmp, w> {
214 inline static Res run(u64 a, u64 b) {
215 using T = w2u<w>;
216 auto x = get<T>(a), y = get<T>(b);
217 bool result = false;
218 auto pm = !(x >> T(w - 1)) && (y >> T(w - 1));
219 auto mp = (x >> T(w - 1)) && !(y >> T(w - 1));
220 result |= ((cmp & icmp::Xygle) != icmp::f) && pm;
221 result |= ((cmp & icmp::xYgle) != icmp::f) && mp;
222 result |= ((cmp & icmp::xyGle) != icmp::f) && x > y && !mp;
223 result |= ((cmp & icmp::xygLe) != icmp::f) && x < y && !pm;
224 result |= ((cmp & icmp::xyglE) != icmp::f) && x == y;
225 return result;
226 }
227};

254template<class Id>
255static const Def* merge_cmps(std::array<std::array<u64, 2>, 2> tab, const Def* a, const Def* b, const Def* dbg) {
256 static_assert(sizeof(sub_t) == 1, "if this ever changes, please adjust the logic below");
257 static constexpr size_t num_bits = std::bit_width(Axiom::Num<Id> - 1_u64);
258
259 auto a_cmp = match<Id>(a);
260 auto b_cmp = match<Id>(b);
261
262 if (a_cmp && b_cmp && a_cmp->args() == b_cmp->args()) {
263 // push sub bits of a_cmp and b_cmp through truth table
264 sub_t res = 0;
265 sub_t a_sub = a_cmp.sub();
266 sub_t b_sub = b_cmp.sub();
267 for (size_t i = 0; i != num_bits; ++i, res >>= 1, a_sub >>= 1, b_sub >>= 1)
268 res |= tab[a_sub & 1][b_sub & 1] << 7_u8;
269 res >>= (7_u8 - u8(num_bits));
270
271 if constexpr (std::is_same_v<Id, rcmp>)
272 return op(rcmp(res), /*rmode*/ a_cmp->decurry()->arg(0), a_cmp->arg(0), a_cmp->arg(1), dbg);
273 else
274 return op(icmp(Axiom::Base<icmp> | res), a_cmp->arg(0), a_cmp->arg(1), dbg);
275 }
276
277 return nullptr;
278}

280/// Reassociates @p a und @p b according to following rules.
281/// We use the following naming convention while literals are prefixed with an 'l':
282/// ```
283/// a op b
284/// (x op y) op (z op w)
285///
286/// (1) la op (lz op w) -> (la op lz) op w
287/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
288/// (3) a op (lz op w) -> lz op (a op w)
289/// (4) (lx op y) op b -> lx op (y op b)
290/// ```
291template<class Id>
292static const Def*
293reassociate(Id id, World& /*world*/, [[maybe_unused]] const App* ab, const Def* a, const Def* b, const Def* dbg) {
294 if (!is_associative(id)) return nullptr;
295
296 auto la = a->isa<Lit>();
297 auto xy = match<Id>(id, a);
298 auto zw = match<Id>(id, b);
299 auto lx = xy ? xy->arg(0)->template isa<Lit>() : nullptr;
300 auto lz = zw ? zw->arg(0)->template isa<Lit>() : nullptr;
301 auto y = xy ? xy->arg(1) : nullptr;
302 auto w = zw ? zw->arg(1) : nullptr;
303
304 std::function<const Def*(const Def*, const Def*)> make_op;
305
306 if constexpr (std::is_same_v<Id, rop>) {
307 // build rmode for all new ops by using the least upper bound of all involved apps
308 nat_t rmode = RMode::bot;
309 auto check_mode = [&](const App* app) {
310 auto app_m = isa_lit(app->arg(0));
311 if (!app_m || !(*app_m & RMode::reassoc)) return false;
312 rmode &= *app_m; // least upper bound
313 return true;
314 };
315
316 if (!check_mode(ab)) return nullptr;
317 if (lx && !check_mode(xy->decurry())) return nullptr;
318 if (lz && !check_mode(zw->decurry())) return nullptr;
319
320 make_op = [&](const Def* a, const Def* b) { return op(id, rmode, a, b, dbg); };
321 } else if constexpr (std::is_same_v<Id, wrap>) {
322 // if we reassociate Wraps, we have to forget about nsw/nuw
323 make_op = [&](const Def* a, const Def* b) { return op(id, WMode::none, a, b, dbg); };
324 } else {
325 make_op = [&](const Def* a, const Def* b) { return op(id, a, b, dbg); };
326 }
327
328 if (la && lz) return make_op(make_op(la, lz), w); // (1)
329 if (lx && lz) return make_op(make_op(lx, lz), make_op(y, w)); // (2)
330 if (lz) return make_op(lz, make_op(a, w)); // (3)
331 if (lx) return make_op(lx, make_op(y, b)); // (4)
332
333 return nullptr;
334}

336template<rop id>
337const Def* normalize_rop(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
338 auto& world = type->world();
339 auto callee = c->as<App>();
340 auto [a, b] = arg->projs<2>();
341 auto [m, w] = callee->args<2>(isa_lit<nat_t>); // mode and width
342
343 if (auto result = fold<rop, id>(world, type, callee, a, b, dbg)) return result;
344
345 // clang-format off
346 // TODO check rmode properly
347 if (m && *m == RMode::fast) {
348 if (auto la = a->isa<Lit>()) {
349 if (la == lit_real(world, *w, 0.0)) {
350 switch (id) {
351 case rop::add: return b; // 0 + b -> b
352 case rop::sub: break;
353 case rop::mul: return la; // 0 * b -> 0
354 case rop::div: return la; // 0 / b -> 0
355 case rop::rem: return la; // 0 % b -> 0
356 }
357 }
358
359 if (la == lit_real(world, *w, 1.0)) {
360 switch (id) {
361 case rop::add: break;
362 case rop::sub: break;
363 case rop::mul: return b; // 1 * b -> b
364 case rop::div: break;
365 case rop::rem: break;
366 }
367 }
368 }
369
370 if (auto lb = b->isa<Lit>()) {
371 if (lb == lit_real(world, *w, 0.0)) {
372 switch (id) {
373 case rop::sub: return a; // a - 0 -> a
374 case rop::div: break;
375 case rop::rem: break;
376 default: unreachable();
377 // add, mul are commutative, the literal has been normalized to the left
378 }
379 }
380 }
381
382 if (a == b) {
383 switch (id) {
384 case rop::add: return core::op(rop::mul, lit_real(world, *w, 2.0), a, dbg); // a + a -> 2 * a
385 case rop::sub: return lit_real(world, *w, 0.0); // a - a -> 0
386 case rop::mul: break;
387 case rop::div: return lit_real(world, *w, 1.0); // a / a -> 1
388 case rop::rem: break;
389 }
390 }
391 }
392 // clang-format on
393
394 if (auto res = reassociate<rop>(id, world, callee, a, b, dbg)) return res;
395
396 return world.raw_app(callee, {a, b}, dbg);
397}

448template<bit2 id>
449const Def* normalize_bit2(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
450 auto& world = type->world();
451 auto callee = c->as<App>();
452 auto [a, b] = arg->projs<2>();
453 auto w = isa_lit(callee->arg());
454
455 commute(id, a, b);
456
457 auto tab = make_truth_table(id);
458 if (auto res = merge_cmps<icmp>(tab, a, b, dbg)) return res;
459 if (auto res = merge_cmps<rcmp>(tab, a, b, dbg)) return res;
460
461 auto la = isa_lit(a);
462 auto lb = isa_lit(b);
463
464 // clang-format off
465 switch (id) {
466 case bit2:: f: return world.lit_idx(*w, 0);
467 case bit2:: t: return world.lit_idx(*w, *w-1_u64);
468 case bit2:: a: return a;
469 case bit2:: b: return b;
470 case bit2:: na: return op_negate(a, dbg);
471 case bit2:: nb: return op_negate(b, dbg);
472 case bit2:: ciff: return op(bit2:: iff, b, a, dbg);
473 case bit2::nciff: return op(bit2::niff, b, a, dbg);
474 default: break;
475 }
476
477 if (la && lb) {
478 switch (id) {
479 case bit2::_and: return world.lit_idx (*w, *la & *lb);
480 case bit2:: _or: return world.lit_idx (*w, *la | *lb);
481 case bit2::_xor: return world.lit_idx (*w, *la ^ *lb);
482 case bit2::nand: return world.lit_idx_mod(*w, ~(*la & *lb));
483 case bit2:: nor: return world.lit_idx_mod(*w, ~(*la | *lb));
484 case bit2::nxor: return world.lit_idx_mod(*w, ~(*la ^ *lb));
485 case bit2:: iff: return world.lit_idx_mod(*w, ~ *la | *lb);
486 case bit2::niff: return world.lit_idx (*w, *la & ~*lb);
487 default: unreachable();
488 }
489 }
490
491 auto unary = [&](bool x, bool y, const Def* a) -> const Def* {
492 if (!x && !y) return world.lit_idx(*w, 0);
493 if ( x && y) return world.lit_idx(*w, *w-1_u64);
494 if (!x && y) return a;
495 if ( x && !y && id != bit2::_xor) return op_negate(a, dbg);
496 return nullptr;
497 };
498 // clang-format on
499
500 if (is_commutative(id) && a == b) {
501 if (auto res = unary(tab[0][0], tab[1][1], a)) return res;
502 }
503
504 if (la) {
505 if (*la == 0) {
506 if (auto res = unary(tab[0][0], tab[0][1], b)) return res;
507 } else if (*la == *w - 1_u64) {
508 if (auto res = unary(tab[1][0], tab[1][1], b)) return res;
509 }
510 }
511
512 if (lb) {
513 if (*lb == 0) {
514 if (auto res = unary(tab[0][0], tab[1][0], b)) return res;
515 } else if (*lb == *w - 1_u64) {
516 if (auto res = unary(tab[0][1], tab[1][1], b)) return res;
517 }
518 }
519
520 if (auto res = reassociate<bit2>(id, world, callee, a, b, dbg)) return res;
521
522 return world.raw_app(callee, {a, b}, dbg);
523}

557template<wrap id>
558const Def* normalize_wrap(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
559 auto& world = type->world();
560 auto callee = c->as<App>();
561 auto [a, b] = arg->projs<2>();
562 auto [m, w] = callee->args<2>(isa_lit<nat_t>); // mode and width
563
564 if (auto result = fold<wrap, id, true>(world, type, callee, a, b, dbg)) return result;
565
566 // clang-format off
567 if (auto la = a->isa<Lit>()) {
568 if (la == world.lit_idx(*w, 0)) {
569 switch (id) {
570 case wrap::add: return b; // 0 + b -> b
571 case wrap::sub: break;
572 case wrap::mul: return la; // 0 * b -> 0
573 case wrap::shl: return la; // 0 << b -> 0
574 }
575 }
576
577 if (la == world.lit_idx(*w, 1)) {
578 switch (id) {
579 case wrap::add: break;
580 case wrap::sub: break;
581 case wrap::mul: return b; // 1 * b -> b
582 case wrap::shl: break;
583 }
584 }
585 }
586
587 if (auto lb = b->isa<Lit>()) {
588 if (lb == world.lit_idx(*w, 0)) {
589 switch (id) {
590 case wrap::sub: return a; // a - 0 -> a
591 case wrap::shl: return a; // a >> 0 -> a
592 default: unreachable();
593 // add, mul are commutative, the literal has been normalized to the left
594 }
595 }
596
597 if (id == wrap::sub)
598 return op(wrap::add, *m, a, world.lit_idx_mod(*w, ~lb->get() + 1_u64)); // a - lb -> a + (~lb + 1)
599 else if (id == wrap::shl && lb->get() > *w)
600 return world.bot(type, dbg);
601 }
602
603 if (a == b) {
604 switch (id) {
605 case wrap::add: return op(wrap::mul, *m, world.lit_idx(*w, 2), a, dbg); // a + a -> 2 * a
606 case wrap::sub: return world.lit_idx(*w, 0); // a - a -> 0
607 case wrap::mul: break;
608 case wrap::shl: break;
609 }
610 }
611 // clang-format on
612
613 if (auto res = reassociate<wrap>(id, world, callee, a, b, dbg)) return res;
614
615 return world.raw_app(callee, {a, b}, dbg);
616}

619const Def* normalize_div(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
620 auto& world = type->world();
621 auto callee = c->as<App>();
622 auto [mem, a, b] = arg->projs<3>();
623 auto w = isa_lit(callee->arg());
624 type = type->as<Sigma>()->op(1); // peel of actual type
625 auto make_res = [&, mem = mem](const Def* res) { return world.tuple({mem, res}, dbg); };
626

628
629 if (auto la = a->isa<Lit>()) {
630 if (la == world.lit_idx(*w, 0)) return make_res(la); // 0 / b -> 0 and 0 % b -> 0
631 }
632
633 if (auto lb = b->isa<Lit>()) {
634 if (lb == world.lit_idx(*w, 0)) return make_res(world.bot(type)); // a / 0 -> and a % 0 ->
635
636 if (lb == world.lit_idx(*w, 1)) {
637 switch (id) {
638 case div::sdiv: return make_res(a); // a / 1 -> a
639 case div::udiv: return make_res(a); // a / 1 -> a
640 case div::srem: return make_res(world.lit_idx(*w, 0)); // a % 1 -> 0
641 case div::urem: return make_res(world.lit_idx(*w, 0)); // a % 1 -> 0
642 }
643 }
644 }
645
646 if (a == b) {
647 switch (id) {
648 case div::sdiv: return make_res(world.lit_idx(*w, 1)); // a / a -> 1
649 case div::udiv: return make_res(world.lit_idx(*w, 1)); // a / a -> 1
650 case div::srem: return make_res(world.lit_idx(*w, 0)); // a % a -> 0
651 case div::urem: return make_res(world.lit_idx(*w, 0)); // a % a -> 0
652 }
653 }
654
655 return world.raw_app(callee, {mem, a, b}, dbg);

666template<nat_t min_sw, nat_t min_dw, conv id>
667static const Def* fold_conv(const Def* dst_type, const App* callee, const Def* src, const Def* dbg) {
668 auto& world = dst_type->world();
669 if (src->isa<Bot>()) return world.bot(dst_type, dbg);
670
671 auto [lit_dw, lit_sw] = callee->args<2>(isa_lit<nat_t>);
672 auto lit_src = src->isa<Lit>();
673 if (lit_src && lit_dw && lit_sw) {
674 if (id == conv::u2u) {
675 if (*lit_dw == 0) return world.lit(dst_type, as_lit(lit_src));
676 return world.lit(dst_type, as_lit(lit_src) % *lit_dw);
677 }
678
679 if (src->type()->isa<Idx>()) *lit_sw = *size2bitwidth(*lit_sw);
680 if (dst_type->isa<Idx>()) *lit_dw = *size2bitwidth(*lit_dw);
681
682 Res res;
683#define CODE(sw, dw) \
684 else if (*lit_dw == dw && *lit_sw == sw) { \
685 if constexpr (dw >= min_dw && sw >= min_sw) res = FoldConv<id, dw, sw>::run(lit_src->get()); \
686 }
687 if (false) {}
688 TABLE(CODE)
689#undef CODE
690 if (res) return world.lit(dst_type, *res, dbg);
691 return world.bot(dst_type, dbg);
692 }
693
694 return nullptr;
695}

697template<conv id>
698const Def* normalize_conv(const Def* dst_type, const Def* c, const Def* src, const Def* dbg) {
699 auto& world = dst_type->world();
700 auto callee = c->as<App>();
701
702 static constexpr auto min_sw = id == conv::r2s || id == conv::r2u || id == conv::r2r ? 16 : 1;
703 static constexpr auto min_dw = id == conv::s2r || id == conv::u2r || id == conv::r2r ? 16 : 1;
704 if (auto result = fold_conv<min_sw, min_dw, id>(dst_type, callee, src, dbg)) return result;
705
706 auto [dw, sw] = callee->args<2>(isa_lit<nat_t>);
707 if (sw == dw && dst_type == src->type()) return src;
708
709 if constexpr (id == conv::s2s) {
710 if (sw && dw && *sw < *dw) return core::op(conv::u2u, dst_type, src, dbg);
711 }
712
713 return world.raw_app(callee, src, dbg);

734inline u64 pad(u64 offset, u64 align) {
735 auto mod = offset % align;
736 if (mod != 0) offset += align - mod;
737 return offset;
738}

740// TODO this currently hard-codes x86_64 ABI
741// TODO in contrast to C, we might want to give singleton types like '.Idx 1' or '[]' a size of 0 and simply nuke each
742// and every occurance of these types in a later phase
743// TODO Pi and others
744template<trait id>
745const Def* normalize_trait(const Def*, const Def* callee, const Def* type, const Def* dbg) {
746 auto& world = type->world();
747 if (auto ptr = match<mem::Ptr>(type)) {
748 return world.lit_nat(8);
749 } else if (type->isa<Pi>()) {
750 return world.lit_nat(8); // Gets lowered to function ptr
751 } else if (auto idx = type->isa<Idx>()) {
752 if (idx->size()->isa<Top>()) return world.lit_nat(8);
753 if (auto w = isa_lit(idx->size())) {
754 if (*w == 0) return world.lit_nat(8);
755 if (*w <= 0x0000'0000'0000'0100_u64) return world.lit_nat(1);
756 if (*w <= 0x0000'0000'0001'0000_u64) return world.lit_nat(2);
757 if (*w <= 0x0000'0001'0000'0000_u64) return world.lit_nat(4);
758 return world.lit_nat(8);
759 }
760 } else if (auto real = match<Real>(type)) {
761 if (auto w = isa_lit(real->arg())) {
762 switch (*w) {
763 case 16: return world.lit_nat(2);
764 case 32: return world.lit_nat(4);
765 case 64: return world.lit_nat(8);
766 default: unreachable();
767 }
768 }
769 } else if (type->isa<Sigma>() || type->isa<Meet>()) {
770 u64 offset = 0;
771 u64 align = 1;
772 for (auto t : type->ops()) {
773 auto a = isa_lit(core::op(trait::align, t));
774 auto s = isa_lit(core::op(trait::size, t));
775 if (!a || !s) goto out;
776
777 align = std::max(align, *a);
778 offset = pad(offset, *a) + *s;
779 }
780
781 offset = pad(offset, align);
782 u64 size = std::max(1_u64, offset);
783
784 switch (id) {
785 case trait::align: return world.lit_nat(align);
786 case trait::size: return world.lit_nat(size);
787 }
788 } else if (auto arr = type->isa_structural<Arr>()) {
789 auto align = core::op(trait::align, arr->body());
790 if constexpr (id == trait::align) return align;
791
792 if (auto b = isa_lit(core::op(trait::size, arr->body()))) {
793 auto i64_t = world.type_int_(64);
794 auto s = op_bitcast(i64_t, arr->shape());
795 auto mul = core::op(wrap::mul, WMode::nsw | WMode::nuw, world.lit_idx(i64_t, *b), s);
796 return op_bitcast(world.type_nat(), mul);
797 }
798 } else if (auto join = type->isa<Join>()) {
799 if (auto sigma = convert(join)) return core::op(id, sigma, dbg);
800 }
801
802out:
803 return world.raw_app(callee, type, dbg);
804}
