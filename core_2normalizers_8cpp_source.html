38template<class Id, Id id, nat_t w>
39Res fold(u64 a, u64 b, [[maybe_unused]] bool nsw, [[maybe_unused]] bool nuw) {
40 using ST = w2s<w>;
41 using UT = w2u<w>;
42 auto s = thorin::bitcast<ST>(a), t = thorin::bitcast<ST>(b);
43 auto u = thorin::bitcast<UT>(a), v = thorin::bitcast<UT>(b);
44
45 if constexpr (std::is_same_v<Id, wrap>) {
46 if constexpr (id == wrap::add) {
47 auto res = u + v;
48 if (nuw && res < u) return {};
49 // TODO nsw
50 return res;
51 } else if constexpr (id == wrap::sub) {
52 auto res = u - v;
53 // TODO nsw
54 return res;
55 } else if constexpr (id == wrap::mul) {
56 if constexpr (std::is_same_v<UT, bool>)
57 return UT(u & v);
58 else
59 return UT(u * v);
60 } else if constexpr (id == wrap::shl) {
61 if (b >= w) return {};
62 decltype(u) res;
63 if constexpr (std::is_same_v<UT, bool>)
64 res = bool(u64(u) << u64(v));
65 else
66 res = u << v;
67 if (nuw && res < u) return {};
68 if (nsw && get_sign(u) != get_sign(res)) return {};
69 return res;
70 } else {
71 []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
72 }
73 } else if constexpr (std::is_same_v<Id, shr>) {
74 if (b >= w) return {};
75 if constexpr (false) {}
76 else if constexpr (id == shr::a) return s >> t;
77 else if constexpr (id == shr::l) return u >> v;
78 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
79 } else if constexpr (std::is_same_v<Id, div>) {
80 if (b == 0) return {};
81 if constexpr (false) {}
82 else if constexpr (id == div::sdiv) return s / t;
83 else if constexpr (id == div::udiv) return u / v;
84 else if constexpr (id == div::srem) return s % t;
85 else if constexpr (id == div::urem) return u % v;
86 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
87 } else if constexpr (std::is_same_v<Id, icmp>) {
88 bool res = false;
89 auto pm = !(u >> UT(w - 1)) && (v >> UT(w - 1));
90 auto mp = (u >> UT(w - 1)) && !(v >> UT(w - 1));
91 res |= ((id & icmp::Xygle) != icmp::f) && pm;
92 res |= ((id & icmp::xYgle) != icmp::f) && mp;
93 res |= ((id & icmp::xyGle) != icmp::f) && u > v && !mp;
94 res |= ((id & icmp::xygLe) != icmp::f) && u < v && !pm;
95 res |= ((id & icmp::xyglE) != icmp::f) && u == v;
96 return res;
97 } else {
98 []<bool flag = false>() { static_assert(flag, "missing tag"); }();
99 }
100}

103/// @attention Note that @p a and @p b are passed by reference as fold also commutes if possible. @sa commute().
104template<class Id, Id id>
105static const Def*
106fold(World& world, const Def* type, const Def*& a, const Def*& b, const Def* dbg, const Def* mode = {}) {
107 auto la = a->isa<Lit>(), lb = b->isa<Lit>();
108
109 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type, dbg);
110
111 if (la && lb) {
112 auto size = as_lit(Idx::size(a->type()));
113 auto width = Idx::size2bitwidth(size);
114 bool nsw = false, nuw = false;
115 if constexpr (std::is_same_v<Id, wrap>) {
116 auto m = as_lit(mode);
117 nsw = m & Mode::nsw;
118 nuw = m & Mode::nuw;
119 }
120
121 Res res;
122 switch (width) {
123#define CODE(i) \
124 case i: res = fold<Id, id, i>(la->get(), lb->get(), nsw, nuw); break;
125 THORIN_1_8_16_32_64(CODE)
126#undef CODE
127 default: unreachable();
128 }
129
130 return res ? world.lit(type, *res, dbg) : world.bot(type, dbg);
131 }
132
133 commute(id, a, b);
134 return nullptr;
135}

137/// Reassociates @p a und @p b according to following rules.
138/// We use the following naming convention while literals are prefixed with an 'l':
139/// ```
140/// a op b
141/// (x op y) op (z op w)
142///
143/// (1) la op (lz op w) -> (la op lz) op w
144/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
145/// (3) a op (lz op w) -> lz op (a op w)
146/// (4) (lx op y) op b -> lx op (y op b)
147/// ```
148template<class Id>
149static const Def*
150reassociate(Id id, World& /*world*/, [[maybe_unused]] const App* ab, const Def* a, const Def* b, const Def* dbg) {
151 if (!is_associative(id)) return nullptr;
152
153 auto la = a->isa<Lit>();
154 auto xy = match<Id>(id, a);
155 auto zw = match<Id>(id, b);
156 auto lx = xy ? xy->arg(0)->template isa<Lit>() : nullptr;
157 auto lz = zw ? zw->arg(0)->template isa<Lit>() : nullptr;
158 auto y = xy ? xy->arg(1) : nullptr;
159 auto w = zw ? zw->arg(1) : nullptr;
160
161 std::function<const Def*(const Def*, const Def*)> make_op;
162
163 if constexpr (std::is_same_v<Id, wrap>) {
164 // if we reassociate Wraps, we have to forget about nsw/nuw
165 make_op = [&](const Def* a, const Def* b) { return op(id, Mode::none, a, b, dbg); };
166 } else {
167 make_op = [&](const Def* a, const Def* b) { return op(id, a, b, dbg); };
168 }
169
170 if (la && lz) return make_op(make_op(la, lz), w); // (1)
171 if (lx && lz) return make_op(make_op(lx, lz), make_op(y, w)); // (2)
172 if (lz) return make_op(lz, make_op(a, w)); // (3)
173 if (lx) return make_op(lx, make_op(y, b)); // (4)
174
175 return nullptr;
176}

261template<class Id>
262static const Def* merge_cmps(std::array<std::array<u64, 2>, 2> tab, const Def* a, const Def* b, const Def* dbg) {
263 static_assert(sizeof(sub_t) == 1, "if this ever changes, please adjust the logic below");
264 static constexpr size_t num_bits = std::bit_width(Axiom::Num<Id> - 1_u64);
265
266 auto a_cmp = match<Id>(a);
267 auto b_cmp = match<Id>(b);
268
269 if (a_cmp && b_cmp && a_cmp->arg() == b_cmp->arg()) {
270 // push sub bits of a_cmp and b_cmp through truth table
271 sub_t res = 0;
272 sub_t a_sub = a_cmp.sub();
273 sub_t b_sub = b_cmp.sub();
274 for (size_t i = 0; i != num_bits; ++i, res >>= 1, a_sub >>= 1, b_sub >>= 1)
275 res |= tab[a_sub & 1][b_sub & 1] << 7_u8;
276 res >>= (7_u8 - u8(num_bits));
277
278 if constexpr (std::is_same_v<Id, math::cmp>)
279 return op(math::cmp(res), /*rmode*/ a_cmp->decurry()->arg(0), a_cmp->arg(0), a_cmp->arg(1), dbg);
280 else
281 return op(icmp(Axiom::Base<icmp> | res), a_cmp->arg(0), a_cmp->arg(1), dbg);
282 }
283
284 return nullptr;
285}

287template<bit2 id>
288const Def* normalize_bit2(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
289 auto& world = type->world();
290 auto callee = c->as<App>();
291 auto [a, b] = arg->projs<2>();
292 auto s = isa_lit(callee->arg());
293
294 commute(id, a, b);
295
296 auto tab = make_truth_table(id);
297 if (auto res = merge_cmps<icmp>(tab, a, b, dbg)) return res;
298 if (auto res = merge_cmps<math::cmp>(tab, a, b, dbg)) return res;
299
300 auto la = isa_lit(a);
301 auto lb = isa_lit(b);
302
303 // clang-format off
304 switch (id) {
305 case bit2:: f: return world.lit(type, 0);
306 case bit2:: t: if (s) return world.lit(type, *s-1_u64); break;
307 case bit2:: a: return a;
308 case bit2:: b: return b;
309 case bit2:: na: return op_negate(a, dbg);
310 case bit2:: nb: return op_negate(b, dbg);
311 case bit2:: ciff: return op(bit2:: iff, b, a, dbg);
312 case bit2::nciff: return op(bit2::niff, b, a, dbg);
313 default: break;
314 }
315
316 if (la && lb) {
317 switch (id) {
318 case bit2::and_: return world.lit_idx (*s, *la & *lb);
319 case bit2:: or_: return world.lit_idx (*s, *la | *lb);
320 case bit2::xor_: return world.lit_idx (*s, *la ^ *lb);
321 case bit2::nand: return world.lit_idx_mod(*s, ~(*la & *lb));
322 case bit2:: nor: return world.lit_idx_mod(*s, ~(*la | *lb));
323 case bit2::nxor: return world.lit_idx_mod(*s, ~(*la ^ *lb));
324 case bit2:: iff: return world.lit_idx_mod(*s, ~ *la | *lb);
325 case bit2::niff: return world.lit_idx (*s, *la & ~*lb);
326 default: unreachable();
327 }
328 }

330 auto unary = [&](bool x, bool y, const Def* a) -> const Def* {
331 if (!x && !y) return world.lit(type, 0);
332 if ( x && y) return s ? world.lit(type, *s-1_u64) : nullptr;
333 if (!x && y) return a;
334 if ( x && !y && id != bit2::xor_) return op_negate(a, dbg);
335 return nullptr;
336 };
337 // clang-format on
338
339 if (is_commutative(id) && a == b) {
340 if (auto res = unary(tab[0][0], tab[1][1], a)) return res;
341 }
342
343 if (la) {
344 if (*la == 0) {
345 if (auto res = unary(tab[0][0], tab[0][1], b)) return res;
346 } else if (*la == *s - 1_u64) {
347 if (auto res = unary(tab[1][0], tab[1][1], b)) return res;
348 }
349 }
350
351 if (lb) {
352 if (*lb == 0) {
353 if (auto res = unary(tab[0][0], tab[1][0], a)) return res;
354 } else if (*lb == *s - 1_u64) {
355 if (auto res = unary(tab[0][1], tab[1][1], a)) return res;
356 }
357 }
358
359 if (auto res = reassociate<bit2>(id, world, callee, a, b, dbg)) return res;
360
361 return world.raw_app(callee, {a, b}, dbg);
362}

364template<shr id>
365const Def* normalize_shr(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
366 auto& world = type->world();
367 auto callee = c->as<App>();
368 auto [a, b] = arg->projs<2>();
369 auto s = Idx::size(arg->type());
370 auto ls = isa_lit(s);
371
372 if (auto result = fold<shr, id>(world, type, a, b, dbg)) return result;
373
374 if (auto la = a->isa<Lit>()) {
375 if (la == world.lit(type, 0)) {
376 switch (id) {
377 case shr::a: return la;
378 case shr::l: return la;
379 }
380 }
381 }
382
383 if (auto lb = b->isa<Lit>()) {
384 if (lb == world.lit(type, 0)) {
385 switch (id) {
386 case shr::a: return a;
387 case shr::l: return a;
388 }
389 }
390
391 if (ls && lb->get() > *ls) return world.bot(type, dbg);
392 }
393
394 return world.raw_app(callee, {a, b}, dbg);
395}

397template<wrap id>
398const Def* normalize_wrap(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
399 auto& world = type->world();
400 auto callee = c->as<App>();
401 auto [a, b] = arg->projs<2>();
402 auto mode = callee->arg();
403 auto s = Idx::size(a->type());
404 auto ls = isa_lit(s);
405
406 if (auto result = fold<wrap, id>(world, type, a, b, dbg, mode)) return result;
407
408 // clang-format off
409 if (auto la = a->isa<Lit>()) {
410 if (la == world.lit(type, 0)) {
411 switch (id) {
412 case wrap::add: return b; // 0 + b -> b
413 case wrap::sub: break;
414 case wrap::mul: return la; // 0 * b -> 0
415 case wrap::shl: return la; // 0 << b -> 0
416 }
417 }
418
419 if (la == world.lit(type, 1)) {
420 switch (id) {
421 case wrap::add: break;
422 case wrap::sub: break;
423 case wrap::mul: return b; // 1 * b -> b
424 case wrap::shl: break;
425 }
426 }
427 }
428
429 if (auto lb = b->isa<Lit>()) {
430 if (lb == world.lit(type, 0)) {
431 switch (id) {
432 case wrap::sub: return a; // a - 0 -> a
433 case wrap::shl: return a; // a >> 0 -> a
434 default: unreachable();
435 // add, mul are commutative, the literal has been normalized to the left
436 }
437 }
438
439 if (id == wrap::sub)
440 return op(wrap::add, mode, a, world.lit_idx_mod(*ls, ~lb->get() + 1_u64)); // a - lb -> a + (~lb + 1)
441 else if (id == wrap::shl && ls && lb->get() > *ls)
442 return world.bot(type, dbg);
443 }
444
445 if (a == b) {
446 switch (id) {
447 case wrap::add: return op(wrap::mul, mode, world.lit(type, 2), a, dbg); // a + a -> 2 * a
448 case wrap::sub: return world.lit(type, 0); // a - a -> 0
449 case wrap::mul: break;
450 case wrap::shl: break;
451 }
452 }
453 // clang-format on
454
455 if (auto res = reassociate<wrap>(id, world, callee, a, b, dbg)) return res;
456
457 return world.raw_app(callee, {a, b}, dbg);
458}

460template<div id>
461const Def* normalize_div(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
462 auto& world = type->world();
463 auto callee = c->as<App>();
464 auto [mem, a, b] = arg->projs<3>();
465 type = type->as<Sigma>()->op(1); // peel off actual type
466 auto make_res = [&, mem = mem](const Def* res) { return world.tuple({mem, res}, dbg); };
467
468 if (auto result = fold<div, id>(world, type, a, b, dbg)) return make_res(result);
469
470 if (auto la = a->isa<Lit>()) {
471 if (la == world.lit(type, 0)) return make_res(la); // 0 / b -> 0 and 0 % b -> 0
472 }
473
474 if (auto lb = b->isa<Lit>()) {
475 if (lb == world.lit(type, 0)) return make_res(world.bot(type)); // a / 0 -> ⊥ and a % 0 -> ⊥
476
477 if (lb == world.lit(type, 1)) {
478 switch (id) {
479 case div::sdiv: return make_res(a); // a / 1 -> a
480 case div::udiv: return make_res(a); // a / 1 -> a
481 case div::srem: return make_res(world.lit(type, 0)); // a % 1 -> 0
482 case div::urem: return make_res(world.lit(type, 0)); // a % 1 -> 0
483 }
484 }
485 }
486
487 if (a == b) {
488 switch (id) {
489 case div::sdiv: return make_res(world.lit(type, 1)); // a / a -> 1
490 case div::udiv: return make_res(world.lit(type, 1)); // a / a -> 1
491 case div::srem: return make_res(world.lit(type, 0)); // a % a -> 0
492 case div::urem: return make_res(world.lit(type, 0)); // a % a -> 0
493 }
494 }
495
496 return world.raw_app(callee, {mem, a, b}, dbg);

568template<trait id>
569const Def* normalize_trait(const Def*, const Def* callee, const Def* type, const Def* dbg) {
570 auto& world = type->world();
571 if (auto ptr = match<mem::Ptr>(type)) {
572 return world.lit_nat(8);
573 } else if (type->isa<Pi>()) {
574 return world.lit_nat(8); // Gets lowered to function ptr
575 } else if (auto size = Idx::size(type)) {
576 if (auto w = Idx::size2bitwidth(size)) return world.lit_nat(std::max(1_n, std::bit_ceil(*w) / 8_n));
577 } else if (auto w = math::isa_f(type)) {
578 switch (*w) {
579 case 16: return world.lit_nat(2);
580 case 32: return world.lit_nat(4);
581 case 64: return world.lit_nat(8);
582 default: unreachable();
583 }
584 } else if (type->isa<Sigma>() || type->isa<Meet>()) {
585 u64 offset = 0;
586 u64 align = 1;
587 for (auto t : type->ops()) {
588 auto a = isa_lit(core::op(trait::align, t));
589 auto s = isa_lit(core::op(trait::size, t));
590 if (!a || !s) goto out;
591
592 align = std::max(align, *a);
593 offset = pad(offset, *a) + *s;
594 }
595
596 offset = pad(offset, align);
597 u64 size = std::max(1_u64, offset);
598
599 switch (id) {
600 case trait::align: return world.lit_nat(align);
601 case trait::size: return world.lit_nat(size);
602 }
603 } else if (auto arr = type->isa_structural<Arr>()) {
604 auto align = op(trait::align, arr->body());
605 if constexpr (id == trait::align) return align;
606 if (auto b = op(trait::size, arr->body())->isa<Lit>()) return core::op(core::nop::mul, arr->shape(), b);
607 } else if (auto join = type->isa<Join>()) {
608 if (auto sigma = convert(join)) return core::op(id, sigma, dbg);
609 }
610
611out:
612 return world.raw_app(callee, type, dbg);
613}

633
634 if (lr && std::ranges::all_of(args, [](const Def* arg) { return arg->isa<Tuple, Pack>(); })) {

641 if (*lr == 1) {
642 return w.app(f, inner_args);
643 } else {

645 return w.app(w.app(app_zip, is_os), inner_args);

648 return w.tuple(elems);

650 }

653 return w.raw_app(callee, arg, dbg);
