42template<class Id, Id id, nat_t w>
43Res fold(u64 a, u64 b, [[maybe_unused]] bool nsw, [[maybe_unused]] bool nuw) {
44 using ST = w2s<w>;
45 using UT = w2u<w>;
46 auto s = thorin::bitcast<ST>(a), t = thorin::bitcast<ST>(b);
47 auto u = thorin::bitcast<UT>(a), v = thorin::bitcast<UT>(b);
48
49 if constexpr (std::is_same_v<Id, wrap>) {
50 if constexpr (id == wrap::add) {
51 auto res = u + v;
52 if (nuw && res < u) return {};
53 // TODO nsw
54 return res;
55 } else if constexpr (id == wrap::sub) {
56 auto res = u - v;
57 // TODO nsw
58 return res;
59 } else if constexpr (id == wrap::mul) {
60 if constexpr (std::is_same_v<UT, bool>)
61 return UT(u & v);
62 else
63 return UT(u * v);
64 } else if constexpr (id == wrap::shl) {
65 if (b >= w) return {};
66 decltype(u) res;
67 if constexpr (std::is_same_v<UT, bool>)
68 res = bool(u64(u) << u64(v));
69 else
70 res = u << v;
71 if (nuw && res < u) return {};
72 if (nsw && get_sign(u) != get_sign(res)) return {};
73 return res;
74 } else {
75 []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
76 }
77 } else if constexpr (std::is_same_v<Id, shr>) {
78 if (b >= w) return {};
79 if constexpr (false) {}
80 else if constexpr (id == shr::a) return s >> t;
81 else if constexpr (id == shr::l) return u >> v;
82 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
83 } else if constexpr (std::is_same_v<Id, div>) {
84 if (b == 0) return {};
85 if constexpr (false) {}
86 else if constexpr (id == div::sdiv) return s / t;
87 else if constexpr (id == div::udiv) return u / v;
88 else if constexpr (id == div::srem) return s % t;
89 else if constexpr (id == div::urem) return u % v;
90 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
91 } else if constexpr (std::is_same_v<Id, icmp>) {
92 bool res = false;
93 auto pm = !(u >> UT(w - 1)) && (v >> UT(w - 1));
94 auto mp = (u >> UT(w - 1)) && !(v >> UT(w - 1));
95 res |= ((id & icmp::Xygle) != icmp::f) && pm;
96 res |= ((id & icmp::xYgle) != icmp::f) && mp;
97 res |= ((id & icmp::xyGle) != icmp::f) && u > v && !mp;
98 res |= ((id & icmp::xygLe) != icmp::f) && u < v && !pm;
99 res |= ((id & icmp::xyglE) != icmp::f) && u == v;
100 return res;
101 } else {
102 []<bool flag = false>() { static_assert(flag, "missing tag"); }();
103 }
104}

107// Note that @p a and @p b are passed by reference as fold also commutes if possible.
108template<class Id, Id id>
109Ref fold(World& world, Ref type, const Def*& a, const Def*& b, Ref mode = {}) {
110 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type);
111
112 if (auto la = Lit::isa(a)) {
113 if (auto lb = Lit::isa(b)) {
114 auto size = Lit::as(Idx::size(a->type()));
115 auto width = Idx::size2bitwidth(size);
116 bool nsw = false, nuw = false;
117 if constexpr (std::is_same_v<Id, wrap>) {
118 auto m = mode ? Lit::as(mode) : 0_n;
119 nsw = m & Mode::nsw;
120 nuw = m & Mode::nuw;
121 }
122
123 Res res;
124 switch (width) {
125#define CODE(i) \
126 case i: res = fold<Id, id, i>(*la, *lb, nsw, nuw); break;
127 THORIN_1_8_16_32_64(CODE)
128#undef CODE
129 default:
130 // TODO this is super rough but at least better than just bailing out
131 res = fold<Id, id, 64>(*la, *lb, false, false);
132 if (res && !std::is_same_v<Id, icmp>) *res %= size;
133 }
134
135 return res ? world.lit(type, *res) : world.bot(type);
136 }
137 }
138
139 commute(id, a, b);
140 return nullptr;
141}

143/// Reassociates @p a und @p b according to following rules.
144/// We use the following naming convention while literals are prefixed with an `l`:
145/// ```
146/// a op b
147/// (x op y) op (z op w)
148///
149/// (1) la op (lz op w) -> (la op lz) op w
150/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
151/// (3) a op (lz op w) -> lz op (a op w)
152/// (4) (lx op y) op b -> lx op (y op b)
153/// ```
154template<class Id>
155Ref reassociate(Id id, World& world, [[maybe_unused]] const App* ab, Ref a, Ref b) {
156 if (!is_associative(id)) return nullptr;
157
158 if (auto xy = match<Id>(id, a)) {
159 if (auto zw = match<Id>(id, b)) {
160 auto la = a->isa<Lit>();
161 auto [x, y] = xy->template args<2>();
162 auto [z, w] = zw->template args<2>();
163 auto lx = Lit::isa(x);
164 auto lz = Lit::isa(z);
165
166 // if we reassociate, we have to forget about nsw/nuw
167 auto make_op = [&world, id](Ref a, Ref b) { return world.call(id, Mode::none, Defs{a, b}); };
168
169 if (la && lz) return make_op(make_op(a, z), w); // (1)
170 if (lx && lz) return make_op(make_op(x, z), make_op(y, w)); // (2)
171 if (lz) return make_op(z, make_op(a, w)); // (3)
172 if (lx) return make_op(x, make_op(y, b)); // (4)
173 }
174 }
175
176 return nullptr;
177}

179template<class Id>
180Ref merge_cmps(std::array<std::array<u64, 2>, 2> tab, Ref a, Ref b) {
181 static_assert(sizeof(sub_t) == 1, "if this ever changes, please adjust the logic below");
182 static constexpr size_t num_bits = std::bit_width(Annex::Num<Id> - 1_u64);
183
184 auto& world = a->world();
185 auto a_cmp = match<Id>(a);
186 auto b_cmp = match<Id>(b);
187
188 if (a_cmp && b_cmp && a_cmp->arg() == b_cmp->arg()) {
189 // push sub bits of a_cmp and b_cmp through truth table
190 sub_t res = 0;
191 sub_t a_sub = a_cmp.sub();
192 sub_t b_sub = b_cmp.sub();
193 for (size_t i = 0; i != num_bits; ++i, res >>= 1, a_sub >>= 1, b_sub >>= 1)
194 res |= tab[a_sub & 1][b_sub & 1] << 7_u8;
195 res >>= (7_u8 - u8(num_bits));
196
197 if constexpr (std::is_same_v<Id, math::cmp>)
198 return world.call(math::cmp(res), /*mode*/ a_cmp->decurry()->arg(), a_cmp->arg());
199 else
200 return world.call(icmp(Annex::Base<icmp> | res), a_cmp->arg());
201 }
202
203 return nullptr;
204}

297template<bit2 id>
298Ref normalize_bit2(Ref type, Ref c, Ref arg) {
299 auto& world = type->world();
300 auto callee = c->as<App>();
301 auto [a, b] = arg->projs<2>();
302 auto s = callee->decurry()->arg();
303 auto ls = Lit::isa(s);
304 // TODO cope with wrap around
305
306 commute(id, a, b);
307
308 auto tab = make_truth_table(id);
309 if (auto res = merge_cmps<icmp>(tab, a, b)) return res;
310 if (auto res = merge_cmps<math::cmp>(tab, a, b)) return res;
311
312 auto la = Lit::isa(a);
313 auto lb = Lit::isa(b);
314
315 // clang-format off
316 switch (id) {
317 case bit2:: f: return world.lit(type, 0);
318 case bit2:: t: if (ls) return world.lit(type, *ls-1_u64); break;
319 case bit2:: fst: return a;
320 case bit2:: snd: return b;
321 case bit2:: nfst: return world.call(bit1::neg, s, a);
322 case bit2:: nsnd: return world.call(bit1::neg, s, b);
323 case bit2:: ciff: return world.call(bit2:: iff, s, Defs{b, a});
324 case bit2::nciff: return world.call(bit2::niff, s, Defs{b, a});
325 default: break;
326 }
327
328 if (la && lb && ls) {
329 switch (id) {
330 case bit2::and_: return world.lit_idx (*ls, *la & *lb);
331 case bit2:: or_: return world.lit_idx (*ls, *la | *lb);
332 case bit2::xor_: return world.lit_idx (*ls, *la ^ *lb);
333 case bit2::nand: return world.lit_idx_mod(*ls, ~(*la & *lb));
334 case bit2:: nor: return world.lit_idx_mod(*ls, ~(*la | *lb));
335 case bit2::nxor: return world.lit_idx_mod(*ls, ~(*la ^ *lb));
336 case bit2:: iff: return world.lit_idx_mod(*ls, ~ *la | *lb);
337 case bit2::niff: return world.lit_idx (*ls, *la & ~*lb);
338 default: unreachable();
339 }
340 }
341
342 // TODO rewrite using bit2
343 auto unary = [&](bool x, bool y, Ref a) -> Ref {
344 if (!x && !y) return world.lit(type, 0);
345 if ( x && y) return ls ? world.lit(type, *ls-1_u64) : nullptr;
346 if (!x && y) return a;
347 if ( x && !y && id != bit2::xor_) return world.call(bit1::neg, s, a);
348 return nullptr;
349 };
350 // clang-format on
351
352 if (is_commutative(id) && a == b) {
353 if (auto res = unary(tab[0][0], tab[1][1], a)) return res;
354 }
355
356 if (la) {
357 if (*la == 0) {
358 if (auto res = unary(tab[0][0], tab[0][1], b)) return res;
359 } else if (ls && *la == *ls - 1_u64) {
360 if (auto res = unary(tab[1][0], tab[1][1], b)) return res;
361 }
362 }
363
364 if (lb) {
365 if (*lb == 0) {
366 if (auto res = unary(tab[0][0], tab[1][0], a)) return res;
367 } else if (ls && *lb == *ls - 1_u64) {
368 if (auto res = unary(tab[0][1], tab[1][1], a)) return res;
369 }
370 }
371
372 if (auto res = reassociate<bit2>(id, world, callee, a, b)) return res;
373
374 return world.raw_app(type, callee, {a, b});
375}

421template<wrap id>
422Ref normalize_wrap(Ref type, Ref c, Ref arg) {
423 auto& world = type->world();
424 auto callee = c->as<App>();
425 auto [a, b] = arg->projs<2>();
426 auto mode = callee->arg();
427 auto s = Idx::size(a->type());
428 auto ls = Lit::isa(s);
429
430 if (auto result = fold<wrap, id>(world, type, a, b)) return result;
431
432 // clang-format off
433 if (auto la = Lit::isa(a)) {
434 if (*la == 0) {
435 switch (id) {
436 case wrap::add: return b; // 0 + b -> b
437 case wrap::sub: break;
438 case wrap::mul: return a; // 0 * b -> 0
439 case wrap::shl: return a; // 0 << b -> 0
440 }
441 } else if (*la == 1) {
442 switch (id) {
443 case wrap::add: break;
444 case wrap::sub: break;
445 case wrap::mul: return b; // 1 * b -> b
446 case wrap::shl: break;
447 }
448 }
449 }
450
451 if (auto lb = Lit::isa(b)) {
452 if (*lb == 0) {
453 switch (id) {
454 case wrap::sub: return a; // a - 0 -> a
455 case wrap::shl: return a; // a >> 0 -> a
456 default: unreachable();
457 // add, mul are commutative, the literal has been normalized to the left
458 }
459 }
460
461 if (id == wrap::sub)
462 return world.call(wrap::add, mode, Defs{a, world.lit_idx_mod(*ls, ~*lb + 1_u64)}); // a - lb -> a + (~lb + 1)
463 else if (id == wrap::shl && ls && *lb > *ls)
464 return world.bot(type);
465 }
466
467 if (a == b) {
468 switch (id) {
469 case wrap::add: return world.call(wrap::mul, mode, Defs{world.lit(type, 2), a}); // a + a -> 2 * a
470 case wrap::sub: return world.lit(type, 0); // a - a -> 0
471 case wrap::mul: break;
472 case wrap::shl: break;
473 }
474 }
475 // clang-format on
476
477 if (auto res = reassociate<wrap>(id, world, callee, a, b)) return res;
478
479 return world.raw_app(type, callee, {a, b});
480}

505 case div::urem: return make_res(world.lit(type, 0)); // a % 1 -> 0
506 }
507 }
508 }
509
510 if (a == b) {
511 switch (id) {
512 case div::sdiv: return make_res(world.lit(type, 1)); // a / a -> 1

583// and every occurance of these types in a later phase
584// TODO Pi and others
585template<trait id>
586Ref normalize_trait(Ref nat, Ref callee, Ref type) {
587 auto& world = type->world();
588 if (auto ptr = match<mem::Ptr>(type)) {
589 return world.lit_nat(8);
590 } else if (type->isa<Pi>()) {
591 return world.lit_nat(8); // Gets lowered to function ptr
592 } else if (auto size = Idx::size(type)) {
593 if (auto w = Idx::size2bitwidth(size)) return world.lit_nat(std::max(1_n, std::bit_ceil(*w) / 8_n));
594 } else if (auto w = math::isa_f(type)) {
595 switch (*w) {
596 case 16: return world.lit_nat(2);
597 case 32: return world.lit_nat(4);
598 case 64: return world.lit_nat(8);
599 default: unreachable();
600 }
601 } else if (type->isa<Sigma>() || type->isa<Meet>()) {
602 u64 offset = 0;
603 u64 align = 1;
604 for (auto t : type->ops()) {
605 auto a = Lit::isa(core::op(trait::align, t));
606 auto s = Lit::isa(core::op(trait::size, t));
607 if (!a || !s) goto out;
608
609 align = std::max(align, *a);
610 offset = pad(offset, *a) + *s;
611 }
612
613 offset = pad(offset, align);
614 u64 size = std::max(1_u64, offset);
615
616 switch (id) {
617 case trait::align: return world.lit_nat(align);
