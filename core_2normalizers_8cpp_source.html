125template<icmp cmp, nat_t w>
126struct Fold<icmp, cmp, w> {
127 inline static Res run(u64 a, u64 b) {
128 using T = w2u<w>;
129 auto x = get<T>(a), y = get<T>(b);
130 bool result = false;
131 auto pm = !(x >> T(w - 1)) && (y >> T(w - 1));
132 auto mp = (x >> T(w - 1)) && !(y >> T(w - 1));
133 result |= ((cmp & icmp::Xygle) != icmp::f) && pm;
134 result |= ((cmp & icmp::xYgle) != icmp::f) && mp;
135 result |= ((cmp & icmp::xyGle) != icmp::f) && x > y && !mp;
136 result |= ((cmp & icmp::xygLe) != icmp::f) && x < y && !pm;
137 result |= ((cmp & icmp::xyglE) != icmp::f) && x == y;
138 return result;
139 }

441template<wrap sub>
442const Def* normalize_wrap(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
443 auto& world = type->world();
444 auto callee = c->as<App>();
445 auto [a, b] = arg->projs<2>();
446 auto [m, w] = callee->args<2>(isa_lit<nat_t>); // mode and width
447
448 if (auto result = fold<wrap, sub, true>(world, type, callee, a, b, dbg)) return result;
449
450 // clang-format off
451 if (auto la = a->isa<Lit>()) {
452 if (la == world.lit_int(*w, 0)) {
453 switch (sub) {
454 case wrap::add: return b; // 0 + b -> b
455 case wrap::sub: break;
456 case wrap::mul: return la; // 0 * b -> 0
457 case wrap::shl: return la; // 0 << b -> 0
458 default: unreachable();
459 }
460 }
461
462 if (la == world.lit_int(*w, 1)) {
463 switch (sub) {
464 case wrap::add: break;
465 case wrap::sub: break;
466 case wrap::mul: return b; // 1 * b -> b
467 case wrap::shl: break;
468 default: unreachable();
469 }
470 }
471 }
472
473 if (auto lb = b->isa<Lit>()) {
474 if (lb == world.lit_int(*w, 0)) {
475 switch (sub) {
476 case wrap::sub: return a; // a - 0 -> a
477 case wrap::shl: return a; // a >> 0 -> a
478 default: unreachable();
479 // add, mul are commutative, the literal has been normalized to the left
480 }
481 }
482
483 if (sub == wrap::sub)
484 return op(wrap::add, *m, a, world.lit_int_mod(*w, ~lb->get() + 1_u64)); // a - lb -> a + (~lb + 1)
485 else if (sub == wrap::shl && lb->get() > *w)
486 return world.bot(type, dbg);
487 }
488
489 if (a == b) {
490 switch (sub) {
491 case wrap::add: return op(wrap::mul, *m, world.lit_int(*w, 2), a, dbg); // a + a -> 2 * a
492 case wrap::sub: return world.lit_int(*w, 0); // a - a -> 0
493 case wrap::mul: break;
494 case wrap::shl: break;
495 default: unreachable();
496 }
497 }
498 // clang-format on
499
500 if (auto res = reassociate<wrap>(sub, world, callee, a, b, dbg)) return res;
501
502 return world.raw_app(callee, {a, b}, dbg);
503}

555template<nat_t min_sw, nat_t min_dw, conv op>
556static const Def* fold_conv(const Def* dst_type, const App* callee, const Def* src, const Def* dbg) {
557 auto& world = dst_type->world();
558 if (src->isa<Bot>()) return world.bot(dst_type, dbg);
559
560 auto [lit_dw, lit_sw] = callee->args<2>(isa_lit<nat_t>);
561 auto lit_src = src->isa<Lit>();
562 if (lit_src && lit_dw && lit_sw) {
563 if (op == conv::u2u) {
564 if (*lit_dw == 0) return world.lit(dst_type, as_lit(lit_src));
565 return world.lit(dst_type, as_lit(lit_src) % *lit_dw);
566 }
567
568 if (isa<Tag::Int>(src->type())) *lit_sw = *mod2width(*lit_sw);
569 if (isa<Tag::Int>(dst_type)) *lit_dw = *mod2width(*lit_dw);
570
571 Res res;
572#define CODE(sw, dw) \
573 else if (*lit_dw == dw && *lit_sw == sw) { \
574 if constexpr (dw >= min_dw && sw >= min_sw) res = FoldConv<op, dw, sw>::run(lit_src->get()); \
575 }
576 if (false) {}
577 TABLE(CODE)
578#undef CODE
579 if (res) return world.lit(dst_type, *res, dbg);
580 return world.bot(dst_type, dbg);
581 }

583 return nullptr;

586template<conv op>
587const Def* normalize_conv(const Def* dst_type, const Def* c, const Def* src, const Def* dbg) {
588 auto& world = dst_type->world();
589 auto callee = c->as<App>();
590
591 static constexpr auto min_sw = op == conv::r2s || op == conv::r2u || op == conv::r2r ? 16 : 1;
592 static constexpr auto min_dw = op == conv::s2r || op == conv::u2r || op == conv::r2r ? 16 : 1;
593 if (auto result = fold_conv<min_sw, min_dw, op>(dst_type, callee, src, dbg)) return result;
594
595 auto [dw, sw] = callee->args<2>(isa_lit<nat_t>);
596 if (sw == dw && dst_type == src->type()) return src;
597
598 if constexpr (op == conv::s2s) {
599 if (sw && dw && *sw < *dw) return core::op(conv::u2u, dst_type, src, dbg);
600 }
601
602 return world.raw_app(callee, src, dbg);
603}

605const Def* normalize_bitcast(const Def* dst_type, const Def* callee, const Def* src, const Def* dbg) {
606 auto& world = dst_type->world();
607
608 if (src->isa<Bot>()) return world.bot(dst_type);
609 if (src->type() == dst_type) return src;
610
611 if (auto other = match<bitcast>(src))
612 return other->arg()->type() == dst_type ? other->arg() : op_bitcast(dst_type, other->arg(), dbg);
613
614 if (auto lit = src->isa<Lit>()) {
615 if (dst_type->isa<Nat>()) return world.lit(dst_type, lit->get(), dbg);
616 if (isa_sized_type(dst_type)) return world.lit(dst_type, lit->get(), dbg);
617 }
618
619 return world.raw_app(callee, src, dbg);
620}
