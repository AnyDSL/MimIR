56/// computes pb type `E* -> A*`
57/// `E` - type of the expression (return type for a function)
58/// `A` - type of the argument (point of orientation resp. derivative - argument type for partial pullbacks)
59const Pi* pullback_type(const Def* E, const Def* A) {
60 auto& world = E->world();
61 auto tang_arg = tangent_type_fun(A);
62 auto tang_ret = tangent_type_fun(E);
63 auto pb_ty = world.cn({tang_ret, world.cn(tang_arg)});
64 return pb_ty;
65}

69const Pi* autodiff_type_fun(const Def* arg, const Def* ret) {
70 auto& world = arg->world();
71 world.DLOG("autodiff type for {} => {}", arg, ret);
72 auto aug_arg = thorin::autodiff::autodiff_type_fun(arg);
73 auto aug_ret = thorin::autodiff::autodiff_type_fun(ret);
74 world.DLOG("augmented types: {} => {}", aug_arg, aug_ret);
75 if (!aug_arg || !aug_ret) return nullptr;
76 // `Q* -> P*`
77 auto pb_ty = pullback_type(ret, arg);
78 world.DLOG("pb type: {}", pb_ty);
79 // `P' -> Q' * (Q* -> P*)`
80
81 auto deriv_ty = world.cn({aug_arg, world.cn({aug_ret, pb_ty})});
82 world.DLOG("autodiff type: {}", deriv_ty);
83 return deriv_ty;
84}

110const Def* autodiff_type_fun(const Def* ty) {
111 auto& world = ty->world();
112 // TODO: handle DS (operators)
113 if (auto pi = ty->isa<Pi>()) return autodiff_type_fun_pi(pi);
114 // Also handles autodiff call from axiom declaration => abstract => leave it.
115 world.DLOG("AutoDiff on type: {} <{}>", ty, ty->node_name());
116 if (Idx::size(ty)) return ty;
117 if (ty == world.type_nat()) return ty;
118 if (auto arr = ty->isa<Arr>()) {
119 auto shape = arr->shape();
120 auto body = arr->body();
121 auto body_ad = autodiff_type_fun(body);
122 if (!body_ad) return nullptr;
123 return world.arr(shape, body_ad);
124 }
125 if (auto sig = ty->isa<Sigma>()) {
126 // TODO: mut sigma

128 world.DLOG("ops: {,}", ops);

130 }
131 // mem
