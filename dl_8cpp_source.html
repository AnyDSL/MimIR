60void* get(void* handle, const std::string& symbol) {
61#ifdef _WIN32
62 if (auto addr = GetProcAddress(static_cast<HMODULE>(handle), symbol.c_str())) {
63 return reinterpret_cast<void*>(addr);
64 } else {
65 err("could not find symbol '{}' in plugin due to error '{}'\n"
66 "see (https://docs.microsoft.com/en-us/windows/win32/debug/system-error-codes)\n",
67 symbol, GetLastError());
68 }
69#else
70 dlerror(); // clear error state
71 void* addr = dlsym(handle, symbol.c_str());
72 if (char* error = dlerror()) {
73 err("could not find symbol '{}' in plugin due to error '{}' \n", symbol, error);
74 } else {
