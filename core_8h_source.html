140/// Use like this:
141/// `a op b = tab[a][b]`
142constexpr std::array<std::array<u64, 2>, 2> make_truth_table(bit2 id) {
143 return {
144 {{sub_t(id) & sub_t(0b0001) ? u64(-1) : 0, sub_t(id) & sub_t(0b0100) ? u64(-1) : 0},
145 {sub_t(id) & sub_t(0b0010) ? u64(-1) : 0, sub_t(id) & sub_t(0b1000) ? u64(-1) : 0}}

167constexpr bool is_commutative(core::bit2 id) {
168 auto tab = make_truth_table(id);
169 return tab[0][1] == tab[1][0];
170}

172constexpr bool is_associative(core::bit2 id) {
173 switch (id) {
174 case core::bit2::t:
175 case core::bit2::xor_:
176 case core::bit2::and_:
177 case core::bit2::nxor:
178 case core::bit2::a:
179 case core::bit2::b:
180 case core::bit2::or_:
181 case core::bit2::f: return true;
182 default: return false;
183 }
184}
