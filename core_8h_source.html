19enum RMode : nat_t {
20 none = 0,
21 nnan = 1 << 0, ///< No NaNs - Allow optimizations to assume the arguments and result are not NaN. Such optimizations
22 ///< are required to retain defined behavior over NaNs, but the value of the result is undefined.
23 ninf =
24 1 << 1, ///< No Infs - Allow optimizations to assume the arguments and result are not +/-Inf. Such optimizations
25 ///< are required to retain defined behavior over +/-Inf, but the value of the result is undefined.
26 nsz = 1
27 << 2, ///< No Signed Zeros - Allow optimizations to treat the sign of a zero argument or result as insignificant.
28 arcp = 1 << 3, ///< Allow Reciprocal - Allow optimizations to use the reciprocal of an argument rather than perform
29 ///< division.
30 contract = 1 << 4, ///< Allow floating-point contraction (e.g. fusing a multiply followed by an addition into a
31 ///< fused multiply-and-add).
32 afn = 1 << 5, ///< Approximate functions - Allow substitution of approximate calculations for functions (sin, log,
33 ///< sqrt, etc). See floating-point intrinsic definitions for places where this can apply to LLVM’s
34 ///< intrinsic math functions.
35 reassoc = 1 << 6, ///< Allow reassociation transformations for floating-point operations. This may dramatically
36 ///< change results in floating point.
37 finite = nnan | ninf,
38 unsafe = nsz | arcp | reassoc,
39 fast = nnan | ninf | nsz | arcp | contract | afn | reassoc,
40 bot = fast,
41 top = none,
42};
43}

186// clang-format off
187template<class R>
188const Lit* lit_real(World& w, R val, const Def* dbg = {}) {
189 static_assert(std::is_floating_point<R>() || std::is_same<R, r16>());
190 if constexpr (false) {}
191 else if constexpr (sizeof(R) == 2) return w.lit(type_real(w, 16), thorin::bitcast<u16>(val), dbg);
192 else if constexpr (sizeof(R) == 4) return w.lit(type_real(w, 32), thorin::bitcast<u32>(val), dbg);
193 else if constexpr (sizeof(R) == 8) return w.lit(type_real(w, 64), thorin::bitcast<u64>(val), dbg);
194 else unreachable();
195}

197inline const Lit* lit_real(World& w, nat_t width, r64 val, const Def* dbg = {}) {
198 switch (width) {
199 case 16: assert(r64(r16(r32(val))) == val && "loosing precision"); return lit_real(w, r16(r32(val)), dbg);
200 case 32: assert(r64(r32( (val))) == val && "loosing precision"); return lit_real(w, r32( (val)), dbg);
201 case 64: assert(r64(r64( (val))) == val && "loosing precision"); return lit_real(w, r64( (val)), dbg);
202 default: unreachable();
203 }
