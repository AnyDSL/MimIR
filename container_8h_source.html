33/// Yields pointer to element (or the element itself if it is already a pointer), if found and `nullptr` otherwise.
34/// @warning If the element is **not** already a pointer, this lookup will simply take the address of this element.
35/// This means that, e.g., a rehash of an `absl::flat_hash_map` will invalidate this pointer.
36template<class C, class K>
37auto lookup(const C& container, const K& key) {
38 auto i = container.find(key);
39 if constexpr (std::is_pointer_v<typename C::mapped_type>)
40 return i != container.end() ? i->second : nullptr;
41 else
42 return i != container.end() ? &i->second : nullptr;
43}
