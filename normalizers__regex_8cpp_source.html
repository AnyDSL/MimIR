21template<quant id> Ref normalize_quant(Ref type, Ref callee, Ref arg) {
22 auto& world = type->world();
23
24 // quantifiers are idempotent
25 if (thorin::match(id, arg)) return arg;
26
27 if constexpr (id == quant::plus) {
28 // (\d?)+ and (\d*)+ == \d*
29 if (auto optional_app = thorin::match(quant::optional, arg))
30 return world.call(quant::star, optional_app->arg());
31 else if (auto star_app = thorin::match(quant::star, arg))
32 return arg;
33 } else if constexpr (id == quant::star) {
34 // (\d?)* and (\d+)* == \d*
35 if (auto quant_app = thorin::match<quant>(arg)) return world.app(callee, quant_app->arg());
36 } else if constexpr (id == quant::optional) {
37 // (\d*)? and (\d+)? == \d*
38 if (auto star_app = thorin::match(quant::star, arg))
39 return arg;
40 else if (auto plus_app = thorin::match(quant::plus, arg))
41 return world.call(quant::star, plus_app->arg());
42 }
43
44 return world.raw_app(type, callee, arg);
45}

82bool compare_re(const Def* lhs, const Def* rhs) {
83 auto lhs_range = thorin::match<range>(lhs);
84 auto rhs_range = thorin::match<range>(rhs);
85 if (lhs_range) {
86 // sort ranges by increasing lower bound
87 if (rhs_range) return Lit::as(lhs_range->arg()->proj(0)) < Lit::as(rhs_range->arg()->proj(0));
88 // ranges to the end
89 return false;
90 } else if (rhs_range) {
91 // ranges to the end
92 return true;
93 }
94 // keep
95 return true;

131void merge_ranges(std::vector<const Def*>& args) {
132 auto ranges_begin = args.begin();
133 while (ranges_begin != args.end() && !thorin::match<range>(*ranges_begin)) ranges_begin++;
134 if (ranges_begin == args.end()) return;
135
136 std::set<const Def*> to_remove;
137 std::vector<std::pair<nat_t, nat_t>> old_ranges, new_ranges;
138 auto& world = (*ranges_begin)->world();
139
140 std::transform(ranges_begin, args.end(), std::back_inserter(old_ranges), get_range);
141
142 for (auto it = old_ranges.begin(); it != old_ranges.end(); ++it) {
143 auto current_range = *it;
144 world.DLOG("old range: {}-{}", current_range.first, current_range.second);
145 for (auto inner = it + 1; inner != old_ranges.end(); ++inner)
146 if (auto merged = merge_ranges(current_range, *inner)) current_range = *merged;
147
148 std::vector<std::vector<std::pair<nat_t, nat_t>>::iterator> de_duplicate;
149 for (auto inner = new_ranges.begin(); inner != new_ranges.end(); ++inner) {
150 if (auto merged = merge_ranges(current_range, *inner)) {
151 current_range = *merged;
152 de_duplicate.push_back(inner);
153 }
154 }
155 for (auto dedup : de_duplicate) {
156 world.DLOG("dedup {}-{}", current_range.first, current_range.second);
157 new_ranges.erase(dedup);
158 }
159 world.DLOG("new range: {}-{}", current_range.first, current_range.second);
160 new_ranges.push_back(std::move(current_range));
161 }
162 // invalidates ranges_begin
163 args.erase(ranges_begin, args.end());
164 std::transform(new_ranges.begin(), new_ranges.end(), std::back_inserter(args), app_range{world});
165
166 make_vector_unique(args);
167}

173bool equals_any(const Def* lhs, const Def* rhs) {
174 auto check_arg_equiv = [](const Def* lhs, const Def* rhs) {
175 if (auto rng_lhs = thorin::match<range>(lhs))
176 if (auto not_rhs = thorin::match<not_>(rhs)) {
177 if (auto rng_rhs = thorin::match<range>(not_rhs->arg())) return rng_lhs == rng_rhs;
178 }
179 return false;
180 };
181
182 return check_arg_equiv(lhs, rhs) || check_arg_equiv(rhs, lhs);
183}

185bool equals_any(const std::vector<const Def*>& lhs, const std::vector<const Def*>& rhs) {
186 std::vector<std::pair<nat_t, nat_t>> lhs_ranges, rhs_ranges;
187 auto only_ranges = std::ranges::views::filter([](auto d) { return match<range>(d); });
188 std::ranges::transform(lhs | only_ranges, std::back_inserter(lhs_ranges), get_range);

190 return std::ranges::includes(lhs_ranges, rhs_ranges) || std::ranges::includes(rhs_ranges, lhs_ranges);

193Ref normalize_disj(Ref type, Ref, Ref arg) {
194 auto& world = type->world();
195 if (arg->as_lit_arity() > 1) {
196 auto contains_any = [](auto args) {
197 return std::ranges::find_if(args, [](const Def* ax) -> bool { return thorin::match<any>(ax); })
198 != args.end();
199 };
200
201 auto new_args = flatten_in_arg<disj>(arg);
202 if (contains_any(new_args)) return world.annex<any>();
203 make_vector_unique(new_args);
204 merge_ranges(new_args);
205
206 const Def* to_remove = nullptr;
207 for (const auto* cls0 : new_args) {
208 for (const auto* cls1 : new_args)
209 if (equals_any(cls0, cls1)) return world.annex<any>();
210
211 if (auto not_rhs = thorin::match<not_>(cls0)) {
212 if (auto disj_rhs = thorin::match<disj>(not_rhs->arg())) {
213 auto rngs = flatten_in_arg<disj>(disj_rhs->arg());
214 make_vector_unique(rngs);
215 if (equals_any(new_args, rngs)) return world.annex<any>();
216 }
217 }
218 }
219
220 std::erase(new_args, to_remove);
221 world.DLOG("final ranges {, }", new_args);
222
223 const Def* retval = new_args.back();
224 if (new_args.size() > 2)
225 retval = make_binary_tree<disj>(type, new_args);
226 else if (new_args.size() > 1)
227 retval = world.raw_app(type, world.call<disj>(world.lit_nat(2)), new_args);
228 world.DLOG("disj app: {}", retval);
229 return retval;
230 }
231 return arg;
232}
