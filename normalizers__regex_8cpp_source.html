21template<quant id> Ref normalize_quant(Ref type, Ref callee, Ref arg) {
22 auto& world = type->world();
23
24 // quantifiers are idempotent
25 if (thorin::match(id, arg)) return arg;
26
27 if constexpr (id == quant::plus) {
28 // (\d?)+ and (\d*)+ == \d*
29 if (auto optional_app = thorin::match(quant::optional, arg))
30 return world.call(quant::star, optional_app->arg());
31 else if (auto star_app = thorin::match(quant::star, arg))
32 return arg;
33 } else if constexpr (id == quant::star) {
34 // (\d?)* and (\d+)* == \d*
35 if (auto quant_app = thorin::match<quant>(arg)) return world.app(callee, quant_app->arg());
36 } else if constexpr (id == quant::optional) {
37 // (\d*)? and (\d+)? == \d*
38 if (auto star_app = thorin::match(quant::star, arg))
39 return arg;
40 else if (auto plus_app = thorin::match(quant::plus, arg))
41 return world.call(quant::star, plus_app->arg());
42 }
43
44 return world.raw_app(type, callee, arg);
45}

147bool equals_any(const Def* lhs, const Def* rhs) {
148 auto check_arg_equiv = [](const Def* lhs, const Def* rhs) {
149 if (auto rng_lhs = thorin::match<range>(lhs))
150 if (auto not_rhs = thorin::match<not_>(rhs)) {
151 if (auto rng_rhs = thorin::match<range>(not_rhs->arg())) return rng_lhs == rng_rhs;
152 }
153 return false;
154 };
155
156 return check_arg_equiv(lhs, rhs) || check_arg_equiv(rhs, lhs);
157}

159bool equals_any(Defs lhs, Defs rhs) {
160 Vector<std::pair<nat_t, nat_t>> lhs_ranges, rhs_ranges;
161 auto only_ranges = std::ranges::views::filter([](auto d) { return match<range>(d); });
162 std::ranges::transform(lhs | only_ranges, std::back_inserter(lhs_ranges), get_range);
163 std::ranges::transform(rhs | only_ranges, std::back_inserter(rhs_ranges), get_range);
164 return std::ranges::includes(lhs_ranges, rhs_ranges) || std::ranges::includes(rhs_ranges, lhs_ranges);

167Ref normalize_disj(Ref type, Ref, Ref arg) {
168 auto& world = type->world();

173 };

176 if (contains_any(new_args)) return world.annex<any>();

179
180 const Def* to_remove = nullptr;
181 for (const auto* cls0 : new_args) {
182 for (const auto* cls1 : new_args)
183 if (equals_any(cls0, cls1)) return world.annex<any>();

188 make_vector_unique(rngs);

191 }
192 }

196
197 const Def* retval = new_args.back();

200 else if (new_args.size() > 1)
