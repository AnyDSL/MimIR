16/// The optimizations proceed in a pipeline ordered by priorities.
17/// Each phase is a sequence of passes that are run interleaved.
18/// The passes are added to the phase ordered by their priority.
19
20/// Order:
21/// * 1-10: initial passes
22/// * 100: Main optimization phase (default for extend_opt_phase)
23/// * 200: Pre-CodeGen Optimization
24/// * 300: CodeGen Preparation (default for extend_codegen_prep_phase)
25///
26/// concrete phases:
27/// * 0 : Scalarize
28/// * 1 : EtaRed
29/// * 2 : TailRecElim
30/// * 100: Optimize (Priority 50)

32/// * BetaRed
33/// * EtaRed
34/// * EtaExp
35/// * Scalarize
36/// * TailRecElim
37/// * + Custom (default priority 100)

40/// * + Custom (default priority 100)

42/// See optimize.h for magic numbers
43void optimize(World& world, PipelineBuilder& builder) {
44 builder.extend_opt_phase(0, [](thorin::PassMan& man) { man.add<Scalerize>(); });
45 builder.extend_opt_phase(1, [](thorin::PassMan& man) { man.add<EtaRed>(); });
