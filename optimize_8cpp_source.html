21void optimize(World& world, Passes& passes, std::vector<Dialect>& dialects) {
22 auto compilation_functions = {world.sym("_compile"), world.sym("_default_compile"), world.sym("_core_compile"),
23 world.sym("_fallback_compile")};
24 const Def* compilation = nullptr;
25 for (auto compilation_function : compilation_functions) {
26 if (auto compilation_ = world.lookup(compilation_function)) {
27 if (!compilation) compilation = compilation_;
28 compilation_->make_internal();
29 }
30 }
31 // make all functions `[] -> Pipeline` internal
32 std::vector<Def*> make_internal;
33 for (auto ext : world.externals()) {
34 auto def = ext.second;
35 if (auto lam = def->isa<Lam>(); lam && lam->num_doms() == 0) {
36 if (*lam->codom()->sym() == "Pipeline") {
37 if (!compilation) compilation = lam;
38 make_internal.push_back(def);
39 }
40 }
41 }
42 for (auto def : make_internal) def->make_internal();
43 assert(compilation && "no compilation function found");
44
45 // We found a compilation directive in the file and use it to build the compilation pipeline.
46 // The general idea is that passes and phases are exposed as axioms.
47 // Each pass/phase axiom is associated with a handler function operating on the PipelineBuilder in the
48 // passes map. This registering is analogous to the normalizers (`code -> code`) but only operated using
49 // side effects that change the pipeline.
50 world.DLOG("compilation using {} : {}", compilation, compilation->type());
51
52 // We can not directly access compile axioms here.
53 // But the compile dialect has not the necessary communication pipeline.
54 // Therefore, we register the handlers and let the compile dialect call them.
55
56 PipelineBuilder pipe_builder(world);
57 // TODO: remove indirections of pipeline builder. Just add passes and phases directly to the pipeline.
58 for (auto& dialect : dialects) pipe_builder.register_dialect(dialect);
59
60 auto pipeline = compilation->as<Lam>()->body();
61 auto [ax, phases] = collect_args(pipeline);
62
63 // handle pipeline like all other pass axioms
64 auto pipeline_axiom = ax->as<Axiom>();
65 auto pipeline_flags = pipeline_axiom->flags();
66 assert(passes.contains(pipeline_flags));
67 world.DLOG("Building pipeline");
68 passes[pipeline_flags](world, pipe_builder, pipeline);
69
70 world.DLOG("Executing pipeline");
71 pipe_builder.run_pipeline();
72
73 return;
74}
