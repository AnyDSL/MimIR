10static bool interesting_type(const Def* type, DefSet& visited) {
11 if (type->isa_nom()) visited.insert(type);
12 if (isa_clos_type(type)) return true;
13 if (auto sigma = type->isa<Sigma>())

15 [&](auto d) { return !visited.contains(d) && interesting_type(d, visited); });

25static void split(DefSet& out, const Def* def, bool as_callee) {
26 if (auto lam = def->isa<Lam>()) {
27 out.insert(lam);
28 } else if (auto [var, lam] = ca_isa_var<Lam>(def); var && lam) {

30 } else if (auto c = isa_clos_lit(def, false)) {

32 } else if (auto q = match<clos>(def)) {
33 split(out, q->arg(), as_callee);
34 } else if (auto proj = def->isa<Extract>()) {
35 split(out, proj->tuple(), as_callee);
36 } else if (auto pack = def->isa<Pack>()) {
37 split(out, pack->body(), as_callee);
38 } else if (auto tuple = def->isa<Tuple>()) {
39 for (auto op : tuple->ops()) split(out, op, as_callee);
40 } else if (as_callee) {

51undo_t LowerTypedClosPrep::set_esc(const Def* def) {
