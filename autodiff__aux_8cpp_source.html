40/// computes pb type `E* -> A*`

44 auto& world = E->world();
45 auto tang_arg = tangent_type_fun(A);
46 auto tang_ret = tangent_type_fun(E);
47 auto pb_ty = world.cn({tang_ret, world.cn(tang_arg)});

51// `A,R` => `(A->R)' = A' -> R' * (R* -> A*)`
52const Pi* autodiff_type_fun(const Def* arg, const Def* ret) {
53 auto& world = arg->world();
54 world.DLOG("autodiff type for {} => {}", arg, ret);
55 auto aug_arg = autodiff_type_fun(arg);
56 auto aug_ret = autodiff_type_fun(ret);
57 world.DLOG("augmented types: {} => {}", aug_arg, aug_ret);
58 if (!aug_arg || !aug_ret) return nullptr;
59 // `Q* -> P*`
60 auto pb_ty = pullback_type(ret, arg);
61 world.DLOG("pb type: {}", pb_ty);
62 // `P' -> Q' * (Q* -> P*)`
63
64 auto deriv_ty = world.cn({aug_arg, world.cn({aug_ret, pb_ty})});
65 world.DLOG("autodiff type: {}", deriv_ty);
66 return deriv_ty;
67}

92const Def* autodiff_type_fun(const Def* ty) {
93 auto& world = ty->world();
94 // TODO: handle DS (operators)
95 if (auto pi = ty->isa<Pi>()) return autodiff_type_fun_pi(pi);
96 // Also handles autodiff call from axiom declaration => abstract => leave it.
97 world.DLOG("AutoDiff on type: {} <{}>", ty, ty->node_name());
98 if (Idx::size(ty)) return ty;
99 if (ty == world.type_nat()) return ty;
100 if (auto arr = ty->isa<Arr>()) {
101 auto shape = arr->shape();
102 auto body = arr->body();
103 auto body_ad = autodiff_type_fun(body);
104 if (!body_ad) return nullptr;
105 return world.arr(shape, body_ad);
106 }
107 if (auto sig = ty->isa<Sigma>()) {
108 // TODO: mut sigma
109 DefArray ops(sig->ops(), [&](const Def* op) { return autodiff_type_fun(op); });
110 world.DLOG("ops: {,}", ops);
111 return world.sigma(ops);
112 }
113 // mem
114 if (match<mem::M>(ty)) return ty;
115 world.WLOG("no-diff type: {}", ty);
116 return nullptr;
117}

198/// The high level view is:
199/// ```
200/// f: B -> C
201/// g: A -> B
202/// f o g := Î» x. f(g(x)) : A -> C
203/// ```
204/// In cps the types look like:
205/// ```
206/// f: cn[B, cn C]
207/// g: cn[A, cn B]
208/// h = f o g
209/// h : cn[A, cn C]
210/// h = Î» a ret_h. g(a,h')
211/// h' : cn B
212/// h' = Î» b. f(b,ret_h)
213/// ```
214const Def* compose_continuation(const Def* f, const Def* g) {
215 assert(is_continuation(f));
216 auto& world = f->world();
217 world.DLOG("compose f (B->C): {} : {}", f, f->type());
218 world.DLOG("compose g (A->B): {} : {}", g, g->type());
219 assert(is_returning_continuation(f));
220 assert(is_returning_continuation(g));
221
222 auto F = f->type()->as<Pi>();
223 auto G = g->type()->as<Pi>();
224
225 auto A = continuation_dom(G);
226 auto B = continuation_codom(G);
227 auto C = continuation_codom(F);
228 // The type check of codom G = dom F is better handled by the application type checking
229
230 world.DLOG("compose f (B->C): {} : {}", f, F);
231 world.DLOG("compose g (A->B): {} : {}", g, G);
232 world.DLOG(" A: {}", A);
233 world.DLOG(" B: {}", B);
234 world.DLOG(" C: {}", C);
235
236 auto H = world.cn({A, world.cn(C)});
237 auto Hcont = world.cn(B);
238
239 auto h = world.mut_lam(H)->set("comp_"s + *f->sym() + "_"s + *g->sym());
240 auto hcont = world.mut_lam(Hcont)->set("comp_"s + *f->sym() + "_"s + *g->sym() + "_cont"s);
241
242 h->app(true, g, {h->var((nat_t)0), hcont});
243
244 hcont->app(true, f,
245 {
246 hcont->var(), // Warning: not var(0) => only one var => normalization flattens tuples down here
247 h->var(1) // ret_var
248 });
249
250 return h;
251}
