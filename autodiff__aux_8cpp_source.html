38/// computes pb type `E* -> A*`

41const Pi* pullback_type(const Def* E, const Def* A) {
42 auto& world = E->world();
43 auto tang_arg = tangent_type_fun(A);
44 auto tang_ret = tangent_type_fun(E);
45 auto pb_ty = world.cn({tang_ret, world.cn({tang_arg})});

49// `A,R` => `(A->R)' = A' -> R' * (R* -> A*)`
50const Pi* autodiff_type_fun(const Def* arg, const Def* ret) {
51 auto& world = arg->world();
52 world.DLOG("autodiff type for {} => {}", arg, ret);
53 auto aug_arg = autodiff_type_fun(arg);
54 auto aug_ret = autodiff_type_fun(ret);
55 world.DLOG("augmented types: {} => {}", aug_arg, aug_ret);
56 if (!aug_arg || !aug_ret) return nullptr;
57 // `Q* -> P*`
58 auto pb_ty = pullback_type(ret, arg);
59 world.DLOG("pb type: {}", pb_ty);
60 // `P' -> Q' * (Q* -> P*)`
61
62 auto deriv_ty = world.cn({aug_arg, world.cn({aug_ret, pb_ty})});
63 world.DLOG("autodiff type: {}", deriv_ty);
64 return deriv_ty;
65}

90const Def* autodiff_type_fun(const Def* ty) {
91 auto& world = ty->world();
92 // TODO: handle DS (operators)
93 if (auto pi = ty->isa<Pi>()) { return autodiff_type_fun_pi(pi); }
94 // Also handles autodiff call from axiom declaration => abstract => leave it.
95 world.DLOG("AutoDiff on type: {} <{}>", ty, ty->node_name());
96 if (Idx::size(ty)) { return ty; }
97 if (ty == world.type_nat()) return ty;
98 if (auto arr = ty->isa<Arr>()) {
99 auto shape = arr->shape();
100 auto body = arr->body();
101 auto body_ad = autodiff_type_fun(body);
102 if (!body_ad) return nullptr;
103 return world.arr(shape, body_ad);
104 }
105 if (auto sig = ty->isa<Sigma>()) {
106 // TODO: nom sigma

108 world.DLOG("ops: {,}", ops);
109 return world.sigma(ops);
110 }
111 // mem
112 if (match<mem::M>(ty)) return ty;
113 world.WLOG("no-diff type: {}", ty);
114 return nullptr;

196/// The high level view is:
197/// ```
198/// f: B -> C
199/// g: A -> B
200/// f o g := λ x. f(g(x)) : A -> C
201/// ```
202/// In cps the types look like:
203/// ```
204/// f: cn[B, cn C]
205/// g: cn[A, cn B]
206/// h = f o g
207/// h : cn[A, cn C]
208/// h = λ a ret_h. g(a,h')
209/// h' : cn B
210/// h' = λ b. f(b,ret_h)
211/// ```
212const Def* compose_continuation(const Def* f, const Def* g) {
213 assert(is_continuation(f));
214 auto& world = f->world();
215 world.DLOG("compose f (B->C): {} : {}", f, f->type());
216 world.DLOG("compose g (A->B): {} : {}", g, g->type());
217 assert(is_returning_continuation(f));
218 assert(is_returning_continuation(g));
219
220 auto F = f->type()->as<Pi>();
221 auto G = g->type()->as<Pi>();
222
223 auto A = continuation_dom(G);
224 auto B = continuation_codom(G);
225 auto C = continuation_codom(F);
226 // The type check of codom G = dom F is better handled by the application type checking
227
228 world.DLOG("compose f (B->C): {} : {}", f, F);
229 world.DLOG("compose g (A->B): {} : {}", g, G);
230 world.DLOG(" A: {}", A);
231 world.DLOG(" B: {}", B);
232 world.DLOG(" C: {}", C);
233
234 auto H = world.cn({A, world.cn(C)});
235 auto Hcont = world.cn(B);
236
237 auto h = world.nom_lam(H, world.dbg("comp_" + f->name() + "_" + g->name()));
238 auto hcont = world.nom_lam(Hcont, world.dbg("comp_" + f->name() + "_" + g->name() + "_cont"));
239
240 h->app(true, g, {h->var((nat_t)0), hcont});
241
242 hcont->app(true, f,
243 {
244 hcont->var(), // Warning: not var(0) => only one var => normalization flattens tuples down here
245 h->var(1) // ret_var
246 });
247
248 return h;
249}
