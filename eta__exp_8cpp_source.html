45/// If a wrapper is somehow reinstantiated again in a different expression, redo eta-conversion.
46/// E.g., say we have `(a, f, g)` and eta-exand to `(a, eta_f, eta_g)`.
47/// But due to beta-reduction we now also have (b, eta_f, eta_g) which renders eta_f and eta_g not unique anymore.
48/// So, we build `(b, eta_f', eta_g')`.
49/// Likewise, we might end up with a call `eta_f (a, b, c)` that we have to eta-reduce again to
50/// `f (a, b, c)`
51const Def* EtaExp::reconvert(const Def* def) {
52 std::vector<std::pair<Lam*, Lam*>> refinements;
53 DefArray new_ops(def->num_ops());
54
55 for (size_t i = 0, e = def->num_ops(); i != e; ++i) {
56 if (auto lam = def->op(i)->isa_nom<Lam>()) {
57 if (auto subst = wrap2subst_.lookup(lam)) {
58 auto [orig, subst_def] = *subst;
59 assert(lam->body()->isa<App>() && lam->body()->as<App>()->callee() == orig);
60 if (isa_callee(def, i)) {
61 new_ops[i] = orig;
62 } else {
63 auto wrap = eta_wrap(orig);
64 refinements.emplace_back(wrap, orig);
65 new_ops[i] = wrap;
66 }
67 continue;
68 }
69 }
70
71 new_ops[i] = def->op(i);
72 }
73
74 auto new_def = def->rebuild(world(), def->type(), new_ops, def->dbg());
75
76 for (auto [wrap, lam] : refinements)
77 wrap2subst_[wrap] = std::pair(lam, new_def);
78
79 return def2exp_[def] = new_def;
80}
