195std::optional<Tok> Lexer::parse_lit() {
196 int base = 10;
197 std::optional<bool> sign;
198
199 if (accept<Append::Off>('+')) {
200 sign = false;
201 } else if (accept<Append::Off>('-')) {
202 if (accept('>')) return tok(Tag::T_arrow);
203 sign = true;
204 }
205
206 // prefix starting with '0'
207 if (accept<Append::Off>('0')) {
208 if (accept<Append::Off>('b')) base = 2;
209 else if (accept<Append::Off>('B')) base = 2;
210 else if (accept<Append::Off>('o')) base = 8;
211 else if (accept<Append::Off>('O')) base = 8;
212 else if (accept<Append::Off>('x')) base = 16;
213 else if (accept<Append::Off>('X')) base = 16;
214 }
215
216 parse_digits(base);
217
218 if (!sign && base == 10) {
219 if (utf8::isrange(ahead(), U'₀', U'₉')) {
220 auto i = strtoull(str_.c_str(), nullptr, 10);
221 std::string mod;
222 while (utf8::isrange(ahead(), U'₀', U'₉')) mod += next() - U'₀' + '0';
223 auto m = strtoull(mod.c_str(), nullptr, 10);
224 return Tok{loc_, world().lit_idx_mod(m, i)};
225 } else if (accept<Append::Off>('_')) {
226 auto i = strtoull(str_.c_str(), nullptr, 10);
227 str_.clear();
228 if (accept(utf8::isdigit)) {
229 parse_digits(10);
230 auto m = strtoull(str_.c_str(), nullptr, 10);
231 return Tok{loc_, world().lit_idx_mod(m, i)};
232 } else {
233 error(loc_, "stray underscore in unsigned literal");
234 auto i = strtoull(str_.c_str(), nullptr, 10);
235 return Tok{loc_, u64(i)};
236 }
237 }
238 }
239
240 bool is_float = false;
241 if (base == 10 || base == 16) {
242 // parse fractional part
243 if (accept('.')) {
244 is_float = true;
245 parse_digits(base);
246 }
247
248 bool has_exp = parse_exp(base);
249 if (base == 16 && is_float && !has_exp) error(loc_, "hexadecimal floating constants require an exponent");
250 is_float |= has_exp;
251 }
252
253 if (sign && str_.empty()) {
254 error(loc_, "stray '{}'", *sign ? "-" : "+");
255 return {};
256 }
257
258 if (is_float && base == 16) str_.insert(0, "0x"sv);
259 if (sign && *sign) str_.insert(0, "-"sv);
260
261 if (is_float) return Tok{loc_, f64(strtod (str_.c_str(), nullptr ))};
262 if (sign) return Tok{loc_, u64(strtoll (str_.c_str(), nullptr, base))};
263 else return Tok{loc_, u64(strtoull(str_.c_str(), nullptr, base))};
264}

266void Lexer::parse_digits(int base /*= 10*/) {
267 switch (base) {
268 // clang-format off
269 case 2: while (accept(utf8::isbdigit)) {} break;
270 case 8: while (accept(utf8::isodigit)) {} break;
271 case 10: while (accept(utf8::isdigit)) {} break;
272 case 16: while (accept(utf8::isxdigit)) {} break;
273 // clang-format on
274 default: fe::unreachable();
275 }
