136Tok Lexer::lex_lit() {

139 // clang-format off
140 auto parse_digits = [&]() {
141 switch (base) {
142 case 2: while (accept_if([](int i) { return '0' <= i && i <= '1'; })) {} break;
143 case 8: while (accept_if([](int i) { return '0' <= i && i <= '7'; })) {} break;
144 case 10: while (accept_if(isdigit)) {} break;
145 case 16: while (accept_if(isxdigit)) {} break;
146 }
147 };
148

150
151 // prefix starting with '0'
152 if (accept('0', false)) {
153 if (accept('b', false)) base = 2;

155 else if (accept('o', false)) base = 8;
156 }

158 parse_digits();

162 // parse fractional part
163 if (accept('.')) {

166 }
167
