186std::optional<Tok> Lexer::parse_lit() {
187 int base = 10;
188 std::optional<bool> sign;
189
190 if (accept('+', false)) {
191 sign = false;
192 } else if (accept('-', false)) {
193 if (accept('>')) return tok(Tag::T_arrow);
194 sign = true;
195 }
196
197 // prefix starting with '0'
198 if (accept('0', false)) {
199 if (accept('b', false)) base = 2;
200 else if (accept('B', false)) base = 2;
201 else if (accept('o', false)) base = 8;
202 else if (accept('O', false)) base = 8;
203 else if (accept('x', false)) base = 16;
204 else if (accept('X', false)) base = 16;
205 }
206
207 parse_digits(base);
208
209 if (!sign && base == 10) {
210 if (issubscsr(ahead())) {
211 auto i = strtoull(str_.c_str(), nullptr, 10);
212 std::string mod;
213 while (issubscsr(ahead())) {
214 mod += ahead() - U'₀' + '0';
215 next();
216 }
217 auto m = strtoull(mod.c_str(), nullptr, 10);
218 return Tok{loc_, world().lit_idx_mod(m, i)};
219 } else if (accept('_', false)) {
220 auto i = strtoull(str_.c_str(), nullptr, 10);
221 str_.clear();
222 if (accept_if(isdigit)) {
223 parse_digits(10);
224 auto m = strtoull(str_.c_str(), nullptr, 10);
225 return Tok{loc_, world().lit_idx_mod(m, i)};
226 } else {
227 error(loc_, "stray underscore in unsigned literal");
228 auto i = strtoull(str_.c_str(), nullptr, 10);
229 return Tok{loc_, u64(i)};
230 }
231 }
232 }
233
234 bool is_float = false;
235 if (base == 10 || base == 16) {
236 // parse fractional part
237 if (accept('.')) {
238 is_float = true;
239 parse_digits(base);
240 }
241
242 bool has_exp = parse_exp(base);
243 if (base == 16 && is_float && !has_exp) error(loc_, "hexadecimal floating constants require an exponent");
244 is_float |= has_exp;
245 }
246
247 if (sign && str_.empty()) {
248 error(loc_, "stray '{}'", *sign ? "-" : "+");
249 return {};
250 }
251
252 if (is_float && base == 16) str_.insert(0, "0x"sv);
253 if (sign && *sign) str_.insert(0, "-"sv);
254
255 if (is_float) return Tok{loc_, f64(strtod (str_.c_str(), nullptr ))};
256 if (sign) return Tok{loc_, u64(strtoll (str_.c_str(), nullptr, base))};
257 else return Tok{loc_, u64(strtoull(str_.c_str(), nullptr, base))};
258}

260void Lexer::parse_digits(int base /*= 10*/) {
261 switch (base) {
262 case 2: while (accept_if([](int i) { return '0' <= i && i <= '1'; })) {} break;
263 case 8: while (accept_if([](int i) { return '0' <= i && i <= '7'; })) {} break;
264 case 10: while (accept_if(isdigit)) {} break;
265 case 16: while (accept_if(isxdigit)) {} break;
266 default: unreachable();
267 }
268};

270bool Lexer::parse_exp(int base /*= 10*/) {
271 if (accept_if(base == 10 ? [](int i) { return i == 'e' || i == 'E'; }
272 : [](int i) { return i == 'p' || i == 'P'; })) {
273 accept_if(issign);
274 if (!isdigit(ahead())) error(loc_, "exponent has no digits");
275 parse_digits();
276 return true;
277 }
