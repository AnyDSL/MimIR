154std::optional<Tok> Lexer::parse_lit() {
155 int base = 10;
156 std::optional<bool> sign;
157
158 if (accept('+', false)) {
159 sign = false;
160 } else if (accept('-', false)) {
161 if (accept('>')) return tok(Tok::Tag::T_arrow);
162 sign = true;
163 }
164
165 // prefix starting with '0'
166 if (accept('0', false)) {
167 if (accept('b', false)) base = 2;
168 else if (accept('B', false)) base = 2;
169 else if (accept('o', false)) base = 8;
170 else if (accept('O', false)) base = 8;
171 else if (accept('x', false)) base = 16;
172 else if (accept('X', false)) base = 16;
173 }
174
175 parse_digits(base);
176
177 if (!sign && base == 10) {
178 if (issubscsr(ahead())) {
179 auto i = strtoull(str_.c_str(), nullptr, 10);
180 std::string mod;
181 while (issubscsr(ahead())) {
182 mod += ahead() - U'₀' + '0';
183 next();
184 }
185 auto m = strtoull(mod.c_str(), nullptr, 10);
186 return Tok{loc_, world().lit_int_mod(m, i)};
187 } else if (accept('_', false)) {
188 auto i = strtoull(str_.c_str(), nullptr, 10);
189 str_.clear();
190 if (accept_if(isdigit)) {
191 parse_digits(10);
192 auto m = strtoull(str_.c_str(), nullptr, 10);
193 return Tok{loc_, world().lit_int_mod(m, i)};
194 } else {
195 err(loc_, "stray underscore in unsigned literal");
196 auto i = strtoull(str_.c_str(), nullptr, 10);
197 return Tok{loc_, u64(i)};
198 }
199 }
200 }
201
202 bool is_float = false;
203 if (base == 10 || base == 16) {
204 // parse fractional part
205 if (accept('.')) {
206 is_float = true;
207 parse_digits(base);
208 }
209
210 is_float |= parse_exp(base);
211 }
212
213 if (sign && str_.empty()) {
214 err(loc_, "stray '{}'", *sign ? "-" : "+");
215 return {};
216 }
217
218 if (is_float && base == 16) str_.insert(0, "0x"sv);
219 if (sign && *sign) str_.insert(0, "-"sv);
220
221 if (is_float) return Tok{loc_, r64(strtod (str_.c_str(), nullptr ))};
222 if (sign) return Tok{loc_, u64(strtoll (str_.c_str(), nullptr, base))};
223 else return Tok{loc_, u64(strtoull(str_.c_str(), nullptr, base))};
224}

226void Lexer::parse_digits(int base /*= 10*/) {
227 switch (base) {
228 case 2: while (accept_if([](int i) { return '0' <= i && i <= '1'; })) {} break;
229 case 8: while (accept_if([](int i) { return '0' <= i && i <= '7'; })) {} break;
230 case 10: while (accept_if(isdigit)) {} break;
231 case 16: while (accept_if(isxdigit)) {} break;
232 default: unreachable();
233 }
234};

236bool Lexer::parse_exp(int base /*= 10*/) {
237 if (accept_if(base == 10 ? [](int i) { return i == 'e' || i == 'E'; }
238 : [](int i) { return i == 'p' || i == 'P'; })) {
239 accept_if(issign);
240 if (!isdigit(ahead())) err(loc_, "exponent has no digits");
241 parse_digits();
242 return true;
243 }
244
245 if (base == 16) {
246 err(loc_, "hexadecimal floating constants require an exponent");
247 return {};
248 }
249 return false;

265void Lexer::emit_md(bool start_of_file) {
266 if (!start_of_file) md_fence();

268 do {
