470 /// @name put into sea of nodes
471 ///@{
472 template<class T, class... Args>
473 const T* unify(size_t num_ops, Args&&... args) {
474 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
475 assert(!def->isa_nom());
476#if THORIN_ENABLE_CHECKS
477 if (flags().trace_gids) outln("{}: {}", def->node_name(), def->gid());
478 if (flags().reeval_breakpoints && state_.breakpoints.contains(def->gid())) thorin::breakpoint();
479#endif
480 if (is_frozen()) {
481 --state_.pod.curr_gid;
482 auto i = move_.defs.find(def);
483 arena_.deallocate<T>(def);
484 if (i != move_.defs.end()) return static_cast<const T*>(*i);
485 return nullptr;
486 }
487
488 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
489 arena_.deallocate<T>(def);
490 return static_cast<const T*>(*i);
491 }
492#if THORIN_ENABLE_CHECKS
493 if (!flags().reeval_breakpoints && state_.breakpoints.contains(def->gid())) thorin::breakpoint();
494#endif
495 def->finalize();
496 return def;
497 }

499 template<class T, class... Args>
500 T* insert(size_t num_ops, Args&&... args) {
501 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
502#if THORIN_ENABLE_CHECKS
503 if (flags().trace_gids) outln("{}: {}", def->node_name(), def->gid());
504 if (state_.breakpoints.contains(def->gid())) thorin::breakpoint();
505#endif
506 auto [_, ins] = move_.defs.emplace(def);
507 assert_unused(ins);
508 return def;
509 }

539 static_assert(sizeof(Def) == sizeof(T),
540 "you are not allowed to introduce any additional data in subclasses of Def");
541 Lock lock;
542 size_t num_bytes = num_bytes_of<T>(num_ops);
543 num_bytes = align(num_bytes);
544 assert(num_bytes < Zone::Size);
545
546 if (index_ + num_bytes >= Zone::Size) {
547 auto zone = new Zone;
548 curr_->next.reset(zone);
549 curr_ = zone;
550 index_ = 0;
551 }
552

554 assert(result->num_ops() == num_ops);
555 index_ += num_bytes;
556 assert(index_ % alignof(T) == 0);
557
558 return result;
559 }
