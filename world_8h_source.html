644 auto num_ops = T::Num_Ops;
645 if constexpr (T::Num_Ops == std::dynamic_extent) {
646 auto&& last = std::get<sizeof...(Args) - 1>(std::forward_as_tuple(std::forward<Args>(args)...));

652 assert(!def->isa_mut());

656#ifdef MIM_ENABLE_CHECKS

663 deallocate<T>(state, def);
664 if (i != move_.defs.end()) return static_cast<const T*>(*i);
665 return nullptr;
666 }

673#ifdef MIM_ENABLE_CHECKS
674 if (!flags().reeval_breakpoints && breakpoints().contains(def->gid())) fe::breakpoint();
