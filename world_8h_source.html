533 struct Zone {
534 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
535 char buffer[Size];
536 std::unique_ptr<Zone> next;
537 };
538
539#ifndef NDEBUG
540 struct Lock {
541 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
542 ~Lock() { guard_ = !guard_; }
543 static bool guard_;
544 };
545#else
546 struct Lock { ~Lock() {} };
547#endif
548 template<class T, class... Args>
549 T* allocate(size_t num_ops, Args&&... args) {
550 static_assert(sizeof(Def) == sizeof(T), "you are not allowed to introduce any additional data in subclasses of Def");
551 Lock lock;
552 size_t num_bytes = num_bytes_of<T>(num_ops);
553 num_bytes = align(num_bytes);
554 assert(num_bytes < Zone::Size);
555
556 if (buffer_index_ + num_bytes >= Zone::Size) {
557 auto zone = new Zone;
558 curr_zone_->next.reset(zone);
559 curr_zone_ = zone;
560 buffer_index_ = 0;
561 }
562
563 auto result = new (curr_zone_->buffer + buffer_index_) T(args...);
564 assert(result->num_ops() == num_ops);
565 buffer_index_ += num_bytes;
566 assert(buffer_index_ % alignof(T) == 0);
567
568 return result;
569 }
570
571 template<class T>
572 void deallocate(const T* def) {
573 size_t num_bytes = num_bytes_of<T>(def->num_ops());
574 num_bytes = align(num_bytes);
575 def->~T();
576 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
577 buffer_index_-= num_bytes;
578 assert(buffer_index_ % alignof(T) == 0);
579 }
580
581 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*)-1); }
582
583 template<class T> static constexpr inline size_t num_bytes_of(size_t num_ops) {
584 size_t result = sizeof(Def) + sizeof(const Def*)*num_ops;
585 return align(result);
586 }
