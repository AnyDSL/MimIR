460 ///@{
461 template<class T, class... Args>
462 const T* unify(size_t num_ops, Args&&... args) {
463 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
464 if (auto loc = emit_loc()) def->set(loc);
465 assert(!def->isa_mut());
466#if THORIN_ENABLE_CHECKS
467 if (flags().trace_gids) outln("{}: {} - {}", def->node_name(), def->gid(), def->flags());
468 if (flags().reeval_breakpoints && breakpoints().contains(def->gid())) thorin::breakpoint();
469#endif
470 if (is_frozen()) {
471 --state_.pod.curr_gid;
472 auto i = move_.defs.find(def);
473 arena_.deallocate<T>(def);
474 if (i != move_.defs.end()) return static_cast<const T*>(*i);
475 return nullptr;
476 }
477
478 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
479 arena_.deallocate<T>(def);
480 return static_cast<const T*>(*i);
481 }
482#if THORIN_ENABLE_CHECKS
483 if (!flags().reeval_breakpoints && breakpoints().contains(def->gid())) thorin::breakpoint();
484#endif
485 def->finalize();
486 return def;
487 }

489 template<class T, class... Args>
490 T* insert(size_t num_ops, Args&&... args) {
491 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
492 if (auto loc = emit_loc()) def->set(loc);

495 if (breakpoints().contains(def->gid())) thorin::breakpoint();

511 struct Zone {
512 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
513 char buffer[Size];
514 std::unique_ptr<Zone> next;
515 };
516
517#if (!defined(_MSC_VER) && defined(NDEBUG))
518 struct Lock {
519 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
520 ~Lock() { guard_ = !guard_; }
521 static bool guard_;
522 };
523#else
524 struct Lock {
525 ~Lock() {}
526 };
527#endif
528 template<class T, class... Args>
529 T* allocate(size_t num_ops, Args&&... args) {
530 static_assert(sizeof(Def) == sizeof(T),
531 "you are not allowed to introduce any additional data in subclasses of Def");
532 Lock lock;
533 size_t num_bytes = num_bytes_of<T>(num_ops);
534 num_bytes = align(num_bytes);
535 assert(num_bytes < Zone::Size);
536
537 if (index_ + num_bytes >= Zone::Size) {
538 auto zone = new Zone;
539 curr_->next.reset(zone);
540 curr_ = zone;
541 index_ = 0;
542 }
543
544 auto result = new (curr_->buffer + index_) T(std::forward<Args&&>(args)...);
545 assert(result->num_ops() == num_ops);
546 index_ += num_bytes;
547 assert(index_ % alignof(T) == 0);
548
549 return result;
550 }
551
552 template<class T>
553 void deallocate(const T* def) {
554 size_t num_bytes = num_bytes_of<T>(def->num_ops());
555 num_bytes = align(num_bytes);
556 def->~T();
557 if (ptrdiff_t(index_ - num_bytes) > 0) // don't care otherwise
558 index_ -= num_bytes;
559 assert(index_ % alignof(T) == 0);
560 }
561
562 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
563
564 template<class T>
565 static constexpr inline size_t num_bytes_of(size_t num_ops) {
566 size_t result = sizeof(Def) + sizeof(void*) * num_ops;
567 return align(result);
568 }
