602 class Arena {
603 public:
604 Arena()
605 : root_zone_(new Zone) // don't use 'new Zone()' - we keep the allocated Zone uninitialized
606 , curr_zone_(root_zone_.get()) {}
607
608 struct Zone {
609 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
610 char buffer[Size];
611 std::unique_ptr<Zone> next;
612 };
613
614#if (!defined(_MSC_VER) && defined(NDEBUG))
615 struct Lock {
616 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
617 ~Lock() { guard_ = !guard_; }
618 static bool guard_;
619 };
620#else
621 struct Lock {
622 ~Lock() {}
623 };
624#endif
625 template<class T, class... Args>
626 T* allocate(size_t num_ops, Args&&... args) {
627 static_assert(sizeof(Def) == sizeof(T),
628 "you are not allowed to introduce any additional data in subclasses of Def");
629 Lock lock;
630 size_t num_bytes = num_bytes_of<T>(num_ops);
631 num_bytes = align(num_bytes);
632 assert(num_bytes < Zone::Size);
633
634 if (buffer_index_ + num_bytes >= Zone::Size) {
635 auto zone = new Zone;
636 curr_zone_->next.reset(zone);
637 curr_zone_ = zone;
638 buffer_index_ = 0;
639 }
640
641 auto result = new (curr_zone_->buffer + buffer_index_) T(std::forward<Args&&>(args)...);
642 assert(result->num_ops() == num_ops);
643 buffer_index_ += num_bytes;
644 assert(buffer_index_ % alignof(T) == 0);
645
646 return result;
647 }
648
649 template<class T>
650 void deallocate(const T* def) {
651 size_t num_bytes = num_bytes_of<T>(def->num_ops());
652 num_bytes = align(num_bytes);
653 def->~T();
654 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
655 buffer_index_ -= num_bytes;
656 assert(buffer_index_ % alignof(T) == 0);
657 }
658
659 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
660
661 template<class T>
662 static constexpr inline size_t num_bytes_of(size_t num_ops) {
663 size_t result = sizeof(Def) + sizeof(const Def*) * num_ops;
664 return align(result);
665 }
