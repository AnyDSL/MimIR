655 class Arena {
656 public:
657 Arena()
658 : root_zone_(new Zone) // don't use 'new Zone()' - we keep the allocated Zone uninitialized
659 , curr_zone_(root_zone_.get()) {}
660
661 struct Zone {
662 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
663 char buffer[Size];
664 std::unique_ptr<Zone> next;
665 };
666
667#if (!defined(_MSC_VER) && defined(NDEBUG))
668 struct Lock {
669 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
670 ~Lock() { guard_ = !guard_; }
671 static bool guard_;
672 };
673#else
674 struct Lock {
675 ~Lock() {}
676 };
677#endif
678 template<class T, class... Args>
679 T* allocate(size_t num_ops, Args&&... args) {
680 static_assert(sizeof(Def) == sizeof(T),
681 "you are not allowed to introduce any additional data in subclasses of Def");
682 Lock lock;
683 size_t num_bytes = num_bytes_of<T>(num_ops);
684 num_bytes = align(num_bytes);
685 assert(num_bytes < Zone::Size);
686
687 if (buffer_index_ + num_bytes >= Zone::Size) {
688 auto zone = new Zone;
689 curr_zone_->next.reset(zone);
690 curr_zone_ = zone;
691 buffer_index_ = 0;
692 }
693
694 auto result = new (curr_zone_->buffer + buffer_index_) T(std::forward<Args&&>(args)...);
695 assert(result->num_ops() == num_ops);
696 buffer_index_ += num_bytes;
697 assert(buffer_index_ % alignof(T) == 0);
698
699 return result;
700 }
701
702 template<class T>
703 void deallocate(const T* def) {
704 size_t num_bytes = num_bytes_of<T>(def->num_ops());
705 num_bytes = align(num_bytes);
706 def->~T();
707 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
708 buffer_index_ -= num_bytes;
709 assert(buffer_index_ % alignof(T) == 0);
710 }
711
712 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
713
714 template<class T>
715 static constexpr inline size_t num_bytes_of(size_t num_ops) {
716 size_t result = sizeof(Def) + sizeof(const Def*) * num_ops;
717 return align(result);
718 }
