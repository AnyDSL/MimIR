24/// The World represents the whole program and manages creation of MimIR nodes (Def%s).
25/// Def%s are hashed into an internal HashSet.
26/// The World's factory methods just calculate a hash and lookup the Def, if it is already present, or create a new one
27/// otherwise. This corresponds to value numbering.
28///
29/// You can create several worlds.
30/// All worlds are completely independent from each other.
31///
32/// Note that types are also just Def%s and will be hashed as well.

555 template<class T, class... Args> const T* unify(size_t num_ops, Args&&... args) {
556 auto state = move_.arena.state();
557 auto def = allocate<T>(num_ops, std::forward<Args&&>(args)...);
558 if (auto loc = get_loc()) def->set(loc);
559 assert(!def->isa_mut());
560#ifdef MIM_ENABLE_CHECKS
561 if (flags().trace_gids) outln("{}: {} - {}", def->node_name(), def->gid(), def->flags());
562 if (flags().reeval_breakpoints && breakpoints().contains(def->gid())) fe::breakpoint();
563#endif
564 if (is_frozen()) {
565 --state_.pod.curr_gid;
566 auto i = move_.defs.find(def);
567 deallocate<T>(state, def);
568 if (i != move_.defs.end()) return static_cast<const T*>(*i);
569 return nullptr;
570 }
571
572 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
573 deallocate<T>(state, def);
574 return static_cast<const T*>(*i);
575 }
576#ifdef MIM_ENABLE_CHECKS
577 if (!flags().reeval_breakpoints && breakpoints().contains(def->gid())) fe::breakpoint();
578#endif
579 return def;
580 }
