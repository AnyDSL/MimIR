452 ///@{
453 template<class T, class... Args> const T* unify(size_t num_ops, Args&&... args) {
454 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
455 if (auto loc = emit_loc()) def->set(loc);
456 assert(!def->isa_mut());
457#ifdef THORIN_ENABLE_CHECKS
458 if (flags().trace_gids) outln("{}: {} - {}", def->node_name(), def->gid(), def->flags());
459 if (flags().reeval_breakpoints && breakpoints().contains(def->gid())) thorin::breakpoint();
460#endif
461 if (is_frozen()) {
462 --state_.pod.curr_gid;
463 auto i = move_.defs.find(def);
464 arena_.deallocate<T>(def);
465 if (i != move_.defs.end()) return static_cast<const T*>(*i);
466 return nullptr;
467 }
468
469 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
470 arena_.deallocate<T>(def);
471 return static_cast<const T*>(*i);
472 }
473#ifdef THORIN_ENABLE_CHECKS
474 if (!flags().reeval_breakpoints && breakpoints().contains(def->gid())) thorin::breakpoint();
475#endif
476 def->finalize();
477 return def;
478 }

480 template<class T, class... Args> T* insert(size_t num_ops, Args&&... args) {
481 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);

485 if (breakpoints().contains(def->gid())) thorin::breakpoint();

501 struct Zone {
502 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
503 char buffer[Size];
504 std::unique_ptr<Zone> next;
505 };
506
507#if (!defined(_MSC_VER) && defined(NDEBUG))
508 struct Lock {
509 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
510 ~Lock() { guard_ = !guard_; }
511 static bool guard_;
512 };
513#else
514 struct Lock {
515 ~Lock() {}
516 };
517#endif
518 template<class T, class... Args> T* allocate(size_t num_ops, Args&&... args) {
519 static_assert(sizeof(Def) == sizeof(T),
520 "you are not allowed to introduce any additional data in subclasses of Def");
521 Lock lock;
522 size_t num_bytes = num_bytes_of<T>(num_ops);
523 num_bytes = align(num_bytes);
524 assert(num_bytes < Zone::Size);
525
526 if (index_ + num_bytes >= Zone::Size) {
527 auto zone = new Zone;
528 curr_->next.reset(zone);
529 curr_ = zone;
530 index_ = 0;
531 }
532
533 auto result = new (curr_->buffer + index_) T(std::forward<Args&&>(args)...);
534 assert(result->num_ops() == num_ops);
535 index_ += num_bytes;
536 assert(index_ % alignof(T) == 0);
537
538 return result;
539 }

541 template<class T> void deallocate(const T* def) {
542 size_t num_bytes = num_bytes_of<T>(def->num_ops());
543 num_bytes = align(num_bytes);
544 def->~T();
545 if (ptrdiff_t(index_ - num_bytes) > 0) // don't care otherwise
546 index_ -= num_bytes;
547 assert(index_ % alignof(T) == 0);
548 }

550 static constexpr size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
551
552 template<class T> static constexpr size_t num_bytes_of(size_t num_ops) {
553 size_t result = sizeof(Def) + sizeof(void*) * num_ops;
554 return align(result);
555 }
