447 template<class T, class... Args>
448 const T* unify(size_t num_ops, Args&&... args) {
449 auto def = arena_.allocate<T>(num_ops, std::forward<Args&&>(args)...);
450 assert(!def->isa_nom());
451#if THORIN_ENABLE_CHECKS
452 if (flags().trace_gids) outln("{}: {}", def->node_name(), def->gid());
453 if (flags().reeval_breakpoints && state_.breakpoints.contains(def->gid())) thorin::breakpoint();
454#endif
455 if (is_frozen()) {
456 --state_.pod.curr_gid;
457 auto i = move_.defs.find(def);
458 arena_.deallocate<T>(def);
459 if (i != move_.defs.end()) return static_cast<const T*>(*i);
460 return nullptr;
461 }
462
463 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
464 arena_.deallocate<T>(def);
465 return static_cast<const T*>(*i);
466 }
467#if THORIN_ENABLE_CHECKS
468 if (!flags().reeval_breakpoints && state_.breakpoints.contains(def->gid())) thorin::breakpoint();
469#endif
470 def->finalize();
471 return def;
472 }

475 T* insert(size_t num_ops, Args&&... args) {

481 auto [_, ins] = move_.defs.emplace(def);

489 class Arena {
490 public:
491 Arena()
492 : root_(new Zone) // don't use 'new Zone()' - we keep the allocated Zone uninitialized
493 , curr_(root_.get()) {}
494
495 struct Zone {
496 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
497 char buffer[Size];
498 std::unique_ptr<Zone> next;
499 };
500
501#if (!defined(_MSC_VER) && defined(NDEBUG))
502 struct Lock {
503 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
504 ~Lock() { guard_ = !guard_; }
505 static bool guard_;
506 };
507#else
508 struct Lock {
509 ~Lock() {}
510 };
511#endif
512 template<class T, class... Args>
513 T* allocate(size_t num_ops, Args&&... args) {
514 static_assert(sizeof(Def) == sizeof(T),
515 "you are not allowed to introduce any additional data in subclasses of Def");
516 Lock lock;
517 size_t num_bytes = num_bytes_of<T>(num_ops);
518 num_bytes = align(num_bytes);
519 assert(num_bytes < Zone::Size);
520
521 if (index_ + num_bytes >= Zone::Size) {
522 auto zone = new Zone;
523 curr_->next.reset(zone);
524 curr_ = zone;
525 index_ = 0;
526 }
527
528 auto result = new (curr_->buffer + index_) T(std::forward<Args&&>(args)...);
529 assert(result->num_ops() == num_ops);
530 index_ += num_bytes;
531 assert(index_ % alignof(T) == 0);
532
533 return result;
534 }
535
536 template<class T>
537 void deallocate(const T* def) {
538 size_t num_bytes = num_bytes_of<T>(def->num_ops());
539 num_bytes = align(num_bytes);
540 def->~T();
541 if (ptrdiff_t(index_ - num_bytes) > 0) // don't care otherwise
542 index_ -= num_bytes;
543 assert(index_ % alignof(T) == 0);
544 }

546 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
547
