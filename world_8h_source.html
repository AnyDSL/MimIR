507 class Arena {
508 public:
509 Arena()
510 : root_zone_(new Zone) // don't use 'new Zone()' - we keep the allocated Zone uninitialized
511 , curr_zone_(root_zone_.get()) {}
512
513 struct Zone {
514 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
515 char buffer[Size];
516 std::unique_ptr<Zone> next;
517 };
518
519#if (!defined(_MSC_VER) && defined(NDEBUG))
520 struct Lock {
521 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
522 ~Lock() { guard_ = !guard_; }
523 static bool guard_;
524 };
525#else
526 struct Lock { ~Lock() {} };
527#endif
528 template<class T, class... Args>
529 T* allocate(size_t num_ops, Args&&... args) {
530 static_assert(sizeof(Def) == sizeof(T), "you are not allowed to introduce any additional data in subclasses of Def");
531 Lock lock;
532 size_t num_bytes = num_bytes_of<T>(num_ops);
533 num_bytes = align(num_bytes);
534 assert(num_bytes < Zone::Size);
535
536 if (buffer_index_ + num_bytes >= Zone::Size) {
537 auto zone = new Zone;
538 curr_zone_->next.reset(zone);
539 curr_zone_ = zone;
540 buffer_index_ = 0;
541 }
542
543 auto result = new (curr_zone_->buffer + buffer_index_) T(args...);
544 assert(result->num_ops() == num_ops);
545 buffer_index_ += num_bytes;
546 assert(buffer_index_ % alignof(T) == 0);
547
548 return result;
549 }
550
551 template<class T>
552 void deallocate(const T* def) {
553 size_t num_bytes = num_bytes_of<T>(def->num_ops());
554 num_bytes = align(num_bytes);
555 def->~T();
556 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
557 buffer_index_-= num_bytes;
558 assert(buffer_index_ % alignof(T) == 0);
559 }
560
561 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*)-1); }
562
563 template<class T> static constexpr inline size_t num_bytes_of(size_t num_ops) {
564 size_t result = sizeof(Def) + sizeof(const Def*)*num_ops;
565 return align(result);
566 }

571 size_t buffer_index_ = 0;
