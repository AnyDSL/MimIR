646 class Arena {
647 public:
648 Arena()
649 : root_zone_(new Zone) // don't use 'new Zone()' - we keep the allocated Zone uninitialized
650 , curr_zone_(root_zone_.get()) {}
651
652 struct Zone {
653 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
654 char buffer[Size];
655 std::unique_ptr<Zone> next;
656 };
657
658#if (!defined(_MSC_VER) && defined(NDEBUG))
659 struct Lock {
660 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
661 ~Lock() { guard_ = !guard_; }
662 static bool guard_;
663 };
664#else
665 struct Lock {
666 ~Lock() {}
667 };
668#endif
669 template<class T, class... Args>
670 T* allocate(size_t num_ops, Args&&... args) {
671 static_assert(sizeof(Def) == sizeof(T),
672 "you are not allowed to introduce any additional data in subclasses of Def");
673 Lock lock;
674 size_t num_bytes = num_bytes_of<T>(num_ops);
675 num_bytes = align(num_bytes);
676 assert(num_bytes < Zone::Size);
677
678 if (buffer_index_ + num_bytes >= Zone::Size) {
679 auto zone = new Zone;
680 curr_zone_->next.reset(zone);
681 curr_zone_ = zone;
682 buffer_index_ = 0;
683 }
684
685 auto result = new (curr_zone_->buffer + buffer_index_) T(std::forward<Args&&>(args)...);
686 assert(result->num_ops() == num_ops);
687 buffer_index_ += num_bytes;
688 assert(buffer_index_ % alignof(T) == 0);
689
690 return result;
691 }
692
693 template<class T>
694 void deallocate(const T* def) {
695 size_t num_bytes = num_bytes_of<T>(def->num_ops());
696 num_bytes = align(num_bytes);
697 def->~T();
698 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
699 buffer_index_ -= num_bytes;
700 assert(buffer_index_ % alignof(T) == 0);
701 }
702
703 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*) - 1); }
704
705 template<class T>
706 static constexpr inline size_t num_bytes_of(size_t num_ops) {
707 size_t result = sizeof(Def) + sizeof(const Def*) * num_ops;
708 return align(result);
709 }

714 size_t buffer_index_ = 0;
