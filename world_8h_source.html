598 const T* unify(Args&&... args) {
599 auto num_ops = T::Num_Ops;
600 if constexpr (T::Num_Ops == std::dynamic_extent) {
601 auto&& last = std::get<sizeof...(Args) - 1>(std::forward_as_tuple(std::forward<Args>(args)...));
602 num_ops = last.size();
603 }

605 auto state = move_.arena.defs.state();
606 auto def = allocate<T>(num_ops, std::forward<Args&&>(args)...);
607 if (auto loc = get_loc()) def->set(loc);
608 assert(!def->isa_mut());
609#ifdef MIM_ENABLE_CHECKS
610 if (flags().trace_gids) outln("{}: {} - {}", def->node_name(), def->gid(), def->flags());
611 if (flags().reeval_breakpoints && breakpoints().contains(def->gid())) fe::breakpoint();
612#endif
613 if (is_frozen()) {
614 auto i = move_.defs.find(def);
615 deallocate<T>(state, def);
616 if (i != move_.defs.end()) return static_cast<const T*>(*i);
617 return nullptr;
618 }
619
620 if (auto [i, ins] = move_.defs.emplace(def); !ins) {
621 deallocate<T>(state, def);
622 return static_cast<const T*>(*i);
623 }
624#ifdef MIM_ENABLE_CHECKS
625 if (!flags().reeval_breakpoints && breakpoints().contains(def->gid())) fe::breakpoint();
626#endif
627 return def;
628 }
