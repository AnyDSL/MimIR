522 struct Zone {
523 static const size_t Size = 1024 * 1024 - sizeof(std::unique_ptr<int>); // 1MB - sizeof(next)
524 char buffer[Size];
525 std::unique_ptr<Zone> next;
526 };
527
528#if (!defined(_MSC_VER) && defined(NDEBUG))
529 struct Lock {
530 Lock() { assert((guard_ = !guard_) && "you are not allowed to recursively invoke allocate"); }
531 ~Lock() { guard_ = !guard_; }
532 static bool guard_;
533 };
534#else
535 struct Lock { ~Lock() {} };
536#endif
537 template<class T, class... Args>
538 T* allocate(size_t num_ops, Args&&... args) {
539 static_assert(sizeof(Def) == sizeof(T), "you are not allowed to introduce any additional data in subclasses of Def");
540 Lock lock;
541 size_t num_bytes = num_bytes_of<T>(num_ops);
542 num_bytes = align(num_bytes);
543 assert(num_bytes < Zone::Size);
544
545 if (buffer_index_ + num_bytes >= Zone::Size) {
546 auto zone = new Zone;
547 curr_zone_->next.reset(zone);
548 curr_zone_ = zone;
549 buffer_index_ = 0;
550 }
551
552 auto result = new (curr_zone_->buffer + buffer_index_) T(args...);
553 assert(result->num_ops() == num_ops);
554 buffer_index_ += num_bytes;
555 assert(buffer_index_ % alignof(T) == 0);
556
557 return result;
558 }
559
560 template<class T>
561 void deallocate(const T* def) {
562 size_t num_bytes = num_bytes_of<T>(def->num_ops());
563 num_bytes = align(num_bytes);
564 def->~T();
565 if (ptrdiff_t(buffer_index_ - num_bytes) > 0) // don't care otherwise
566 buffer_index_-= num_bytes;
567 assert(buffer_index_ % alignof(T) == 0);
568 }
569
570 static constexpr inline size_t align(size_t n) { return (n + (sizeof(void*) - 1)) & ~(sizeof(void*)-1); }
571
572 template<class T> static constexpr inline size_t num_bytes_of(size_t num_ops) {
573 size_t result = sizeof(Def) + sizeof(const Def*)*num_ops;
574 return align(result);
575 }
