147/// Organizes several Phase%s as a pipeline.
148class Pipeline : public Phase {
149public:
150 Pipeline(World& world)
151 : Phase(world, "pipeline", true) {}
152
153 void start() override;
154
155 /// @name phases
156 ///@{
157 const auto& phases() const { return phases_; }
158 /// Add a Phase.
159 /// You don't need to pass the World to @p args - it will be passed automatically.
160 /// If @p P is a Pass, this method will wrap this in a PassPhase.
161 template<class P, class... Args>
162 auto add(Args&&... args) {
163 if constexpr (std::is_base_of_v<Pass, P>) {
164 return add<PassPhase<P>>(std::forward<Args>(args)...);
165 } else {
166 auto p = std::make_unique<P>(world(), std::forward<Args&&>(args)...);
167 auto phase = p.get();
168 phases_.emplace_back(std::move(p));
169 if (phase->is_dirty()) phases_.emplace_back(std::make_unique<Cleanup>(world()));
170 return phase;
171 }
172 }
173 ///@}
174
175private:
176 std::deque<std::unique_ptr<Phase>> phases_;
177};
