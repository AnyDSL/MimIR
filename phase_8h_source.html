114/// Organizes several Phase%s as a pipeline.

122 /// @name phases
123 ///@{
124 const auto& phases() const { return phases_; }
125 /// Add a Phase.
126 /// You don't need to pass the World to @p args - it will be passed automatically.
127 /// If @p P is a Pass, this method will wrap this in a PassPhase.
128 template<class P, class... Args>
129 auto add(Args&&... args) {
130 if constexpr (std::is_base_of_v<Pass, P>) {
131 return add<PassPhase<P>>(std::forward<Args>(args)...);
132 } else {
133 auto p = std::make_unique<P>(world(), std::forward<Args&&>(args)...);
134 auto phase = p.get();
135 phases_.emplace_back(std::move(p));
136 if (phase->is_dirty()) phases_.emplace_back(std::make_unique<Cleanup>(world()));
137 return phase;
138 }
139 }
140 ///@}
141
142private:
143 std::deque<std::unique_ptr<Phase>> phases_;
144};
