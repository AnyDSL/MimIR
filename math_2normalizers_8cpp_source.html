81template<class Id, Id id, nat_t w>
82Res fold(u64 a, u64 b) {
83 using T = w2f<w>;
84 auto x = bitcast<T>(a), y = bitcast<T>(b);
85 if constexpr (std::is_same_v<Id, arith>) {
86 if constexpr (false) {}
87 else if constexpr (id == arith::add) return x + y;
88 else if constexpr (id == arith::sub) return x - y;
89 else if constexpr (id == arith::mul) return x * y;
90 else if constexpr (id == arith::div) return x / y;
91 else if constexpr (id == arith::rem) return rem(x, y);
92 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
93 } else if constexpr (std::is_same_v<Id, math::extrema>) {
94 if (x == T(-0.0) && y == T(+0.0)) return (id == extrema::fmin || id == extrema::ieee754min) ? x : y;
95 if (x == T(+0.0) && y == T(-0.0)) return (id == extrema::fmin || id == extrema::ieee754min) ? y : x;
96
97 if constexpr (id == extrema::fmin || id == extrema::fmax) {
98 return id == extrema::fmin ? std::fmin(x, y) : std::fmax(x, y);
99 } else if constexpr (id == extrema::ieee754min || id == extrema::ieee754max) {
100 if (std::isnan(x)) return x;
101 if (std::isnan(y)) return y;
102 return id == extrema::ieee754min ? std::fmin(x, y) : std::fmax(x, y);
103 } else {
104 []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
105 }
106 } else if constexpr (std::is_same_v<Id, pow>) {
107 return std::pow(a, b);
108 } else if constexpr (std::is_same_v<Id, cmp>) {
109 bool res = false;
110 res |= ((id & cmp::u) != cmp::f) && std::isunordered(x, y);
111 res |= ((id & cmp::g) != cmp::f) && x > y;
112 res |= ((id & cmp::l) != cmp::f) && x < y;
113 res |= ((id & cmp::e) != cmp::f) && x == y;
114 return res;
115 } else {
116 []<bool flag = false>() { static_assert(flag, "missing tag"); }();
117 }
118}

121template<class Id, Id id>
122static const Def* fold(World& world, const Def* type, const Def* a, const Def* dbg) {
123 if (a->isa<Bot>()) return world.bot(type, dbg);
124 auto la = a->isa<Lit>();
125
126 if (la) {
127 nat_t width = *isa_f(a->type());
128 Res res;
129 switch (width) {
130#define CODE(i) \
131 case i: res = fold<Id, id, i>(la->get()); break;
132 THORIN_16_32_64(CODE)
133#undef CODE
134 default: unreachable();
135 }
136
137 return world.lit(type, *res, dbg);
138 }
139
140 return nullptr;
141}

143/// @attention Note that @p a and @p b are passed by reference as fold also commutes if possible. @sa commute().
144template<class Id, Id id>
145static const Def* fold(World& world, const Def* type, const Def*& a, const Def*& b, const Def* dbg) {
146 auto la = a->isa<Lit>(), lb = b->isa<Lit>();
147
148 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type, dbg);
149
150 if (la && lb) {
151 nat_t width = *isa_f(a->type());
152 Res res;
153 switch (width) {
154#define CODE(i) \
155 case i: res = fold<Id, id, i>(la->get(), lb->get()); break;
156 THORIN_16_32_64(CODE)
157#undef CODE
158 default: unreachable();
159 }
160
161 return world.lit(type, *res, dbg);
162 }
163
164 commute(id, a, b);
165 return nullptr;
166}

168/// Reassociates @p a und @p b according to following rules.
169/// We use the following naming convention while literals are prefixed with an 'l':
170/// ```
171/// a op b
172/// (x op y) op (z op w)
173///
174/// (1) la op (lz op w) -> (la op lz) op w
175/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
176/// (3) a op (lz op w) -> lz op (a op w)
177/// (4) (lx op y) op b -> lx op (y op b)
178/// ```
179template<class Id>
180static const Def*
181reassociate(Id id, World& /*world*/, [[maybe_unused]] const App* ab, const Def* a, const Def* b, const Def* dbg) {
182 if (!is_associative(id)) return nullptr;
183
184 auto la = a->isa<Lit>();
185 auto xy = match<Id>(id, a);
186 auto zw = match<Id>(id, b);
187 auto lx = xy ? xy->arg(0)->template isa<Lit>() : nullptr;
188 auto lz = zw ? zw->arg(0)->template isa<Lit>() : nullptr;
189 auto y = xy ? xy->arg(1) : nullptr;
190 auto w = zw ? zw->arg(1) : nullptr;
191
192 std::function<const Def*(const Def*, const Def*)> make_op;
193
194 // build mode for all new ops by using the least upper bound of all involved apps
195 nat_t mode = Mode::bot;
196 auto check_mode = [&](const App* app) {
197 auto app_m = isa_lit(app->arg(0));
198 if (!app_m || !(*app_m & Mode::reassoc)) return false;
199 mode &= *app_m; // least upper bound
200 return true;
201 };
202
203 if (!check_mode(ab)) return nullptr;
204 if (lx && !check_mode(xy->decurry())) return nullptr;
205 if (lz && !check_mode(zw->decurry())) return nullptr;
206
207 make_op = [&](const Def* a, const Def* b) { return op(id, mode, a, b, dbg); };
208
209 if (la && lz) return make_op(make_op(la, lz), w); // (1)
210 if (lx && lz) return make_op(make_op(lx, lz), make_op(y, w)); // (2)
211 if (lz) return make_op(lz, make_op(a, w)); // (3)
212 if (lx) return make_op(lx, make_op(y, b)); // (4)
213
214 return nullptr;
215}

217template<arith id>
218const Def* normalize_arith(const Def* type, const Def* c, const Def* arg, const Def* dbg) {
219 auto& world = type->world();
220 auto callee = c->as<App>();
221 auto [a, b] = arg->projs<2>();
222 auto mode = callee->arg();
223 auto lm = isa_lit(mode);
224 auto w = isa_f(a->type());
225
226 if (auto result = fold<arith, id>(world, type, a, b, dbg)) return result;
227
228 // clang-format off
229 // TODO check mode properly
230 if (lm && *lm == Mode::fast) {
231 if (auto la = a->isa<Lit>()) {
232 if (la == lit_f(world, *w, 0.0)) {
233 switch (id) {
234 case arith::add: return b; // 0 + b -> b
235 case arith::sub: break;
236 case arith::mul: return la; // 0 * b -> 0
237 case arith::div: return la; // 0 / b -> 0
238 case arith::rem: return la; // 0 % b -> 0
239 }
240 }
241
242 if (la == lit_f(world, *w, 1.0)) {
243 switch (id) {
244 case arith::add: break;
245 case arith::sub: break;
246 case arith::mul: return b; // 1 * b -> b
247 case arith::div: break;
248 case arith::rem: break;
249 }
250 }
251 }
252
253 if (auto lb = b->isa<Lit>()) {
254 if (lb == lit_f(world, *w, 0.0)) {
255 switch (id) {
256 case arith::sub: return a; // a - 0 -> a
257 case arith::div: break;
258 case arith::rem: break;
259 default: unreachable();
260 // add, mul are commutative, the literal has been normalized to the left
261 }
262 }
263 }
264
265 if (a == b) {
266 switch (id) {
267 case arith::add: return math::op(arith::mul, mode, lit_f(world, *w, 2.0), a, dbg); // a + a -> 2 * a
268 case arith::sub: return lit_f(world, *w, 0.0); // a - a -> 0
269 case arith::mul: break;
270 case arith::div: return lit_f(world, *w, 1.0); // a / a -> 1
271 case arith::rem: break;
272 }
273 }
274 }
275 // clang-format on
276
277 if (auto res = reassociate<arith>(id, world, callee, a, b, dbg)) return res;
278
279 return world.raw_app(type, callee, {a, b}, dbg);
280}

365template<conv id>
366const Def* normalize_conv(const Def* dst_t, const Def* c, const Def* x, const Def* dbg) {
367 auto& world = dst_t->world();
368 auto callee = c->as<App>();
369 auto s_t = x->type()->as<App>();
370 auto d_t = dst_t->as<App>();
371 auto s = s_t->arg();
372 auto d = d_t->arg();
373 auto ls = isa_lit(s);
374 auto ld = isa_lit(d);
375
376 if (s_t == d_t) return x;
377 if (x->isa<Bot>()) return world.bot(d_t, dbg);
378
379 if (auto l = isa_lit(x); l && ls && ld) {
380 constexpr bool sf = id == conv::f2f || id == conv::f2s || id == conv::f2u;
381 constexpr bool df = id == conv::f2f || id == conv::s2f || id == conv::u2f;
382 constexpr nat_t min_s = sf ? 16 : 1;
383 constexpr nat_t min_d = df ? 16 : 1;
384 auto sw = sf ? isa_f(s_t) : Idx::size2bitwidth(*ls);
385 auto dw = df ? isa_f(d_t) : Idx::size2bitwidth(*ld);
386
387 if (sw && dw) {
388 Res res;
389 // clang-format off
390 if (false) {}
391#define M(S, D) \
392 else if (S == *sw && D == *dw) { \
393 if constexpr (S >= min_s && D >= min_d) \
394 res = fold<conv, id, S, D>(*l); \
395 else \
396 goto out; \
397 }
398 M( 1, 1) M( 1, 8) M( 1, 16) M( 1, 32) M( 1, 64)
399 M( 8, 1) M( 8, 8) M( 8, 16) M( 8, 32) M( 8, 64)
400 M(16, 1) M(16, 8) M(16, 16) M(16, 32) M(16, 64)
401 M(32, 1) M(32, 8) M(32, 16) M(32, 32) M(32, 64)
402 M(64, 1) M(64, 8) M(64, 16) M(64, 32) M(64, 64)
403
404 else unreachable();
405 // clang-format on
406 return world.lit(d_t, *res, dbg);
407 }
408 }
409out:
410 return world.raw_app(dst_t, callee, x, dbg);
411}
