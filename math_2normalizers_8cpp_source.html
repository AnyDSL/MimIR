121template<class Id, Id id, nat_t w>
122Res fold(u64 a, u64 b) {
123 using T = w2f<w>;
124 auto x = bitcast<T>(a), y = bitcast<T>(b);
125 if constexpr (std::is_same_v<Id, arith>) {
126 if constexpr (false) {}
127 else if constexpr (id == arith::add) return x + y;
128 else if constexpr (id == arith::sub) return x - y;
129 else if constexpr (id == arith::mul) return x * y;
130 else if constexpr (id == arith::div) return x / y;
131 else if constexpr (id == arith::rem) return rem(x, y);
132 else []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
133 } else if constexpr (std::is_same_v<Id, math::extrema>) {
134 if (x == T(-0.0) && y == T(+0.0)) return (id == extrema::fmin || id == extrema::ieee754min) ? x : y;
135 if (x == T(+0.0) && y == T(-0.0)) return (id == extrema::fmin || id == extrema::ieee754min) ? y : x;
136
137 if constexpr (id == extrema::fmin || id == extrema::fmax) {
138 return id == extrema::fmin ? std::fmin(x, y) : std::fmax(x, y);
139 } else if constexpr (id == extrema::ieee754min || id == extrema::ieee754max) {
140 if (std::isnan(x)) return x;
141 if (std::isnan(y)) return y;
142 return id == extrema::ieee754min ? std::fmin(x, y) : std::fmax(x, y);
143 } else {
144 []<bool flag = false>() { static_assert(flag, "missing sub tag"); }();
145 }
146 } else if constexpr (std::is_same_v<Id, pow>) {
147 return std::pow(a, b);
148 } else if constexpr (std::is_same_v<Id, cmp>) {
149 bool res = false;
150 res |= ((id & cmp::u) != cmp::f) && std::isunordered(x, y);
151 res |= ((id & cmp::g) != cmp::f) && x > y;
152 res |= ((id & cmp::l) != cmp::f) && x < y;
153 res |= ((id & cmp::e) != cmp::f) && x == y;
154 return res;
155 } else {
156 []<bool flag = false>() { static_assert(flag, "missing tag"); }();
157 }
158}

207/// Reassociates @p a und @p b according to following rules.
208/// We use the following naming convention while literals are prefixed with an 'l':
209/// ```
210/// a op b
211/// (x op y) op (z op w)
212///
213/// (1) la op (lz op w) -> (la op lz) op w
214/// (2) (lx op y) op (lz op w) -> (lx op lz) op (y op w)
215/// (3) a op (lz op w) -> lz op (a op w)
216/// (4) (lx op y) op b -> lx op (y op b)
217/// ```
218template<class Id>
219Ref reassociate(Id id, World& world, [[maybe_unused]] const App* ab, Ref a, Ref b) {
220 if (!is_associative(id)) return nullptr;
221
222 if (auto xy = match<Id>(id, a)) {
223 if (auto zw = match<Id>(id, b)) {
224 auto la = Lit::isa(a);
225 auto [x, y] = xy->template args<2>();
226 auto [z, w] = zw->template args<2>();
227 auto lx = Lit::isa(x);
228 auto lz = Lit::isa(z);
229
230 std::function<Ref(Ref, Ref)> make_op;
231
232 // build mode for all new ops by using the least upper bound of all involved apps
233 auto mode = (nat_t)Mode::bot;
234 auto check_mode = [&](const App* app) {
235 auto app_m = Lit::isa(app->arg(0));
236 if (!app_m || !(*app_m & Mode::reassoc)) return false;
237 mode &= *app_m; // least upper bound
238 return true;
239 };
240
241 if (!check_mode(ab)) return nullptr;
242 if (lx && !check_mode(xy->decurry())) return nullptr;
243 if (lz && !check_mode(zw->decurry())) return nullptr;
244
245 make_op = [&](Ref a, Ref b) { return world.call(id, mode, Defs{a, b}); };
246
247 if (la && lz) return make_op(make_op(a, z), w); // (1)
248 if (lx && lz) return make_op(make_op(x, z), make_op(y, w)); // (2)
249 if (lz) return make_op(z, make_op(a, w)); // (3)
250 if (lx) return make_op(x, make_op(y, b)); // (4)
251 }
252 }
253
254 return nullptr;
255}

266template<arith id>
267Ref normalize_arith(Ref type, Ref c, Ref arg) {
268 auto& world = type->world();
269 auto callee = c->as<App>();
270 auto [a, b] = arg->projs<2>();
271 auto mode = callee->arg();
272 auto lm = Lit::isa(mode);
273 auto w = isa_f(a->type());
274
275 if (auto result = fold<arith, id>(world, type, a, b)) return result;
276
277 // clang-format off
278 // TODO check mode properly
279 if (lm && *lm == Mode::fast) {
280 if (auto la = a->isa<Lit>()) {
281 if (la == lit_f(world, *w, 0.0)) {
282 switch (id) {
283 case arith::add: return b; // 0 + b -> b
284 case arith::sub: break;
285 case arith::mul: return la; // 0 * b -> 0
286 case arith::div: return la; // 0 / b -> 0
287 case arith::rem: return la; // 0 % b -> 0
288 }
289 }
290
291 if (la == lit_f(world, *w, 1.0)) {
292 switch (id) {
293 case arith::add: break;
294 case arith::sub: break;
295 case arith::mul: return b; // 1 * b -> b
296 case arith::div: break;
297 case arith::rem: break;
298 }
299 }
300 }
301
302 if (auto lb = b->isa<Lit>()) {
303 if (lb == lit_f(world, *w, 0.0)) {
304 switch (id) {
305 case arith::sub: return a; // a - 0 -> a
306 case arith::div: break;
307 case arith::rem: break;
308 default: unreachable();
309 // add, mul are commutative, the literal has been normalized to the left
310 }
311 }
312 }
313
314 if (a == b) {
315 switch (id) {
316 case arith::add: return world.call(arith::mul, mode, Defs{lit_f(world, *w, 2.0), a}); // a + a -> 2 * a
317 case arith::sub: return lit_f(world, *w, 0.0); // a - a -> 0
318 case arith::mul: break;
319 case arith::div: return lit_f(world, *w, 1.0); // a / a -> 1
320 case arith::rem: break;
321 }
322 }
323 }
324 // clang-format on
325
326 if (auto res = reassociate<arith>(id, world, callee, a, b)) return res;
327
328 return world.raw_app(type, callee, {a, b});
329}
