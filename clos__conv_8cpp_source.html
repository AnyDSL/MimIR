139void ClosConv::rewrite_body(Lam* new_lam, Def2Def& subst) {
140 auto& w = world();
141 auto it = closures_.find(new_lam);
142 assert(it != closures_.end() && "closure should have a stub if rewrite_body is called!");
143 auto [old_fn, num_fvs, env, new_fn] = it->second;
144
145 if (!old_fn->is_set()) return;
146
147 w.DLOG("rw body: {} [old={}, env={}]\nt", new_fn, old_fn, env);
148 auto env_param = new_fn->var(Clos_Env_Param, w.dbg("closure_env"));
149 if (num_fvs == 1) {
150 subst.emplace(env, env_param);
151 } else {
152 for (size_t i = 0; i < num_fvs; i++) {

154 auto dbg = (fv->name().empty()) ? w.dbg("fv_" + std::to_string(i)) : w.dbg("fv_" + env->op(i)->name());

156 }
157 }
158
159 auto params =
160 w.tuple(DefArray(old_fn->num_doms(), [&](auto i) { return new_lam->var(skip_env(i), old_fn->var(i)->dbg()); }));
161 subst.emplace(old_fn->var(), params);
162
163 auto filter = rewrite(new_fn->filter(), subst);
164 auto body = rewrite(new_fn->body(), subst);
165 new_fn->set(filter, body);

168const Def* ClosConv::rewrite(const Def* def, Def2Def& subst) {
169 switch (def->node()) {
170 case Node::Type:
171 case Node::Univ:
172 case Node::Nat:
173 case Node::Bot: // TODO This is used by the AD stuff????
174 case Node::Top: return def;
175 default: break;
176 }
177
178 auto& w = world();
179 auto map = [&](const Def* new_def) {
180 subst[def] = new_def;
181 assert(subst[def] == new_def);
182 return new_def;
183 };
184
185 if (auto i = subst.find(def); i != subst.end()) {
186 return i->second;
187 } else if (auto pi = def->isa<Pi>(); pi && pi->is_cn()) {
188 return map(closure_type(pi, subst));
189 } else if (auto lam = def->isa_nom<Lam>(); lam && lam->type()->is_cn()) {
190 auto& w = world();
191 auto [_, __, fv_env, new_lam] = make_stub(lam, subst);
192 auto closure_type = rewrite(lam->type(), subst);
193 auto env = rewrite(fv_env, subst);
194 auto closure = clos_pack(env, new_lam, closure_type);
195 w.DLOG("RW: pack {} ~> {} : {}", lam, closure, closure_type);
196 return map(closure);
197 } else if (auto q = match(clos::ret, def)) {
198 if (auto ret_lam = q->arg()->isa_nom<Lam>()) {
199 // assert(ret_lam && ret_lam->is_basicblock());
200 // Note: This should be cont_lam's only occurance after η-expansion, so its okay to
201 // put into the local subst only
202 auto new_doms = DefArray(ret_lam->num_doms(), [&](auto i) { return rewrite(ret_lam->dom(i), subst); });
203 auto new_lam = ret_lam->stub(w, w.cn(new_doms), ret_lam->dbg());
204 subst[ret_lam] = new_lam;
205 if (ret_lam->is_set()) {
206 new_lam->set_filter(rewrite(ret_lam->filter(), subst));
207 new_lam->set_body(rewrite(ret_lam->body(), subst));
208 }
209 return new_lam;
210 }
211 } else if (auto q = match<clos>(def); q && (q.id() == clos::fstclassBB || q.id() == clos::freeBB)) {
212 // Note: Same thing about η-conversion applies here
213 auto bb_lam = q->arg()->isa_nom<Lam>();
214 assert(bb_lam && bb_lam->is_basicblock());
215 auto [_, __, ___, new_lam] = make_stub({}, bb_lam, subst);
216 subst[bb_lam] = clos_pack(w.tuple(), new_lam, rewrite(bb_lam->type(), subst));
217 rewrite_body(new_lam, subst);
218 return map(subst[bb_lam]);
219 } else if (auto [var, lam] = ca_isa_var<Lam>(def); var && lam && lam->ret_var() == var) {
220 // HACK to rewrite a retvar that is defined in an enclosing lambda
221 // If we put external bb's into the env, this should never happen
222 auto new_lam = make_stub(lam, subst).fn;
223 auto new_idx = skip_env(as_lit(var->index()));
224 return map(new_lam->var(new_idx));
225 }
226
227 auto new_type = rewrite(def->type(), subst);
228 auto new_dbg = (def->dbg()) ? rewrite(def->dbg(), subst) : nullptr;
229
230 if (auto nom = def->isa_nom()) {
231 if (auto global = def->isa_nom<Global>()) {
232 if (auto i = glob_noms_.find(global); i != glob_noms_.end()) return i->second;
233 auto subst = Def2Def();
234 return glob_noms_[nom] = rewrite_nom(global, new_type, new_dbg, subst);
235 }
236 assert(!isa_clos_type(nom));
237 w.DLOG("RW: nom {}", nom);
238 auto new_nom = rewrite_nom(nom, new_type, new_dbg, subst);
239 // Try to reduce the amount of noms that are created
240 if (!nom->isa_nom<Global>() && Checker(w).equiv(nom, new_nom, nullptr)) return map(nom);
241 if (auto restruct = new_nom->restructure()) return map(restruct);
242 return map(new_nom);
243 } else {
244 auto new_ops = DefArray(def->num_ops(), [&](auto i) { return rewrite(def->op(i), subst); });
245 if (auto app = def->isa<App>(); app && new_ops[0]->type()->isa<Sigma>()) {
246 return map(clos_apply(new_ops[0], new_ops[1]));
247 }else if(def->isa<Axiom>()){
248 return def;
249 }else{
250 return map(def->rebuild(w, new_type, new_ops, new_dbg));
251 }
252 }
253
254 thorin::unreachable();
255}

291 auto& w = world();
292 auto env = w.tuple(DefArray(fvs.begin(), fvs.end()));
293 auto num_fvs = fvs.size();
294 auto env_type = rewrite(env->type(), subst);

301 w.DLOG("wrap ext lam: {} -> stub: {}, ext: {}", old_lam, new_lam, new_ext_lam);

303 old_lam->make_internal();
304 new_ext_lam->make_external();
305 new_ext_lam->app(false, new_lam, clos_insert_env(env, new_ext_lam->var()));
306 new_lam->set(old_lam->filter(), old_lam->body());
307 } else {
308 new_ext_lam->set(nullptr, nullptr);
309 new_lam->app(false, new_ext_lam, clos_remove_env(new_lam->var()));
310 }
311 } else {
312 new_lam->set(old_lam->filter(), old_lam->body());
313 }
314 w.DLOG("STUB {} ~~> ({}, {})", old_lam, env, new_lam);
315 auto closure = ClosureStub{old_lam, num_fvs, env, new_lam};
316 closures_.emplace(old_lam, closure);

342void FreeDefAna::split_fd(Node* node, const Def* fd, bool& init_node, NodeQueue& worklist) {
343 assert(!match<mem::M>(fd) && "mem tokens must not be free");
344 if (is_toplevel(fd)) return;
345 if (auto [var, lam] = ca_isa_var<Lam>(fd); var && lam) {
346 if (var != lam->ret_var()) { node->add_fvs(fd); }
347 } else if (auto q = match(clos::freeBB, fd)) {
348 node->add_fvs(q);
349 } else if (auto pred = fd->isa_nom()) {
350 if (pred != node->nom) {
351 auto [pnode, inserted] = build_node(pred, worklist);
352 node->preds.push_back(pnode);
353 pnode->succs.push_back(node);
354 init_node |= inserted;
355 }
356 } else if (fd->dep() == Dep::Var && !fd->isa<Tuple>()) {
357 // Note: Var's can still have Def::Top, if their type is a nom!
358 // So the first case is *not* redundant
359 node->add_fvs(fd);
360 } else if (is_memop_res(fd)) {
361 // Results of memops must not be floated down
362 node->add_fvs(fd);
363 } else {
364 for (auto op : fd->ops()) split_fd(node, op, init_node, worklist);
365 }
366}

368std::pair<FreeDefAna::Node*, bool> FreeDefAna::build_node(Def* nom, NodeQueue& worklist) {
369 auto& w = world();
370 auto [p, inserted] = lam2nodes_.emplace(nom, nullptr);
371 if (!inserted) return {p->second.get(), false};
372 w.DLOG("FVA: create node: {}", nom);
373 p->second = std::make_unique<Node>(Node{nom, {}, {}, {}, 0});
374 auto node = p->second.get();
375 auto scope = Scope(nom);
376 bool init_node = false;
377 for (auto v : scope.free_defs()) { split_fd(node, v, init_node, worklist); }
378 if (!init_node) {
379 worklist.push(node);
380 w.DLOG("FVA: init {}", nom);
381 }
382 return {node, true};
383}

385void FreeDefAna::run(NodeQueue& worklist) {
386 // auto& w = world();

388 while (!worklist.empty()) {
389 auto node = worklist.front();
390 worklist.pop();
391 // w.DLOG("FA: iter {}: {}", iter, node->nom);
392 if (is_done(node)) continue;
393 auto changed = is_bot(node);
394 mark(node);
395 for (auto p : node->preds) {
396 auto& pfvs = p->fvs;
397 for (auto&& pfv : pfvs) { changed |= node->add_fvs(pfv).second; }
398 // w.DLOG("\tFV({}) ∪= FV({}) = {{{, }}}\b", node->nom, p->nom, pfvs);
399 }
400 if (changed) {
401 for (auto s : node->succs) { worklist.push(s); }
402 }
403 iter++;
404 }
405 // w.DLOG("FVA: done");
