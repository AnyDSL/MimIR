22 if (!proj) return false;

27void FreeDefAna::split_fd(Node* node, const Def* fd, bool& init_node, NodeQueue& worklist) {
28 assert(!match<mem::M>(fd) && "mem tokens must not be free");
29 if (is_toplevel(fd)) return;
30 if (auto [var, lam] = ca_isa_var<Lam>(fd); var && lam) {
31 if (var != lam->ret_var()) { node->add_fvs(fd); }
32 } else if (auto q = match(attr::freeBB, fd)) {
33 node->add_fvs(q);
34 } else if (auto pred = fd->isa_nom()) {
35 if (pred != node->nom) {
36 auto [pnode, inserted] = build_node(pred, worklist);
37 node->preds.push_back(pnode);
38 pnode->succs.push_back(node);
39 init_node |= inserted;
40 }
41 } else if (fd->dep() == Dep::Var && !fd->isa<Tuple>()) {
42 // Note: Var's can still have Def::Top, if their type is a nom!
43 // So the first case is *not* redundant
44 node->add_fvs(fd);
45 } else if (is_memop_res(fd)) {
46 // Results of memops must not be floated down
47 node->add_fvs(fd);
48 } else {
49 for (auto op : fd->ops()) split_fd(node, op, init_node, worklist);
50 }
51}

53std::pair<FreeDefAna::Node*, bool> FreeDefAna::build_node(Def* nom, NodeQueue& worklist) {
54 auto& w = world();
55 auto [p, inserted] = lam2nodes_.emplace(nom, nullptr);
56 if (!inserted) return {p->second.get(), false};
57 w.DLOG("FVA: create node: {}", nom);
58 p->second = std::make_unique<Node>(Node{nom, {}, {}, {}, 0});
59 auto node = p->second.get();
60 auto scope = Scope(nom);
61 bool init_node = false;
62 for (auto v : scope.free_defs()) { split_fd(node, v, init_node, worklist); }
63 if (!init_node) {
64 worklist.push(node);
65 w.DLOG("FVA: init {}", nom);
66 }
67 return {node, true};
68}

70void FreeDefAna::run(NodeQueue& worklist) {
71 int iter = 0;
72 while (!worklist.empty()) {
73 auto node = worklist.front();
74 worklist.pop();
75 // w.DLOG("FA: iter {}: {}", iter, node->nom);
76 if (is_done(node)) continue;
77 auto changed = is_bot(node);
78 mark(node);
79 for (auto p : node->preds) {
80 auto& pfvs = p->fvs;
81 for (auto&& pfv : pfvs) { changed |= node->add_fvs(pfv).second; }
82 // w.DLOG("\tFV({}) ∪= FV({}) = {{{, }}}\b", node->nom, p->nom, pfvs);
83 }
84 if (changed) {
85 for (auto s : node->succs) { worklist.push(s); }
86 }
87 iter++;
88 }
89}

123void ClosConv::rewrite_body(Lam* new_lam, Def2Def& subst) {
124 auto& w = world();
125 auto it = closures_.find(new_lam);
126 assert(it != closures_.end() && "closure should have a stub if rewrite_body is called!");
127 auto [old_fn, num_fvs, env, new_fn] = it->second;
128
129 if (!old_fn->is_set()) return;
130
131 w.DLOG("rw body: {} [old={}, env={}]\nt", new_fn, old_fn, env);
132 auto env_param = new_fn->var(Clos_Env_Param, w.dbg("closure_env"));
133 if (num_fvs == 1) {
134 subst.emplace(env, env_param);
135 } else {
136 for (size_t i = 0; i < num_fvs; i++) {
137 auto fv = env->op(i);
138 auto dbg = (fv->name().empty()) ? w.dbg("fv_" + std::to_string(i)) : w.dbg("fv_" + env->op(i)->name());
139 subst.emplace(fv, env_param->proj(i, dbg));
140 }
141 }
142
143 auto params =
144 w.tuple(DefArray(old_fn->num_doms(), [&](auto i) { return new_lam->var(skip_env(i), old_fn->var(i)->dbg()); }));
145 subst.emplace(old_fn->var(), params);
146
147 auto filter = rewrite(new_fn->filter(), subst);
148 auto body = rewrite(new_fn->body(), subst);
149 new_fn->set(filter, body);
150}

152const Def* ClosConv::rewrite(const Def* def, Def2Def& subst) {
153 switch (def->node()) {
154 case Node::Type:
155 case Node::Univ:
156 case Node::Nat:
157 case Node::Bot: // TODO This is used by the AD stuff????
158 case Node::Top: return def;
159 default: break;
160 }
161
162 auto& w = world();
163 auto map = [&](const Def* new_def) {
164 subst[def] = new_def;
165 assert(subst[def] == new_def);
166 return new_def;
167 };
168
169 if (auto i = subst.find(def); i != subst.end()) {
170 return i->second;
171 } else if (auto pi = def->isa<Pi>(); pi && pi->is_cn()) {
172 return map(type_clos(pi, subst));
173 } else if (auto lam = def->isa_nom<Lam>(); lam && lam->type()->is_cn()) {
174 auto [_, __, fv_env, new_lam] = make_stub(lam, subst);
175 auto clos_ty = rewrite(lam->type(), subst);
176 auto env = rewrite(fv_env, subst);
177 auto closure = clos_pack(env, new_lam, clos_ty);
178 world().DLOG("RW: pack {} ~> {} : {}", lam, closure, clos_ty);
179 return map(closure);
180 } else if (auto a = match<attr>(def)) {
181 switch (a.id()) {
182 case attr::ret:
183 if (auto ret_lam = a->arg()->isa_nom<Lam>()) {
184 // assert(ret_lam && ret_lam->is_basicblock());
185 // Note: This should be cont_lam's only occurance after η-expansion, so its okay to
186 // put into the local subst only
187 auto new_doms =
188 DefArray(ret_lam->num_doms(), [&](auto i) { return rewrite(ret_lam->dom(i), subst); });
189 auto new_lam = ret_lam->stub(w, w.cn(new_doms), ret_lam->dbg());
190 subst[ret_lam] = new_lam;
191 if (ret_lam->is_set()) {
192 new_lam->set_filter(rewrite(ret_lam->filter(), subst));
193 new_lam->set_body(rewrite(ret_lam->body(), subst));
194 }
195 return new_lam;
196 }
197 break;
198 case attr::fstclassBB:
199 case attr::freeBB: {
200 // Note: Same thing about η-conversion applies here
201 auto bb_lam = a->arg()->isa_nom<Lam>();
202 assert(bb_lam && bb_lam->is_basicblock());
203 auto [_, __, ___, new_lam] = make_stub({}, bb_lam, subst);
204 subst[bb_lam] = clos_pack(w.tuple(), new_lam, rewrite(bb_lam->type(), subst));
205 rewrite_body(new_lam, subst);
206 return map(subst[bb_lam]);
207 }
208 default: break;
209 }
210 } else if (auto [var, lam] = ca_isa_var<Lam>(def); var && lam && lam->ret_var() == var) {
211 // HACK to rewrite a retvar that is defined in an enclosing lambda
212 // If we put external bb's into the env, this should never happen
213 auto new_lam = make_stub(lam, subst).fn;
214 auto new_idx = skip_env(as_lit(var->index()));
215 return map(new_lam->var(new_idx));
216 }
217
218 auto new_type = rewrite(def->type(), subst);
219 auto new_dbg = (def->dbg()) ? rewrite(def->dbg(), subst) : nullptr;
220
221 if (auto nom = def->isa_nom()) {
222 if (auto global = def->isa_nom<Global>()) {
223 if (auto i = glob_noms_.find(global); i != glob_noms_.end()) return i->second;
224 auto subst = Def2Def();
225 return glob_noms_[nom] = rewrite_nom(global, new_type, new_dbg, subst);
226 }
227 assert(!isa_clos_type(nom));
228 w.DLOG("RW: nom {}", nom);
229 auto new_nom = rewrite_nom(nom, new_type, new_dbg, subst);
230 // Try to reduce the amount of noms that are created
231 if (!nom->isa_nom<Global>() && Checker(w).equiv(nom, new_nom, nullptr)) return map(nom);
232 if (auto restruct = new_nom->restructure()) return map(restruct);
233 return map(new_nom);
234 } else {
235 auto new_ops = DefArray(def->num_ops(), [&](auto i) { return rewrite(def->op(i), subst); });
236 if (auto app = def->isa<App>(); app && new_ops[0]->type()->isa<Sigma>()) {
237 return map(clos_apply(new_ops[0], new_ops[1]));
238 } else if (def->isa<Axiom>()) {
239 return def;
240 } else {
241 return map(def->rebuild(w, new_type, new_ops, new_dbg));
242 }
243 }
244
245 thorin::unreachable();
246}

264const Def* ClosConv::type_clos(const Pi* pi, Def2Def& subst, const Def* env_type) {
265 if (auto i = glob_noms_.find(pi); i != glob_noms_.end() && !env_type) return i->second;
266 auto& w = world();
267 auto new_doms = DefArray(pi->num_doms(), [&](auto i) {
268 return (i == pi->num_doms() - 1 && pi->is_returning()) ? rewrite_type_cn(pi->ret_pi(), subst)
269 : rewrite(pi->dom(i), subst);
270 });
271 auto ct = ctype(w, new_doms, env_type);
272 if (!env_type) {
273 glob_noms_.emplace(pi, ct);
274 w.DLOG("C-TYPE: pct {} ~~> {}", pi, ct);
275 } else {
276 w.DLOG("C-TYPE: ct {}, env = {} ~~> {}", pi, env_type, ct);
277 }
278 return ct;

281ClosConv::Stub ClosConv::make_stub(const DefSet& fvs, Lam* old_lam, Def2Def& subst) {
282 auto& w = world();
283 auto env = w.tuple(DefArray(fvs.begin(), fvs.end()));
284 auto num_fvs = fvs.size();
285 auto env_type = rewrite(env->type(), subst);
286 auto new_fn_type = type_clos(old_lam->type(), subst, env_type)->as<Pi>();
287 auto new_lam = old_lam->stub(w, new_fn_type, w.dbg(old_lam->name()));
288 new_lam->set_debug_name((old_lam->is_external() || !old_lam->is_set()) ? "cc_" + old_lam->name() : old_lam->name());
289 if (!isa_workable(old_lam)) {
290 auto new_ext_type = w.cn(clos_remove_env(new_fn_type->dom()));
291 auto new_ext_lam = old_lam->stub(w, new_ext_type, w.dbg(old_lam->name()));
292 w.DLOG("wrap ext lam: {} -> stub: {}, ext: {}", old_lam, new_lam, new_ext_lam);
293 if (old_lam->is_set()) {
294 old_lam->make_internal();
295 new_ext_lam->make_external();
296 new_ext_lam->app(false, new_lam, clos_insert_env(env, new_ext_lam->var()));
297 new_lam->set(old_lam->filter(), old_lam->body());
298 } else {
299 new_ext_lam->set(nullptr, nullptr);
300 new_lam->app(false, new_ext_lam, clos_remove_env(new_lam->var()));
301 }
302 } else {
303 new_lam->set(old_lam->filter(), old_lam->body());
304 }
305 w.DLOG("STUB {} ~~> ({}, {})", old_lam, env, new_lam);
306 auto closure = Stub{old_lam, num_fvs, env, new_lam};
307 closures_.emplace(old_lam, closure);
308 closures_.emplace(closure.fn, closure);
309 return closure;
310}
