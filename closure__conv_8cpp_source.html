13void ClosureConv::run() {
14 auto externals = std::vector(world().externals().begin(), world().externals().end());
15 auto subst = Def2Def();
16 world().DLOG("===== ClosureConv: start =====");
17 for (auto [_, ext_def]: externals) {
18 rewrite(ext_def, subst);
19 }
20 while (!worklist_.empty()) {
21 auto def = worklist_.front();
22 subst = Def2Def();
23 worklist_.pop();
24 if (auto closure = closures_.lookup(def)) {
25 auto new_fn = closure->fn;
26 auto env = closure->env;
27 auto num_fvs = closure->num_fvs;
28 auto old_fn = closure->old_fn;
29
30 world().DLOG("RUN: closure body {} [old={}, env={}]\n\t", new_fn, old_fn, env);
31 auto env_param = new_fn->var(0_u64);
32 if (num_fvs == 1) {
33 subst.emplace(env, env_param);
34 } else {
35 for (size_t i = 0; i < num_fvs; i++) {
36 subst.emplace(env->op(i), world().extract(env_param, i, world().dbg("cc_fv")));
37 }
38 }
39
40 auto params =
41 world().tuple(DefArray(old_fn->num_doms(), [&] (auto i) {
42 return new_fn->var(i + 1);
43 }), world().dbg("cc_param"));
44 subst.emplace(old_fn->var(), params);
45
46 auto filter = (new_fn->filter())
47 ? rewrite(new_fn->filter(), subst)
48 : nullptr; // extern function
49
50 auto body = (new_fn->body())
51 ? rewrite(new_fn->body(), subst)
52 : nullptr;
53
54 new_fn->set_body(body);
55 new_fn->set_filter(filter);
56 }
57 else {
58 world().DLOG("RUN: rewrite def {}\t", def);
59 rewrite(def, subst);
60 }
61 world().DLOG("\b");
62 }
63 world().DLOG("===== ClosureConv: done ======");
64 // world().debug_stream();
65}

67const Def* ClosureConv::rewrite(const Def* def, Def2Def& subst) {
68 switch(def->node()) {
69 case Node::Kind:
70 case Node::Space:
71 case Node::Nat:
72 case Node::Bot:
73 case Node::Top:
74 case Node::Axiom:
75 return def;
76 default:
77 break;
78 }
79
80 auto map = [&](const Def* new_def) {
81 subst.emplace(def, new_def);
82 return new_def;
83 };
84
85 if (subst.contains(def)) {
86 return map(*subst.lookup(def));
87 } else if (auto pi = def->isa<Pi>(); pi && pi->is_cn()) {
88 /* Types: rewrite dom, codom \w susbt here */
89 return map(closure_type(pi, subst));
90 } else if (auto lam = def->isa_nom<Lam>(); lam && lam->type()->is_cn()) {
91 auto [old, num, fv_env, fn] = make_closure(lam, subst);
92 auto closure_type = rewrite(lam->type(), subst);
93 auto env = rewrite(fv_env, subst);
94 auto closure = world().tuple(closure_type, {env, fn});
95 world().DLOG("RW: pack {} ~> {} : {}", lam, closure, closure_type);
96 return map(closure);
97 }
98
99 auto new_type = rewrite(def->type(), subst);
100 auto new_dbg = (def->dbg()) ? rewrite(def->dbg(), subst) : nullptr;
101
102 if (auto nom = def->isa_nom()) {
103 // TODO: Test this
104 world().DLOG("RW: nom {}", nom);
105 auto new_nom = nom->stub(world(), new_type, new_dbg);
106 subst.emplace(nom->var(), new_nom->var());
107 for (size_t i = 0; i < nom->num_ops(); i++) {
108 if (def->op(i))
109 new_nom->set(i, rewrite(def->op(i), subst));
110 }
111 if (auto restruct = new_nom->restructure())
112 return map(restruct);
113 return map(new_nom);
114 } else {
115 auto new_ops = DefArray(def->num_ops(), [&](auto i) {
116 return rewrite(def->op(i), subst);
117 });
118 if (auto app = def->isa<App>(); app && new_ops[0]->type()->isa<Sigma>()) {
119 auto closure = new_ops[0];
120 auto args = new_ops[1];
121 auto env = world().extract(closure, 0_u64, world().dbg("cc_app_env"));
122 auto fn = world().extract(closure, 1_u64, world().dbg("cc_app_f"));
123 world().DLOG("RW: call {} ~> APP {} {} {}", closure, fn, env, args);
124 return map(world().app(fn, DefArray(num_doms(fn), [&](auto i) {
125 return (i == 0) ? env : world().extract(args, i - 1);
126 })));
127 } else {
128 return map(def->rebuild(world(), new_type, new_ops, new_dbg));
129 }
130 }

133const Def* ClosureConv::closure_type(const Pi* pi, Def2Def& subst, const Def* env_type) {
134 if (!env_type) {
135 if (auto pct = closure_types_.lookup(pi))
136 return* pct;
137 auto sigma = world().nom_sigma(world().kind(), 2_u64, world().dbg("cc_pct"));
138 auto new_pi = closure_type(pi, subst, sigma->var());
139 sigma->set(0, sigma->var());
140 sigma->set(1, new_pi);
141 closure_types_.emplace(pi, sigma);
142 world().DLOG("C-TYPE: pct {} ~~> {}", pi, sigma);
143 return sigma;
144 } else {
145 auto dom = world().sigma(DefArray(pi->num_doms() + 1, [&](auto i) {
146 return (i == 0) ? env_type : rewrite(pi->dom(i - 1), subst);
147 }));
148 auto new_pi = world().cn(dom, world().dbg("cc_ct"));
149 world().DLOG("C-TYPE: ct {}, env = {} ~~> {}", pi, env_type, new_pi);
150 return new_pi;
151 }

155void FVA::split_fv(const Def* def, DefSet& out) {
156 if (def->no_dep() || def->isa<Lam>() || def->is_external() || def->isa<Axiom>()) {
157 return;
158 } else if (auto tuple = def->isa<Tuple>()) {
159 for (auto op: tuple->ops())
160 split_fv(op, out);
161 } else {
162 out.emplace(def);

166std::pair<FVA::Node*, bool> FVA::build_node(Lam *lam, NodeQueue& worklist) {
167 auto [p, inserted] = lam2nodes_.emplace(lam, nullptr);
168 if (!inserted)
169 return {p->second.get(), false};
170 world().DLOG("FVA: create node: {}", lam);
171 p->second = std::make_unique<Node>();
172 auto node = p->second.get();
173 node->lam = lam;
174 node->pass_id = 0;
175 auto scope = Scope(lam);
176 node->fvs = DefSet();
177 for (auto v: scope.free_defs()) {
178 split_fv(v, node->fvs);
179 }
180 node->preds = Nodes();
181 node->succs = Nodes();
182 bool init_node = false;
183 for (auto n: scope.free_noms()) {
184 if (auto pred = n->isa_nom<Lam>(); pred && pred != lam) {
185 auto [pnode, inserted] = build_node(pred, worklist);
186 node->preds.push_back(pnode);
187 pnode->succs.push_back(node);
188 init_node |= inserted;
189 }
190 }
191 if (!init_node) {
192 worklist.push(node);
193 world().DLOG("FVA: init {}", lam);
194 }
195 return {node, true};
196}

199void FVA::run(NodeQueue& worklist) {
200 int iter = 0;
201 while(!worklist.empty()) {
202 auto node = worklist.front();
203 worklist.pop();
204 world().DLOG("FA: iter {}: {}", iter, node->lam);
205 if (is_done(node))
206 continue;
207 auto changed = is_bot(node);
208 mark(node);
209 for (auto p: node->preds) {
210 auto& pfvs = p->fvs;
211 changed |= node->fvs.insert(pfvs.begin(), pfvs.end());
212 world().DLOG("\tFV({}) ∪= FV({}) = {{{, }}}\b", node->lam, p->lam, pfvs);
213 }
214 if (changed) {
215 for (auto s: node->succs) {
216 worklist.push(s);
217 }
218 }
219 iter++;
220 }
221 world().DLOG("FVA: done");
222}

235ClosureConv::Closure ClosureConv::make_closure(Lam* fn, Def2Def& subst) {
236 if (auto closure = closures_.lookup(fn))
237 return* closure;
238
239 auto& fv_set = fva_.run(fn);
240 auto fvs = DefVec();
241 auto fvs_types = DefVec();
242 for (auto fv: fv_set) {
243 fvs.emplace_back(fv);
244 fvs_types.emplace_back(rewrite(fv->type(), subst));
245 }
246 auto env = world().tuple(fvs);
247 auto env_type = world().sigma(fvs_types);
248
249 /* Types: rewrite function type here \w fv s */
250 auto new_fn_type = closure_type(fn->type(), subst, env_type)->as<Pi>();
251 auto new_lam = world().nom_lam(new_fn_type, world().dbg(fn->name()));
252 new_lam->set_body(fn->body());
253 new_lam->set_filter(fn->filter());
254 if (fn->is_external()) {
255 fn->make_internal();
256 new_lam->make_external();
257 }
258
259 world().DLOG("STUB {} ~~> ({}, {})", fn, new_lam, env);
260
261 auto closure = Closure{fn, fv_set.size(), env, new_lam};
262 closures_.emplace(fn, closure);
263 closures_.emplace(new_lam, closure);
264 worklist_.emplace(new_lam);
265 return closure;
266}
