116 auto new_lam = reshape_lam(lam);
117 world().DLOG("rewrote lam {} : {}", def, def->type());

119 return new_lam;
120 } else if (auto tuple = def->isa<Tuple>()) {
121 DefArray elements(tuple->ops(), [&](const Def* op) { return rewrite_def(op); });
122 return world().tuple(elements);
123 } else {

125 // Warning: if the new_type is not correct, inconcistencies will arise.
126 auto new_type = rewrite_def(def->type());

184 return world().pi(new_dom, new_cod);

189 [&](auto T) { return reshape_type(T); });

191 const Def* mem = nullptr;
192 // find mem
193 for (auto i = new_types.begin(); i != new_types.end(); i++)
194 if (is_mem_ty(*i) && !mem) mem = *i;

227const Def* Reshape::reshape(std::vector<const Def*>& defs, const Def* T, const Def* mem) {
228 auto& world = T->world();
229 if (should_flatten(T)) {
230 DefArray tuples(T->projs(), [&](auto P) { return reshape(defs, P, mem); });
231 return world.tuple(tuples);
232 } else {
233 const Def* def;
234 if (is_mem_ty(T)) {
235 assert(mem != nullptr && "Reshape: mems not found");
236 def = mem;
237 } else {
238 do {
239 assert(defs.size() > 0 && "Reshape: not enough arguments");
240 def = defs.front();
241 defs.erase(defs.begin());
242 } while (is_mem_ty(def->type()));
243 }
244 // For inner function types, we override the type
245 if (!def->type()->isa<Pi>()) {
246 if (!Check::alpha(def->type(), T)) world.ELOG("reconstruct T {} from def {}", T, def->type());
247 assert(Check::alpha(def->type(), T) && "Reshape: argument type mismatch");
248 }
249 return def;
250 }
