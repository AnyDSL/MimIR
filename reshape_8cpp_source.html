225const Def* Reshape::reshape(std::vector<const Def*>& defs, const Def* T, const Def* mem) {
226 auto& world = T->world();
227 if (should_flatten(T)) {
228 DefArray tuples(T->projs(), [&](auto P) { return reshape(defs, P, mem); });
229 return world.tuple(tuples);
230 } else {
231 const Def* def;
232 if (is_mem_ty(T)) {
233 assert(mem != nullptr && "Reshape: mems not found");
234 def = mem;
235 } else {
236 do {
237 assert(defs.size() > 0 && "Reshape: not enough arguments");
238 def = defs.front();
239 defs.erase(defs.begin());
240 } while (is_mem_ty(def->type()));
241 }
242 // For inner function types, we override the type
243 if (!def->type()->isa<Pi>()) {
244 if (!world.checker().equiv(def->type(), T)) world.ELOG("reconstruct T {} from def {}", T, def->type());
245 assert(world.checker().equiv(def->type(), T) && "Reshape: argument type mismatch");
246 }
247 return def;
248 }
