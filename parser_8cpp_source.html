27Tok Parser::lex() {
28 auto result = ahead();
29 prev() = result.loc();
30 for (size_t i = 0; i < Max_Ahead - 1; ++i) ahead(i) = ahead(i + 1);
31 ahead(Max_Ahead - 1) = lexer().lex();
32 return result;
33}

71 world().VLOG("import: {}", name);
72 auto filename = name;
73
74 if (!filename.has_extension()) filename.replace_extension("thorin"); // TODO error cases

89void Parser::import(std::istream& is, const fs::path* path, std::ostream* md) {
90 world().VLOG("reading: {}", path ? path->string() : "<unknown file>"s);
91 if (!is) error("cannot read file '{}'", *path);
92
93 lexers_.emplace(world(), is, path, md);
94 auto state = state_;
95
96 for (size_t i = 0; i != Max_Ahead; ++i) ahead(i) = lexer().lex();
97 prev() = Loc(path, {1, 1});
98
99 parse_module();
100 state_ = state;
101 lexers_.pop();

176Ref Parser::parse_infix_expr(Tracker track, const Def* lhs, Tok::Prec p) {
177 while (true) {
178 // If operator in ahead has less left precedence: reduce (break).
179 if (ahead().isa(Tag::T_extract)) {
180 if (auto extract = parse_extract_expr(track, lhs, p))
181 lhs = extract;
182 else
183 break;
184 } else if (ahead().isa(Tag::T_arrow)) {
185 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
186 if (l < p) break;
187 lex();
188 auto rhs = parse_expr("right-hand side of an function type", r);
189 lhs = world().pi(lhs, rhs)->set(track.loc());
190 } else {
191 auto [l, r] = Tok::prec(Tok::Prec::App);
192 if (l < p) break;
193 bool is_explicit = accept(Tag::T_at).has_value();
194 if (auto rhs = parse_expr({}, r)) // if we can parse an expression, it's an App
195 lhs = (is_explicit ? world().app(lhs, rhs) : world().iapp(lhs, rhs))->set(track.loc());
196 else
197 return lhs;
198 }
199 }
200
201 return lhs;
202}

204Ref Parser::parse_extract_expr(Tracker track, const Def* lhs, Tok::Prec p) {
205 auto [l, r] = Tok::prec(Tok::Prec::Extract);
206 if (l < p) return nullptr;
207 lex();
208
209 if (ahead().isa(Tag::M_id)) {
210 if (auto sigma = lhs->type()->isa_mut<Sigma>()) {
211 auto tok = eat(Tag::M_id);
212 if (tok.sym() == '_') error(tok.loc(), "you cannot use special symbol '_' as field access");
213
214 if (auto i = def2fields_.find(sigma); i != def2fields_.end()) {
215 if (auto& fields = i->second; fields.size() == sigma->num_ops()) {
216 for (size_t i = 0, n = sigma->num_ops(); i != n; ++i)
217 if (fields[i] == tok.sym()) return world().extract(lhs, n, i)->set(track.loc());
218 }
219 }
220 error(tok.loc(), "could not find elemement '{}' to extract from '{}' of type '{}'", tok.sym(), lhs, sigma);
221 }
222 }
223
224 auto rhs = parse_expr("right-hand side of an extract", r);
225 return world().extract(lhs, rhs)->set(track.loc());
226}

243Ref Parser::parse_primary_expr(std::string_view ctxt) {
244 // clang-format off
245 switch (ahead().tag()) {
246 case Tag::D_quote_l: return parse_arr_expr();
247 case Tag::D_angle_l: return parse_pack_expr();
248 case Tag::D_brace_l: return parse_block_expr();
249 case Tag::D_brckt_l: return parse_sigma_expr();
250 case Tag::D_paren_l: return parse_tuple_expr();
251 case Tag::K_Type: return parse_type_expr();
252 case Tag::K_Univ: lex(); return world().univ();
253 case Tag::K_Bool: lex(); return world().type_bool();
254 case Tag::K_Idx: lex(); return world().type_idx();
255 case Tag::K_Nat: lex(); return world().type_nat();
256 case Tag::K_ff: lex(); return world().lit_ff();
257 case Tag::K_tt: lex(); return world().lit_tt();
258 case Tag::K_Cn:
259 case Tag::K_Fn:
260 case Tag::T_Pi: return parse_pi_expr();
261 case Tag::K_cn:
262 case Tag::K_fn:
263 case Tag::T_lm: return parse_lam();
264 case Tag::T_star: lex(); return world().type();
265 case Tag::T_box: lex(); return world().type<1>();
266 case Tag::T_bot:
267 case Tag::T_top:
268 case Tag::L_s:
269 case Tag::L_u:
270 case Tag::L_f: return parse_lit_expr();
271 case Tag::L_c: return world().lit_int(8, lex().lit_c());
272 case Tag::L_i: return lex().lit_i();
273 case Tag::K_ins: return parse_insert_expr();
274 case Tag::K_ret: return parse_ret_expr();
275 case Tag::M_anx:
276 case Tag::M_id: return scopes_.find(lex().dbg());
277 case Tag::M_str: return world().tuple(lex().sym())->set(prev());
278 default:
279 if (ctxt.empty()) return nullptr;
280 syntax_err("primary expression", ctxt);
281 }
282 // clang-format on
283 return nullptr;
284}

286Ref Parser::parse_arr_expr() {
287 auto track = tracker();
288 scopes_.push();
289 eat(Tag::D_quote_l);
290
291 const Def* shape = nullptr;
292 Arr* arr = nullptr;
293 if (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon)) {
294 auto id = eat(Tag::M_id);
295 eat(Tag::T_colon);
296
297 auto shape = parse_expr("shape of an array");
298 auto type = world().mut_infer_univ();
299 arr = world().mut_arr(type)->set_shape(shape);
300 scopes_.bind(id.dbg(), arr->var()->set(id.dbg()));
301 } else {
302 shape = parse_expr("shape of an array");
303 }
304
305 expect(Tag::T_semicolon, "array");
306 auto body = parse_expr("body of an array");
307 expect(Tag::D_quote_r, "closing delimiter of an array");
308 scopes_.pop();
309
310 if (arr) return arr->set_body(body);
311 return world().arr(shape, body)->set(track.loc());
312}

314Ref Parser::parse_pack_expr() {
315 auto track = tracker();
316 scopes_.push();
317 eat(Tag::D_angle_l);
318
319 if (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon)) {
320 auto id = eat(Tag::M_id);
321 eat(Tag::T_colon);
322
323 auto shape = parse_expr("shape of a pack");
324 auto infer = world().mut_infer(world().type_idx(shape))->set(id.sym());
325 scopes_.bind(id.dbg(), infer);
326
327 expect(Tag::T_semicolon, "pack");
328 auto body = parse_expr("body of a pack");
329 expect(Tag::D_angle_r, "closing delimiter of a pack");
330 scopes_.pop();
331

333 auto pack = world().mut_pack(arr)->set(body);
334 auto var = pack->var();
335 infer->set(var);
336 Scope scope(pack);
337 ScopeRewriter rw(scope);
338 rw.map(infer, var);
339 return pack->reset(rw.rewrite(pack->body()));
340 }
341
342 auto shape = parse_expr("shape of a pack");
343 expect(Tag::T_semicolon, "pack");
344 auto body = parse_expr("body of a pack");
345 expect(Tag::D_angle_r, "closing delimiter of a pack");
346 scopes_.pop();
347 return world().pack(shape, body)->set(track.loc());

380Pi* Parser::parse_pi_expr(Pi* outer) {
381 auto track = tracker();
382 auto tok = lex();
383
384 std::string entity;
385 switch (tok.tag()) {
386 case Tag::T_Pi: entity = "dependent function type"; break;
387 case Tag::K_Cn: entity = "continuation type"; break;
388 case Tag::K_Fn: entity = "returning continuation type"; break;
389 default: unreachable();
390 }
391
392 Pi* first = nullptr;
393 std::deque<Pi*> pis;
394 scopes_.push();
395 do {
396 auto implicit = accept(Tag::T_dot).has_value();
397 auto prec = tok.isa(Tag::K_Cn) ? Tok::Prec::Bot : Tok::Prec::App;
398 auto dom = parse_ptrn(Tag::D_brckt_l, "domain of a "s + entity, prec);
399 auto dom_t = dom->type(world(), def2fields_);
400 auto pi = (outer ? outer : world().mut_pi(world().type_infer_univ()))->set_dom(dom_t)->set(dom->dbg());
401 auto var = pi->var()->set(dom->sym());
402 first = first ? first : pi;
403
404 if (implicit) pi->make_implicit();
405 dom->bind(scopes_, var);
406 pis.emplace_back(pi);
407 } while (ahead().isa(Tag::T_dot) || ahead().isa(Tag::D_brckt_l) || ahead().isa(Tag::T_backtick)
408 || (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon_colon)));
409
410 Ref codom;
411 switch (tok.tag()) {
412 case Tag::T_Pi:
413 expect(Tag::T_arrow, entity);
414 codom = parse_expr("codomain of a dependent function type", Tok::Prec::Arrow);
415 break;
416 case Tag::K_Cn: codom = world().type_bot(); break;
417 case Tag::K_Fn: {
418 expect(Tag::T_arrow, entity);
419 codom = world().type_bot();
420 auto ret = parse_expr("domain of return continuation", Tok::Prec::Arrow);
421 auto pi = pis.back();
422 auto last = world().sigma({pi->dom(), world().cn(ret)});
423 pi->unset()->set_dom(last);
424 break;
425 }
426 default: unreachable();
427 }
428
429 for (auto pi : pis | std::ranges::views::reverse) {
430 pi->set_codom(codom);
431 codom = pi;
432 }
433
434 first->set(track.loc());
435 scopes_.pop();
436 return first;
437}

439Lam* Parser::parse_lam(bool is_decl) {
440 auto track = tracker();
441 auto tok = lex();
442 auto prec = tok.isa(Tag::K_cn) || tok.isa(Tag::K_con) ? Tok::Prec::Bot : Tok::Prec::Pi;
443 bool external = is_decl && accept(Tag::K_extern).has_value();
444
445 std::string entity;
446 // clang-format off
447 switch (tok.tag()) {
448 case Tag::T_lm: entity = "function expression"; break;
449 case Tag::K_cn: entity = "continuation expression"; break;
450 case Tag::K_fn: entity = "returning continuation expression"; break;
451 case Tag::K_lam: entity = "function declaration"; break;
452 case Tag::K_con: entity = "continuation declaration"; break;
453 case Tag::K_fun: entity = "returning continuation declaration"; break;
454 default: unreachable();
455 }
456 // clang-format on
457
458 auto [dbg, anx] = is_decl ? parse_name(entity) : std::pair(Dbg{prev(), anonymous_}, false);
459 auto outer = scopes_.curr();
460 Lam* decl = nullptr;
461
462 if (auto def = scopes_.query(dbg)) {
463 if (auto lam = def->isa_mut<Lam>())
464 decl = lam;
465 else
466 error(dbg.loc, "'{}' has not been declared as a function", dbg.sym);
467 }
468
469 std::unique_ptr<Ptrn> dom_p;
470 scopes_.push();
471 std::deque<std::tuple<Pi*, Lam*, const Def*>> funs;
472 do {
473 const Def* filter = accept(Tag::T_bang) ? world().lit_tt() : nullptr;
474 bool implicit = accept(Tag::T_dot).has_value();
475 dom_p = parse_ptrn(Tag::D_paren_l, "domain pattern of a "s + entity, prec);
476 auto dom_t = dom_p->type(world(), def2fields_);
477 auto pi = world().mut_pi(world().type_infer_univ(), implicit)->set_dom(dom_t);
478 auto lam = (funs.empty() && decl) ? decl : world().mut_lam(pi);
479 auto var = lam->var()->set<true>(dom_p->loc(), dom_p->sym());
480 dom_p->bind(scopes_, var);
481
482 if (auto tok = accept(Tag::T_at)) {
483 if (filter) error(tok->loc(), "filter already specified via '!'");
484 expect(Tag::D_paren_l, "opening parenthesis of a filter");
485 filter = parse_expr("filter");
486 expect(Tag::D_paren_r, "closing parenthesis of a filter");
487 }
488
489 funs.emplace_back(std::tuple(pi, lam, filter));
490 } while (!ahead().isa(Tag::T_colon) && !ahead().isa(Tag::T_assign) && !ahead().isa(Tag::T_semicolon));
491
492 Ref codom;
493 switch (tok.tag()) {
494 case Tag::T_lm:
495 case Tag::K_lam: {
496 codom = accept(Tag::T_colon) ? parse_expr("return type of a "s + entity) : world().mut_infer_type();
497 break;
498 }
499 case Tag::K_cn:
500 case Tag::K_con: codom = world().type_bot(); break;
501 case Tag::K_fn:
502 case Tag::K_fun: {
503 auto& [pi, lam, filter] = funs.back();
504
505 codom = world().type_bot();
506 auto track = tracker();
507 auto ret = accept(Tag::T_colon) ? parse_expr("return type of a "s + entity) : world().mut_infer_type();
508 auto ret_loc = dom_p->loc() + track.loc();
509 auto sigma = world().mut_sigma(2)->set(0, pi->dom());
510
511 // Fix wrong dependencies:
512 // Connect old filter to lam and ret to pi. Otherwise, scope will not find it.
513 // 1. ret depends on lam->var() instead of sigma->var(2, 0)
514 pi->set_codom(ret);
515 if (filter) lam->set_filter(filter);
516 Scope scope(lam);
517 ScopeRewriter rw(scope);
518 rw.map(lam->var(), sigma->var(2, 0));
519 sigma->set(1, world().cn({rw.rewrite(ret)}));
520
521 auto new_pi = world().mut_pi(pi->type(), pi->is_implicit())->set(ret_loc)->set_dom(sigma);
522 auto new_lam = world().mut_lam(new_pi);
523 auto new_var = new_lam->var()->set(ret_loc);
524
525 if (filter) {
526 // 2. filter depends on lam->var() instead of new_lam->var(2, 0)
527 ScopeRewriter rw(scope);
528 rw.map(lam->var(), new_lam->var(2, 0)->set(lam->var()->dbg()));
529 auto new_filter = rw.rewrite(filter);
530 filter = new_filter;
531 }
532
533 pi->unset();
534 pi = new_pi;
535 lam->unset();
536 lam = new_lam;
537
538 dom_p->bind(scopes_, new_var->proj(2, 0), true);
539 scopes_.bind({ret_loc, return_}, new_var->proj(2, 1)->set(Dbg{ret_loc, return_}));
540 break;
541 }
542 default: unreachable();
543 }
544
545 auto [_, first, __] = funs.front();
546 first->set(dbg.sym);
547
548 for (auto [pi, lam, _] : funs | std::ranges::views::reverse) {
549 // First, connect old codom to lam. Otherwise, scope will not find it.
550 pi->set_codom(codom);
551 Scope scope(lam);
552 ScopeRewriter rw(scope);
553 rw.map(lam->var(), pi->var()->set(lam->var()->dbg()));
554
555 // Now update.
556 auto dom = pi->dom();
557 codom = rw.rewrite(codom);
558 pi->reset({dom, codom});
559 codom = pi;
560 }
561
562 if (!decl) {
563 scopes_.bind(outer, dbg, first);
564 if (external) first->make_external();
565 }
566
567 auto body = accept(Tag::T_assign) ? parse_decls("body of a "s + entity) : nullptr;
568 if (!body) {
569 if (!is_decl) error(prev(), "body of a {}", entity);
570 if (auto [_, __, filter] = funs.back(); filter) error(prev(), "cannot specify filter of a {}", entity);
571 }
572
573 // filter defaults to .tt for everything except the actual continuation of con/cn/fun/fn; here we use .ff as default
574 bool last = true;
575 for (auto [_, lam, filter] : funs | std::ranges::views::reverse) {
576 bool is_cn
577 = last
578 && (tok.tag() == Tag::K_con || tok.tag() == Tag::K_cn || tok.tag() == Tag::K_fun || tok.tag() == Tag::K_fn);
579 if (body) lam->set(filter ? filter : is_cn ? world().lit_ff() : world().lit_tt(), body);
580 body = lam;
581 last = false;
582 }
583
584 if (is_decl) expect(Tag::T_semicolon, "end of "s + entity);
585 if (anx) register_annex(dbg, first);
586
587 first->set(track.loc());
588
589 scopes_.pop();
590 return first;
591}

593Ref Parser::parse_ret_expr() {
594 eat(Tag::K_ret);
595 auto ptrn = parse_ptrn(Tag::D_paren_l, "binding pattern of a ret expression");
596 expect(Tag::T_assign, "let expression");
597
598 auto cn = parse_expr("continuation expression of a ret expression");
599 expect(Tag::T_dollar, "separator of a ret expression");
600 if (auto ret_pi = Pi::ret_pi(cn->type())) {
601 auto arg = parse_expr("argument of ret expression");
602 expect(Tag::T_semicolon, "let expression");
603 auto lam = world().mut_lam(ret_pi);
604 ptrn->bind(scopes_, lam->var());
605 auto body = parse_decls("body of a ret expression");
606 lam->set(false, body);
607 return world().app(cn, {arg, lam});
608 }
609
610 error(prev(), "continuation of the ret expression is not a returning continuation but has type '{}'", cn->type());

613Ref Parser::parse_lit_expr() {
614 auto track = tracker();
615 auto tok = lex();
616 auto [_, r] = Tok::prec(Tok::Prec::Lit);
617
618 if (accept(Tag::T_colon)) {
619 auto type = parse_expr("literal", r);
620
621 // clang-format off
622 switch (tok.tag()) {
623 case Tag::L_s:
624 case Tag::L_u:
625 case Tag::L_f: break;
626 case Tag::T_bot: return world().bot(type)->set(track.loc());
627 case Tag::T_top: return world().top(type)->set(track.loc());
628 default: unreachable();
629 }
630 // clang-format on
631 return world().lit(type, tok.lit_u())->set(track.loc());
632 }
633
634 if (tok.tag() == Tag::T_bot) return world().bot(world().type())->set(track.loc());
635 if (tok.tag() == Tag::T_top) return world().top(world().type())->set(track.loc());
636 if (tok.tag() == Tag::L_s) error(prev(), ".Nat literal specified as signed but must be unsigned");
637 if (tok.tag() == Tag::L_f) error(prev(), ".Nat literal specified as floating-point but must be unsigned");
638
639 return world().lit_nat(tok.lit_u())->set(track.loc());
640}

646std::unique_ptr<Ptrn> Parser::parse_ptrn(Tag delim_l, std::string_view ctxt, Tok::Prec prec /*= Tok::Prec::Bot*/) {
647 auto track = tracker();
648 auto sym = anonymous_;
649 bool p = delim_l == Tag::D_paren_l;
650 bool b = delim_l == Tag::D_brckt_l;
651 assert((p ^ b) && "left delimiter must either be '(' or '['");
652 // p -> (p, ..., p)
653 // p -> [b, ..., b] b -> [b, ..., b]
654 // p -> s::(p, ..., p)
655 // p -> s::[b, ..., b] b -> s::[b, ..., b]
656 // p -> s: e b -> s: e
657 // p -> s b -> e
658 // p -> 's::(p, ..., p)
659 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
660 // p -> 's: e b -> 's: e
661 // p -> 's
662
663 if (p && ahead().isa(Tag::D_paren_l)) {
664 // p -> (p, ..., p)
665 return parse_tuple_ptrn(track, false, sym);
666 } else if (ahead().isa(Tag::D_brckt_l)) {
667 // p -> [b, ..., b] b -> [b, ..., b]
668 return parse_tuple_ptrn(track, false, sym);
669 }
670
671 auto backtick = accept(Tag::T_backtick);
672 bool rebind = backtick.has_value();
673
674 if (ahead(0).isa(Tag::M_id)) {
675 // p -> s::(p, ..., p)
676 // p -> s::[b, ..., b] b -> s::[b, ..., b]
677 // p -> s: e b -> s: e
678 // p -> s b -> e where e == id
679 // p -> 's::(p, ..., p)
680 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
681 // p -> 's: e b -> 's: e
682 // p -> 's
683 if (ahead(1).isa(Tag::T_colon_colon)) {
684 sym = eat(Tag::M_id).sym();
685 eat(Tag::T_colon_colon);
686 if (b && ahead().isa(Tag::D_paren_l))
687 error(ahead().loc(), "switching from []-style patterns to ()-style patterns is not allowed");
688 // b -> s::(p, ..., p)
689 // b -> s::[b, ..., b] b -> s::[b, ..., b]
690 // b -> 's::(p, ..., p)
691 // b -> 's::[b, ..., b] b -> 's::[b, ..., b]
692 if (ahead().isa(Tag::D_paren_l) || ahead().isa(Tag::D_brckt_l))
693 return parse_tuple_ptrn(track, rebind, sym);
694 else
695 syntax_err("tuple pattern after '" + *sym + "::'", ctxt);
696 } else if (ahead(1).isa(Tag::T_colon)) {
697 // p -> s: e b -> s: e
698 // p -> 's: e b -> 's: e
699 sym = eat(Tag::M_id).sym();
700 eat(Tag::T_colon);
701 auto type = parse_expr(ctxt, prec);
702 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
703 } else {
704 // p -> s b -> e where e == id
705 // p -> 's
706 if (p) {
707 // p -> s
708 // p -> 's
709 sym = eat(Tag::M_id).sym();
710 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, nullptr);
711 } else {
712 // b -> e where e == id
713 auto type = parse_expr(ctxt, prec);
714 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
715 }
716 }
717 } else if (b) {
718 // b -> e where e != id
719 if (backtick) error(backtick->loc(), "you can only prefix identifiers with backtick for rebinding");
720 auto type = parse_expr(ctxt, prec);
721 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
722 } else if (!ctxt.empty()) {
723 // p -> ↯
724 syntax_err("pattern", ctxt);
725 }
726
727 return nullptr;
728}

730std::unique_ptr<TuplePtrn> Parser::parse_tuple_ptrn(Tracker track, bool rebind, Sym sym, Def* decl) {
731 auto delim_l = ahead().tag();
732 bool p = delim_l == Tag::D_paren_l;
733 bool b = delim_l == Tag::D_brckt_l;
734 assert(p ^ b);
735
736 std::deque<std::unique_ptr<Ptrn>> ptrns;
737 std::vector<Infer*> infers;
738 DefVec ops;
739
740 scopes_.push();
741 parse_list("tuple pattern", delim_l, [&]() {
742 parse_decls({});
743 auto track = tracker();
744 if (!ptrns.empty()) ptrns.back()->bind(scopes_, infers.back());
745 std::unique_ptr<Ptrn> ptrn;
746
747 if (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::M_id)) {
748 std::vector<Tok> sym_toks;
749 while (auto tok = accept(Tag::M_id)) sym_toks.emplace_back(*tok);
750
751 if (accept(Tag::T_colon)) { // identifier group: x y x: T
752 auto type = parse_expr("type of an identifier group within a tuple pattern");
753
754 for (size_t i = 0, e = sym_toks.size(); i != e; ++i) {
755 auto tok = sym_toks[i];
756 infers.emplace_back(world().mut_infer(type)->set(tok.dbg()));
757 ops.emplace_back(type);
758 auto ptrn = std::make_unique<IdPtrn>(tok.dbg(), false, type);
759 if (i != e - 1) ptrn->bind(scopes_, infers.back()); // last element will be bound above
760 ptrns.emplace_back(std::move(ptrn));
761 }
762 return;
763 }
764
765 // "x y z" is a curried app and maybe the prefix of a longer type expression
766 auto lhs = scopes_.find(sym_toks.front().dbg());
767 for (auto sym_tok : sym_toks | std::views::drop(1)) {
768 auto rhs = scopes_.find(sym_tok.dbg());
769 lhs = world().iapp(lhs, rhs)->set(lhs->loc() + sym_tok.loc());
770 }
771 auto [_, r] = Tok::prec(Tok::Prec::App);
772 auto expr = parse_infix_expr(track, lhs, r);
773 ptrn = std::make_unique<IdPtrn>(track.dbg(anonymous_), false, expr);
774 } else {
775 ptrn = parse_ptrn(delim_l, "element of a tuple pattern");
776 auto type = ptrn->type(world(), def2fields_);
777
778 if (b) { // If we are able to parse more stuff, we got an expression instead of just a binder.
779 if (auto expr = parse_infix_expr(track, type); expr != type)
780 ptrn = std::make_unique<IdPtrn>(track.dbg(anonymous_), false, expr);
781 }
782 }
783
784 auto type = ptrn->type(world(), def2fields_);
785 infers.emplace_back(world().mut_infer(type)->set(ptrn->sym()));
786 ops.emplace_back(type);
787 ptrns.emplace_back(std::move(ptrn));
788 });
789 scopes_.pop();
790
791 // TODO parse type
792 return std::make_unique<TuplePtrn>(track.dbg(sym), rebind, std::move(ptrns), nullptr, std::move(infers), decl);
793}

799Ref Parser::parse_decls(std::string_view ctxt) {
800 while (true) {
801 // clang-format off
802 switch (ahead().tag()) {
803 case Tag::T_semicolon: lex(); break; // eat up stray semicolons
804 case Tag::K_ax: parse_ax_decl(); break;
805 case Tag::K_let: parse_let_decl(); break;
806 case Tag::K_Sigma: parse_sigma_decl(); break;
807 case Tag::K_Pi: parse_pi_decl(); break;
808 case Tag::K_con:
809 case Tag::K_fun:
810 case Tag::K_lam: parse_lam(true); break;
811 default: return ctxt.empty() ? nullptr : parse_expr(ctxt);
812 }
813 // clang-format on
814 }

817void Parser::parse_ax_decl() {
818 auto track = tracker();
819 eat(Tag::K_ax);
820 auto dbg = expect(Tag::M_anx, "annex name of an axiom").dbg();
821 auto [plugin, tag, sub] = Annex::split(world(), dbg.sym);
822 auto&& [annex, is_new] = driver().name2annex(dbg.sym, plugin, tag, dbg.loc);
823
824 if (!plugin) error(dbg.loc, "invalid axiom name '{}'", dbg.sym);
825 if (sub) error(dbg.loc, "axiom '{}' must not have a subtag", dbg.sym);
826
827 std::deque<std::deque<Sym>> new_subs;
828 if (ahead().isa(Tag::D_paren_l)) {
829 parse_list("tag list of an axiom", Tag::D_paren_l, [&]() {
830 auto& aliases = new_subs.emplace_back();
831 auto [_, tag] = parse_id("tag of an axiom");
832 aliases.emplace_back(tag);
833 while (accept(Tag::T_assign)) {
834 auto [_, alias] = parse_id("alias of an axiom tag");
835 aliases.emplace_back(alias);
836 }
837 });
838 }
839
840 if (!is_new && new_subs.empty() && !annex.subs.empty())
841 error(dbg.loc, "redeclaration of axiom '{}' without specifying new subs", dbg.sym);
842 else if (!is_new && !new_subs.empty() && annex.subs.empty())
843 error(dbg.loc, "cannot extend subs of axiom '{}' because it was declared as a subless axiom", dbg.sym);
844
845 auto type = parse_type_ascr("type ascription of an axiom");
846 if (!is_new && annex.pi != (type->isa<Pi>() != nullptr))
847 error(dbg.loc, "all declarations of annex '{}' have to be function types if any is", dbg.sym);
848 annex.pi = type->isa<Pi>() != nullptr;
849
850 Sym normalizer;
851 if (ahead().isa(Tag::T_comma) && ahead(1).isa(Tag::M_id)) {
852 lex();
853 normalizer = parse_id("normalizer of an axiom").sym;
854 }
855
856 if (!is_new && (annex.normalizer && normalizer) && annex.normalizer != normalizer)
857 error(dbg.loc, "all declarations of axiom '{}' must use the same normalizer name", dbg.sym);
858 annex.normalizer = normalizer;
859
860 auto [curry, trip] = Axiom::infer_curry_and_trip(type);
861
862 if (accept(Tag::T_comma)) {
863 auto c = expect(Tag::L_u, "curry counter for axiom");
864 if (c.lit_u() > curry) error(c.loc(), "curry counter cannot be greater than {}", curry);
865 curry = c.lit_u();
866 }
867
868 if (accept(Tag::T_comma)) trip = expect(Tag::L_u, "trip count for axiom").lit_u();
869
870 plugin_t p = *Annex::mangle(plugin);
871 tag_t t = annex.tag_id;
872 sub_t s = annex.subs.size();
873 if (new_subs.empty()) {
874 auto norm = driver().normalizer(p, t, 0);
875 auto axiom = world().axiom(norm, curry, trip, type, p, t, 0)->set(dbg);
876 world().register_annex(p | (flags_t(t) << 8_u64), axiom);
877 scopes_.bind(dbg, axiom);
878 } else {
879 for (const auto& sub : new_subs) {
880 auto name = world().sym(*dbg.sym + "."s + *sub.front());
881 auto norm = driver().normalizer(p, t, s);
882 auto axiom = world().axiom(norm, curry, trip, type, p, t, s)->set(track.loc(), name);
883 world().register_annex(p | (flags_t(t) << 8_u64) | flags_t(s), axiom);
884 for (auto& alias : sub) {
885 auto sym = world().sym(*dbg.sym + "."s + *alias);
886 scopes_.bind({prev(), sym}, axiom);
887 }
888 ++s;
889 }
890 annex.subs.insert(annex.subs.end(), new_subs.begin(), new_subs.end());
891 }
892 expect(Tag::T_semicolon, "end of an axiom");
893}

895void Parser::parse_let_decl() {
896 auto track = tracker();
897 eat(Tag::K_let);
898
899 std::variant<std::unique_ptr<Ptrn>, Dbg> name;
900 Ref type;
901 if (auto tok = accept(Tag::M_anx)) {
902 name = tok->dbg();
903 type = parse_type_ascr();
904 } else {
905 auto ptrn = parse_ptrn(Tag::D_paren_l, "binding pattern of a let declaration");
906 type = ptrn->type(world(), def2fields_);
907 name = std::move(ptrn);
908 }
909
910 expect(Tag::T_assign, "let");
911 auto value = parse_expr("value of a let declaration");
912
913 if (type && !Check::assignable(type, value))
914 error(track.loc(), "cannot assign value `{}` of type `{}` to its declared type `{}` in a let declaration",
915 value, value->type(), type);
916
917 if (auto dbg = std::get_if<Dbg>(&name)) {
918 scopes_.bind(*dbg, value);
919 register_annex(*dbg, value);
920 } else {
921 std::get<std::unique_ptr<Ptrn>>(name)->bind(scopes_, value);
922 }
923
924 expect(Tag::T_semicolon, "let declaration");
925}

927void Parser::parse_sigma_decl() {
928 auto track = tracker();
929 eat(Tag::K_Sigma);
930 auto [dbg, anx] = parse_name("sigma declaration");
931 auto type = accept(Tag::T_colon) ? parse_expr("type of a sigma declaration") : world().type();
932 auto arity = std::optional<nat_t>{};
933 if (accept(Tag::T_comma)) arity = expect(Tag::L_u, "arity of a sigma").lit_u();
934
935 if (accept(Tag::T_assign)) {
936 Def* decl;
937 if (auto def = scopes_.query(dbg)) {
938 if ((!def->isa_mut<Sigma>() && !def->isa<Infer>()) || !def->isa_lit_arity())
939 error(dbg.loc, "'{}' has not been declared as a sigma", dbg.sym);
940 if (!Check::alpha(def->type(), type))
941 error(dbg.loc, "'{}' of type '{}' has been redeclared with a different type '{}'; here: {}", dbg.sym,
942 def->type(), type, def->loc());
943 if (arity && *arity != def->as_lit_arity())
944 error(dbg.loc, "sigma '{}' redeclared with a different arity '{}'; previous arity was '{}' here: {}",
945 dbg.sym, *arity, def->arity(), def->loc());
946 decl = def->as_mut();
947 if (decl->is_set())
948 error(dbg.loc, "redefinition of '{}'; previous definition here: '{}'", dbg.sym, decl->loc());
949 } else {
950 decl = (arity ? (Def*)world().mut_sigma(type, *arity) : (Def*)world().mut_infer(type))->set(dbg);
951 scopes_.bind(dbg, decl);
952 }
953
954 if (!ahead().isa(Tag::D_brckt_l)) syntax_err("sigma expression", "definition of a sigma declaration");
955
956 auto ptrn = parse_tuple_ptrn(track, false, dbg.sym, decl);
957 auto t = ptrn->type(world(), def2fields_);
958
959 assert(t->isa_mut<Sigma>());
960 t->set<true>(track.loc());
961 if (anx) register_annex(dbg, t);
962
963 if (auto infer = decl->isa<Infer>()) {
964 assert(infer->op() == t);
965 infer->set<true>(track.loc());
966 scopes_.bind(dbg, t, true); // rebind dbg to point to sigma instead of infer
967 } else {
968 assert(t == decl);
969 }
970 } else {
971 auto decl = (arity ? (Def*)world().mut_sigma(type, *arity) : (Def*)world().mut_infer(type))->set(dbg);
972 scopes_.bind(dbg, decl);
973 }
974
975 expect(Tag::T_semicolon, "end of a sigma declaration");
976}

978void Parser::parse_pi_decl() {
979 auto track = tracker();
980 eat(Tag::K_Pi);
981 auto [dbg, anx] = parse_name("pi declaration");
982 auto type = accept(Tag::T_colon) ? parse_expr("type of a pi declaration") : world().type();
983
984 if (accept(Tag::T_assign)) {
985 Pi* pi;
986 if (auto def = scopes_.query(dbg)) {
987 if (auto mut = def->isa_mut<Pi>()) {
988 if (!Check::alpha(mut->type(), type))
989 error(dbg.loc, "'{}' of type '{}' has been redeclared with a different type '{}'; here: {}",
990 dbg.sym, mut->type(), type, mut->loc());
991 if (mut->is_set())
992 error(dbg.loc, "redefinition of '{}'; previous definition here: '{}'", dbg.sym, mut->loc());
993 pi = mut;
994 } else {
995 error(dbg.loc, "'{}' has not been declared as a pi", dbg.sym);
996 }
997 } else {
998 pi = world().mut_pi(type)->set(dbg);
999 scopes_.bind(dbg, pi);
1000 }
1001 if (anx) register_annex(dbg, pi);
1002
1003 if (!ahead().isa(Tag::T_Pi) && !ahead().isa(Tag::K_Cn) && !ahead().isa(Tag::K_Fn))
1004 syntax_err("pi expression", "definition of a pi declaration");
1005
1006 auto other = parse_pi_expr(pi);
1007 assert_unused(other == pi);
1008 pi->set<true>(track.loc());
1009 } else {
1010 auto pi = world().mut_pi(type)->set(dbg);
1011 scopes_.bind(dbg, pi);
1012 }
1013
1014 expect(Tag::T_semicolon, "end of a pi declaration");
1015}
