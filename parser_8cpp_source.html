27Tok Parser::lex() {
28 auto result = ahead();
29 prev() = result.loc();
30 for (size_t i = 0; i < Max_Ahead - 1; ++i) ahead(i) = ahead(i + 1);
31 ahead(Max_Ahead - 1) = lexer().lex();
32 return result;
33}

71 world().VLOG("import: {}", name);

89void Parser::import(std::istream& is, const fs::path* path, std::ostream* md) {
90 world().VLOG("reading: {}", path ? path->string() : "<unknown file>"s);
91 if (!is) error("cannot read file '{}'", *path);
92
93 lexers_.emplace(world(), is, path, md);
94 auto state = state_;
95
96 for (size_t i = 0; i != Max_Ahead; ++i) ahead(i) = lexer().lex();
97 prev() = Loc(path, {1, 1});

100 state_ = state;
101 lexers_.pop();

176Ref Parser::parse_infix_expr(Tracker track, const Def* lhs, Tok::Prec p) {
177 while (true) {
178 // If operator in ahead has less left precedence: reduce (break).
179 if (ahead().isa(Tag::T_extract)) {
180 if (auto extract = parse_extract_expr(track, lhs, p))
181 lhs = extract;
182 else
183 break;
184 } else if (ahead().isa(Tag::T_arrow)) {
185 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
186 if (l < p) break;
187 lex();
188 auto rhs = parse_expr("right-hand side of an function type", r);
189 lhs = world().pi(lhs, rhs)->set(track.loc());
190 } else {
191 auto [l, r] = Tok::prec(Tok::Prec::App);
192 if (l < p) break;
193 bool is_explicit = accept(Tag::T_at).has_value();
194 if (auto rhs = parse_expr({}, r)) // if we can parse an expression, it's an App
195 lhs = (is_explicit ? world().app(lhs, rhs) : world().iapp(lhs, rhs))->set(track.loc());
196 else
197 return lhs;
198 }
199 }
200
201 return lhs;
202}

204Ref Parser::parse_extract_expr(Tracker track, const Def* lhs, Tok::Prec p) {
205 auto [l, r] = Tok::prec(Tok::Prec::Extract);
206 if (l < p) return nullptr;
207 lex();
208
209 if (ahead().isa(Tag::M_id)) {
210 if (auto sigma = lhs->type()->isa_mut<Sigma>()) {
211 auto tok = eat(Tag::M_id);
212 if (tok.sym() == '_') error(tok.loc(), "you cannot use special symbol '_' as field access");
213
214 if (auto i = def2fields_.find(sigma); i != def2fields_.end()) {
215 if (auto& fields = i->second; fields.size() == sigma->num_ops()) {
216 for (size_t i = 0, n = sigma->num_ops(); i != n; ++i)
217 if (fields[i] == tok.sym()) return world().extract(lhs, n, i)->set(track.loc());
218 }
219 }
220 error(tok.loc(), "could not find elemement '{}' to extract from '{}' of type '{}'", tok.sym(), lhs, sigma);
221 }
222 }
223
224 auto rhs = parse_expr("right-hand side of an extract", r);
225 return world().extract(lhs, rhs)->set(track.loc());
226}

243Ref Parser::parse_primary_expr(std::string_view ctxt) {
244 // clang-format off
245 switch (ahead().tag()) {
246 case Tag::D_quote_l: return parse_arr_expr();
247 case Tag::D_angle_l: return parse_pack_expr();
248 case Tag::D_brace_l: return parse_block_expr();
249 case Tag::D_brckt_l: return parse_sigma_expr();
250 case Tag::D_paren_l: return parse_tuple_expr();
251 case Tag::K_Type: return parse_type_expr();
252 case Tag::K_Univ: lex(); return world().univ();
253 case Tag::K_Bool: lex(); return world().type_bool();
254 case Tag::K_Idx: lex(); return world().type_idx();
255 case Tag::K_Nat: lex(); return world().type_nat();
256 case Tag::K_ff: lex(); return world().lit_ff();
257 case Tag::K_tt: lex(); return world().lit_tt();
258 case Tag::K_Cn:
259 case Tag::K_Fn:
260 case Tag::T_Pi: return parse_pi_expr();
261 case Tag::K_cn:
262 case Tag::K_fn:
263 case Tag::T_lm: return parse_lam();
264 case Tag::T_star: lex(); return world().type();
265 case Tag::T_box: lex(); return world().type<1>();
266 case Tag::T_bot:
267 case Tag::T_top:
268 case Tag::L_s:
269 case Tag::L_u:
270 case Tag::L_f: return parse_lit_expr();
271 case Tag::L_c: return world().lit_int(8, lex().lit_c());
272 case Tag::L_i: return lex().lit_i();
273 case Tag::K_ins: return parse_insert_expr();
274 case Tag::K_ret: return parse_ret_expr();
275 case Tag::M_anx:
276 case Tag::M_id: return scopes_.find(lex().dbg());
277 case Tag::M_str: return world().tuple(lex().sym())->set(prev());
278 default:
279 if (ctxt.empty()) return nullptr;
280 syntax_err("primary expression", ctxt);
281 }
282 // clang-format on
283 return nullptr;
284}

286Ref Parser::parse_arr_expr() {
287 auto track = tracker();
288 scopes_.push();
289 eat(Tag::D_quote_l);
290
291 const Def* shape = nullptr;
292 Arr* arr = nullptr;
293 if (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon)) {
294 auto id = eat(Tag::M_id);
295 eat(Tag::T_colon);
296
297 auto shape = parse_expr("shape of an array");
298 auto type = world().mut_infer_univ();
299 arr = world().mut_arr(type)->set_shape(shape);
300 scopes_.bind(id.dbg(), arr->var()->set(id.dbg()));
301 } else {
302 shape = parse_expr("shape of an array");
303 }
304
305 expect(Tag::T_semicolon, "array");
306 auto body = parse_expr("body of an array");
307 expect(Tag::D_quote_r, "closing delimiter of an array");
308 scopes_.pop();
309
310 if (arr) return arr->set_body(body);
311 return world().arr(shape, body)->set(track.loc());
312}

314Ref Parser::parse_pack_expr() {
315 auto track = tracker();
316 scopes_.push();
317 eat(Tag::D_angle_l);
318
319 if (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon)) {
320 auto id = eat(Tag::M_id);
321 eat(Tag::T_colon);
322
323 auto shape = parse_expr("shape of a pack");
324 auto infer = world().mut_infer(world().type_idx(shape))->set(id.sym());
325 scopes_.bind(id.dbg(), infer);
326
327 expect(Tag::T_semicolon, "pack");
328 auto body = parse_expr("body of a pack");
329 expect(Tag::D_angle_r, "closing delimiter of a pack");
330 scopes_.pop();

334 auto var = pack->var();
335 infer->set(var);
336 Scope scope(pack);
337 ScopeRewriter rw(scope);
338 rw.map(infer, var);
339 return pack->reset(rw.rewrite(pack->body()));
340 }
341
342 auto shape = parse_expr("shape of a pack");
343 expect(Tag::T_semicolon, "pack");
344 auto body = parse_expr("body of a pack");
345 expect(Tag::D_angle_r, "closing delimiter of a pack");
346 scopes_.pop();
347 return world().pack(shape, body)->set(track.loc());

380Pi* Parser::parse_pi_expr(Pi* outer) {
381 auto track = tracker();
382 auto tok = lex();
383
384 std::string entity;
385 switch (tok.tag()) {
386 case Tag::T_Pi: entity = "dependent function type"; break;
387 case Tag::K_Cn: entity = "continuation type"; break;
388 case Tag::K_Fn: entity = "returning continuation type"; break;
389 default: unreachable();
390 }
391
392 Pi* first = nullptr;
393 std::deque<Pi*> pis;
394 scopes_.push();
395 do {
396 auto implicit = accept(Tag::T_dot).has_value();
397 auto prec = tok.isa(Tag::K_Cn) ? Tok::Prec::Bot : Tok::Prec::App;
398 auto dom = parse_ptrn(Tag::D_brckt_l, "domain of a "s + entity, prec);
399 auto dom_t = dom->type(world(), def2fields_);
400 auto pi = (outer ? outer : world().mut_pi(world().type_infer_univ()))->set_dom(dom_t)->set(dom->dbg());
401 auto var = pi->var()->set(dom->sym());
402 first = first ? first : pi;
403
404 if (implicit) pi->make_implicit();
405 dom->bind(scopes_, var);
406 pis.emplace_back(pi);
407 } while (ahead().isa(Tag::T_dot) || ahead().isa(Tag::D_brckt_l) || ahead().isa(Tag::T_backtick)
408 || (ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::T_colon_colon)));
409
410 Ref codom;
411 switch (tok.tag()) {
412 case Tag::T_Pi:
413 expect(Tag::T_arrow, entity);
414 codom = parse_expr("codomain of a dependent function type", Tok::Prec::Arrow);
415 break;
416 case Tag::K_Cn: codom = world().type_bot(); break;
417 case Tag::K_Fn: {
418 expect(Tag::T_arrow, entity);
419 codom = world().type_bot();
420 auto ret = parse_expr("domain of return continuation", Tok::Prec::Arrow);
421 auto pi = pis.back();
422 auto last = world().sigma({pi->dom(), world().cn(ret)});
423 pi->unset()->set_dom(last);
424 break;
425 }
426 default: unreachable();
427 }
428
429 for (auto pi : pis | std::ranges::views::reverse) {
430 pi->set_codom(codom);
431 codom = pi;
432 }
433
434 first->set(track.loc());
435 scopes_.pop();
436 return first;
437}

439Lam* Parser::parse_lam(bool is_decl) {
440 auto track = tracker();
441 auto tok = lex();
442 auto prec = tok.isa(Tag::K_cn) || tok.isa(Tag::K_con) ? Tok::Prec::Bot : Tok::Prec::Pi;
443 bool external = is_decl && accept(Tag::K_extern).has_value();
444
445 std::string entity;
446 // clang-format off
447 switch (tok.tag()) {
448 case Tag::T_lm: entity = "function expression"; break;
449 case Tag::K_cn: entity = "continuation expression"; break;
450 case Tag::K_fn: entity = "returning continuation expression"; break;
451 case Tag::K_lam: entity = "function declaration"; break;
452 case Tag::K_con: entity = "continuation declaration"; break;
453 case Tag::K_fun: entity = "returning continuation declaration"; break;
454 default: unreachable();
455 }
456 // clang-format on
457
458 auto [dbg, anx] = is_decl ? parse_name(entity) : std::pair(Dbg{prev(), anonymous_}, false);
459 auto outer = scopes_.curr();
460 Lam* decl = nullptr;
461
462 if (auto def = scopes_.query(dbg)) {
463 if (auto lam = def->isa_mut<Lam>())
464 decl = lam;
465 else
466 error(dbg.loc, "'{}' has not been declared as a function", dbg.sym);
467 }
468
469 std::unique_ptr<Ptrn> dom_p;
470 scopes_.push();
471 std::deque<std::tuple<Pi*, Lam*, const Def*>> funs;
472 do {
473 const Def* filter = accept(Tag::T_bang) ? world().lit_tt() : nullptr;
474 bool implicit = accept(Tag::T_dot).has_value();
475 dom_p = parse_ptrn(Tag::D_paren_l, "domain pattern of a "s + entity, prec);
476 auto dom_t = dom_p->type(world(), def2fields_);
477 auto pi = world().mut_pi(world().type_infer_univ(), implicit)->set_dom(dom_t);
478 auto lam = (funs.empty() && decl) ? decl : world().mut_lam(pi);
479 auto var = lam->var()->set<true>(dom_p->loc(), dom_p->sym());
480 dom_p->bind(scopes_, var);
481
482 if (auto tok = accept(Tag::T_at)) {
483 if (filter) error(tok->loc(), "filter already specified via '!'");
484 if (accept(Tag::T_at)) {
485 filter = world().lit_tt();
486 } else {
487 expect(Tag::D_paren_l, "opening parenthesis of a filter");
488 filter = parse_expr("filter");
489 expect(Tag::D_paren_r, "closing parenthesis of a filter");
490 }
491 }
492
493 funs.emplace_back(std::tuple(pi, lam, filter));
494 } while (!ahead().isa(Tag::T_colon) && !ahead().isa(Tag::T_assign) && !ahead().isa(Tag::T_semicolon));
495
496 Ref codom;
497 switch (tok.tag()) {
498 case Tag::T_lm:
499 case Tag::K_lam: {
500 codom = accept(Tag::T_colon) ? parse_expr("return type of a "s + entity) : world().mut_infer_type();
501 break;
502 }
503 case Tag::K_cn:
504 case Tag::K_con: codom = world().type_bot(); break;
505 case Tag::K_fn:
506 case Tag::K_fun: {
507 auto& [pi, lam, filter] = funs.back();
508
509 codom = world().type_bot();
510 auto track = tracker();
511 auto ret = accept(Tag::T_colon) ? parse_expr("return type of a "s + entity) : world().mut_infer_type();
512 auto ret_loc = dom_p->loc() + track.loc();
513 auto sigma = world().mut_sigma(2)->set(0, pi->dom());
514
515 // Fix wrong dependencies:
516 // Connect old filter to lam and ret to pi. Otherwise, scope will not find it.
517 // 1. ret depends on lam->var() instead of sigma->var(2, 0)
518 pi->set_codom(ret);
519 if (filter) lam->set_filter(filter);
520 Scope scope(lam);
521 ScopeRewriter rw(scope);
522 rw.map(lam->var(), sigma->var(2, 0));
523 sigma->set(1, world().cn({rw.rewrite(ret)}));
524
525 auto new_pi = world().mut_pi(pi->type(), pi->is_implicit())->set(ret_loc)->set_dom(sigma);
526 auto new_lam = world().mut_lam(new_pi);
527 auto new_var = new_lam->var()->set(ret_loc);
528
529 if (filter) {
530 // 2. filter depends on lam->var() instead of new_lam->var(2, 0)
531 ScopeRewriter rw(scope);
532 rw.map(lam->var(), new_lam->var(2, 0)->set(lam->var()->dbg()));
533 auto new_filter = rw.rewrite(filter);
534 filter = new_filter;
535 }
536
537 pi->unset();
538 pi = new_pi;
539 lam->unset();
540 lam = new_lam;
541
542 dom_p->bind(scopes_, new_var->proj(2, 0), true);
543 scopes_.bind({ret_loc, return_}, new_var->proj(2, 1)->set(Dbg{ret_loc, return_}));
544 break;
545 }
546 default: unreachable();
547 }
548
549 auto [_, first, __] = funs.front();
550 first->set(dbg.sym);
551
552 for (auto [pi, lam, _] : funs | std::ranges::views::reverse) {
553 // First, connect old codom to lam. Otherwise, scope will not find it.
554 pi->set_codom(codom);
555 Scope scope(lam);
556 ScopeRewriter rw(scope);
557 rw.map(lam->var(), pi->var()->set(lam->var()->dbg()));
558
559 // Now update.
560 auto dom = pi->dom();
561 codom = rw.rewrite(codom);
562 pi->reset({dom, codom});
563 codom = pi;
564 }
565
566 if (!decl) {
567 scopes_.bind(outer, dbg, first);
568 if (external) first->make_external();
569 }
570
571 auto body = accept(Tag::T_assign) ? parse_decls("body of a "s + entity) : nullptr;
572 if (!body) {
573 if (!is_decl) error(prev(), "body of a {}", entity);
574 if (auto [_, __, filter] = funs.back(); filter) error(prev(), "cannot specify filter of a {}", entity);
575 }
576
577 for (auto [_, lam, filter] : funs | std::ranges::views::reverse) {
578 if (body) lam->set(filter ? filter : world().lit_ff(), body);
579 body = lam;
580 }
581
582 if (is_decl) expect(Tag::T_semicolon, "end of "s + entity);
583 if (anx) register_annex(dbg, first);
584
585 first->set(track.loc());
586
587 scopes_.pop();
588 return first;
589}

591Ref Parser::parse_ret_expr() {
592 eat(Tag::K_ret);
593 auto ptrn = parse_ptrn(Tag::D_paren_l, "binding pattern of a ret expression");
594 expect(Tag::T_assign, "let expression");
595
596 auto cn = parse_expr("continuation expression of a ret expression");
597 expect(Tag::T_dollar, "separator of a ret expression");
598 if (auto ret_pi = Pi::ret_pi(cn->type())) {
599 auto arg = parse_expr("argument of ret expression");
600 expect(Tag::T_semicolon, "let expression");
601 auto lam = world().mut_lam(ret_pi);
602 ptrn->bind(scopes_, lam->var());
603 auto body = parse_decls("body of a ret expression");
604 lam->set(false, body);

606 }

611Ref Parser::parse_lit_expr() {
612 auto track = tracker();
613 auto tok = lex();
614 auto [_, r] = Tok::prec(Tok::Prec::Lit);
615
616 if (accept(Tag::T_colon)) {
617 auto type = parse_expr("literal", r);
618
619 // clang-format off
620 switch (tok.tag()) {
621 case Tag::L_s:
622 case Tag::L_u:
623 case Tag::L_f: break;
624 case Tag::T_bot: return world().bot(type)->set(track.loc());
625 case Tag::T_top: return world().top(type)->set(track.loc());
626 default: unreachable();
627 }
628 // clang-format on
629 return world().lit(type, tok.lit_u())->set(track.loc());
630 }
631
632 if (tok.tag() == Tag::T_bot) return world().bot(world().type())->set(track.loc());
633 if (tok.tag() == Tag::T_top) return world().top(world().type())->set(track.loc());
634 if (tok.tag() == Tag::L_s) error(prev(), ".Nat literal specified as signed but must be unsigned");
635 if (tok.tag() == Tag::L_f) error(prev(), ".Nat literal specified as floating-point but must be unsigned");
636
637 return world().lit_nat(tok.lit_u())->set(track.loc());
638}

644std::unique_ptr<Ptrn> Parser::parse_ptrn(Tag delim_l, std::string_view ctxt, Tok::Prec prec /*= Tok::Prec::Bot*/) {
645 auto track = tracker();
646 auto sym = anonymous_;
647 bool p = delim_l == Tag::D_paren_l;
648 bool b = delim_l == Tag::D_brckt_l;
649 assert((p ^ b) && "left delimiter must either be '(' or '['");
650 // p -> (p, ..., p)
651 // p -> [b, ..., b] b -> [b, ..., b]
652 // p -> s::(p, ..., p)
653 // p -> s::[b, ..., b] b -> s::[b, ..., b]
654 // p -> s: e b -> s: e
655 // p -> s b -> e
656 // p -> 's::(p, ..., p)
657 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
658 // p -> 's: e b -> 's: e
659 // p -> 's
660
661 if (p && ahead().isa(Tag::D_paren_l)) {
662 // p -> (p, ..., p)
663 return parse_tuple_ptrn(track, false, sym);
664 } else if (ahead().isa(Tag::D_brckt_l)) {
665 // p -> [b, ..., b] b -> [b, ..., b]
666 return parse_tuple_ptrn(track, false, sym);
667 }
668
669 auto backtick = accept(Tag::T_backtick);
670 bool rebind = backtick.has_value();
671
672 if (ahead(0).isa(Tag::M_id)) {
673 // p -> s::(p, ..., p)
674 // p -> s::[b, ..., b] b -> s::[b, ..., b]
675 // p -> s: e b -> s: e
676 // p -> s b -> e where e == id
677 // p -> 's::(p, ..., p)
678 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
679 // p -> 's: e b -> 's: e
680 // p -> 's
681 if (ahead(1).isa(Tag::T_colon_colon)) {
682 sym = eat(Tag::M_id).sym();
683 eat(Tag::T_colon_colon);
684 if (b && ahead().isa(Tag::D_paren_l))
685 error(ahead().loc(), "switching from []-style patterns to ()-style patterns is not allowed");
686 // b -> s::(p, ..., p)
687 // b -> s::[b, ..., b] b -> s::[b, ..., b]
688 // b -> 's::(p, ..., p)
689 // b -> 's::[b, ..., b] b -> 's::[b, ..., b]
690 if (ahead().isa(Tag::D_paren_l) || ahead().isa(Tag::D_brckt_l))
691 return parse_tuple_ptrn(track, rebind, sym);
692 else
693 syntax_err("tuple pattern after '" + *sym + "::'", ctxt);
694 } else if (ahead(1).isa(Tag::T_colon)) {
695 // p -> s: e b -> s: e
696 // p -> 's: e b -> 's: e
697 sym = eat(Tag::M_id).sym();
698 eat(Tag::T_colon);
699 auto type = parse_expr(ctxt, prec);
700 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
701 } else {
702 // p -> s b -> e where e == id
703 // p -> 's
704 if (p) {
705 // p -> s
706 // p -> 's
707 sym = eat(Tag::M_id).sym();
708 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, nullptr);
709 } else {
710 // b -> e where e == id
711 auto type = parse_expr(ctxt, prec);
712 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
713 }
714 }
715 } else if (b) {
716 // b -> e where e != id
717 if (backtick) error(backtick->loc(), "you can only prefix identifiers with backtick for rebinding");
718 auto type = parse_expr(ctxt, prec);
719 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
720 } else if (!ctxt.empty()) {
721 // p ->
722 syntax_err("pattern", ctxt);
723 }
724
725 return nullptr;
726}

728std::unique_ptr<TuplePtrn> Parser::parse_tuple_ptrn(Tracker track, bool rebind, Sym sym, Def* decl) {
729 auto delim_l = ahead().tag();
730 bool p = delim_l == Tag::D_paren_l;
731 bool b = delim_l == Tag::D_brckt_l;
732 assert(p ^ b);
733
734 std::deque<std::unique_ptr<Ptrn>> ptrns;
735 std::vector<Infer*> infers;
736 DefVec ops;
737
738 scopes_.push();
739 parse_list("tuple pattern", delim_l, [&]() {
740 parse_decls({});
741 auto track = tracker();
742 if (!ptrns.empty()) ptrns.back()->bind(scopes_, infers.back());
743
744 if (p && ahead(0).isa(Tag::M_id) && ahead(1).isa(Tag::M_id)) {
745 std::vector<Tok> sym_toks;
746 while (auto tok = accept(Tag::M_id)) sym_toks.emplace_back(*tok);
747
748 expect(Tag::T_colon, "type ascription of an identifier group within a tuple pattern");
749 auto type = parse_expr("type of an identifier group within a tuple pattern");
750
751 for (size_t i = 0, e = sym_toks.size(); i != e; ++i) {
752 auto tok = sym_toks[i];
753 infers.emplace_back(world().mut_infer(type)->set(tok.dbg()));
754 ops.emplace_back(type);
755 auto ptrn = std::make_unique<IdPtrn>(tok.dbg(), false, type);
756 if (i != e - 1) ptrn->bind(scopes_, infers.back()); // last element will be bound above
757 ptrns.emplace_back(std::move(ptrn));
758 }
759 } else {
760 auto ptrn = parse_ptrn(delim_l, "element of a tuple pattern");
761 auto type = ptrn->type(world(), def2fields_);
762
763 if (b) {
764 // If we are able to parse more stuff, we got an expression instead of just a binder.
765 if (auto expr = parse_infix_expr(track, type); expr != type) {
766 ptrn = std::make_unique<IdPtrn>(track.dbg(anonymous_), false, expr);
767 type = ptrn->type(world(), def2fields_);
768 }
769 }
770
771 infers.emplace_back(world().mut_infer(type)->set(ptrn->sym()));
772 ops.emplace_back(type);
773 ptrns.emplace_back(std::move(ptrn));
774 }
775 });
776 scopes_.pop();
777
778 // TODO parse type
779 return std::make_unique<TuplePtrn>(track.dbg(sym), rebind, std::move(ptrns), nullptr, std::move(infers), decl);
780}

786Ref Parser::parse_decls(std::string_view ctxt) {
787 while (true) {
788 // clang-format off
789 switch (ahead().tag()) {
790 case Tag::T_semicolon: lex(); break; // eat up stray semicolons
791 case Tag::K_ax: parse_ax_decl(); break;
792 case Tag::K_let: parse_let_decl(); break;
793 case Tag::K_Sigma: parse_sigma_decl(); break;
794 case Tag::K_Pi: parse_pi_decl(); break;
795 case Tag::K_con:
796 case Tag::K_fun:
797 case Tag::K_lam: parse_lam(true); break;
798 default: return ctxt.empty() ? nullptr : parse_expr(ctxt);
799 }
800 // clang-format on
801 }

804void Parser::parse_ax_decl() {
805 auto track = tracker();
806 eat(Tag::K_ax);
807 auto dbg = expect(Tag::M_anx, "annex name of an axiom").dbg();
808 auto [plugin, tag, sub] = Annex::split(world(), dbg.sym);
809 auto&& [annex, is_new] = driver().name2annex(dbg.sym, plugin, tag, dbg.loc);
810
811 if (!plugin) error(dbg.loc, "invalid axiom name '{}'", dbg.sym);
812 if (sub) error(dbg.loc, "axiom '{}' must not have a subtag", dbg.sym);
813
814 std::deque<std::deque<Sym>> new_subs;
815 if (ahead().isa(Tag::D_paren_l)) {
816 parse_list("tag list of an axiom", Tag::D_paren_l, [&]() {
817 auto& aliases = new_subs.emplace_back();
818 auto [_, tag] = parse_id("tag of an axiom");
819 aliases.emplace_back(tag);
820 while (accept(Tag::T_assign)) {
821 auto [_, alias] = parse_id("alias of an axiom tag");
822 aliases.emplace_back(alias);
823 }
824 });
825 }
826
827 if (!is_new && new_subs.empty() && !annex.subs.empty())
828 error(dbg.loc, "redeclaration of axiom '{}' without specifying new subs", dbg.sym);
829 else if (!is_new && !new_subs.empty() && annex.subs.empty())
830 error(dbg.loc, "cannot extend subs of axiom '{}' because it was declared as a subless axiom", dbg.sym);
831
832 auto type = parse_type_ascr("type ascription of an axiom");
833 if (!is_new && annex.pi != (type->isa<Pi>() != nullptr))
834 error(dbg.loc, "all declarations of annex '{}' have to be function types if any is", dbg.sym);
835 annex.pi = type->isa<Pi>() != nullptr;
836
837 Sym normalizer;
838 if (ahead().isa(Tag::T_comma) && ahead(1).isa(Tag::M_id)) {
839 lex();
840 normalizer = parse_id("normalizer of an axiom").sym;
841 }
842
843 if (!is_new && (annex.normalizer && normalizer) && annex.normalizer != normalizer)
844 error(dbg.loc, "all declarations of axiom '{}' must use the same normalizer name", dbg.sym);
845 annex.normalizer = normalizer;
846
847 auto [curry, trip] = Axiom::infer_curry_and_trip(type);
848
849 if (accept(Tag::T_comma)) {
850 auto c = expect(Tag::L_u, "curry counter for axiom");
851 if (c.lit_u() > curry) error(c.loc(), "curry counter cannot be greater than {}", curry);
852 curry = c.lit_u();
853 }
854
855 if (accept(Tag::T_comma)) trip = expect(Tag::L_u, "trip count for axiom").lit_u();
856
857 plugin_t p = *Annex::mangle(plugin);
858 tag_t t = annex.tag_id;
859 sub_t s = annex.subs.size();
860 if (new_subs.empty()) {
861 auto norm = driver().normalizer(p, t, 0);
862 auto axiom = world().axiom(norm, curry, trip, type, p, t, 0)->set(dbg);
863 world().register_annex(p | (flags_t(t) << 8_u64), axiom);
864 scopes_.bind(dbg, axiom);
865 } else {
866 for (const auto& sub : new_subs) {
867 auto name = world().sym(*dbg.sym + "."s + *sub.front());
868 auto norm = driver().normalizer(p, t, s);
869 auto axiom = world().axiom(norm, curry, trip, type, p, t, s)->set(track.loc(), name);
870 world().register_annex(p | (flags_t(t) << 8_u64) | flags_t(s), axiom);
871 for (auto& alias : sub) {
872 auto sym = world().sym(*dbg.sym + "."s + *alias);
873 scopes_.bind({prev(), sym}, axiom);
874 }
875 ++s;
876 }
877 annex.subs.insert(annex.subs.end(), new_subs.begin(), new_subs.end());
878 }
879 expect(Tag::T_semicolon, "end of an axiom");
880}

883 eat(Tag::K_let);

904void Parser::parse_sigma_decl() {
905 auto track = tracker();
906 eat(Tag::K_Sigma);
907 auto [dbg, anx] = parse_name("sigma declaration");
908 auto type = accept(Tag::T_colon) ? parse_expr("type of a sigma declaration") : world().type();
909 auto arity = std::optional<nat_t>{};
910 if (accept(Tag::T_comma)) arity = expect(Tag::L_u, "arity of a sigma").lit_u();
911
912 if (accept(Tag::T_assign)) {
913 Def* decl;
914 if (auto def = scopes_.query(dbg)) {
915 if ((!def->isa_mut<Sigma>() && !def->isa<Infer>()) || !def->isa_lit_arity())
916 error(dbg.loc, "'{}' has not been declared as a sigma", dbg.sym);
917 if (!Check::alpha(def->type(), type))
918 error(dbg.loc, "'{}' of type '{}' has been redeclared with a different type '{}'; here: {}", dbg.sym,
919 def->type(), type, def->loc());
920 if (arity && *arity != def->as_lit_arity())
921 error(dbg.loc, "sigma '{}' redeclared with a different arity '{}'; previous arity was '{}' here: {}",
922 dbg.sym, *arity, def->arity(), def->loc());
923 decl = def->as_mut();
924 if (decl->is_set())
925 error(dbg.loc, "redefinition of '{}'; previous definition here: '{}'", dbg.sym, decl->loc());
926 } else {
927 decl = (arity ? (Def*)world().mut_sigma(type, *arity) : (Def*)world().mut_infer(type))->set(dbg);
928 scopes_.bind(dbg, decl);
929 }
930
931 if (!ahead().isa(Tag::D_brckt_l)) syntax_err("sigma expression", "definition of a sigma declaration");
932
933 auto ptrn = parse_tuple_ptrn(track, false, dbg.sym, decl);
934 auto t = ptrn->type(world(), def2fields_);
935
936 assert(t->isa_mut<Sigma>());
937 t->set<true>(track.loc());
938 if (anx) register_annex(dbg, t);
939
940 if (auto infer = decl->isa<Infer>()) {
941 assert(infer->op() == t);
942 infer->set<true>(track.loc());
943 scopes_.bind(dbg, t, true); // rebind dbg to point to sigma instead of infer
944 } else {
945 assert(t == decl);
946 }
947 } else {
948 auto decl = (arity ? (Def*)world().mut_sigma(type, *arity) : (Def*)world().mut_infer(type))->set(dbg);
949 scopes_.bind(dbg, decl);
950 }
951
952 expect(Tag::T_semicolon, "end of a sigma declaration");
953}

955void Parser::parse_pi_decl() {
956 auto track = tracker();
957 eat(Tag::K_Pi);
958 auto [dbg, anx] = parse_name("pi declaration");
959 auto type = accept(Tag::T_colon) ? parse_expr("type of a pi declaration") : world().type();
960
961 if (accept(Tag::T_assign)) {
962 Pi* pi;
963 if (auto def = scopes_.query(dbg)) {
964 if (auto mut = def->isa_mut<Pi>()) {
965 if (!Check::alpha(mut->type(), type))
966 error(dbg.loc, "'{}' of type '{}' has been redeclared with a different type '{}'; here: {}",
967 dbg.sym, mut->type(), type, mut->loc());
968 if (mut->is_set())
969 error(dbg.loc, "redefinition of '{}'; previous definition here: '{}'", dbg.sym, mut->loc());
970 pi = mut;
971 } else {
972 error(dbg.loc, "'{}' has not been declared as a pi", dbg.sym);
973 }
974 } else {
975 pi = world().mut_pi(type)->set(dbg);
976 scopes_.bind(dbg, pi);
977 }
978 if (anx) register_annex(dbg, pi);
979
980 if (!ahead().isa(Tag::T_Pi) && !ahead().isa(Tag::K_Cn) && !ahead().isa(Tag::K_Fn))
981 syntax_err("pi expression", "definition of a pi declaration");
982
983 auto other = parse_pi_expr(pi);
984 assert_unused(other == pi);
985 pi->set<true>(track.loc());
986 } else {
987 auto pi = world().mut_pi(type)->set(dbg);
988 scopes_.bind(dbg, pi);
989 }
990
991 expect(Tag::T_semicolon, "end of a pi declaration");
992}
