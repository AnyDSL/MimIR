94Parser Parser::import_module(World& world,
95 std::string_view name,
96 ArrayRef<std::string> user_search_paths,
97 const Normalizers* normalizers) {
98 auto search_paths = get_plugin_search_paths(user_search_paths);
99
100 auto file_name = std::string(name) + ".thorin";
101
102 std::string input_path{};
103 for (const auto& path : search_paths) {
104 auto full_path = path / file_name;
105
106 std::error_code ignore;
107 if (bool reg_file = std::filesystem::is_regular_file(full_path, ignore); reg_file && !ignore) {
108 input_path = full_path.string();
109 break;
110 }
111 }
112 std::ifstream ifs(input_path);
113
114 if (!ifs) throw std::runtime_error("could not find file '" + file_name + "'");
115
116 thorin::fe::Parser parser(world, input_path, ifs, user_search_paths, normalizers);
117 parser.parse_module();
118
119 world.add_imported(name);
120
121 return parser;
122}

180const Def* Parser::parse_infix_expr(Tracker track, const Def* lhs, Tok::Prec p /*= Tok::Prec::Bot*/) {
181 while (true) {
182 // If operator in ahead has less left precedence: reduce (break).
183 if (ahead().isa(Tok::Tag::T_extract)) {
184 if (auto extract = parse_extract(track, lhs, p))
185 lhs = extract;
186 else
187 break;
188 } else if (ahead().isa(Tok::Tag::T_arrow)) {
189 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
190 if (l < p) break;
191 lex();
192 auto rhs = parse_expr("right-hand side of an function type", r);
193 lhs = world().pi(lhs, rhs, track);
194 } else {
195 auto [l, r] = Tok::prec(Tok::Prec::App);
196 if (l < p) break;
197 if (auto rhs = parse_expr({}, r)) {
198 lhs = world().app(lhs, rhs, track); // if we can parse an expression, it's an App
199 } else {
200 return lhs;
201 }
202 }
203 }
204
205 return lhs;
206}

208const Def* Parser::parse_extract(Tracker track, const Def* lhs, Tok::Prec p) {
209 auto [l, r] = Tok::prec(Tok::Prec::Extract);
210 if (l < p) return nullptr;
211 lex();
212
213 if (ahead().isa(Tok::Tag::M_id)) {
214 if (auto sigma = lhs->type()->isa_nom<Sigma>()) {
215 auto sym = eat(Tok::Tag::M_id).sym();
216 if (sym.is_anonymous()) err(sym.loc(), "you cannot use special symbol '_' as field access");
217
218 auto meta = sigma->meta();
219 if (meta->arity() == sigma->arity()) {
220 size_t a = sigma->num_ops();
221 for (size_t i = 0; i != a; ++i) {
222 if (meta->proj(a, i) == sym) return world().extract(lhs, a, i, track);
223 }
224 }
225 err(sym.loc(), "could not find elemement '{}' to extract from '{} of type '{}'", sym, lhs, sigma);
226 }
227 }
228
229 auto rhs = parse_expr("right-hand side of an extract", r);
230 return world().extract(lhs, rhs, track);
231}

249 // clang-format off

253 case Tok::Tag::D_angle_l: return parse_pack();
254 case Tok::Tag::D_brace_l: return parse_block();
255 case Tok::Tag::D_brckt_l: return parse_sigma();
256 case Tok::Tag::D_paren_l: return parse_tuple();
257 case Tok::Tag::K_Cn: return parse_Cn();
258 case Tok::Tag::K_Type: return parse_type();
259 case Tok::Tag::K_Bool: lex(); return world().type_bool();
260 case Tok::Tag::K_Nat: lex(); return world().type_nat();
261 case Tok::Tag::K_ff: lex(); return world().lit_ff();
262 case Tok::Tag::K_tt: lex(); return world().lit_tt();
263 case Tok::Tag::T_Pi: return parse_pi();
264 case Tok::Tag::T_lam: return parse_lam();
265 case Tok::Tag::T_at: return parse_var();
266 case Tok::Tag::T_star: lex(); return world().type();
267 case Tok::Tag::T_box: lex(); return world().type<1>();
268 case Tok::Tag::T_bot:
269 case Tok::Tag::T_top:
270 case Tok::Tag::L_s:
271 case Tok::Tag::L_u:
272 case Tok::Tag::L_r: return parse_lit();
273 case Tok::Tag::M_id: return scopes_.find(parse_sym());
274 case Tok::Tag::M_i: return lex().index();
275 case Tok::Tag::K_ins: return parse_insert();
276 case Tok::Tag::M_ax: {
277 // HACK hard-coded some built-in axioms
278 auto tok = lex();
279 auto s = tok.sym().to_string();
280 // if (s == "%Mem") return world().ax();

282 if (s == "%Real" ) return world().type_real();
283 if (s == "%Wrap_add") return world().ax(Wrap::add);
284 if (s == "%Wrap_sub") return world().ax(Wrap::sub);
285 return scopes_.find(tok.sym());
286 }
287 default:
288 if (ctxt.empty()) return nullptr;
289 syntax_err("primary expression", ctxt);

428 auto track = tracker();
429 auto lit = lex();

431

461std::unique_ptr<Ptrn> Parser::parse_ptrn(Tok::Tag delim_l, std::string_view ctxt, Tok::Prec p /*= Tok::Prec::Bot*/) {
462 auto track = tracker();
463 auto sym = anonymous_sym();
464 // p -> (p, ..., p)
465 // p -> [b, ..., b] b -> [b, ..., b]
466 // p -> s::(p, ..., p) b -> s::(e)
467 // p -> s::[b, ..., b] b -> s::[b, ..., b]
468 // p -> s: e b -> s: e
469 // p -> s b -> e
470 if (ahead().isa(Tok::Tag::D_brckt_l)) {
471 // p -> [b, ..., b] b -> [b, ..., b]
472 return parse_tuple_ptrn(track, sym);
473 } else if (delim_l == Tok::Tag::D_paren_l && ahead().isa(Tok::Tag::D_paren_l)) {
474 // p -> (p, ..., p)
475 return parse_tuple_ptrn(track, sym);
476 } else if (ahead(0).isa(Tok::Tag::M_id)) {
477 // p -> s::(p, ..., p)
478 // p -> s::[b, ..., b] b -> s::[b, ..., b]
479 // p -> s: e b -> s: e
480 // p -> s b -> e where e == id
481 if (ahead(1).isa(Tok::Tag::T_colon_colon)) {
482 sym = eat(Tok::Tag::M_id).sym();
483 eat(Tok::Tag::T_colon_colon);
484 if (delim_l == Tok::Tag::D_brckt_l && ahead().isa(Tok::Tag::D_paren_l))
485 err(ahead().loc(), "switching from []-style patterns to ()-style patterns is not allowed");
486 // b -> s::(p, ..., p)
487 // b -> s::[b, ..., b] b -> s::[b, ..., b]
488 return parse_tuple_ptrn(track, sym);
489 } else if (ahead(1).isa(Tok::Tag::T_colon)) {
490 // p -> s: e b -> s: e
491 sym = eat(Tok::Tag::M_id).sym();
492 eat(Tok::Tag::T_colon);
493 auto type = parse_expr(ctxt, p);
494 return std::make_unique<IdPtrn>(track.loc(), sym, type);
495 } else {
496 if (delim_l == Tok::Tag::D_brckt_l) {
497 // b -> e where e == id
498 auto type = parse_expr(ctxt, p);
499 return std::make_unique<IdPtrn>(track.loc(), sym, type);
500 } else {
501 // p -> s
502 sym = eat(Tok::Tag::M_id).sym();
503 return std::make_unique<IdPtrn>(track.loc(), sym, nullptr);
504 }
505 }
506 } else if (delim_l == Tok::Tag::D_brckt_l) {
507 // b -> (e)
508 // b -> e where e != id
509 auto type = parse_expr(ctxt, p);
510 return std::make_unique<IdPtrn>(track.loc(), sym, type);
511 } else if (!ctxt.empty()) {
512 // p -> ↯
513 syntax_err("pattern", ctxt);
514 }
515
516 return nullptr;
517}

519std::unique_ptr<TuplePtrn> Parser::parse_tuple_ptrn(Tracker track, Sym sym) {
520 auto delim_l = ahead().tag();
521 std::deque<std::unique_ptr<Ptrn>> ptrns;
522 std::vector<const Def*> fields;
523 std::vector<Infer*> infers;
524 DefVec ops;
525
526 scopes_.push();
527 parse_list("tuple pattern", delim_l, [&]() {
528 auto track = tracker();
529 if (!ptrns.empty()) ptrns.back()->bind(scopes_, infers.back());
530
531 auto ptrn = parse_ptrn(delim_l, "element of a tuple pattern");
532 auto type = ptrn->type(world());
533
534 if (delim_l == Tok::Tag::D_brckt_l) {
535 // If we are able to parse more stuff, we got an expression instead of just a binder.
536 if (auto expr = parse_infix_expr(track, type); expr != type) {
537 ptrn = std::make_unique<IdPtrn>(track.loc(), anonymous_sym(), expr);
538 type = ptrn->type(world());
539 }
540 }
541
542 infers.emplace_back(world().nom_infer(type, ptrn->sym()));
543 fields.emplace_back(ptrn->sym().str());
544 ops.emplace_back(type);
545 ptrns.emplace_back(std::move(ptrn));
546 });
547 scopes_.pop();
548
549 // TODO parse type
550 return std::make_unique<TuplePtrn>(track.loc(), sym, std::move(ptrns), nullptr, std::move(infers));
551}

577void Parser::parse_ax() {
578 using namespace std::string_view_literals;
579 auto track = tracker();
580 eat(Tok::Tag::K_ax);
581 auto ax = expect(Tok::Tag::M_ax, "name of an axiom");
582 auto ax_str = ax.sym().to_string();
583 auto split = Axiom::split(ax_str);
584 if (!split) err(ax.loc(), "invalid axiom name '{}'", ax);
585
586 auto [dialect, tag, sub] = *split;
587
588 if (!sub.empty()) err(ax.loc(), "definition of axiom '{}' must not have sub in tag name", ax);
589
590 auto [it, is_new] = bootstrapper_.axioms.emplace(ax_str, h::AxiomInfo{});
591 auto& [key, info] = *it;
592 if (is_new) {
593 info.dialect = dialect;
594 info.tag = tag;
595 info.tag_id = bootstrapper_.axioms.size() - 1;
596 }
597
598 if (dialect != bootstrapper_.dialect()) {
599 // TODO
600 // err(ax.loc(), "axiom name `{}` implies a dialect name of `{}` but input file is named `{}`", ax,
601 // info.dialect, lexer_.file());
602 }
603
604 if (bootstrapper_.axioms.size() >= std::numeric_limits<tag_t>::max())
605 err(ax.loc(), "exceeded maxinum number of axioms in current dialect");
606
607 std::deque<std::deque<std::string>> new_subs;
608 if (ahead().isa(Tok::Tag::D_paren_l)) {
609 parse_list("tag list of an axiom", Tok::Tag::D_paren_l, [&]() {
610 auto& aliases = new_subs.emplace_back();
611 aliases.emplace_back(parse_sym("tag of an axiom"));
612 while (accept(Tok::Tag::T_assign)) aliases.emplace_back(parse_sym("alias of an axiom tag"));
613 });
614 }
615
616 if (!is_new && new_subs.empty() && !info.subs.empty())
617 err(ax.loc(), "redeclaration of axiom '{}' without specifying new subs", ax);
618 else if (!is_new && !new_subs.empty() && info.subs.empty())
619 err(ax.loc(), "cannot extend subs of axiom '{}' which does not have subs", ax);
620
621 auto type = parse_type_ascr("an axiom");
622 if (!is_new && info.pi != (type->isa<Pi>() != nullptr))
623 err(ax.loc(), "all declarations of axiom '{}' have to be PIs if any is", ax);
624 info.pi = type->isa<Pi>() != nullptr;
625
626 auto normalizer_name = (accept(Tok::Tag::T_comma) ? parse_sym("normalizer of an axiom") : Sym()).to_string();
627 if (!is_new && !(info.normalizer.empty() || normalizer_name.empty()) && info.normalizer != normalizer_name)
628 err(ax.loc(), "all declarations of axiom '{}' have use the same normalizer name", ax);
629 info.normalizer = normalizer_name;
630
631 auto normalizer = [this](dialect_t d, tag_t t, sub_t s) -> Def::NormalizeFn {
632 if (normalizers_)

642 scopes_.bind(ax.sym(), axiom);

654 }
655 expect(Tok::Tag::T_semicolon, "end of an axiom");

722void Parser::parse_nom_fun() {
723 auto track = tracker();
724 auto key = lex().tag();
725
726 scopes_.push(); // pi scope
727
728 bool external = accept(Tok::Tag::K_extern).has_value();
729 auto sym = parse_sym("nominal lambda");
730 auto dom_p = parse_ptrn(Tok::Tag::D_paren_l, "domain pattern of a lambda", Tok::Prec::Pi);
731 auto dom_t = dom_p->type(world());
732 auto pi = world().nom_pi(world().nom_infer_univ())->set_dom(dom_t);
733 auto var_dbg = world().dbg(dom_p->sym());
734 auto pi_var = pi->var(var_dbg);
735
736 dom_p->bind(scopes_, pi_var);
737
738 auto codom = key == Tok::Tag::K_cn ? world().type_bot()
739 : accept(Tok::Tag::T_arrow) ? parse_expr("return type of a lambda", Tok::Prec::Arrow)
740 : world().nom_infer_of_infer_level();
741 pi->set_codom(codom);
742 pi->set_type(codom->unfold_type());
743 pi->set_dbg(track);
744
745 scopes_.pop(); // pi scope
746
747 auto outer = scopes_.curr();
748 scopes_.push(); // lam scope
749
750 auto lam = world().nom_lam(pi, track.named(sym));
751 auto lam_var = lam->var(var_dbg);
752 if (external) lam->make_external();
753 scopes_.bind(outer, sym, lam);
754 dom_p->bind(scopes_, lam_var);
755
756 if (accept(Tok::Tag::T_assign)) {
757 auto body = parse_expr("body of a lambda");
758 lam->set(false, body);
759 }
760 expect(Tok::Tag::T_semicolon, "end of lambda");
761
762 scopes_.pop(); // lam scope
763}

765void Parser::parse_def(Sym sym /*= {}*/) {
766 if (!sym) {
767 eat(Tok::Tag::K_def);
768 sym = parse_sym("nominal definition");
769 }
770
771 auto nom = scopes_.find(sym)->as_nom();
772 expect(Tok::Tag::T_assign, "nominal definition");
773
774 size_t i = nom->first_dependend_op();
775 size_t n = nom->num_ops();
776
777 if (ahead().isa(Tok::Tag::D_brace_l)) {
778 scopes_.push();
779 parse_list("nominal definition", Tok::Tag::D_brace_l, [&]() {
780 if (i == n) err(prev_, "too many operands");
781 nom->set(i++, parse_expr("operand of a nominal"));
782 });
783 scopes_.pop();
784 } else if (n - i == 1) {
785 nom->set(i, parse_expr("operand of a nominal"));
786 } else {
787 err(prev_, "expected operands for nominal definition");
788 }
789
790 expect(Tok::Tag::T_semicolon, "end of a nominal definition");
791}
