59const Def* Parser::parse_def(std::string_view ctxt, Tok::Prec p /*= Tok::Prec::Bottom*/) {
60 auto track = tracker();
61 auto lhs = parse_primary_def(ctxt);
62
63 while (true) {
64 // If operator in ahead has less left precedence: reduce (break).
65 if (ahead().isa(Tok::Tag::T_extract)) {
66 auto [l, r] = Tok::prec(Tok::Prec::Extract);
67 if (l < p) break;
68 lex();
69 auto rhs = parse_def("right-hand side of an extract", r);
70 lhs = world().extract(lhs, rhs, track);
71 } else if (ahead().isa(Tok::Tag::T_arrow)) {
72 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
73 if (l < p) break;
74 lex();
75 auto rhs = parse_def("right-hand side of an function type", r);
76 lhs = world().pi(lhs, rhs, track);
77 lhs->dump(0);
78 } else {
79 auto [l, r] = Tok::prec(Tok::Prec::App);
80 if (l < p) break;
81 if (auto rhs = parse_def({}, r)) {
82 lhs = world().app(lhs, rhs, track); // if we can parse an expression, it's an App
83 } else {
84 return lhs;
85 }
86 }
87 }
88
89 return lhs;
90}

92const Def* Parser::parse_primary_def(std::string_view ctxt) {
93 // clang-format off
94 switch (ahead().tag()) {
95 case Tok::Tag::D_angle_l: return parse_pack_or_array(true);
96 case Tok::Tag::D_quote_l: return parse_pack_or_array(false);
97 case Tok::Tag::D_brace_l: return parse_block();
98 case Tok::Tag::D_bracket_l: return parse_sigma();
99 case Tok::Tag::D_paren_l: return parse_tuple();
100 case Tok::Tag::K_Nat: lex(); return world().type_nat();
101 case Tok::Tag::T_bot: return parse_ext(false);
102 case Tok::Tag::T_top: return parse_ext(true);
103 case Tok::Tag::T_Pi: return parse_Pi();
104 case Tok::Tag::T_lam: return parse_lam();
105 case Tok::Tag::T_star: lex(); return world().type();
106 case Tok::Tag::T_space: lex(); return world().type<1>();
107 case Tok::Tag::L_s:
108 case Tok::Tag::L_u:
109 case Tok::Tag::L_r: return parse_lit();
110 case Tok::Tag::M_i: return lex().index();
111 case Tok::Tag::M_ax: {
112 // HACK hard-coded some built-in axioms
113 auto tok = lex();
114 auto s = tok.sym().to_string();
115 if (s == ":Int" ) return world().type_int();
116 if (s == ":Real" ) return world().type_real();
117 if (s == ":Wrap_add") return world().ax(Wrap::add);
118 if (s == ":Wrap_sub") return world().ax(Wrap::sub);
119 assert(false && "TODO");
120 }
121 case Tok::Tag::M_id: {
122 if (ahead(1).isa(Tok::Tag::T_assign) || ahead(1).isa(Tok::Tag::T_colon)) return parse_let();
123 auto sym = parse_sym();
124 if (auto def = find(sym)) return def;
125 errln("symbol '{}' not found", sym);
126 return nullptr;
127 }
128 default:
129 if (ctxt.empty()) return nullptr;
130 err("primary def", ctxt);
131 }
132 // clang-format on
133 return nullptr;
134}
