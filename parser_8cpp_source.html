75const Def* Parser::parse_primary_def(std::string_view ctxt) {
76 // clang-format off
77 switch (ahead().tag()) {
78 case Tok::Tag::D_angle_l: return parse_pack_or_array(true);
79 case Tok::Tag::D_quote_l: return parse_pack_or_array(false);
80 case Tok::Tag::D_paren_l: return parse_tuple();
81 case Tok::Tag::D_bracket_l: return parse_sigma();
82 case Tok::Tag::D_brace_l: return parse_block();
83 case Tok::Tag::K_Nat: lex(); return world().type_nat();
84 case Tok::Tag::P_star: lex(); return world().kind();
85 case Tok::Tag::M_id: {
86 if (ahead(1).isa(Tok::Tag::P_assign) || ahead(1).isa(Tok::Tag::P_colon)) return parse_let();
87 auto sym = parse_sym();
88 return find(sym);
89 }
90 case Tok::Tag::L_s:
91 case Tok::Tag::L_u:
92 case Tok::Tag::L_r: return parse_lit();
93 default: err("primary def", ctxt);
94 }
95 // clang-format on
96 return nullptr;
97}
