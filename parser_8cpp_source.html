67const Def* Parser::parse_expr(std::string_view ctxt, Tok::Prec p /*= Tok::Prec::Bottom*/) {
68 auto track = tracker();
69 auto lhs = parse_primary_expr(ctxt);
70
71 while (true) {
72 // If operator in ahead has less left precedence: reduce (break).
73 if (ahead().isa(Tok::Tag::T_extract)) {
74 auto [l, r] = Tok::prec(Tok::Prec::Extract);
75 if (l < p) break;
76 lex();
77 auto rhs = parse_expr("right-hand side of an extract", r);
78 lhs = world().extract(lhs, rhs, track);
79 } else if (ahead().isa(Tok::Tag::T_arrow)) {
80 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
81 if (l < p) break;
82 lex();
83 auto rhs = parse_expr("right-hand side of an function type", r);
84 lhs = world().pi(lhs, rhs, track);
85 } else {
86 auto [l, r] = Tok::prec(Tok::Prec::App);
87 if (l < p) break;
88 if (auto rhs = parse_expr({}, r)) {
89 lhs = world().app(lhs, rhs, track); // if we can parse an expression, it's an App
90 } else {
91 return lhs;
92 }
93 }
94 }
95
96 return lhs;
97}

99const Def* Parser::parse_primary_expr(std::string_view ctxt) {
100 // clang-format off
101 switch (ahead().tag()) {
102 case DECL: return parse_decls();
103 case Tok::Tag::D_angle_l: return parse_pack_or_arr(true);
104 case Tok::Tag::D_quote_l: return parse_pack_or_arr(false);
105 case Tok::Tag::D_brace_l: return parse_block();
106 case Tok::Tag::D_bracket_l: return parse_sigma();
107 case Tok::Tag::D_paren_l: return parse_tuple();
108 case Tok::Tag::K_Cn: return parse_Cn();
109 case Tok::Tag::K_Type: return parse_type();
110 case Tok::Tag::K_Bool: lex(); return world().type_bool();
111 case Tok::Tag::K_Nat: lex(); return world().type_nat();
112 case Tok::Tag::K_ff: lex(); return world().lit_false();
113 case Tok::Tag::K_tt: lex(); return world().lit_true();
114 case Tok::Tag::T_Pi: return parse_pi();
115 case Tok::Tag::T_lam: return parse_lam();
116 case Tok::Tag::T_at: return parse_var();
117 case Tok::Tag::T_star: lex(); return world().type();
118 case Tok::Tag::T_box: lex(); return world().type<1>();
119 case Tok::Tag::T_bot:
120 case Tok::Tag::T_top:
121 case Tok::Tag::L_s:
122 case Tok::Tag::L_u:
123 case Tok::Tag::L_r: return parse_lit();
124 case Tok::Tag::M_id: return find(parse_sym());
125 case Tok::Tag::M_i: return lex().index();
126 case Tok::Tag::M_ax: {
127 // HACK hard-coded some built-in axioms
128 auto tok = lex();
129 auto s = tok.sym().to_string();
130 if (s == ":Mem") return world().type_mem();
131 if (s == ":Int" ) return world().type_int();
132 if (s == ":Real" ) return world().type_real();
133 if (s == ":Wrap_add") return world().ax(Wrap::add);
134 if (s == ":Wrap_sub") return world().ax(Wrap::sub);
135 return find(tok.sym());
136 }
137 default:
138 if (ctxt.empty()) return nullptr;
139 err("primary expression", ctxt);
140 }
141 // clang-format on
142 return nullptr;
143}

204 auto infer = world().nom_infer(type, id->sym(), id->loc());

206 insert(id->sym(), infer);

330void Parser::parse_ax() {
331 auto track = tracker();
332 eat(Tok::Tag::K_ax);
333 auto& info = bootstrapper_.axioms.emplace_back();

337

339 if (!dialect_and_group) err(ax.loc(), "invalid axiom name '{}'", ax);
340 info.dialect = dialect_and_group->first;
341 info.group = dialect_and_group->second;
342
343 if (info.dialect != bootstrapper_.dialect()) {
344 // TODO

348
349 if (ahead().isa(Tok::Tag::D_paren_l)) {
350 parse_list("tag list of an axiom", Tok::Tag::D_paren_l, [&]() {
351 auto& aliases = info.tags.emplace_back();
352 aliases.emplace_back(parse_sym("tag of an axiom"));

378void Parser::parse_nom() {
379 auto track = tracker();
380 auto tag = lex().tag();
381 bool external = accept(Tok::Tag::K_extern).has_value();
382 auto sym = parse_sym("nominal");
383 auto type = accept(Tok::Tag::T_colon) ? parse_expr("type of a nominal") : world().type();
384
385 Def* nom;
386 switch (tag) {
387 case Tok::Tag::K_Sigma: {
388 expect(Tok::Tag::T_comma, "nominal Sigma");
389 auto arity = expect(Tok::Tag::L_u, "arity of a nominal Sigma");
390 nom = world().nom_sigma(type, arity.u(), track.named(sym));
391 break;
392 }
393 case Tok::Tag::K_Arr: {
394 expect(Tok::Tag::T_comma, "nominal array");
395 auto shape = parse_expr("shape of a nominal array");
396 nom = world().nom_arr(type, track)->set_shape(shape);
397 break;
398 }
399 case Tok::Tag::K_pack: nom = world().nom_pack(type, track.named(sym)); break;
400 case Tok::Tag::K_Pi: {
401 expect(Tok::Tag::T_comma, "nominal Pi");
402 auto dom = parse_expr("domain of a nominal Pi");
403 nom = world().nom_pi(type, track.named(sym))->set_dom(dom);
404 break;
405 }
406 case Tok::Tag::K_lam: {
407 const Pi* pi = type->isa<Pi>();
408 if (!pi) err(type->loc(), "type of lambda must be a Pi");
409 nom = world().nom_lam(pi, track.named(sym));
410 break;
411 }
412 default: unreachable();
413 }
414
415 insert(sym, nom);
416 if (external) nom->make_external();
417 if (ahead().isa(Tok::Tag::T_assign)) return parse_def(sym);
418 expect(Tok::Tag::T_semicolon, "end of a nominal");
419}

421void Parser::parse_def(Sym sym /*= {}*/) {
422 if (!sym) {
423 eat(Tok::Tag::K_def);
424 sym = parse_sym("nominal definition");
425 }
426
427 auto nom = find(sym)->as_nom();
428 expect(Tok::Tag::T_assign, "nominal definition");
429
430 size_t i = nom->first_dependend_op();
431 size_t n = nom->num_ops();
432
433 if (ahead().isa(Tok::Tag::D_brace_l)) {
434 parse_list("nominal definition", Tok::Tag::D_brace_l, [&]() {
435 if (i == n) err(prev_, "too many operands");
436 nom->set(i++, parse_expr("operand of a nominal"));
437 });
438 } else if (n - i == 1) {
439 nom->set(i, parse_expr("operand of a nominal"));
440 } else {
441 err(prev_, "expected operands for nominal definition");
442 }
443
444 nom->dump(0);
445 expect(Tok::Tag::T_semicolon, "end of a nominal definition");
446}
