38Tok Parser::lex() {
39 auto result = ahead();
40 prev() = result.loc();
41 for (size_t i = 0; i < Max_Ahead - 1; ++i) ahead(i) = ahead(i + 1);
42 ahead(Max_Ahead - 1) = lexer().lex();
43 return result;
44}

160Ref Parser::parse_infix_expr(Tracker track, const Def* lhs, Tok::Prec p) {
161 while (true) {
162 // If operator in ahead has less left precedence: reduce (break).
163 if (ahead().isa(Tok::Tag::T_extract)) {
164 if (auto extract = parse_extract(track, lhs, p))
165 lhs = extract;
166 else
167 break;
168 } else if (ahead().isa(Tok::Tag::T_arrow)) {
169 auto [l, r] = Tok::prec(Tok::Prec::Arrow);
170 if (l < p) break;
171 lex();
172 auto rhs = parse_expr("right-hand side of an function type", r);
173 lhs = world().pi(lhs, rhs)->set(track.loc());
174 } else {
175 auto [l, r] = Tok::prec(Tok::Prec::App);
176 if (l < p) break;
177 if (auto rhs = parse_expr({}, r)) // if we can parse an expression, it's an App
178 lhs = world().iapp(lhs, rhs)->set(track.loc());
179 else
180 return lhs;
181 }
182 }
183
184 return lhs;
185}

187Ref Parser::parse_extract(Tracker track, const Def* lhs, Tok::Prec p) {
188 auto [l, r] = Tok::prec(Tok::Prec::Extract);
189 if (l < p) return nullptr;
190 lex();
191
192 if (ahead().isa(Tok::Tag::M_id)) {
193 if (auto sigma = lhs->type()->isa_mut<Sigma>()) {
194 auto tok = eat(Tok::Tag::M_id);
195 if (tok.sym() == '_') error(tok.loc(), "you cannot use special symbol '_' as field access");
196
197 if (auto i = def2fields_.find(sigma); i != def2fields_.end()) {
198 if (auto& fields = i->second; fields.size() == sigma->num_ops()) {
199 for (size_t i = 0, n = sigma->num_ops(); i != n; ++i)
200 if (fields[i] == tok.sym()) return world().extract(lhs, n, i)->set(track.loc());
201 }
202 }
203 error(tok.loc(), "could not find elemement '{}' to extract from '{}' of type '{}'", tok.sym(), lhs, sigma);
204 }
205 }
206
207 auto rhs = parse_expr("right-hand side of an extract", r);
208 return world().extract(lhs, rhs)->set(track.loc());
209}

230 case Tok::Tag::D_angle_l: return parse_pack();
231 case Tok::Tag::D_brace_l: return parse_block();
232 case Tok::Tag::D_brckt_l: return parse_sigma();
233 case Tok::Tag::D_paren_l: return parse_tuple();
234 case Tok::Tag::K_Cn: return parse_Cn();
235 case Tok::Tag::K_Type: return parse_type();
236 case Tok::Tag::K_Univ: lex(); return world().univ();
237 case Tok::Tag::K_Bool: lex(); return world().type_bool();
238 case Tok::Tag::K_Idx: lex(); return world().type_idx();
239 case Tok::Tag::K_Nat: lex(); return world().type_nat();
240 case Tok::Tag::K_ff: lex(); return world().lit_ff();
241 case Tok::Tag::K_tt: lex(); return world().lit_tt();
242 case Tok::Tag::T_Pi: return parse_pi();
243 case Tok::Tag::T_at: return parse_var();
244 case Tok::Tag::K_cn:
245 case Tok::Tag::K_fn:
246 case Tok::Tag::T_lm: return parse_lam();
247 case Tok::Tag::T_star: lex(); return world().type();
248 case Tok::Tag::T_box: lex(); return world().type<1>();

252 case Tok::Tag::L_u:

257 case Tok::Tag::M_ax: return scopes_.find(lex().dbg());

365Ref Parser::parse_pi() {
366 auto track = tracker();
367 eat(Tok::Tag::T_Pi);
368 scopes_.push();
369
370 Pi* first = nullptr;
371 std::deque<Pi*> pis;
372 do {
373 auto implicit = accept(Tok::Tag::T_dot).has_value();
374 auto dom = parse_ptrn(Tok::Tag::D_brckt_l, "domain of a dependent function type", Tok::Prec::App);
375 auto pi = world().mut_pi(world().type_infer_univ(), implicit)->set_dom(dom->type(world(), def2fields_));
376 auto var = pi->var()->set(dom->sym());
377 first = first ? first : pi;
378 pi->set(dom->dbg());
379
380 dom->bind(scopes_, var);
381 pis.emplace_back(pi);
382 } while (!ahead().isa(Tok::Tag::T_arrow));
383
384 expect(Tok::Tag::T_arrow, "dependent function type");
385 auto codom = parse_expr("codomain of a dependent function type", Tok::Prec::Arrow);
386
387 for (auto pi : pis | std::ranges::views::reverse) {
388 pi->set_codom(codom);
389 codom = pi;
390 }
391
392 first->set(track.loc());
393 scopes_.pop();
394 return first;
395}

400 auto [_, r] = Tok::prec(Tok::Prec::Lit);

430std::unique_ptr<Ptrn> Parser::parse_ptrn(Tok::Tag delim_l, std::string_view ctxt, Tok::Prec prec /*= Tok::Prec::Bot*/) {
431 auto track = tracker();
432 auto sym = anonymous_;
433 bool p = delim_l == Tok::Tag::D_paren_l;
434 bool b = delim_l == Tok::Tag::D_brckt_l;
435 assert((p ^ b) && "left delimiter must either be '(' or '['");
436 // p -> (p, ..., p)
437 // p -> [b, ..., b] b -> [b, ..., b]
438 // p -> s::(p, ..., p)
439 // p -> s::[b, ..., b] b -> s::[b, ..., b]
440 // p -> s: e b -> s: e
441 // p -> s b -> e
442 // p -> 's::(p, ..., p)
443 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
444 // p -> 's: e b -> 's: e
445 // p -> 's
446
447 if (p && ahead().isa(Tok::Tag::D_paren_l)) {
448 // p -> (p, ..., p)
449 return parse_tuple_ptrn(track, false, sym);
450 } else if (ahead().isa(Tok::Tag::D_brckt_l)) {
451 // p -> [b, ..., b] b -> [b, ..., b]
452 return parse_tuple_ptrn(track, false, sym);
453 }
454
455 auto apos = accept(Tok::Tag::T_apos);
456 bool rebind = apos.has_value();
457
458 if (ahead(0).isa(Tok::Tag::M_id)) {
459 // p -> s::(p, ..., p)
460 // p -> s::[b, ..., b] b -> s::[b, ..., b]
461 // p -> s: e b -> s: e
462 // p -> s b -> e where e == id
463 // p -> 's::(p, ..., p)
464 // p -> 's::[b, ..., b] b -> 's::[b, ..., b]
465 // p -> 's: e b -> 's: e
466 // p -> 's
467 if (ahead(1).isa(Tok::Tag::T_colon_colon)) {
468 sym = eat(Tok::Tag::M_id).sym();
469 eat(Tok::Tag::T_colon_colon);
470 if (b && ahead().isa(Tok::Tag::D_paren_l))
471 error(ahead().loc(), "switching from []-style patterns to ()-style patterns is not allowed");
472 // b -> s::(p, ..., p)
473 // b -> s::[b, ..., b] b -> s::[b, ..., b]
474 // b -> 's::(p, ..., p)
475 // b -> 's::[b, ..., b] b -> 's::[b, ..., b]
476 return parse_tuple_ptrn(track, rebind, sym);
477 } else if (ahead(1).isa(Tok::Tag::T_colon)) {
478 // p -> s: e b -> s: e
479 // p -> 's: e b -> 's: e
480 sym = eat(Tok::Tag::M_id).sym();
481 eat(Tok::Tag::T_colon);
482 auto type = parse_expr(ctxt, prec);
483 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
484 } else {
485 // p -> s b -> e where e == id
486 // p -> 's
487 if (p) {
488 // p -> s
489 // p -> 's
490 sym = eat(Tok::Tag::M_id).sym();
491 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, nullptr);
492 } else {
493 // b -> e where e == id
494 auto type = parse_expr(ctxt, prec);
495 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
496 }
497 }
498 } else if (b) {
499 // b -> e where e != id
500 if (apos) error(apos->loc(), "you can only prefix identifiers with apostrophe for rebinding");
501 auto type = parse_expr(ctxt, prec);
502 return std::make_unique<IdPtrn>(track.dbg(sym), rebind, type);
503 } else if (!ctxt.empty()) {
504 // p -> ↯
505 syntax_err("pattern", ctxt);
506 }
507
508 return nullptr;
509}

511std::unique_ptr<TuplePtrn> Parser::parse_tuple_ptrn(Tracker track, bool rebind, Sym sym) {
512 auto delim_l = ahead().tag();
513 bool p = delim_l == Tok::Tag::D_paren_l;
514 bool b = delim_l == Tok::Tag::D_brckt_l;
515 assert(p ^ b);
516
517 std::deque<std::unique_ptr<Ptrn>> ptrns;
518 std::vector<Infer*> infers;
519 DefVec ops;
520
521 scopes_.push();
522 parse_list("tuple pattern", delim_l, [&]() {
523 auto track = tracker();
524 if (!ptrns.empty()) ptrns.back()->bind(scopes_, infers.back());
525
526 if (p && ahead(0).isa(Tok::Tag::M_id) && ahead(1).isa(Tok::Tag::M_id)) {
527 std::vector<Tok> sym_toks;
528 while (auto tok = accept(Tok::Tag::M_id)) sym_toks.emplace_back(*tok);
529
530 expect(Tok::Tag::T_colon, "type ascription of an identifier group within a tuple pattern");
531 auto type = parse_expr("type of an identifier group within a tuple pattern");
532
533 for (auto tok : sym_toks) {
534 infers.emplace_back(world().mut_infer(type)->set(tok.dbg()));
535 ops.emplace_back(type);
536 ptrns.emplace_back(std::make_unique<IdPtrn>(tok.dbg(), false, type));
537 }
538 } else {
539 auto ptrn = parse_ptrn(delim_l, "element of a tuple pattern");
540 auto type = ptrn->type(world(), def2fields_);
541
542 if (b) {
543 // If we are able to parse more stuff, we got an expression instead of just a binder.
544 if (auto expr = parse_infix_expr(track, type); expr != type) {
545 ptrn = std::make_unique<IdPtrn>(track.dbg(anonymous_), false, expr);
546 type = ptrn->type(world(), def2fields_);
547 }
548 }
549
550 infers.emplace_back(world().mut_infer(type)->set(ptrn->sym()));
551 ops.emplace_back(type);
552 ptrns.emplace_back(std::move(ptrn));
553 }
554 });
555 scopes_.pop();
556
557 // TODO parse type
558 return std::make_unique<TuplePtrn>(track.dbg(sym), rebind, std::move(ptrns), nullptr, std::move(infers));
559}

586void Parser::parse_ax() {
587 auto track = tracker();
588 eat(Tok::Tag::K_ax);
589 auto ax = expect(Tok::Tag::M_ax, "name of an axiom");
590 auto [plugin, tag, sub] = Axiom::split(world(), ax.sym());
591 auto&& [info, is_new] = driver().axiom2info(ax.sym(), plugin, tag, ax.loc());
592
593 if (!plugin) error(ax.loc(), "invalid axiom name '{}'", ax);
594 if (sub) error(ax.loc(), "definition of axiom '{}' must not have sub in tag name", ax);
595
596 // if (plugin != bootstrapper_.plugin()) {
597 // TODO
598 // error(ax.loc(), "axiom name `{}` implies a plugin name of `{}` but input file is named `{}`", ax,
599 // info.plugin, lexer_.file());
600 //}
601
602 std::deque<std::deque<Sym>> new_subs;
603 if (ahead().isa(Tok::Tag::D_paren_l)) {
604 parse_list("tag list of an axiom", Tok::Tag::D_paren_l, [&]() {
605 auto& aliases = new_subs.emplace_back();
606 auto [_, tag] = parse_sym("tag of an axiom");
607 aliases.emplace_back(tag);
608 while (accept(Tok::Tag::T_assign)) {
609 auto [_, alias] = parse_sym("alias of an axiom tag");
610 aliases.emplace_back(alias);
611 }
612 });
613 }
614
615 if (!is_new && new_subs.empty() && !info.subs.empty())
616 error(ax.loc(), "redeclaration of axiom '{}' without specifying new subs", ax);
617 else if (!is_new && !new_subs.empty() && info.subs.empty())
618 error(ax.loc(), "cannot extend subs of axiom '{}' because it was declared as a subless axiom", ax);
619
620 auto type = parse_type_ascr("type ascription of an axiom");
621 if (!is_new && info.pi != (type->isa<Pi>() != nullptr))
622 error(ax.loc(), "all declarations of axiom '{}' have to be function types if any is", ax);
623 info.pi = type->isa<Pi>() != nullptr;
624
625 auto normalizer_name = accept(Tok::Tag::T_comma) ? parse_sym("normalizer of an axiom").sym : Sym();
626 if (!is_new && (info.normalizer && normalizer_name) && info.normalizer != normalizer_name)
627 error(ax.loc(), "all declarations of axiom '{}' must use the same normalizer name", ax);
628 info.normalizer = normalizer_name;
629
630 auto [curry, trip] = Axiom::infer_curry_and_trip(type);
631
632 if (accept(Tok::Tag::T_comma)) {
633 auto c = expect(Tok::Tag::L_u, "curry counter for axiom");
634 if (c.u() > curry) error(c.loc(), "curry counter cannot be greater than {}", curry);
635 curry = c.u();
636 }
637
638 if (accept(Tok::Tag::T_comma)) trip = expect(Tok::Tag::L_u, "trip count for axiom").u();
639
640 plugin_t p = *Axiom::mangle(plugin);
641 tag_t t = info.tag_id;
642 sub_t s = info.subs.size();
643 if (new_subs.empty()) {
644 auto norm = driver().normalizer(p, t, 0);
645 auto axiom = world().axiom(norm, curry, trip, type, p, t, 0)->set(ax.loc(), ax.sym());
646 scopes_.bind(ax.dbg(), axiom);
647 } else {
648 for (const auto& sub : new_subs) {
649 auto name = world().sym(*ax.sym() + "."s + *sub.front());
650 auto norm = driver().normalizer(p, t, s);
651 auto axiom = world().axiom(norm, curry, trip, type, p, t, s)->set(track.loc(), name);
652 for (auto& alias : sub) {
653 auto sym = world().sym(*ax.sym() + "."s + *alias);
654 scopes_.bind({prev(), sym}, axiom);
655 }
656 ++s;
657 }
658 info.subs.insert(info.subs.end(), new_subs.begin(), new_subs.end());
659 }
660 expect(Tok::Tag::T_semicolon, "end of an axiom");
661}

672void Parser::parse_mut() {
673 auto track = tracker();
674 auto tag = lex().tag();
675 bool external = accept(Tok::Tag::K_extern).has_value();
676 auto dbg = parse_sym("mutable");
677 auto type = accept(Tok::Tag::T_colon) ? parse_expr("type of a mutable") : world().type();
678
679 Def* mut;
680 switch (tag) {
681 case Tok::Tag::K_Sigma: {
682 expect(Tok::Tag::T_comma, "mutable Sigma");
683 auto arity = expect(Tok::Tag::L_u, "arity of a mutable Sigma");
684 mut = world().mut_sigma(type, arity.u())->set(dbg);
685 break;
686 }
687 case Tok::Tag::K_Arr: {
688 expect(Tok::Tag::T_comma, "mutable array");
689 auto shape = parse_expr("shape of a mutable array");
690 mut = world().mut_arr(type)->set(track.loc())->set_shape(shape);
691 break;
692 }
693 case Tok::Tag::K_pack: mut = world().mut_pack(type)->set(dbg); break;
694 case Tok::Tag::K_Pi: {
695 expect(Tok::Tag::T_comma, "mutable Pi");
696 auto dom = parse_expr("domain of a mutable Pi");
697 mut = world().mut_pi(type)->set(dbg)->set_dom(dom);
698 break;
699 }
700 default: unreachable();
701 }
702 scopes_.bind(dbg, mut);
703
704 scopes_.push();
705 if (external) mut->make_external();
706
707 scopes_.push();
708 if (ahead().isa(Tok::Tag::T_assign))
709 parse_def(dbg);
710 else
711 expect(Tok::Tag::T_semicolon, "end of a mutable");
712
713 scopes_.pop();
714 scopes_.pop();
715}

717Lam* Parser::parse_lam(bool decl) {
718 // TODO .fn/.fun
719 auto track = tracker();
720 auto tok = lex();
721 bool is_cn = tok.isa(Tok::Tag::K_cn) || tok.isa(Tok::Tag::K_con);
722 auto prec = is_cn ? Tok::Prec::Bot : Tok::Prec::Pi;
723 bool external = decl && accept(Tok::Tag::K_extern).has_value();
724 auto dbg = decl ? parse_sym("mutable lambda") : Dbg{prev(), anonymous_};
725
726 auto outer = scopes_.curr();
727 scopes_.push();
728
729 std::deque<std::pair<Pi*, Lam*>> funs;
730 Lam* first = nullptr;
731 do {
732 const Def* filter = world().lit_bool(accept(Tok::Tag::T_bang).has_value());
733 bool implicit = accept(Tok::Tag::T_dot).has_value();
734 auto dom_p = parse_ptrn(Tok::Tag::D_paren_l, "domain pattern of a lambda", prec);
735 auto dom_t = dom_p->type(world(), def2fields_);
736 auto pi = world().mut_pi(world().type_infer_univ(), implicit)->set_dom(dom_t);
737 auto lam = world().mut_lam(pi);
738 auto lam_var = lam->var()->set(dom_p->loc(), dom_p->sym());
739
740 if (first == nullptr) {
741 first = lam;
742 lam->set(dbg.sym);
743 if (external) first->make_external();
744 }
745
746 dom_p->bind(scopes_, lam_var);
747
748 if (accept(Tok::Tag::T_at)) {
749 if (accept(Tok::Tag::T_at)) {
750 filter = world().lit_tt();
751 } else {
752 expect(Tok::Tag::D_paren_l, "opening parenthesis of a filter");
753 filter = parse_expr("filter");
754 expect(Tok::Tag::D_paren_r, "closing parenthesis of a filter");
755 }
756 }
757 lam->set_filter(filter);
758
759 funs.emplace_back(std::pair(pi, lam));
760 } while (!ahead().isa(Tok::Tag::T_arrow) && !ahead().isa(Tok::Tag::T_assign) &&
761 !ahead().isa(Tok::Tag::T_semicolon));
762
763 auto codom = is_cn ? world().type_bot()
764 : accept(Tok::Tag::T_arrow) ? parse_expr("return type of a lambda", Tok::Prec::Arrow)
765 : world().mut_infer_type();
766 for (auto [pi, lam] : funs | std::ranges::views::reverse) {
767 // First, connect old codom to lam. Otherwise, scope will not find it.
768 pi->set_codom(codom);
769 Scope scope(lam);
770 ScopeRewriter rw(world(), scope);
771 rw.map(lam->var(), pi->var()->set(lam->var()->dbg()));
772
773 // Now update.
774 codom = rw.rewrite(codom);
775 pi->set_codom(codom);
776
777 codom = pi;
778 }
779
780 scopes_.bind(outer, dbg, first);
781
782 auto body = accept(Tok::Tag::T_assign) ? parse_decls("body of a lambda") : nullptr;
783 if (!body) {
784 if (!decl) error(prev(), "body of a lambda expression is mandatory");
785 // TODO error message if filter is non .ff
786 funs.back().second->unset(0);
787 }
788
789 for (auto [_, lam] : funs | std::ranges::views::reverse) {
790 lam->set_body(body);
791 body = lam;
792 }
793
794 if (decl) expect(Tok::Tag::T_semicolon, "end of lambda");
795
796 first->set(track.loc());
797
798 scopes_.pop();
799 return first;
800}

802void Parser::parse_def(Dbg dbg /*= {}*/) {
803 if (!dbg.sym) {
804 eat(Tok::Tag::K_def);
805 dbg = parse_sym("mutable definition");
806 }
807
808 auto mut = scopes_.find(dbg)->as_mut();
809 expect(Tok::Tag::T_assign, "mutable definition");
810
811 size_t i = mut->isa<Arr, Pi>() ? 1 : 0; // first dependend op
812 size_t n = mut->num_ops();
813
814 if (ahead().isa(Tok::Tag::D_brace_l)) {
815 scopes_.push();
816 parse_list("mutable definition", Tok::Tag::D_brace_l, [&]() {
817 if (i == n) error(prev(), "too many operands");
818 mut->set(i++, parse_decls("operand of a mutable"));
819 });
820 scopes_.pop();
821 } else if (n - i == 1) {
822 mut->set(i, parse_decls("operand of a mutable"));
823 } else {
824 error(prev(), "expected operands for mutable definition");
825 }
826
827 expect(Tok::Tag::T_semicolon, "end of a mutable definition");
828}
