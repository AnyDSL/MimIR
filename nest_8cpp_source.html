33void Nest::populate() {
34 std::queue<Node*> queue;
35
36 if (root()->mut())
37 queue.push(root_);
38 else
39 for (auto child : root_->children().nodes())
40 queue.push(child);
41
42 while (!queue.empty()) {
43 auto curr_node = pop(queue);
44 for (auto op : curr_node->mut()->deps()) {
45 for (auto local_mut : op->local_muts()) {
46 if ((*this)[local_mut] || !contains(local_mut)) continue;
47
48 if (curr_node->level() < local_mut->free_vars().size()) {
49 for (auto node = curr_node;; node = node->inest_) {
50 if (auto var = node->mut()->has_var()) {
51 if (local_mut->free_vars().contains(var)) {
52 queue.push(make_node(local_mut, node));
53 break;
54 }
55 }
56 }
57 } else {
58 uint32_t max = 0;
59 auto inest = root_;
60 for (auto var : local_mut->free_vars()) {
61 if (auto node = (*this)[var->mut()]; node && node->level() > max) {
62 max = node->level();
63 inest = node;
64 }
65 }
66 queue.push(make_node(local_mut, inest));
67 }
68 }
69 }
70 }

83void Nest::assign_postorder_numbers() const {
84 if (root()->postorder_number_.has_value()) return;
85
86 auto number = 0;
87
88 std::function<void(const Nest::Node*)> visit = [&](const Nest::Node* node) {
89 if (node->postorder_number_.has_value()) return; // already visited
90 node->postorder_number_ = 0; // mark in progress
91 for (auto op : node->mut()->deps()) {
92 for (auto mut : op->local_muts())
93 if (auto succ = node->nest()[mut]) visit(succ);
94 }
95 node->postorder_number_ = ++number;
96 };
97
98 if (root()->mut()) {
99 // not virtual root, visit
100 visit(root());
101 } else {
102 // virtual root, visit children
103 for (auto [_, node] : root()->children())
104 visit(node);
105 root()->postorder_number_ = ++number;
106 }

109template<bool bootstrapping>
110const Nest::Node* Nest::lca(const Node* n, const Node* m) {
111 while (n != m) {
112 // Nest::lca is also used within with_dominance and should not call it recursively
113 if constexpr (!bootstrapping) {
114 n->calc_dominance();
115 m->calc_dominance();
116 }
117 if (n->postorder_number_ < m->postorder_number_)
118 n = n->idom_ ? n->idom_ : n->inest();
119 else if (m->postorder_number_ < n->postorder_number_)
120 m = m->idom_ ? m->idom_ : m->inest();
121 }

123 return n;

165uint32_t Nest::Node::tarjan(uint32_t i, Node* inest, Stack& stack) {
166 this->idx_ = this->low_ = i++;
167 this->on_stack_ = true;
168 stack.emplace(this);
169
170 for (auto dep : this->sibl_deps_.nodes_) {
171 if (dep->idx_ == Unvisited) i = dep->tarjan(i, inest, stack);
172 if (dep->on_stack_) this->low_ = std::min(this->low_, dep->low_);
173 }
174
175 if (this->idx_ == this->low_) {
176 inest_->topo_.emplace_front(std::make_unique<SCC>());
177 SCC* scc = inest_->topo_.front().get();
178 Node* node;
179 int num = 0;
180 do {
181 node = pop(stack);
182 node->on_stack_ = false;
183 node->recursive_ = true;
184 node->low_ = this->idx_;
185 ++num;
186
187 scc->emplace(node);
188 auto [_, ins] = inest_->SCCs_.emplace(node, scc);
189 assert_unused(ins);
190 } while (node != this);
191
192 if (num == 1 && !this->sibl_deps().contains(this)) this->recursive_ = false;
193 }
194
195 return i;
196}

198/// Calculates dominance using Cooper-Harvey-Kennedy algorithm
199/// from Cooper et al, "A Simple, Fast Dominance Algorithm".
200/// https://www.clear.rice.edu/comp512/Lectures/Papers/TR06-33870-Dom.pdf
201const Nest::Node* Nest::Node::calc_dominance() const {
202 if (idom_ || is_root() || !inest()->mut()) return this;
203 nest().assign_postorder_numbers();
204
205 if (!inest()->mut()) idom_ = inest();
206
207 // Holds all siblings in reverse post-order coming from the parent
208 absl::flat_hash_set<const Node*> visited;
209 Vector<const Node*> nodes;
210
211 // Initialize entry nodes directly referenced by the parent
212 for (auto op : inest()->mut()->deps()) {
213 for (auto local_mut : op->local_muts())
214 if (auto node = nest()[local_mut]; node && node->inest() == inest()) node->idom_ = inest();
215 }
216
217 std::function<void(const Node*)> visit = [&](const Node* node) {
218 if (visited.contains(node)) return; // already visited
219 visited.insert(node);
220 for (auto child : node->sibl_deps())
221 visit(child);
222 nodes.push_back(node);
223 };
224
225 // Traverse siblings in postorder
226 for (auto op : inest()->mut()->deps()) {
227 for (auto mut : op->local_muts())
228 if (auto node = nest()[mut]; node && node->inest() == inest()) visit(node);
229 }
230
231 // Actual dominance algorithm
232 for (bool todo = true; todo;) {
233 todo = false;
234 for (auto node : nodes | reverse) {
235 // skip entry nodes
236 if (node->idom_ == inest()) continue;
237
238 const Node* new_idom = nullptr;
239 for (auto user : node->sibl_deps<false>())
240 if (user->idom_) new_idom = new_idom ? Nest::lca<true>(new_idom, user) : user;
241 if (node->idom_ != new_idom) {
242 node->idom_ = new_idom;
243 todo = true;
244 }
245 }
246 }
247
248 return this;
249}
