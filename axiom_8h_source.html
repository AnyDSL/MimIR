28 /// @name Normalization
29 /// @anchor normalization
30 ///@{
31 /// For a curried App of an Axiom, you only want to trigger normalization at specific spots.
32 /// For this reason, Thorin maintains a Def::curry_ counter that each App decrements.
33 /// The Axiom::normalizer() will be triggered when Axiom::curry() becomes `0`.
34 /// These are also the spots that you can thorin::match/thorin::force/Match.
35 /// After that, the counter will be set to Axiom::trip().
36 /// E.g., let's say an Axiom has this type:
37 /// ```
38 /// A -> B -> C -> D -> E
39 /// ^ |
40 /// | |
41 /// +---------+
42 /// ```
43 /// Using an initial value as `5` for Axiom::curry and `3` as Axiom::trip has the effect that here
44 /// ```
45 /// x a b c1 d1 e1 c2 d2 e2 c3 d3 e3
46 /// ```
47 /// the Axiom::normalizer will be triggered after App'ing `e1`, `e2`, and `e3`.
48 NormalizeFn normalizer() const { return normalizer_; }
49 u8 curry() const { return curry_; }
50 u8 trip() const { return trip_; }

60 /// @name Axiom Name
61 /// @anchor anatomy
62 ///@{
63 /// Anatomy of an Axiom name:
64 /// ```
65 /// %plugin.tag.sub
66 /// | 48 | 8 | 8 | <-- Number of bits per field.
67 /// ```
68 /// * Def::name() retrieves the full name as Sym.
69 /// * Def::flags() retrieves the full name as Axiom::mangle%d 64-bit integer.
70
71 /// Yields the `plugin` part of the name as integer.
72 /// It consists of 48 relevant bits that are returned in the highest 6 bytes of a 64-bit integer.
73 plugin_t plugin() const { return flags() & Global_Plugin; }
74
75 /// Yields the `tag` part of the name as integer.
76 tag_t tag() const { return tag_t((flags() & 0x0000'0000'0000'ff00_u64) >> 8_u64); }
77
78 /// Yields the `sub` part of the name as integer.

90 /// Mangles @p s into a dense 48-bit representation.
91 /// The layout is as follows:
92 /// ```
93 /// |---7--||---6--||---5--||---4--||---3--||---2--||---1--||---0--|
94 /// 7654321076543210765432107654321076543210765432107654321076543210
95 /// Char67Char66Char65Char64Char63Char62Char61Char60|---reserved---|
96 /// ```
97 /// The `reserved` part is used for the Axiom::tag and the Axiom::sub.
98 /// Each `Char6x` is 6-bit wide and hence a plugin name has at most Axiom::Max_Plugin_Size = 8 chars.
99 /// It uses this encoding:
100 /// | `Char6` | ASCII |
101 /// |---------|---------|
102 /// | 1: | `_` |
103 /// | 2-27: | `a`-`z` |
104 /// | 28-53: | `A`-`Z` |
105 /// | 54-63: | `0`-`9` |
106 /// The 0 is special and marks the end of the name if the name has less than 8 chars.
107 /// @returns `std::nullopt` if encoding is not possible.
108 static std::optional<plugin_t> mangle(Sym s);

177/// @name match/force

180template<class Id, bool DynCast = true>
181auto match(Ref def) {
182 using D = typename Axiom::Match<Id>::type;
183 auto [axiom, curry, _] = Axiom::get(def);
184 bool cond = axiom && curry == 0 && axiom->base() == Axiom::Base<Id>;
185
186 if constexpr (DynCast) return cond ? Match<Id, D>(axiom, def->as<D>()) : Match<Id, D>();
