13// * Inline: These Defs are *always* displayed with all of its operands "inline".

47/// This is a wrapper to dump a Def "inline" and print it with all of its operands.
48struct Inline {
49 Inline(const Def* def, int dump_gid)
50 : def_(def)
51 , dump_gid_(dump_gid) {}
52 Inline(const Def* def)
53 : Inline(def, def->world().flags().dump_gid) {}
54
55 const Def* operator->() const { return def_; };
56 const Def* operator*() const { return def_; };
57 explicit operator bool() const {
58 if (def_->dep_const()) return true;
59
60 if (auto nom = def_->isa_nom()) {
61 if (isa_decl(nom)) return false;
62 return true;
63 }
64
65 if (auto app = def_->isa<App>()) {
66 if (app->type()->isa<Pi>()) return true; // curried apps are printed inline
67 if (app->type()->isa<Type>()) return true;
68 if (app->callee()->isa<Axiom>()) { return app->callee_type()->num_doms() <= 1; }
69 return false;
70 }
71
72 return true;
73 }

209 auto nom_prefix = [&](const Def* def) {

211 if (def->isa<Arr>()) return ".Arr";
212 if (def->isa<Pack>()) return ".pack";
213 if (def->isa<Pi>()) return ".Pi";

304void Dumper::recurse(const Def* def, bool first /*= false*/) {
305 if (auto nom = isa_decl(def)) {
306 if (!dep) noms.push(nom);
307 return;
308 }

310 if (!defs.emplace(def).second) return;

332std::ostream& Def::stream(std::ostream& os, int max) const {
333 auto freezer = World::Freezer(world());
