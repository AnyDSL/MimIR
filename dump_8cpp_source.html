12// During dumping, we classify Defs according to the following logic:
13// * Inline: These Defs are *always* displayed with all of its operands "inline".

56 if (def_->dep_const()) return true;

307void Dumper::recurse(const Def* def, bool first /*= false*/) {
308 if (auto mut = isa_decl(def)) {
309 if (!dep) muts.push(mut);
310 return;
311 }
312
313 if (!defs.emplace(def).second) return;
314
315 for (auto op : def->partial_ops()) {

317 recurse(op);
318 }

320 if (!first && !Inline(def)) dump_let(def);

341std::ostream& Def::stream(std::ostream& os, int max) const {
342 auto freezer = World::Freezer(world());
343 auto dumper = Dumper(os);
344
345 if (max == 0) {
346 os << this << std::endl;
347 } else if (auto mut = isa_decl(this)) {
348 dumper.muts.push(mut);
349 } else {
350 dumper.recurse(this);
351 dumper.tab.print(os, "{}\n", Inline(this));

353 }

355 for (; !dumper.muts.empty() && max > 0; --max) dumper.dump(dumper.muts.pop());
