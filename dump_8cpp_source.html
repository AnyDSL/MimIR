306void Dumper::recurse(const Def* def, bool first /*= false*/) {
307 if (auto mut = isa_decl(def)) {
308 if (!dep) muts.push(mut);
309 return;
310 }
311
312 if (!defs.emplace(def).second) return;
313
314 for (auto op : def->partial_ops()) {
315 if (!op) continue;
316 recurse(op);
317 }
318
319 if (!first && !Inline(def)) dump_let(def);
320}

341std::ostream& Def::stream(std::ostream& os, int max) const {
342 auto freezer = World::Freezer(world());
343 auto dumper = Dumper(os);

345 if (max == 0) {
346 os << this << std::endl;
