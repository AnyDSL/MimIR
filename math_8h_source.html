11enum Mode : nat_t {
12 none = 0,
13 nnan = 1 << 0, ///< No NaNs.
14 ///< Allow optimizations to assume the arguments and result are not NaN.
15 ///< Such optimizations are required to retain defined behavior over NaNs,
16 ///< but the value of the result is undefined.
17 ninf = 1 << 1, ///< No Infs.
18 ///> Allow optimizations to assume the arguments and result are not +/-Inf.
19 ///< Such optimizations are required to retain defined behavior over +/-Inf,
20 ///< but the value of the result is undefined.
21 nsz = 1 << 2, ///< No Signed Zeros.
22 ///< Allow optimizations to treat the sign of a zero argument or result as insignificant.
23 arcp = 1 << 3, ///< Allow Reciprocal.
24 ///< Allow optimizations to use the reciprocal of an argument rather than perform division.
25 contract = 1 << 4, ///< Allow floating-point contraction.
26 ///< (e.g. fusing a multiply followed by an addition into a fused multiply-and-add).
27 afn = 1 << 5, ///< Approximate functions.
28 ///< Allow substitution of approximate calculations for functions (sin, log, sqrt, etc).
29 reassoc = 1 << 6, ///< Allow reassociation transformations for floating-point operations.
30 ///< This may dramatically change results in floating point.
31 finite = nnan | ninf,
32 unsafe = nsz | arcp | reassoc,
33 fast = nnan | ninf | nsz | arcp | contract | afn | reassoc,
34 bot = fast,
35 top = none,
36};

139inline auto match_f(const Def* def) {
140 if (auto f_ty = match<F>(def)) {
141 if (auto [p, e] = f_ty->arg()->projs<2>(isa_lit<nat_t>); p && e && *p == P && *e == E) return f_ty;

148inline std::optional<nat_t> isa_f(const Def* def) {

180template<class R>
181const Lit* lit_f(World& w, R val, const Def* dbg = {}) {
182 static_assert(std::is_floating_point<R>() || std::is_same<R, f16>());
183 if constexpr (false) {}
184 else if constexpr (sizeof(R) == 2) return w.lit(type_f16(w), thorin::bitcast<u16>(val), dbg);
185 else if constexpr (sizeof(R) == 4) return w.lit(type_f32(w), thorin::bitcast<u32>(val), dbg);
186 else if constexpr (sizeof(R) == 8) return w.lit(type_f64(w), thorin::bitcast<u64>(val), dbg);
187 else unreachable();
188}

191 switch (width) {

193 case 32: assert(f64(f32( (val))) == val && "loosing precision"); return lit_f(w, f32( (val)), dbg);
