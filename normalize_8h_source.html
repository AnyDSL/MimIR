97template<class Op, Op op, bool isaWrap = std::is_same_v<Op, Wrap>>
98static const Def* fold(World& world, const Def* type, const App* callee, const Def*& a, const Def*& b, const Def* dbg) {
99 static constexpr int min_w = std::is_same_v<Op, ROp> || std::is_same_v<Op, RCmp> ? 16 : 1;
100 auto la = a->isa<Lit>(), lb = b->isa<Lit>();
101
102 if (a->isa<Bot>() || b->isa<Bot>()) return world.bot(type, dbg);
103
104 if (la && lb) {
105 nat_t width;
106 [[maybe_unused]] bool nsw = false, nuw = false;
107 if constexpr (std::is_same_v<Op, Wrap>) {
108 auto [mode, w] = callee->args<2>(as_lit<nat_t>);
109 nsw = mode & WMode::nsw;
110 nuw = mode & WMode::nuw;
111 width = w;
112 } else {
113 width = as_lit(a->type()->as<App>()->arg());
114 }
115
116 if (is_int<Op>()) width = *mod2width(width);
117
118 Res res;
119 switch (width) {
120#define CODE(i) \
121 case i: \
122 if constexpr (i >= min_w) { \
123 if constexpr (isaWrap) \
124 res = Fold<Op, op, i>::run(la->get(), lb->get(), nsw, nuw); \
125 else \
126 res = Fold<Op, op, i>::run(la->get(), lb->get()); \
127 } \
128 break;
129 THORIN_1_8_16_32_64(CODE)
130#undef CODE
131 default: unreachable();
132 }
133
134 return res ? world.lit(type, *res, dbg) : world.bot(type, dbg);
135 }

137 commute(op, a, b);
138 return nullptr;
