343void AxiomDecl::emit(Emitter& e) const {
344 mim_type_ = type()->emit(e);
345 auto& id = annex_->id;
346
347 std::tie(id.curry, id.trip) = Axiom::infer_curry_and_trip(mim_type_);
348 if (curry_) {
349 if (curry_.lit_u() > id.curry)
350 error(curry_.loc(), "curry counter cannot be greater than {}", id.curry);
351 else
352 id.curry = curry_.lit_u();
353 }
354
355 if (trip_) {
356 if (trip_.lit_u() > id.curry)
357 error(trip_.loc(), "trip counter cannot be greater than curry counter '{}'", (int)id.curry);
358 else
359 id.trip = trip_.lit_u();
360 }
361
362 auto pi = mim_type_->isa<Pi>();
363 if (!annex_->pi)
364 annex_->pi = pi;
365 else if (bool(pi) ^ bool(*annex_->pi))
366 error(dbg().loc(), "all declarations of annex '{}' have to be function types if any is", dbg().sym());
367
368 if (num_subs() == 0) {
369 auto norm = e.driver().normalizer(id.plugin, id.tag, 0);
370 auto axiom = e.world().axiom(norm, id.curry, id.trip, mim_type_, id.plugin, id.tag, 0)->set(dbg());
371 def_ = axiom;
372 e.world().register_annex(id.plugin, id.tag, 0, axiom);
373 } else {
374 sub_t offset = annex_->subs.size();
375 for (sub_t i = 0, n = num_subs(); i != n; ++i) {
376 auto& aliases = annex_->subs.emplace_back(std::deque<Sym>());
377 sub_t s = i + offset;
378 auto norm = e.driver().normalizer(id.plugin, id.tag, s);
379 auto name = e.world().sym(dbg().sym().str() + "."s + sub(i).front()->dbg().sym().str());
380 auto axiom = e.world().axiom(norm, id.curry, id.trip, mim_type_, id.plugin, id.tag, s)->set(name);
381 e.world().register_annex(id.plugin, id.tag, s, axiom);
382
383 for (const auto& alias : sub(i)) {
384 alias->def_ = axiom;
385 aliases.emplace_back(alias->dbg().sym());
386 }
387 }
388 }

428void LamDecl::emit_decl(Emitter& e) const {
429 auto _ = e.world().push(loc());
430 bool is_cps = tag_ == Tag::K_cn || tag_ == Tag::K_con || tag_ == Tag::K_fn || tag_ == Tag::K_fun;
431
432 // Iterate over all doms: Build a Lam for cur dom, by first building a curried Pi for the remaining doms.
433 for (size_t i = 0, n = num_doms(); i != n; ++i) {
434 for (const auto& dom : doms() | std::ranges::views::drop(i)) dom->emit_type(e);
435 auto cod = codom() ? codom()->emit(e) : is_cps ? e.world().type_bot() : e.world().mut_infer_type();
436
437 for (const auto& dom : doms() | std::ranges::views::drop(i) | std::ranges::views::reverse) {
438 dom->pi_->set_codom(cod);
439 cod = dom->pi_;
440 }
441
442 auto cur = dom(i);
443 auto lam = cur->emit_value(e);
444 auto filter = cur->filter() ? cur->filter()->emit(e)
445 : i + 1 == n && is_cps ? e.world().lit_ff()
446 : e.world().lit_tt();
447 lam->set_filter(filter);
448
449 if (i == 0)
450 def_ = lam->set(dbg().sym());
451 else
452 dom(i - 1)->lam_->set_body(lam);
453 }
454}
