77const Def* LowerTypedClos::rewrite(const Def* def) {
78 switch (def->node()) {
79 case Node::Bot:
80 case Node::Top:
81 case Node::Type:
82 case Node::Univ:
83 case Node::Nat: return def;
84 }
85
86 auto& w = world();
87
88 if (auto i = old2new_.find(def); i != old2new_.end()) return i->second;
89 if (auto var = def->isa<Var>();
90 var && var->mut()->isa_mut<Lam>()) // TODO put this conditions inside the assert below
91 assert(false && "Lam vars should appear in a map!");
92
93 auto new_type = rewrite(def->type());
94
95 if (auto ct = isa_clos_type(def)) {
96 auto pi = rewrite(ct->op(1))->as<Pi>();
97 if (Pi::isa_basicblock(pi)) pi = w.cn(insert_ret(pi->dom(), dummy_ret_->type()));
98 auto env_type = rewrite(ct->op(2));
99 return map(def, w.sigma({pi, env_type}));
100 } else if (auto proj = def->isa<Extract>()) {
101 auto tuple = proj->tuple();
102 auto idx = Lit::isa(proj->index());
103 if (isa_clos_type(tuple->type())) {
104 assert(idx && idx <= 2 && "unknown proj from closure tuple");
105 if (*idx == 0)
106 return map(def, env_type());
107 else
108 return map(def, rewrite(tuple)->proj(*idx - 1));
109 } else if (auto var = tuple->isa<Var>(); var && isa_clos_type(var->mut())) {
110 assert(false && "proj fst type form closure type");
111 }
112 }
113
114 if (auto c = isa_clos_lit(def)) {
115 auto env = rewrite(c.env());
116 auto mode = (env->type()->isa<Idx>() || match<mem::Ptr>(env->type())) ? Unbox : Box;
117 const Def* fn = make_stub(c.fnc_as_lam(), mode, true);
118 if (env->type() == w.sigma()) {
119 // Optimize empty env
120 env = w.bot(env_type());
121 } else if (!mode) {
122 auto mem_ptr = (c.get() == attr::esc) ? mem::op_alloc(env->type(), lcm_) : mem::op_slot(env->type(), lcm_);
123 auto mem = w.extract(mem_ptr, 0_u64);
124 auto env_ptr = mem_ptr->proj(1_u64); //, w.dbg(fn->sym() + "_env"));
125 lcm_ = w.call<mem::store>(Defs{mem, env_ptr, env});
126 map(lvm_, lcm_);
127 env = env_ptr;
128 }
129 fn = w.call<core::bitcast>(new_type->op(0), fn);
130 env = w.call<core::bitcast>(new_type->op(1), env);
131 return map(def, w.tuple({fn, env}));
132 } else if (auto lam = def->isa_mut<Lam>()) {
133 return make_stub(lam, No_Env, false);
134 } else if (auto mut = def->isa_mut()) {
135 assert(!isa_clos_type(mut));
136 auto new_mut = mut->stub(w, new_type);
137 map(mut, new_mut);
138 for (size_t i = 0; i < mut->num_ops(); i++)
139 if (mut->op(i)) new_mut->set(i, rewrite(mut->op(i)));
140 if (!def->isa_mut<Global>() && Check::alpha(mut, new_mut)) return map(mut, mut);
141 if (auto imm = new_mut->immutabilize()) return map(mut, imm);
142 return new_mut;
143 } else if (def->isa<Axiom>()) {
144 return def;
145 } else {
146 auto new_ops = Array<const Def*>(def->num_ops(), [&](auto i) { return rewrite(def->op(i)); });
147
148 if (auto app = def->isa<App>()) {
149 // Add dummy retcont to first-class BB
150 if (auto p = app->callee()->isa<Extract>();
151 p && isa_clos_type(p->tuple()->type()) && Pi::isa_basicblock(app->callee_type()))
152 new_ops[1] = insert_ret(new_ops[1], dummy_ret_);
153 }
154
155 auto new_def = def->rebuild(w, new_type, new_ops);
156
157 // We may need to update the mem token after all ops have been rewritten:
158 // F (m, a1, ..., (env, f):pct)
159 // ~>
160 // let [m', env_ptr] = :alloc T m'
161 // let m'' = :store env_ptr env
162 // F (m, a1', ..., (env_ptr, f'))
163 // ~>
164 // let ...
165 // F (m'', a1', ..., (env_ptr, f'))
166 for (size_t i = 0; i < new_def->num_ops(); i++)
167 if (new_def->op(i)->type() == w.annex<mem::M>()) new_def = new_def->refine(i, lcm_);
168
169 if (new_type == w.annex<mem::M>()) { // :store
170 lcm_ = new_def;
171 lvm_ = def;
172 } else if (new_type->isa<Sigma>()) { // :alloc, :slot, ...
173 for (size_t i = 0; i < new_type->num_ops(); i++) {
174 if (new_type->op(i) == w.annex<mem::M>()) {
175 lcm_ = w.extract(new_def, i);
176 lvm_ = w.extract(def, i);
177 break;
178 }
179 }
180 }
181
182 return map(def, new_def);
183 }
