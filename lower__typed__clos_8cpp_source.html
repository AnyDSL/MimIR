9void LowerTypedClos::start() {
10 auto externals = std::vector(world().externals().begin(), world().externals().end());
11 for (auto [_, n] : externals) rewrite(n);
12 while (!worklist_.empty()) {
13 auto [lvm, lcm, lam] = worklist_.front();
14 worklist_.pop();
15 lcm_ = lcm;
16 lvm_ = lvm;
17 world().DLOG("in {} (lvm={}, lcm={})", lam, lvm_, lcm_);
18 if (lam->is_set()) {
19 lam->set_body(rewrite(lam->body()));
20 lam->set_filter(rewrite(lam->filter()));
21 }
22 }
23}

25static const Def* insert_ret(const Def* def, const Def* ret) {
26 auto new_ops = DefArray(def->num_projs() + 1, [&](auto i) { return (i == def->num_projs()) ? ret : def->proj(i); });
27 auto& w = def->world();
28 return (def->sort() == Sort::Term) ? w.tuple(new_ops) : w.sigma(new_ops);

76const Def* LowerTypedClos::rewrite(const Def* def) {
77 switch (def->node()) {
78 case Node::Bot:
79 case Node::Top:
80 case Node::Type:
81 case Node::Univ:
82 case Node::Nat: return def;
83 }
84
85 auto& w = world();
86
87 if (auto i = old2new_.find(def); i != old2new_.end()) return i->second;
88 if (auto var = def->isa<Var>();
89 var && var->nom()->isa_nom<Lam>()) // TODO put this conditions inside the assert below
90 assert(false && "Lam vars should appear in a map!");
91
92 auto new_type = rewrite(def->type());
93 auto new_dbg = def->dbg() ? rewrite(def->dbg()) : nullptr;
94
95 if (auto ct = isa_clos_type(def)) {
96 auto pi = rewrite(ct->op(1))->as<Pi>();
97 if (pi->is_basicblock()) pi = w.cn(insert_ret(pi->dom(), dummy_ret_->type()));
98 auto env_type = rewrite(ct->op(2));
99 return map(def, w.sigma({pi, env_type}));
100 } else if (auto proj = def->isa<Extract>()) {
101 auto tuple = proj->tuple();
102 auto idx = isa_lit(proj->index());
103 if (isa_clos_type(tuple->type())) {
104 assert(idx && idx <= 2 && "unknown proj from closure tuple");
105 if (*idx == 0)
106 return map(def, env_type());
107 else
108 return map(def, rewrite(tuple)->proj(*idx - 1));
109 } else if (auto var = tuple->isa<Var>(); var && isa_clos_type(var->nom())) {
110 assert(false && "proj fst type form closure type");
111 }
112 }
113
114 if (auto c = isa_clos_lit(def)) {
115 auto env = rewrite(c.env());
116 auto mode = (env->type()->isa<Idx>() || isa<Tag::Ptr>(env->type())) ? Unbox : Box;
117 const Def* fn = make_stub(c.fnc_as_lam(), mode, true);
118 if (env->type() == w.sigma()) {
119 // Optimize empty env
120 env = w.bot(env_type());
121 } else if (!mode) {
122 auto mem_ptr = (c.get() == clos::esc) ? mem::op_alloc(env->type(), lcm_) : mem::op_slot(env->type(), lcm_);
123 auto mem = w.extract(mem_ptr, 0_u64);
124 auto env_ptr = mem_ptr->proj(1_u64, w.dbg(fn->name() + "_env"));
125 lcm_ = mem::op_store(mem, env_ptr, env);
126 map(lvm_, lcm_);
127 env = env_ptr;
128 }
129 fn = core::op_bitcast(new_type->op(0), fn);
130 env = core::op_bitcast(new_type->op(1), env);
131 return map(def, w.tuple({fn, env}));
132 } else if (auto lam = def->isa_nom<Lam>()) {
133 return make_stub(lam, No_Env, false);
134 } else if (auto nom = def->isa_nom()) {
135 assert(!isa_clos_type(nom));
136 auto new_nom = nom->stub(w, new_type, new_dbg);
137 map(nom, new_nom);
138 for (size_t i = 0; i < nom->num_ops(); i++)
139 if (nom->op(i)) new_nom->set(i, rewrite(nom->op(i)));
140 if (!def->isa_nom<Global>() && Checker(w).equiv(nom, new_nom, nullptr)) return map(nom, nom);
141 if (auto restruct = new_nom->restructure()) return map(nom, restruct);
142 return new_nom;
143 } else {
144 auto new_ops = Array<const Def*>(def->num_ops(), [&](auto i) { return rewrite(def->op(i)); });
145
146 if (auto app = def->isa<App>()) {
147 // Add dummy retcont to first-class BB
148 if (auto p = app->callee()->isa<Extract>();
149 p && isa_clos_type(p->tuple()->type()) && app->callee_type()->is_basicblock())
150 new_ops[1] = insert_ret(new_ops[1], dummy_ret_);
151 }
152
153 auto new_def = def->rebuild(w, new_type, new_ops, new_dbg);
154
155 // We may need to update the mem token after all ops have been rewritten:
156 // F (m, a1, ..., (env, f):pct)
157 // ~>
158 // let [m', env_ptr] = :alloc T m'
159 // let m'' = :store env_ptr env
160 // F (m, a1', ..., (env_ptr, f'))
161 // ~>
162 // let ...
163 // F (m'', a1', ..., (env_ptr, f'))
164 for (size_t i = 0; i < new_def->num_ops(); i++)
165 if (new_def->op(i)->type() == mem::type_mem(w)) new_def = new_def->refine(i, lcm_);
166
167 if (new_type == mem::type_mem(w)) { // :store
168 lcm_ = new_def;
169 lvm_ = def;
170 } else if (new_type->isa<Sigma>()) { // :alloc, :slot, ...
171 for (size_t i = 0; i < new_type->num_ops(); i++) {
172 if (new_type->op(i) == mem::type_mem(w)) {
173 lcm_ = w.extract(new_def, i);
174 lvm_ = w.extract(def, i);
175 break;
176 }
177 }
178 }
179
180 return map(def, new_def);
181 }
