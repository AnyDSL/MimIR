74const Def* LowerTypedClos::rewrite(const Def* def) {
75 switch (def->node()) {
76 case Node::Bot:
77 case Node::Top:
78 case Node::Type:
79 case Node::Univ:
80 case Node::Nat: return def;
81 }
82
83 auto& w = world();
84
85 if (auto i = old2new_.find(def); i != old2new_.end()) return i->second;
86 if (auto var = def->isa<Var>();
87 var && var->mut()->isa_mut<Lam>()) // TODO put this conditions inside the assert below
88 assert(false && "Lam vars should appear in a map!");
89
90 auto new_type = rewrite(def->type());
91
92 if (auto ct = isa_clos_type(def)) {
93 auto pi = rewrite(ct->op(1))->as<Pi>();
94 if (Pi::isa_basicblock(pi)) pi = w.cn(insert_ret(pi->dom(), dummy_ret_->type()));
95 auto env_type = rewrite(ct->op(2));
96 return map(def, w.sigma({pi, env_type}));
97 } else if (auto proj = def->isa<Extract>()) {
98 auto tuple = proj->tuple();
99 auto idx = Lit::isa(proj->index());
100 if (isa_clos_type(tuple->type())) {
101 assert(idx && idx <= 2 && "unknown proj from closure tuple");
102 if (*idx == 0)
103 return map(def, env_type());
104 else
105 return map(def, rewrite(tuple)->proj(*idx - 1));
106 } else if (auto var = tuple->isa<Var>(); var && isa_clos_type(var->mut())) {
107 assert(false && "proj fst type form closure type");
108 }
109 }
110
111 if (auto c = isa_clos_lit(def)) {
112 auto env = rewrite(c.env());
113 auto mode = (env->type()->isa<Idx>() || match<mem::Ptr>(env->type())) ? Unbox : Box;
114 const Def* fn = make_stub(c.fnc_as_lam(), mode, true);
115 if (env->type() == w.sigma()) {
116 // Optimize empty env
117 env = w.bot(env_type());
118 } else if (!mode) {
119 auto mem_ptr = (c.get() == attr::esc) ? mem::op_alloc(env->type(), lcm_) : mem::op_slot(env->type(), lcm_);
120 auto mem = w.extract(mem_ptr, 0_u64);
121 auto env_ptr = mem_ptr->proj(1_u64); //, w.dbg(fn->sym() + "_env"));
122 lcm_ = w.call<mem::store>(Defs{mem, env_ptr, env});
123 map(lvm_, lcm_);
124 env = env_ptr;
125 }
126 fn = w.call<core::bitcast>(new_type->op(0), fn);
127 env = w.call<core::bitcast>(new_type->op(1), env);
128 return map(def, w.tuple({fn, env}));
129 } else if (auto lam = def->isa_mut<Lam>()) {
130 return make_stub(lam, No_Env, false);
131 } else if (auto mut = def->isa_mut()) {
132 assert(!isa_clos_type(mut));
133 auto new_mut = mut->stub(w, new_type);
134 map(mut, new_mut);
135 for (size_t i = 0; i < mut->num_ops(); i++)
136 if (mut->op(i)) new_mut->set(i, rewrite(mut->op(i)));
137 if (!def->isa_mut<Global>() && Check::alpha(mut, new_mut)) return map(mut, mut);
138 if (auto imm = new_mut->immutabilize()) return map(mut, imm);
139 return new_mut;
140 } else if (def->isa<Axiom>()) {
141 return def;
142 } else {
143 auto new_ops = Array<const Def*>(def->num_ops(), [&](auto i) { return rewrite(def->op(i)); });
144
145 if (auto app = def->isa<App>()) {
146 // Add dummy retcont to first-class BB
147 if (auto p = app->callee()->isa<Extract>();
148 p && isa_clos_type(p->tuple()->type()) && Pi::isa_basicblock(app->callee_type()))
149 new_ops[1] = insert_ret(new_ops[1], dummy_ret_);
150 }
151
152 auto new_def = def->rebuild(w, new_type, new_ops);
153
154 // We may need to update the mem token after all ops have been rewritten:
155 // F (m, a1, ..., (env, f):pct)
156 // ~>
157 // let [m', env_ptr] = :alloc T m'
158 // let m'' = :store env_ptr env
159 // F (m, a1', ..., (env_ptr, f'))
160 // ~>
161 // let ...
162 // F (m'', a1', ..., (env_ptr, f'))
163 for (size_t i = 0; i < new_def->num_ops(); i++)
164 if (new_def->op(i)->type() == w.annex<mem::M>()) new_def = new_def->refine(i, lcm_);
165
166 if (new_type == w.annex<mem::M>()) { // :store
167 lcm_ = new_def;
168 lvm_ = def;
169 } else if (new_type->isa<Sigma>()) { // :alloc, :slot, ...
170 for (size_t i = 0; i < new_type->num_ops(); i++) {
171 if (new_type->op(i) == w.annex<mem::M>()) {
172 lcm_ = w.extract(new_def, i);
173 lvm_ = w.extract(def, i);
174 break;
175 }
176 }
177 }
178
