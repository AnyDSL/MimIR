9void LowerTypedClos::start() {
10 auto externals = std::vector(world().externals().begin(), world().externals().end());
11 for (auto [_, n] : externals) rewrite(n);
12 while (!worklist_.empty()) {
13 auto [lvm, lcm, lam] = worklist_.front();
14 worklist_.pop();
15 lcm_ = lcm;
16 lvm_ = lvm;
17 world().DLOG("in {} (lvm={}, lcm={})", lam, lvm_, lcm_);
18 if (lam->is_set()) {
19 lam->set_body(rewrite(lam->body()));
20 lam->set_filter(rewrite(lam->filter()));
21 }
22 }
23}

25static const Def* insert_ret(const Def* def, const Def* ret) {

67 }));

75const Def* LowerTypedClos::rewrite(const Def* def) {
76 switch (def->node()) {
77 case Node::Bot:
78 case Node::Top:
79 case Node::Type:
80 case Node::Univ:
81 case Node::Nat: return def;
82 }
83
84 auto& w = world();
85
86 if (auto i = old2new_.find(def); i != old2new_.end()) return i->second;
87 if (auto var = def->isa<Var>();
88 var && var->nom()->isa_nom<Lam>()) // TODO put this conditions inside the assert below
89 assert(false && "Lam vars should appear in a map!");
90
91 auto new_type = rewrite(def->type());
92 auto new_dbg = def->dbg() ? rewrite(def->dbg()) : nullptr;
93
94 if (auto ct = isa_clos_type(def)) {
95 auto pi = rewrite(ct->op(1))->as<Pi>();
96 if (pi->is_basicblock()) pi = w.cn(insert_ret(pi->dom(), dummy_ret_->type()));
97 auto env_type = rewrite(ct->op(2));
98 return map(def, w.sigma({pi, env_type}));
99 } else if (auto proj = def->isa<Extract>()) {
100 auto tuple = proj->tuple();
101 auto idx = isa_lit(proj->index());
102 if (isa_clos_type(tuple->type())) {
103 assert(idx && idx <= 2 && "unknown proj from closure tuple");
104 if (*idx == 0)
105 return map(def, env_type());
106 else
107 return map(def, rewrite(tuple)->proj(*idx - 1));
108 } else if (auto var = tuple->isa<Var>(); var && isa_clos_type(var->nom())) {
109 assert(false && "proj fst type form closure type");
110 }
111 }
112
113 if (auto c = isa_clos_lit(def)) {
114 auto env = rewrite(c.env());
115 auto mode = (env->type()->isa<Idx>() || match<mem::Ptr>(env->type())) ? Unbox : Box;
116 const Def* fn = make_stub(c.fnc_as_lam(), mode, true);
117 if (env->type() == w.sigma()) {
118 // Optimize empty env
119 env = w.bot(env_type());
120 } else if (!mode) {
121 auto mem_ptr = (c.get() == attr::esc) ? mem::op_alloc(env->type(), lcm_) : mem::op_slot(env->type(), lcm_);
122 auto mem = w.extract(mem_ptr, 0_u64);
123 auto env_ptr = mem_ptr->proj(1_u64, w.dbg(fn->name() + "_env"));
124 lcm_ = w.call<mem::store>(Defs{mem, env_ptr, env});
125 map(lvm_, lcm_);
126 env = env_ptr;
127 }
128 fn = core::op_bitcast(new_type->op(0), fn);
129 env = core::op_bitcast(new_type->op(1), env);
130 return map(def, w.tuple({fn, env}));
131 } else if (auto lam = def->isa_nom<Lam>()) {
132 return make_stub(lam, No_Env, false);
133 } else if (auto nom = def->isa_nom()) {
134 assert(!isa_clos_type(nom));
135 auto new_nom = nom->stub(w, new_type, new_dbg);
136 map(nom, new_nom);
137 for (size_t i = 0; i < nom->num_ops(); i++)
138 if (nom->op(i)) new_nom->set(i, rewrite(nom->op(i)));
139 if (!def->isa_nom<Global>() && Checker(w).equiv(nom, new_nom, nullptr)) return map(nom, nom);
140 if (auto restruct = new_nom->restructure()) return map(nom, restruct);
141 return new_nom;
142 } else if (def->isa<Axiom>()) {
143 return def;
144 } else {
145 auto new_ops = Array<const Def*>(def->num_ops(), [&](auto i) { return rewrite(def->op(i)); });
146
147 if (auto app = def->isa<App>()) {
148 // Add dummy retcont to first-class BB
149 if (auto p = app->callee()->isa<Extract>();
150 p && isa_clos_type(p->tuple()->type()) && app->callee_type()->is_basicblock())
151 new_ops[1] = insert_ret(new_ops[1], dummy_ret_);
152 }
153
154 auto new_def = def->rebuild(w, new_type, new_ops, new_dbg);
155
156 // We may need to update the mem token after all ops have been rewritten:
157 // F (m, a1, ..., (env, f):pct)
158 // ~>
159 // let [m', env_ptr] = :alloc T m'
160 // let m'' = :store env_ptr env
161 // F (m, a1', ..., (env_ptr, f'))
162 // ~>
163 // let ...
164 // F (m'', a1', ..., (env_ptr, f'))
165 for (size_t i = 0; i < new_def->num_ops(); i++)
166 if (new_def->op(i)->type() == mem::type_mem(w)) new_def = new_def->refine(i, lcm_);
167
168 if (new_type == mem::type_mem(w)) { // :store
169 lcm_ = new_def;
170 lvm_ = def;
171 } else if (new_type->isa<Sigma>()) { // :alloc, :slot, ...
172 for (size_t i = 0; i < new_type->num_ops(); i++) {
173 if (new_type->op(i) == mem::type_mem(w)) {
174 lcm_ = w.extract(new_def, i);
175 lvm_ = w.extract(def, i);
176 break;
177 }
178 }
179 }
180
181 return map(def, new_def);
182 }
